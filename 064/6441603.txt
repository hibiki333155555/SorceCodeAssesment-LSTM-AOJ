#include<iostream>
#include<vector>
using namespace std;

int main() {
    int Wnum, Tlim;
    long L = 0, U = 1000000000, mid;
    cin >> Wnum >> Tlim;
    vector<int> w(Wnum);
    for (int i = 0; i < Wnum; i++)cin >> w[i];

    int load, Tcnt, NG=0;
    while (U != L) {
        mid = (L + U) / 2;
        load = 0;
        Tcnt = 1;
        NG = 0;
        for (int i = 0; i < Wnum; i++) {
            load += w[i];
            if (w[i] > mid) {
                NG = 1;
                break;
            }else if(load > mid){
                Tcnt++;
                load = w[i];
            }
        }

        if (NG == 1) L = mid+1;
        else if (Tcnt <= Tlim)U = mid;
        else L = mid + 1;
    }
    cout << L << endl;

    return 0;
}
