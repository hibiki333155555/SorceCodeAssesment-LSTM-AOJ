#include <iostream>
using namespace std;

int main() {
    while (true) {
        int h, w;
        cin >> h >> w;
        if (h == w == 0) break;
        for (int i = 0; i < w; i++) {
            for (int j = 0; j < h; j++) {
                cout << "#";
            }
            cout << endl;
        }
        cout << endl;
    }

    return 0;
}
