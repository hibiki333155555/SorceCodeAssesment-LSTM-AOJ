#include <bits/stdc++.h>

using namespace std;

int main() {
    int n,k;
    cin>>n>>k;
    vector<int> a(n);
    map<int,int> m;
    int ans=0;
    for (int i = 0; i < n; ++i) {
        cin>>a[i];
        ans+=a[i];
        ++m[a[i]];
    }
    if(m[1]<=k) {
        ans+=4*m[1];
        k-=m[1];
        m[5]+=m[1];

    }
    else {
        ans+=4*k;
        k=0;
    }
    vector<int> x={2,3,4,5};
    for (int i = 0; i < 4; ++i) {
        int z=x[i];
        if(m[z]<=k) {
            ans+=(6-z)*m[z];
            k-=m[z];
        }
        else {
            ans+=(6-z)*k;
            k=0;
        }
        if(k==0)break;
    }
    cout<<ans<<endl;
}
