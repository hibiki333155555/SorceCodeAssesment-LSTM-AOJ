// F - Counting Characters

#include <iostream>
#include <string>
using namespace std;

int main() {
    string x;
    int count[26]={};
    while (cin >> x) 
    {
        for (char c : x)
        {
            if (c >= 'a' and c <= 'z') 
                count[c-'a']++;
            else if (c >= 'A' and c <= 'Z') 
                count[c-'A']++;
        }
    }

    for (int i=0; i<26; i++)
    {
        char c = 'a' + i;
        cout << c << " : " << count[i] << endl;
    }
}
