import scala.io.StdIn

object Main {
  def main(args: Array[String]): Unit = {
    val Array(l, n) = StdIn.readLine().split(" ").map(_.toInt)
    val point = StdIn
      .readLine()
      .split(" ")
      .map(_.toInt)
      .scanLeft(0)(_ + _)
      .indexWhere(_ >= l)
    println(if (point >= 0) point.toString() else "No")
  }
}
