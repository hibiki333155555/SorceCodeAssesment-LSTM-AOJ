//int N, M, K, T, H, W, L, R;
long long int N, M, K, T, H, W, L, R;

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);

	cin >> N;
	map<int, pair<int, int>>mp;
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			cin >> K;
			if (mp.find(K) == mp.end()) {
				mp[K] = { i + 1,j + 1 };
			}
		}
	}
	cin >> K;
	while (K--) {
		cin >> M;
		if (mp.find(M) == mp.end()) {
			cout << -1 << endl;
		}
		else {
			cout << mp[M].first << " " << mp[M].second << endl;
		}
	}
}
