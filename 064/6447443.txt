#include <stdio.h>
#include <string.h>
int main()
{
    int i, j, k, len;
    char str[1200];
    fgets(str, 1200, stdin);
    len = strlen(str);
    char alph[27] = "abcdefghijklmnopqrstuvwxyz";
    int count[26] = {0};
    // CONVERTING the uppercase letter to lowerCase
    for (i = 0; i < len; i++)
    {
        if (str[i] >= 'A' && str[i] <= 'Z')
        {
            str[i] = str[i] + 32;
        }
    }
    // checking for the matching of alphabets
    for (i = 0; i < len; i++)
    {
        for (j = 0; j < 27; j++)
        {
            if (str[i] == alph[j])
            {
                count[j]++;
            }
        }
    }
    // printing the output
    for (j = 0; j < 25; j++)
    {
        printf("%c : %d\n", alph[j], count[j]);
    }

    // printf("%s", str);
    return 0;
}
