#define rep(i,n) for(int i=0;i<(int)(n);i++)
#define ALL(v) v.begin(),v.end()
typedef long long ll;
 
#include <bits/stdc++.h>
using namespace std;

int main(){
  ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n;
  cin>>n;
  map<int,pair<int,int>> m;
  rep(i,n){
    rep(j,n){
      int a;
      cin>>a;
      if(m.count(a)==0) m[a]={i,j};
    }
  }
  int q;
  cin>>q;
  rep(i,q){
    int t;
    cin>>t;
    if(m.count(t)==0) cout<<-1<<endl;
    else cout<<m[t].first+1<<" "<<m[t].second+1<<endl;
  }
        
  
  
  
  
  return 0;
}
