#include <bits/stdc++.h>
using namespace std;
int main(){
    int i;
    bool b=1;
    while(b){
        int ans;
        int n;
        cin>>n;
        if(n == 0) {
            b = 0;
            break;
        }
        vector<int> v(n);
        for(int i=0;i<n;i++){
            cin>>v[i];
        }
        sort(v.begin(),v.end());

            }
        }
    }
