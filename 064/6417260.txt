#include<bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < (int)(n); i++) //指定した回数だけ処理を繰り返す」という機能です。
using namespace std;

int main()
{
        int h,w;
        bool ishash;
        while(1)
        {
                cin >> h >> w;
                if(h==0&&w == 0)
                        break;
                for(int i=0;i<h;i++)
                {
                        if(i%2 ==0)
                                ishash=true;
                        else ishash=false;
                        //行の描画
                        for(int j = 0;j<w;j++)
                        {
                                if(ishash==true)
                                {
                                        cout << "#";
                                        ishash=false;
                                }
                                else
                                {
                                        cout << ".";
                                        ishash=true;
                                }
                        }
                        cout <<endl;
                }
                cout <<endl;
        }
}
