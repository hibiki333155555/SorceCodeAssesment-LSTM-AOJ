template <typename Cost>
struct Edge
{
  int from, to;
  Cost cost;
  Edge(int s, int t, Cost c = 1) : from(s), to(t), cost(c) {}
  operator int() const { return to; }
};
template <typename Cost>
struct Graph : vector<vector<Edge<Cost>>>
{
  Graph(int n) : vector<vector<Edge<Cost>>>(n) {}
  void add_edge(int s, int t, Cost c = 1) { (*this)[s].emplace_back(s, t, c); }
  void add_edge2(int s, int t, Cost c = 1) { add_edge(s, t, c), add_edge(t, s, c); }
};

template<typename Cost, typename Data>
struct ShortestPath : Graph<Cost>
{
  vector<Cost> dists;
  vector<Data> pathcnt;
  vector<int> prev;
  vector<int> path;

  ShortestPath(int n) : Graph<Cost>::Graph(n) {}

  vector<Cost> bfs(int sv)
  {
    dists.assign((*this).size(), INF);
    pathcnt.assign((*this).size(), 0);
    prev.assign((*this).size(), -1);
    dists[sv] = 0;
    pathcnt[sv] = 1;
    prev[sv] = -2;
    queue<int> que;
    que.emplace(sv);
    while (!que.empty())
    {
      auto v = que.front();
      que.pop();

      for (auto nv : (*this)[v])
      {
        if (chmin(dists[nv], dists[v] + nv.cost))
        {
          pathcnt[nv] = pathcnt[v];
          prev[nv] = v;
          que.emplace(nv);
        }
        else if (dists[nv] == dists[v] + nv.cost)
          pathcnt[nv] += pathcnt[v];
      }
    }
    return dists;
  }

  vector<Cost> _01bfs(int sv, int smallercost = 0)
  {
    dists.assign((*this).size(), INF);
    pathcnt.assign((*this).size(), 0);
    prev.assign((*this).size(), -1);
    dists[sv] = 0;
    pathcnt[sv] = 1;
    prev[sv] = -2;
    deque<int> deq;
    deq.emplace_front(sv);
    while (!deq.empty())
    {
      auto v = deq.front();
      deq.pop_front();

      for (auto nv : (*this)[v])
      {
        if (chmin(dists[nv], dists[v] + nv.cost))
        {
          pathcnt[nv] = pathcnt[v];
          prev[nv] = v;
          if (nv.cost == smallercost)
            deq.emplace_front(nv);
          else
            deq.emplace_back(nv);
        }
        else if (dists[nv] == dists[v] + nv.cost)
          pathcnt[nv] += pathcnt[v];
      }
    }
    return dists;
  }

  vector<Cost> dijkstra(int sv)
  {
    dists.assign((*this).size(), INF);
    pathcnt.assign((*this).size(), 0);
    prev.assign((*this).size(), -1);
    dists[sv] = 0;
    pathcnt[sv] = 1;
    prev[sv] = -2;
    priority_queue<pair<Cost, int>, vector<pair<Cost, int>>, greater<pair<Cost, int>>> pque;
    pque.emplace(make_pair(0, sv));
    while (!pque.empty())
    {
      auto [d, v] = pque.top();
      pque.pop();

      if (dists[v] != d)
        continue;

      for (auto nv : (*this)[v])
      {
        if (chmin(dists[nv], dists[v] + nv.cost))
        {
          pathcnt[nv] = pathcnt[v];
          prev[nv] = v;
          pque.emplace(make_pair(dists[nv], nv));
        }
        else if (dists[nv] == dists[v] + nv.cost)
          pathcnt[nv] += pathcnt[v];
      }
    }
    return dists;
  }

  bool bellman_ford(int sv) // 負の閉路があれば false, なければ true
  {
    dists.assign((*this).size(), INF);
    pathcnt.assign((*this).size(), 0);
    prev.assign((*this).size(), -1);
    dists[sv] = 0;
    pathcnt[sv] = 1;
    prev[sv] = -2;
    for (int i = 0; i < int((*this).size()); i++)
    {
      for (int v = 0; v < int((*this).size()); v++)
      {
        for (auto nv : (*this)[v])
        {
          if (chmin(dists[nv], dists[v] + nv.cost))
          {
            if (i == int((*this).size()) - 1)
              return false;
            pathcnt[nv] = pathcnt[v];
            prev[nv] = v;
          }
          else if (dists[nv] == dists[v] + nv.cost)
            pathcnt[nv] += pathcnt[v];
        }
      }
    }
    return true;
  }

  vector<int> restore_path(int gv)
  {
    path.clear();
    if (prev[gv] == -1)
      return {};

    int v = gv;
    while (prev[v] != -2)
    {
      path.emplace_back(v);
      v = prev[v];
    }
    path.emplace_back(v);
    reverse(path.begin(), path.end());
    return path;
  }
};

int main()
{
  ll N, M, sv;
  cin >> N >> M >> sv;
  ShortestPath<ll, ll> G(N);
  for (ll i = 0; i < M; i++)
  {
    ll s, t, d;
    cin >> s >> t >> d;
    G.add_edge(s, t, d);
  }

  if (!G.bellman_ford(sv))
    FINALANS("NEGATIVE CYCLE");
  for (auto d : G.dists)
  {
    if (d == INF)
      cout << "INF\n";
    else
      cout << d << '\n';
  }
}
