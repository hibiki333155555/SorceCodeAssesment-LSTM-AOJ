void pline(vector<ll> lis){
    rep(i,0,lis.size()){
        printf ("%lld",lis[i]);
        if (i != lis.size()-1) printf(" ");
        else printf("\n");
    }
}

int main(){

    ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    //ModFac mf(200000);

    ll H,W,N;
    cin >> H >> W >> N;

    vector<vector<ll>> lis(H,vector<ll> (W,0));

    ll sx,sy,gx,gy;
    cin >> sx >> sy >> gx >> gy;
    sx--;sy--;gx--;gy--;

    rep(i,0,N){
        ll x,y,k;
        cin >> x >> y >> k;
        x--; y--;
        if (llabs(x-sx) + llabs(y-sy) <= k){
            cout << "No\n";
            return 0;
        }else if (llabs(x-gx) + llabs(y-gy) <= k){
            cout << "No\n";
            return 0;
        }

        rep(xd,-k,k+1){
            ll yd = k - llabs(xd);
            lis[x+xd][y+yd] = 1;
            lis[x+xd][y-yd] = 1;
        }

    }

    list<pair<ll,ll>> q;
    lis[sx][sy] = 2;
    q.push_back(make_pair(sx,sy));

    while (!q.empty()){
        auto xy = q.front();
        q.pop_front();
        ll x = xy.first;
        ll y = xy.second;

        rep(i,0,4){
            ll newx,newy;
            if (i == 0){
                newx = x-1; newy = y;
            }else if (i == 1){
                newx = x+1; newy = y;
            }else if (i == 2){
                newx = x; newy = y+1;
            }else{
                newx = x; newy = y-1;
            }

            if (newx < 0 || H <= newx || newy < 0 || W <= newy) continue;

            if (lis[newx][newy] == 0){
                q.push_back(make_pair(newx,newy));
                lis[newx][newy] = 2;
            }

        }
    }

    if (lis[gx][gy] == 2){
        cout << "Yes\n";
    }else{
        cout << "No\n";
    }

}
