typedef long long ll;
using namespace std;
const int MN = 505;
const int MAXN = 1000005;
const int INF = 0x3f3f3f3f;


int fa[MAXN];
int find(int x) {
	//printf("find:fa[%d]=%d\n", x, fa[x]);
	if (fa[x] == x) {
		return x;
	} else {
		return find(fa[x]);
	}
}

int main() {
	int n, q;
	while (scanf("%d %d", &n, &q)) {
		if (n == 0 && q == 0) {
			break;
		}
		fa[1] = 1;
		for (int i = 2; i <= n; i++) {
			scanf("%d", fa + i);
		}
		getchar();
		int ans = 0;
		while (q--) {
			char ch;
			int x;
			cin >> ch >> x;
			if (ch == 'M') {
				fa[x] = x;
			} else if (ch == 'Q') {
				ans += find(x);
			}
		}
		printf("%d\n", ans);
	}
	return 0;
}
