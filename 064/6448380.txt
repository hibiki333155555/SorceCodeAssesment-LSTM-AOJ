#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    map<char, int> a;
    while(true){
    getline(cin, s);
    if(s.size()==0){
        break;
    }
        for(int i=0;i<s.size();i++){
            if('A'<= s[i] && s[i]<='z'){
                if('A'<=s[i]&&s[i]<='Z'){
                    s[i]+='a'-'A';
                }
                if(a.find(s[i]) == a.end()){
                    a[s[i]] = 1;
                }else{
                    a[s[i]]++;
                }
            }
        }
        for(char i='a'; i<='z';i++){
            cout<<i<<" : "<<a[i]<<"\n";
        }
    }
    return(0);
}
