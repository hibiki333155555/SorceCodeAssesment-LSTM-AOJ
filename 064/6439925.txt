#include <bits/stdc++.h>
#define REP(i, n) for(int i = 0; i < (int)(n); i++)
using namespace std;

int main(void) {
    int n; cin >> n;
    vector<string> report(n);
    REP(i, n) cin >> report[i];
    int m = report[0].length();
    vector<int> cheating(n);
    REP(i, n) {
        REP(j, i) {
            REP(k, m) {
                char c = report[j][k];
                report[j][k] = report[i][k];
                if(report[i] == report[j]) {
                    cheating[i] = cheating[j] = 1;
                }
                report[j][k] = c;
            }
        }
    }
    int ans = 0;
    REP(i, n) ans += cheating[i];
    cout << ans << '\n';
}
