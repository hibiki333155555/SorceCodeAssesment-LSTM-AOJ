import java.util.Scanner;

public class Main {
    void solve() {
        Scanner scanner = new Scanner(System.in);
        int h = scanner.nextInt();
        int r = scanner.nextInt();

        if (h + r == 0) {
            System.out.println(0);
        } else if (h + r > 0) {
            System.out.println(1);
        } else {
            System.out.println(-1);
        }
    }

    public static void main(String[] args) {
        new Main().solve();
    }
}
