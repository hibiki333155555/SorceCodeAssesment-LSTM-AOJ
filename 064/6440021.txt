#include<iostream>
#include<string>
using namespace std;
int main(){
    int a[2001];
    for(int i=0; i<2001; i++){
        a[i]=0;
    }
    int N;
    int A[N];
    cin >> N;
    for(int i=0; i<N; i++){
        cin >> A[i];
        a[A[i]]++;
    }
    int min=101;
    for(int i=0; i<2001; i++){
        if(min>=a[i] && a[i]!=0){
            min=a[i];
        }
    }
    for(int i=0; i<2001; i++){
        if(a[i]==min){
            cout << i << endl;
            return 0;
        }
    }
}
