#include <iostream>
using namespace std;

void printArray(int a[], int n)
{
    for (int i = 0; i < n; ++i) {
        cout << a[i];
	if (i < n - 1)
	    cout << " ";
	else
	    cout << endl;
    }
}

void insertionSort(int A[], int N)
{
    for (int i = 1; i < N; ++i) {   // A[0] < A[1] < ... < A[i - 1]
	int v = A[i];
	/*
	int j;
	for (j = i - 1; j >= 0; --j) {
	    if (A[j] <= v)         // A[j] <= v < A[j + 1]
		break;
	    else   
		A[j + 1] = A[j]; 
	}
	A[j + 1] = v;
	*/
	int j = i - 1;
	while (j >= 0 && v < A[j]) {
	    A[j + 1] = A[j];
	    --j;
	}                          // A[j] <= v < A[j+1]
	A[j + 1] = v;
	    
	printArray(A, N);
    }
}

int main()
{
    int N;
    int A[100];
    
    cin >> N;
    for (int i = 0; i < N; ++i) {
	cin >> A[i];
    }

    printArray(A, N);
    insertionSort(A, N);

    return 0;
}
