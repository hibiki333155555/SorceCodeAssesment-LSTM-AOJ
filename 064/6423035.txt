#include<iostream>
using namespace std;

int main()
{ 
	int a[100][100];//行列Aの要素
	int b[100];//列ベクトルbの要素
	int c[100];
	int n, m;

	cin >> n >> m;

	//行列Aのインプット
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			cin >> a[i][j];
		}
	}

	//列ベクトルbのインプット
	for (int i = 0; i < m; i++) {
		cin >> b[i];
	}
	//列cの初期値を入力
	for (int i = 0; i < n; i++) {
		c[i] = 0;
	}

	//掛け算
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			c[i] += a[i][j] * b[j];
		}
	}

	//行列cのアウトプット
	for (int i = 0; i < n; i++) {
		cout << c[i] << "\n";
	}
}
