#include <stdio.h>

void chmin(int* a, int b)
{
	if (*a > b) *a = b;
}

int main()
{
	int i, M, N;
	char S[102][102];
	scanf("%d %d", &M, &N);
	for (i = 1; i <= M; i++) scanf("%s", &(S[i][1]));
	
	int j, ii, jj, ans = N * M, tmp;
	for (i = 1; i < M; i++) {
		for (ii = 1, tmp = 0; ii <= i; ii++) for (jj = 1; jj <= N; jj++) if (S[ii][jj] == 'o') tmp++;
		for (; ii <= M; ii++) for (jj = 1; jj <= N; jj++) if (S[ii][jj] == 'x') tmp++;
		chmin(&ans, tmp);
		chmin(&ans, N * M - tmp);
	}
	for (j = 1; j < N; j++) {
		for (ii = 1, tmp = 0; ii <= M; ii++) for (jj = 1; jj <= j; jj++) if (S[ii][jj] == 'o') tmp++;
		for (ii = 1; ii <= M; ii++) for (jj = j + 1; jj <= N; jj++) if (S[ii][jj] == 'x') tmp++;
		chmin(&ans, tmp);
		chmin(&ans, N * M - tmp);
	}
	printf("%d\n", ans);
	fflush(stdout);
	return 0;
}
