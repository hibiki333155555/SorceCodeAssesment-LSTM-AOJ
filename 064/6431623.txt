#include<bits/stdc++.h>
using namespace std;
int bubblesort(vector<int>&a){
    int n=a.size();
    int c=0;
    for(int i=0;i<n-1;i++){
        for(int j=0;j<n-1-i;j++){
            if(a[j]>a[j+1]){
                swap(a[j],a[j+1]);
                c++;
            }
        }
    }
    return c;
}

    cin>>n;

}
