#include <iostream>
using namespace std;
void LIS(int arr[], int n) {
    int len = n;
    if (len == 0) {
        //cout << 0 << endl;
        return;
    }
    int L[len];
    int f[len];
    L[0] = 1;
    f[0] = -1;
    f[1] = arr[0];
    for (int i = 1; i < len; i++) {
        if (arr[i] < f[L[i-1]] && arr[i] > f[L[i]-1]) {
            f[L[i]] = arr[i];
            L[i] = L[i-1];
        }
        else if (arr[i] > f[L[i-1]]) {
            L[i] = L[i-1]+1;
            f[L[i]] = arr[i];
        }
        //cout << L[i] << " " << f[L[i]] << endl;
    }
    int max = L[0];
    for (int i = 1; i < len; i++) {
        if (L[i] > max)
            max = L[i];
    }
    cout << max << endl;
}

int main(int argc, char *argv[]) {
    int n;
    int arr[n];
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> arr[i];
    LIS(arr, n);
    return 0;
}
