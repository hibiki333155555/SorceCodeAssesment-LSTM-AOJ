#include <iostream>
#include <string>
using namespace std;

int main()
{
string x;
int count[26] = {0};
cin >> x;
for  (int i = 0; i < x.length(); i++)
  {  
    char c = x[i];
    if (c >= 'a' and c <='z')
    {
      count[c - 'a']++;
    }
    else if (c >= 'A' and c <= 'z')
    {
      count[c - 'A']++;
    }
  }
for (int j = 0; j < 26; j++)
  {
    char c = 'a' + j;
    cout << c << " : " << count[j] << endl;
  }
}
