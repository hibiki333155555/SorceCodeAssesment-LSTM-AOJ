#include <bits/stdc++.h>
using namespace std;
string s;
int a[5] = {1, 5, 10, 25, 50};
int dp[100][1001], v[100], w[100];
int f(int i, int W){
    if(i < 0) return 0;
    if(dp[i][W] != -1) return dp[i][W];
    if(w[i] > W) return dp[i][W] = f(i-1, W);
    return dp[i][W] = max(v[i] + f(i-1, W-w[i]), f(i-1, W));
}
void solve(){
    int n, W;
    cin >> n >> W;
    for(int i = 0;i < n;i++) {
        cin >> v[i] >> w[i];
        fill(dp[i], dp[i]+1001, -1);
    }
    cout << f(n-1, W) << '\n';
}
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t = 1;
    //cin >> t;
    while(t--) solve();
}
