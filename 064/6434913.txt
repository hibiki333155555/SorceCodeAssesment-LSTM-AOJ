while True:
    H, W = map(int,input().split())
    if H == 0 and W == 0:
        break
    for i in range(H):
        for j in range(W):
            if j == 0 and j == W:
                print("#",end = "")
            else:
                print(".",end = "")
            print()
        print()
    print()
