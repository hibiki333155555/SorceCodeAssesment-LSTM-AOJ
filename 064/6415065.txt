#include <bits/stdc++.h>
using namespace std;
int main() {
    int a,b;
    char ch;
    while (true) {
        cin >> a >> ch >> b;
        if (ch == '?') {
            break;
        } else if (ch == '+') {
            cout << a + b << endl;
        } else if (ch == '-') {
            cout << a - b << endl;
        } else if (ch == '-') {
            cout << a - b << endl;
        } else if (ch == '/') {
            cout << a / b << endl;
        } else if (ch == '*') {
            cout << a * b << endl;
        }
    }
    return 0;
}
