#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin >> n;
    int g[n + 1][n + 1];
    for(int i = 0; i <= n; i++){
        for(int j = 0; j <= n; j++){
            g[i][j] = 0;
        }
    }
    for(int i = 1; i <= n; i++){
        int u, k;
        cin >> u >> k;
        for(int j = 0; j < k; j++){
            int v;
            cin >> v;
            g[u][v] = 1;
        }
    }
    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
            if(j > 1){
                cout << " ";
            }
            cout << g[i][j];
        }
        cout << endl;
    }
}
