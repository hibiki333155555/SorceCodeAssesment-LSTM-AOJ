import java.util.*;


class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        List<Integer> list = new ArrayList<>();
        
        for(int i=0;i<n;i++){
            list.add(Integer.valueOf(sc.nextInt()));
        }
        
        for(int j=n-1;j>=0;j--){
            if(j==0) System.out.printf("%d\n",list.get(j));
            else System.out.printf("%d ",list.get(j));
        }
    }
}
