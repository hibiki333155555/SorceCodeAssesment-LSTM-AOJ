#include <iostream>
using namespace std;

int main(){
    int a, b;
    int ans;
    char op;
    cin >> a >> op >> b;
    
   while(op != '?'){
        switch(op){
            case '+': 
                ans = a + b;
                break;
            case '-':
                ans = a - b;
                break;
            case '*':
                ans = a * b;
                break;
            case '/':
                ans = a / b;
                break;
        }
        cout << ans << endl;
    }
    
    return 0;
}
