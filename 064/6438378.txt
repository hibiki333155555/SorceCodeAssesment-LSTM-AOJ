#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

struct P
{
    double x, y;
};

P p0, p1, p2, p3;

void cal()
{
    if ((p1.y - p0.y) * (p3.x - p2.x) == (p3.y - p2.y) * (p1.x - p0.x))
    {
        cout << 2 << endl;
    }
    else if (((p1.x - p0.x) * (p3.x - p2.x) + (p1.y - p0.y) * (p3.y - p2.y)) == 0)
    {
        cout << 1 << endl;
    }
    else
    {
        cout << 0 << endl;
    }
}

signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    int q;
    cin >> q;
    while (q--)
    {
        cin >> p0.x >> p0.y >> p1.x >> p1.y >> p2.x >> p2.y >> p3.x >> p3.y;
        cal();
    }
    return 0;
}
