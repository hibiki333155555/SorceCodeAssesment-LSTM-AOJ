template <typename T>
struct manacher{
	vector<int> A;
	manacher(T &s){
		A.resize(s.size(),-1);
		int i=0,j=0;
		while(i<s.size()) {
			while(i-j>=0&&i+j<s.size()&&s[i-j]==s[i+j])j++;
			A[i] = j;
			int k = 1;
			while(i-k>=0&&i+k<s.size()&&k+A[i-k]<j){
				A[i+k] = A[i-k];
				k++;
			}
			i+=k;
			j-=k;
		}
	}

	
	//posを中心とした回文の半径((回文の長さ+1)/2)
	int get_length(int pos){
		return A[pos];
	}
	
};

int main(){
	int n;
	cin>>n;
	string s;
	cin>>s;
	{
		string cur = "$";
		rep(i,s.size()){
			cur += s[i];
			cur += '$';
		}
		swap(cur,s);
	}
	int ans = 0;
	manacher M(s);
	for(int i=2;i<s.size()-1;i+=2){
		int l = i,r = s.size() - i - 1;
		if(M.get_length(i)-1 == min(l,r))ans++;
	}
	cout<<ans<<endl;
	
	
	return 0;
}
