#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    vector<int>v;
    int q; cin>>q;
    while(q--)
    {
        int x; cin>>x;
        if(x==0)
        {
            int a; cin>>a; v.push_back(a);
        }
        else if(x==1)
        {
            int b; cin>>b;
            cout<<v.at(b)<<endl;
        }
        else if(x==2)
            v.pop_back();
    }
}
