#include <stdio.h>

int main(void){
    int x,y,z,a1,a2,a3,m1,m2,m3,ctr;
    
    while(1){
        scanf("%d %d %d %d %d %d",&a1,&m1,&a2,&m2,&a3,&m3);
        if(a1*a2*a3*m1*m2*m3==0) break;
        
        x=y=z=1;
        ctr=0;
        while(1){
            x=a1*x%m1;
            y=a2*y%m2;
            z=a3*z%m3;
            ctr++;
            
            //printf("%5d %d %d %d\n",ctr,x,y,z);
            if(x*y*z==1) break;
        }
        
        printf("%d\n",ctr);
    }
    
    return 0;
}
