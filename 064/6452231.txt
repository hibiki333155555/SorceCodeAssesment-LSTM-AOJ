#include<bits/stdc++.h>
//#include<atcoder/all>

using namespace std;
//using namespace atcoder;
using ll = long long;

int main() {
    int r1, r2;
    string c1, c2;
    cin >> c1 >> r1 >> c2 >> r2;

    auto ch_to_num = [&](string s) {
        int ret = 0;
        if ((int)s.size() == 2) ret += (s[0] - 'A' + 1) * 26;
        ret += (s.back() - 'A' + 1);
        return ret;
    };

    int c = r2 - r1 + 1, r = ch_to_num(c2) - ch_to_num(c1) + 1;

    cout << c * r << endl;
    return 0;
}
