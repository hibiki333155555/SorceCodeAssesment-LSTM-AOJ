#include <bits/stdc++.h>
using namespace std;

/* alias */
using ull = unsigned long long;
using ll = long long;
using vi = vector<int>;
using vl = vector<long>;
using vll = vector<long long>;
using vvi = vector<vi>;
using vvl = vector<vl>;
using vvll = vector<vll>;
using vs = vector<string>;
using pii = pair<int, int>;

//Answer
int main() {
    ll n;cin>>n;
    vll r(n);
    for(int i = 0;i<n;i++) cin >> r[i];

    ll maxv = -2000000000;
    ll minv = r[0];

    //現在探索中までのr[i]について、それまでに出てきたr[]の中で
    //最も最小のものとのみ比較すれば良い
    for(int i = 1;i<n;i++) {
        maxv = max(maxv, r[i] - minv);
        minv = min(minv,r[i]);
    }
    cout << maxv << endl;
    return 0;
}
