#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int top; int s[1000];
void push(int x) {
    s[++top] = x;

}
int pop() {
    top--;
    return s[top + 1];


}

int main() {

    char m[100];
    top = 0;
    int a, b;
    while (scanf("%s", m) != EOF) {
        if (a[0] == '+')
        {
            a = pop();
            b = pop();
            push(a + b);
        }
        else if (m[0] == '-')
        {
            a = pop();
            b = pop();
            push(a - b);
        }
        else if (m[0] == '*')
        {
            a = pop();
            b = pop();
            push(a * b);
        }
        else
        {
            push(atoi(m));
        }

    }
    cout << pop() << endl;
}
