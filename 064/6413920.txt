int main() {
    ll n;
    cin >> n;
    vector<ll> a(n);
    for (ll i = 0; i < n; i++) cin >> a[i];
    // dp[i] = iの長さの右端
    vector<ll> dp(n + 1, INF);
    rep(i, n) {
        auto iter = lower_bound(dp.begin(), dp.end(), a[i]);
        auto dis = distance(dp.begin(), iter);
        // cout << *iter << " " << dis << "\n";
        dp[dis] = a[i];
        // for (auto a : dp) cout << a << " ";
        // cout << endl;
    }

    rep(i, n + 1) {
        if (dp[i] == INF) {
            cout << i << "\n";
            return 0;
        }
    }
    return 0;
}
