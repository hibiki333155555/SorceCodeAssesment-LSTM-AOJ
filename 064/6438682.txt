#include <iostream>
#include <vector> // vector
#include <utility> // pair, make_pair
using namespace std;


int main() {
    int  n;

    cin >> n;
    vector<int> R(n);

    for(int i = 0;i < n;i++)cin >> R.at(i);

    for (int i = 1; i < n; i++) {
        for (int j = i - 1 ; j >= 0; j--) {
            if (R.at(j+1) < R.at(j))swap(R.at(j), R.at(j+1));

        }
        for(int j = 0;j < n;j++)cout << R.at(j) << " ";
        cout << endl;
    }

}
