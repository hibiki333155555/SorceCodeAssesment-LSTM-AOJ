int main()
{	
	int n,i;
	int a[200005];
	int ma=-1000000000,mi;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	
	mi=a[0];
	
	for(i=1;i<n;i++)
	{
		ma=max(ma,a[i]-mi);
		mi=min(mi,a[i]);
	}
	 
	cout<<ma<<endl;
	
return 0;
}
