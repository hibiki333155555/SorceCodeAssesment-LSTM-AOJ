#include "bits/stdc++.h"
using namespace std;
using ll=long long;

int main(){
    ll n,k; cin>>n>>k;
    vector<int>a(n);
    for(int i=0;i<n;i++)cin>>a[i];
    sort(a.begin(),a.end());
    ll c=0,sum=0,i;
    for(i=0;i<k;i++){
        if(i==n||a[i]==6){
            i--;break;
        }
        if(a[i]==1){
            c++;sum+=5;
        }
        else sum+=6;
    }
    if(c>0&&i<k){
        int co=k-i-1;
        sum+=co;
    }
    if(i==k)i--;
    for(int j=0;j<n-i-1;j++){
        sum+=a[n-j-1];
    }
    cout<<sum<<endl;
}
