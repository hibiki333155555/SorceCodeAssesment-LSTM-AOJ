#include <bits/stdc++.h>
using namespace std;

int main(){
  int a;
  cin >> a;
  int Min,Max,Sum;
  int x;
  cin >> x;
  Min = x;
  Max = x;
  Sum = x;
  for(int i=1; i<a; i++){
    cin >> x;
    Min = min(Min,x);
    Max = max(Max,x);
    Sum += x;
  }

  cout << Min << " " << Max << " " << Sum << endl;
  return 0;
}
