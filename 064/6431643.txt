#include <bits/stdc++.h>

using namespace std;

int main() {

  
    int n,c=0;
    cin>>n;
    vector<int> a(n,0);
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    for(int k=0;k<n-1;k++){
        for(int j=0;j<n-1-k;j++){
            if(a[j]>a[j+1]){
                swap(a[j],a[j+1]);
                c++;
            }                                                                           
        }
    }
    for(int i=0;i<n;i++){
    cout<<a[i];
    if(i!=n-1)   cout<<" ";
    }
    cout<<"\n";
    cout<<c;
   
    
    
}
