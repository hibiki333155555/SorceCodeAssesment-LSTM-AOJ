#include<stdio.h>
#include<string.h>
int main()
{

    char w[10];
    char t[1000];
    char cheak[20]="end_of_text";
    int count=0;
    //printf("Enter search:");
    scanf("%s",w);
    while(1)

    {
        int i,j;
        char t1[1000];

        //printf("Enter text:");
        scanf("%s",t);

        for(i=0,j=0;i<strlen(t),j<strlen(t);i++,j++)
            t1[j]=tolower(t[i]);
        t1[j]='\0';

        int br=strcmp(t1,cheak);
        if(br==0)
            break;



        int d=strcmp(w,t1);
        if(d==0)

            count++;

    }
    printf("%d\n",count);


//TANVIR
    return 0;
}
