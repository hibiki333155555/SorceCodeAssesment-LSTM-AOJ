n = int(input())
A = list(map(int, input().split()))

for i in range(1, n):
    v = A[i]
    j = i - 1
    print(f'{" ".join(str(i) for i in A)}')
    # v より大きい数は全て入れかえる
    while j >= 0 and A[j] > v:
        A[j+1] = A[j] # 後ろに移す
        j -= 1
    # 入れかえが終わったら、その一つ前にvを入れる    
    A[j+1] = v
    
print(f'{" ".join(str(i) for i in A)}')
