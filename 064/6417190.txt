#include <iostream>
#include <cassert>

using namespace std;

int nw = 0;
int sm = 0;
string s;
string ans;

void x(){
  assert('0' <= s[nw] && s[nw] <= '9');
  int z = s[nw]-'0';
  sm += z;
  nw+=2;
  while(s[nw] != ')'){
    if('0' <= s[nw] && s[nw] <= '9'){
      x();
    }else{
      char k = s[nw]-('z' < s[nw]+(sm%26)?26:0)+(sm%26);
      ans.push_back(k);
      nw++;
    }
  }
  sm -= z;
  nw++;
}

int main(){
  cin>>s;
  s = "0(" + s + ")";
  x();
  cout << ans << endl;
}
