#include<bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	vector<int>a(n);
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	
	int p;
	cin>>p;
	vector<int>b(p);
	for(int i=0;i<p;i++){
		cin>>b[i];
	}
	
	int count=0;
	for(int i=0;i<p;i++){
		for(int j=0;j<n;j++){
			if(a[j]==b[i]){
				count++;
			}
		}
		
	}
if(count==p){
		cout<<1<<" ";
	}
	else{
		cout<<0<<" ";
	}
	
	
	
}
