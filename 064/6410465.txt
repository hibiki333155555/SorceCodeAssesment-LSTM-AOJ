#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

int main(){
    int n, a,b,c,d,e;
    
    cin>>n;
    vector<tuple<int, int, char, int, string>> v;
    while(n--){
        cin>>a>>b>>c>>d>>e;
        v.push_back(make_tuple(a,b,c,d,e));
    }
    sort(v.begin(), v.end());
    for(int i = 0; i < v.size(); i++){
        cout << v.at(i).get<0>() << " " << v.at(i).get<1>() << " " << v.at(i).get<2>() << " " << v.at(i).get<3>() << " " << v.at(i).get<4>() << endl;
    }
    return 0;
}
