void com(int a,int b,int c){
    if(b<c)
    printf("%d %d %d\n",a,b,c);
    else
    {
        if(c<a)
        printf("%d %d %d\n",c,a,b);
        else
        printf("%d %d %d\n",a,c,b);
    }
}

int main() {
    int a,b,c,temp;
    scanf("%d %d %d",&a,&b,&c);
    if(a>b)
    {
        temp=a;
        a=b;
        b=temp;
    }
    com(a,b,c);

}
