#include<stdio.h>
#include<vector>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<set>
#include<map>
using namespace std;

int main(){
  int n;
  scanf("%d",&n);
  set<string> set;
  char order[100];
  char moji[100];
  for (int i = 0; i < n; i++){
      if(scanf("%s",order)=="insert"){
          set.insert(scanf("%s",moji));
      }
      else if(scanf("%s",order)=="find"){
          if(set.count(scanf("%s",moji))){
              printf("%s","yes");
          }
          else{
              printf("%s","no");
          }
      }
      
  }
}
