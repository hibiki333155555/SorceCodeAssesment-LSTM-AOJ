#include<stdio.h>

int main(void)
{
    int n;
    long x,min,max;
    long long  sum=0;
    scanf("%d\n",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d\n",&x);
        if(i==0||min>x)
        {
            min=x;
        }
        if(i==0||max<x)
        {
            max=x;
        }
        sum=sum+x;
    }
    printf("%d %d %10d\n",min,max,sum);
    return 0;
}
