#include <iostream>
#include <vector>

using namespace std;

void printVector(vector<int> A) {
    for (int i = 0; i < A.size(); i++){
        if (i < A.size() - 1) {
            cout << A.at(i) << " ";
        } else {
            cout << A.at(i) << endl;
        }
    }
}

void insertionSort(vector<int> A, int N) {
    printVector(A);
    for (int i = 1; i < N; i++) {
        int v = A.at(i);
        int j = i - 1;
        // cout << i << "-------" << endl;
        while (j >= 0 && A.at(j) > v) {
            A.at(j+1) = A.at(j);
            j--;
            // printVector(A);
        }
        A.at(j+1) = v;
        printVector(A);
    }
}

int main(){
    int N;
    cin >> N;

    vector<int> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A.at(i);
    }

    insertionSort(A, N);
}
