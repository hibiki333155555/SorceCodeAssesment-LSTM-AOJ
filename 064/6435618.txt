#include <bits/stdc++.h>
using namespace std;

int insertion_sort(int A[1009],int N) {
    for(int i = 1;i < N;i++) {
        int v = A[i];
        int j = i - 1;
        while(j >= 0 && A[j] > v) {
            A[j+1] = A[j];
            j--;
        }
        A[j+1] = v;
        return A[j+1];
    }
}

int main() {
    int n;cin >> n;
    int * a[n];

    for(int i = 0;i<n;i++) insertion_sort(a[i],n);
}
