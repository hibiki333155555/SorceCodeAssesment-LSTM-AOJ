template <class T>
inline bool chmin(T &a, T b) {
    if (a > b) {
        a = b;
        return true;
    }
    return false;
}

template <class T>
inline bool chmax(T &a, T b) {
    if (a < b) {
        a = b;
        return true;
    }
    return false;
}

template <class T>
T gcd(T a, T b) {
    if (b == 0)
        return a;
    else
        return gcd(b, a % b);
}

template <class T>
T lcm(T a, T b) {
    a /= gcd(a, b);
    return a * b;
}

template <class T>
T Pow(T x, i64 n) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = res * x;
        x = x * x;
        n >>= 1;
    }
    return res;
}

template <class T>
T mod_pow(T x, i64 n, i64 mod) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = (res * x) % mod;
        x = (x * x) % mod;
        n >>= 1;
    }
    return res;
}

void main_() {
    int n;
    while (std::cin >> n, n != 0) {
        std::map<std::string, int> map;
        std::vector<int> powers(n);
        std::vector table(n, std::vector<std::string>());
        rep(i, 0, n) {
            std::string s;
            std::cin >> s >> powers[i];
            map[s] = i;
            int c;
            std::cin >> c;
            while (c--) {
                std::string t;
                std::cin >> t;
                table[i].emplace_back(t);
            }
        }
        if (n == 1) {
            std::cout << powers.back() << '\n';
            continue;
        }
        const int sz = n / 2;
        graph front(n), back(n);
        rep(i, 0, n) {
            for (auto s : table[i]) {
                assert(map.find(s) != map.end());
                int idx = map[s];
                if (idx < sz) {
                    front[i].emplace_back(idx);
                } else {
                    back[i].emplace_back(idx);
                }
            }
        }
        std::vector<int> dp(1 << sz, 0);
        {
            std::vector<int> a(sz, 0);
            rep(i, 0, sz) {
                for (auto v : front[i]) {
                    a[i] |= 1 << v;
                }
            }
            rep(bit, 0, 1 << sz) {
                int sum = 0;
                int ret = 0;
                rep(i, 0, sz) {
                    if (bit & (1 << i)) {
                        sum += powers[i];
                        ret |= a[i];
                        chmax(dp[bit], dp[bit ^ (1 << i)]);
                    }
                }
                if ((bit & ret) == 0 && (bit & 1)) chmax(dp[bit], sum);
            }
        }
        int ans = 0;
        {
            const int m = n - sz;
            const int mask = (1 << sz) - 1;
            std::vector<int> a(m, 0);
            std::vector<int> b(m, 0);
            rep(i, sz, n) {
                for (auto v : back[i]) {
                    a[i - sz] |= 1 << (v - sz);
                }
                for (auto v : front[i]) {
                    b[i - sz] |= 1 << v;
                }
            }
            rep(bit, 0, 1 << m) {
                int sum = 0;
                int ret = 0;
                int cobit = 0;
                rep(i, 0, m) {
                    if (bit & (1 << i)) {
                        sum += powers[i + sz];
                        ret |= a[i];
                        cobit |= b[i];
                    }
                }
                if ((bit & ret) == 0) {
                    cobit = mask ^ cobit;
                    chmax(ans, dp[cobit] + sum);
                }
            }
        }
        std::cout << ans << '\n';
    }
}
