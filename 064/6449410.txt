#include <iostream>
#include <vector>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<string> S(n);
  for(int i = 0; i < n; ++i)
    cin >> S[i];
  int ans = 0;
  for(int i = 0; i < n; ++i){
    for(int j = 0; j < n; ++j){
      if(i == j) continue;
      int d = 0;
      for(int k = 0; k < S[i].size(); ++k){
        d += S[i][k] != S[j][k];
      }
      if(d <= 1){
        ++ans;
        break;
      }
    }
  }
  cout << ans << endl;
}
