#include <bits/stdc++.h>
using namespace std;
const int INF = 1001001001;
const int MOD = 1000000007;

int main(){
  int x;
  char s[100];
  vector<int> a(0);

  while( scanf("%s", s) != EOF ){
      if ( s[0] == '+' ){
          int l = a.at(a.size()-1);
          a.pop_back();
          int r = a.at(a.size()-1);
          a.pop_back();
          int tmp = l+r;
          a.push_back(tmp);
      } else if ( s[0] == '-' ){
          int l = a.at(a.size()-1);
          a.pop_back();
          int r = a.at(a.size()-1);
          a.pop_back();
          int tmp = r-l;
          a.push_back(tmp);
      } else if ( s[0] == '*' ){
          int l = a.at(a.size()-1);
          a.pop_back();
          int r = a.at(a.size()-1);
          a.pop_back();
          int tmp = l*r;
          a.push_back(tmp);
      } else {
          x = atoi(s);
          a.push_back(x);
      }
  }
  
  cout << a.at(0) << endl;

  return 0;
}
