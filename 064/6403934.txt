#include <iostream>
#include <math.h>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;
int main() {
	int a=0;
	string ans;
	while (true) {
		int n, x;
		cin >> n >> x;
		if (n == 0 && x == 0)break;
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				for (int k = 1; k <= n; k++) {
					if (i == k || k == j || i == j)continue;
					if (i + j + k == x)a++;
				}
			}
		}
		ans += to_string(a / 6)+" ";
	}
	ans.erase(ans.end() - 1);
	cout << ans << endl;
	return 0;
}
