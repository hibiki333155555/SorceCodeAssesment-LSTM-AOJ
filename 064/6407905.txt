#include <bits/stdc++.h>
using namespace std;
const double PI =acos(-1);
const long double EPS = 1e-14;

int main(){
  long N, cnt = 0;
  cin >> N;
  vector<long> A(N);
  for(long i = 0; i < N; i++){
    cin >> A[i];
  }
  for(long i = 0; i < N; i++){
    long mn = i;
    for(long j = i + 1; j < N; j++){
      if(A[j] < A[mn]){
        mn = j;
      }
    }
    if(mn != i){
      cnt++;
    }
    long tmp = A[i];
    A[i] = A[mn];
    A[mn] = tmp;
  }
  for(long i = 0; i < N; i++){
    cout << A[i];
    if(i == N - 1){
      cout << endl;
    }else{
      cout << " ";
    }
  }
  cout << cnt << endl;
  return 0;
}
