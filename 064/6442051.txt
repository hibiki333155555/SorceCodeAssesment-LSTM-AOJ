#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;

int main()
{	
	
	int H,W,i,j;
	
	cin>>H>>W;
	while(!(H==0&&W==0))
	{
	
		for(i=0;i<H;i++)
		{
			if(i%2)
			{
				for(j=0;j<W;j++)
				{
					if(j%2)
						cout<<"#";
					else
						cout<<".";
				}
			}
			else
			{
				for(j=0;j<W;j++)
				{
					if(j%2)
						cout<<".";
					else
						cout<<"#";
				}
			}
			
			
			
			cout<<endl;
		
		}
		
		
		cout<<endl;
		cin>>H>>W;
		
	}
	
return 0;
}
