import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner t = new Scanner(System.in);
        int h =t.nextInt();
        int l = t.nextInt();
        for(int i=1;i<=h;i++){
            for(int j=1;j<=l;j++){
                System.out.println("#");
            }
            System.out.println();
        }
    }
}
