#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    unsigned int a[100][100], b[100][100];


    int n, m, l;
    cin >> n >> m >> l;

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            cin >> a[i][j];
        }
    }

    for (int i = 0; i < m; i++)
    {
        for (int j = 0; j < l; j++)
        {
            cin >> b[i][j];
        }
    }

    for(int i = 0; i < n; i++){
        for(int j = 0; j < l; j++){
            int c = 0;
            for(int k = 0; k < m; k++){
                c += a[i][k] * b[k][j];
            }

            cout<<c;
            if(j != l-1){
                cout<<' ';
            }
        }

        cout<<endl;
    }

    return 0;
}
