#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main(){
	cin.tie(nullptr);
	ios::sync_with_stdio(false);
	int n; cin >> n;
	vector<int> a(n);
	int id=0;
	for(int i=0;i<n;i++){
		cin >> a[i];
		if(a[id]<a[i]){
			id=i;
		}
	}
	int sum=0;
	int sum2=0;
	for(int i=0;i<n;i++){
		if(i<id)sum+=a[i];
		if(id<i)sum2+=a[i];
	}
	cout << sum << endl;
	cout << sum2 << endl;
}
