#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <math.h>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;
int main() {
	int a=0;
	bool isf = true;
	std::vector<int>anss;
	while (true) {
		int n, x;
		cin >> n >> x;
		if (n == 0 && x == 0)break;
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				for (int k = 1; k <= n; k++) {
					if (i == k || k == j || i == j)continue;
					if (i + j + k == x)a++;
				}
			}
		}
		anss.push_back(a / 6);
	}
	for (int i = 0; i < anss.size(); i++) {
		if (i == 0)cout << anss[i];
		else cout << " " << anss[i];
	}
	cout << endl;
	return 0;
}
