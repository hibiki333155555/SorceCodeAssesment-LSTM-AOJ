#include<stdio.h>
#include<stdbool.h>
#define MAX 20

int N; 
int A[MAX];

int solve(int n, int m){

  if(m == 0) return 1;
  if(n >= N) return 0;
  return solve(n + 1, m - A[n]) || solve(n + 1, m);
}

int main()
{   
  int q;
  int m;
  int sum;
  bool flag;

  scanf("%d", &N);
  for(int i = 0; i < N; i++ ){
    scanf("%d", &A[i]);
  }

  scanf("%d", &q);
  for(int i = 0; i < q; i++ ){
    scanf("%d", &m);
    if(solve(0, m)) printf("yes\n");
    else printf("no\n");
  }

  return 0;
}
