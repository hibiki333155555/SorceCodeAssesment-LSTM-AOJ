#include<bits/stdc++.h>

using namespace std;

int forb[105][105][105]={0};

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int n;
  cin >> n;
  vector<string> x(n),y(n),z(n);
  for(auto &nx : x){cin >> nx;}
  for(auto &nx : y){cin >> nx;}
  for(auto &nx : z){cin >> nx;}
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(x[n-i-1][j]=='0'){
        for(int k=0;k<n;k++){
          forb[k][j][i]=1;
        }
      }
    }
  }
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(y[n-i-1][j]=='0'){
        for(int k=0;k<n;k++){
          forb[i][k][j]=1;
        }
      }
    }
  }
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(z[n-i-1][j]=='0'){
        for(int k=0;k<n;k++){
          forb[j][i][k]=1;
        }
      }
    }
  }

  // cout << '\n';
  // for(int i=0;i<n;i++){
  //   for(int j=0;j<n;j++){
  //     for(int k=0;k<n;k++){
  //       cout << forb[i][j][k];
  //     }cout << '\n';
  //   }cout << '\n';
  // }

  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(x[n-i-1][j]=='1'){
        for(int k=0;k<n;k++){
          if(forb[k][j][i]==0){break;}
          if(k==n-1){cout << "No\n";return 0;}
        }
      }
      //cout << i << ' ' << j << '\n';
      if(y[n-i-1][j]=='1'){
        for(int k=0;k<n;k++){
          if(forb[i][k][j]==0){break;}
          if(k==n-1){cout << "No\n";return 0;}
        }
      }
      //cout << i << ' ' << j << '\n';
      if(z[n-i-1][j]=='1'){
        for(int k=0;k<n;k++){
          if(forb[j][i][k]==0){break;}
          if(k==n-1){cout << "No\n";return 0;}
        }
      }
      //cout << i << ' ' << j << '\n';
    }
  }
  cout << "Yes\n";
  return 0;
}
