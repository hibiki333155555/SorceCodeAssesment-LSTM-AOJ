#include <bits/stdc++.h>

#define ll long long
#define ld long double
#define rep(i,a,b) for (int i = a; i < b; i++)

using namespace std;

int main(){
    int H,W;
    string s1,s2;   
    while (true){
        cin >> H >> W;
        if (H==0 && W == 0) break;

        s1 = "";
        s2 = "";
        rep(i,0,W){
            if (i%2 == 0){
                s1 += "#";
                s2 += ".";
            }else{
                s1 += ".";
                s2 += "#";
            }
        }
        rep(i,0,H){
            if (i%2 == 0) cout << s1 << endl;
            else cout << s2 << endl;
        }
        cout << endl;
    }
    //system("pause");
    return 0;
}
