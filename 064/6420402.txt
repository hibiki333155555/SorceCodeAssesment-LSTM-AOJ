H = int(input())
heap = [int(x) for x in input().split()]

for index,i in enumerate(heap):
    print(f'node {index}: key = {i}')
    if index != 0:
        print(f', parent key = {head[index//2]}')
    if 2*index < H:
        print(f', left key = {heap[2*index]}')
    if 2*index + 1 < H:
        print(f',right key = {heap[2*index + 1]}')
    print()
