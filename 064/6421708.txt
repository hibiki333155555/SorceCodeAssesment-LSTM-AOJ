#include <bits/stdc++.h>
using namespace std;

#define REP(i,n) for(int i=0;i<(n);i++)

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  
  string s;cin>>s;
  stack<int> sta;
  int now=0;
  for(const char&c:s){
    if(c>='0' and c<='9'){
      now+=c-'0';
      sta.push(c-'0');
    }
    if(c==')'){
      now-=sta.top();
      sta.pop();
    }
    if(c>='a' and c<='z')cout<<char('a'+(c-'a'+now)%26);
  }
  cout<<endl;
}
