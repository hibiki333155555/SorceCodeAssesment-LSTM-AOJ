#include <iostream>
#include <list>

using namespace std;
int main(void)
{
    int n;
    cin >> n;
    
    list<int> lst;
    list<int>::iterator itr;
    itr = lst.begin();
    
    list<int>::iterator itrtemp;
    int q;
    int x;
    int d;
    
    for( int i = 0; i < n; i++)
    {
        cin >> q;
        switch(q)
        {
            case 0:
                cin >> x;
                //itrの前に要素を挿入し、挿入した要素のitrを返す
                itr = lst.insert(itr, x);
                break;
            case 1:
                cin >> d;
                // dだけitrをずらす
                if( d >= 0)
                {             
                    for(int j = 0; j<d; j++) itr++;
                }
                else
                {
                    for(int j = 0; j>d; j--) itr--;
                }
                break;
            case 2:
                //itrの要素を消去し、消去した要素の次の要素のitrを返す
                itr = lst.erase(itr); 
                break;
        }
    }
    
    for(itr = lst.begin(); itr != lst.end(); ++itr) {
        cout << *itr << "\n";
    }
}
