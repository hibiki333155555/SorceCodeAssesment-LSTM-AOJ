void is_factor(int a, int b, int N)
{
    int c = 0;
    int tmp;
    if (N == 1)
    {
        if (a == 1)
        {
            cout << 1;
            return;
        }
        else
        {
            cout << 0;
            return;
        }
    }
    for (int i = 2; i * i <= N; ++i)
    {
        if (N % i == 0)
        {
            if (a <= i && i <= b)
            {
                c++;
            }
            tmp = N / i;
            if (a <= tmp && tmp <= b)
            {
                c++;
            }
        };
    }
    cout << c << endl;
    return;
}
