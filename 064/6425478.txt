#include<stdio.h>
#include<string.h>
#include <ctype.h>

int main(void){
    int cnt[26]={};
    char moji[1200]; 
    char *p,c;
    int i=0;

    scanf("%s",moji);

    p=moji;

    while(*p!='\0'){
    c=*p;
    if(islower(c)){
        cnt[((int)p[i]-97)]++;
    }
    else if(isupper(c)){
        cnt[((int)p[i]-65)]++;
    }
    *p++;
    }
    for(i=0;i<26;i++){
        printf("%c : %d\n",i+97,cnt[i]);
    }
    return 0;
}
