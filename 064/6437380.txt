#include <stdio.h>
#define N 10000
#define INF 1 << 30
int n;
int from, m, to, cost;

int G[N][1000];
int count[N];
int cost_list[N] = {[0 ... N - 1] = INF};
int visited[N];

void dijkstra(int start) {
  cost_list[start] = 0;
  while (1) {
    int min_cost = INF;
    int min_v = -1;
    for (int i = 0; i < n; i++) {
      if (!visited[i] && min_cost > cost_list[i]) {
        min_cost = cost_list[i];
        min_v = i;
      }
    }
    if (min_v == -1) {
      break;
    }
    visited[min_v] = 1;
    for (int i = 0; i < n; i++) {
      int cost_to_i = G[min_v][i];
      if (!visited[i] && cost_to_i != INF) {
        if (cost_list[i] > cost_list[min_v] + cost_to_i)
          cost_list[i] = cost_list[min_v] + cost_to_i;
      }
    }
  }
}

void create_list() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf("%d %d", &from, &m);
    for (int k = 0; k < m; k++) {
      scanf("%d %d", &to, &cost);
      G[from][count[from]++] = cost;
    }
  }
}

void create_matrix() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    for (int k = 0; k < n; k++) {
      G[i][k] = INF;
    }
  }
  for (int i = 0; i < n; i++) {
    scanf("%d %d", &from, &m);
    for (int k = 0; k < m; k++) {
      scanf("%d %d", &to, &cost);
      G[from][to] = cost;
    }
  }
}

void print() {
  for (int i = 0; i < n; i++) {
    printf("%d %d", i, cost_list[i]);
    puts("");
  }
}

int main(void) {
  create_matrix();
  dijkstra(0);
  print();
}
