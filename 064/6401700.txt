#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(int)(n);i++)

bool isPrime(int x){
    if(x==2)    return true;
    if(x<2||x%2==0) return false;
    int a=sqrt(x);
    for(int i=3;i<=a;i+=2){
        if(x%i==0)  return false;
    }
    return true;
}

int main(){
    int n;
    cin>>n;
    vector<int> num(n);
    rep(i,n){
        cin>>num[i];
    }
    int cnt=0;
    rep(i,n){
        if(isPrime(num[i]))   cnt++;
    }
    cout<<cnt<<endl;
    
    return 0;
}
