#include <bits/stdc++.h>
using namespace std;

int main(){
    string str,ord,p;
    int q,a,b;
    cin>>str>>q; 
    for(int i=0; i<q; i++){
        cin>>ord;
        if(ord=="print"){
            cin>>a>>b;
            cout<<str.substr(a,b-a+1)<<endl;
        }
        else if(ord=="reverse"){
            cin>>a>>b;
            reverse(str.begin()+a,str.begin()+b+1); 
        }
        else if(ord=="replace"){
          cin>>a>>b>>p;
          str.replace(a,b-a+1,p);  
        }
    }
}
