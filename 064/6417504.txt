void solve() {
    int n, m, v, a, b;
    cin >> n >> m >> v >> a >> b;
    if (n > m) {
        cout << "Yes" << endl;
        return;
    }
    vector<bool> used(m, false);
    v %= m;
    used[v] = true;
    n--;
    while (n--) {
        v = a * v + b;
        v %= m;
        if (used[v]) {
            cout << "Yes" << endl;
            return;
        }
        used[v] = true;
    }
    cout << "No" << endl;
}

    int t; cin >> t;
    while (t--) {
        solve();
