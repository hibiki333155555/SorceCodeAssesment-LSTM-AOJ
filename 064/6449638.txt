#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)
#define Inf 1000000001

int main(){
	
	int _t;
	cin>>_t;
	rep(_,_t){
		int n,S;
		cin>>n>>S;
		set<int> S1;
		int cc = 0;
		bool F = false;
		rep(i,n){
			int c1 = 0;
			rep(j,4){
				int a;
				cin>>a;
				if(a==1)c1++;
				else{
					if(a==2)cc++;
					else cc--;
				}
			}
			if(S1.count(c1))S1.erase(c1);
			else S1.insert(c1);
			if(c1==3){
				F = true;
			}
		}
		
		if(F){
			vector<int> ans;
			if(S1.count(3)){
				rep(i,3)ans.push_back(1);
				if(cc>0)ans.push_back(3);
				else ans.push_back(2);
			}
			else{
				ans.push_back(1);
				if(cc==0){
					ans.push_back(2);
					ans.push_back(1);
					ans.push_back(3);
				}
				else{
					if(cc>0){
						ans.push_back(3);
						ans.push_back(1);
						ans.push_back(3);
					}
					else{
						ans.push_back(2);
						ans.push_back(1);
						ans.push_back(2);
					}
				}
			}
			rep(i,4){
				if(i!=0)cout<<' ';
				cout<<ans[i];
			}
			cout<<endl;
			
			
			
			continue;
		}
		
		vector<int> ans;
		if(cc>0){
			rep(i,cc)ans.push_back(3);
		}
		else{
			rep(i,-cc)ans.push_back(2);
		}
		if(S1.size()!=0){
			rep(i,*(S1.begin()))ans.push_back(1);
		}
		
		
		while(ans.size()!=4){
			ans.push_back(2);
			ans.push_back(3);
			
		}
		//cout<<ans.size()<<endl;
		sort(ans.begin(),ans.end());
		vector<int> T(4,Inf);
		
		do{
			int t = 0;
			if(ans[0]==ans[2] && ans[1]==ans[3])t = 1;
			if(S!=t)continue;
			T = min(T,ans);
		}
		while(next_permutation(ans.begin(),ans.end()));
	
		rep(i,4){
			if(i!=0)cout<<' ';
			cout<<T[i];
		}
		cout<<endl;
	}
	
	return 0;
}
