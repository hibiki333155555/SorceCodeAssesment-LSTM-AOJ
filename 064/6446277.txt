#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;
const int INF = 1e9;
const ll LINF = 1e18;
const ld pi = 3.141592653589793238;
double Edis(ll x1,ll y1,ll x2,ll y2){
    ll x,y;
    x=(x1-x2),y=(y1-y2);
    double Edis=sqrt(x*x+y*y);
    return Edis;
}
int main(){
    ll x1,y1,x2,y2;
    cin >> x1 >> y1 >> x2 >> y2 ;
    double dis=Edis(x1,y1,x2,y2);
    printf("%.12lf\n", dis);
}
