#include <bits/stdc++.h>
using namespace std;
int main(){
    int N;
    cin >> N;
    vector<vector<long long>> A(N, vector<long long>(N));
    map<int, pair<int, int>> m;
    for(int i = 0; i < N; i++){
        for(int j = 0; j < N; j++) {
            cin >> A[i][j];
            if(!m.count(A[i][j])) m[A[i][j]] = {i, j};
            else{
                auto [l, c] = m[A[i][j]];
                if(l > i) m[A[i][j]] = {i, j};
                else if(l == i && c > j) m[A[i][j]] = {i, j};
            }
        }
    }
    int Q;
    cin >> Q;
    for(int i = 0; i < Q; i++){
        int T;
        cin >> T;
        if(!m.count(T)) cout << -1 << '\n';
        else cout << m[T].first + 1 << ' ' << m[T].second + 1 << '\n';
    }
}
