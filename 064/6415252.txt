#include<iostream>
#include<queue>
#include<cstring>
using namespace std;
struct Renwu
{
	string id;
	int time;
}renwus[100005];
int main()
{
	int n,t;
	cin>>n>>t;
	for(int i=0;i<n;i++)
		cin>>renwus[i].id>>renwus[i].time;
	 queue<Renwu>q;
	 int nowt=0;
	 for(int i=0;i<n;i++)
	 {
	 	q.push(renwus[i]);
	 }
	while(!q.empty())
	{
		Renwu now=q.front();
		q.pop();
		if(now.time<=t)
		{
			
			cout<<now.id<<" "<<nowt+now.time<<endl;
			nowt+=now.time;
		}
		else
		{
			now.time-=t;
			nowt+=t;
			q.push(now);
		}
		//cout<<nowt<<" "<<t;
	}
	return 0;
}
