#include <iostream>
#include <set>
using namespace std;

int main(){
    set<int> s;
    int choice, x;
    int q;
    cin >> q;
    for(int i=0; i<q; i++){
        cin >> choice >> x;
        if(choice == 0){
            s.insert(x);
            cout << s.size() << endl;
        }
        else if(choice == 1){
            cout << s.count(x) << endl;
        }
        else if(choice == 2){
            s.erase(x);
        }
    }
    return 0;
}
