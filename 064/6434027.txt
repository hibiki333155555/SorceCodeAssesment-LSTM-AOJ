#include<bits/stdc++.h>

using namespace std;

int main(){
    vector<int> num(12);
    for(int i=0;i<12;i++){
        cin >> num[i];
    }
    sort(num.begin(),num.end());
    if(num[0]==num[3]&&num[4]==num[7]&&num[8]==num[11]){
        cout << "yes" << endl;
    }else{
        cout << "no" << endl;
    }
    return(0);
}
