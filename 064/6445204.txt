int partition(vector<int> &v, int p, int n) {
	int x = v.at(n - 1);
	int i = p - 1;
	for (int j = p; j < n-1; j++) {
		if (v.at(j) <= x) {
			i++;
			swap(v.at(i), v.at(j));
		}
	}
	swap(v.at(i + 1), v.at(n - 1));
	return i + 1;
}

int main() {
	int n;
	cin >> n;
	vector<int> v(n);
	for (int i = 0; i < n; i++) {
		cin >> v.at(i);
	}
	int num = partition(v, 0, n);
	for (int i = 0; i < n; i++) {
		if (i == num) {
			cout << "[" << v.at(i) << "] ";
		}
		else if (i == n - 1) {
			cout << v.at(i) << endl;
		}
		else {
			cout << v.at(i) << " ";
		}
	}
}
