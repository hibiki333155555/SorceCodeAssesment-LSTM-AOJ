#include <iostream>
#include <iomanip>
#include <vector>

struct Point {
  double x, y;
};

int main() {
  int n; std::cin >> n;
  std::vector<Point> ps(n);
  for (int i=0; i<n; i++) std::cin >> ps[i].x >> ps[i].y;
  double area = 0.0;
  for (int i=0; i<n-1; i++) {
    area += ps[i].x * ps[i+1].y - ps[i+1].x * ps[i].y;
  }
  std::cout << std::fixed << std::setprecision(1) << area / 2.0 << std::endl;
}
