#include <iostream>
#include <math.h>
#include <vector>
#include <string>
#include <algorithm>
#include <regex>

using namespace std;
int main() {
	int a=0;
	while (true) {
		int n, x;
		cin >> n >> x;
		if (n == 0 && x == 0)break;
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				for (int k = 1; k <= n; k++) {
					if (i == k || k == j || i == j)continue;
					if (i + j + k == x)a++;
				}
			}
		}
	}
	std::cout << a/6;
	return 0;
}
