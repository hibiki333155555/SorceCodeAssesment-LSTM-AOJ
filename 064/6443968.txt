#include <iostream>
using namespace std;

int main()
{
    int i = 1;
    int x;
    
    while(1) {
        cin >> x;
        
        if ( x != 0 ) {
            cout << "Case" << " " << i << ":" << " " << x << endl;
            i += 1;
        }
        else {
            break;
        }
    }
    
    return 0;
}
