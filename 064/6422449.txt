#include <stdio.h>

int main(void)
{
    char c,nC;
    int i,n;
    
    while(1){
        while(scanf("%c",&c)!=EOF)
        {
            switch(c){
                case '\n':
                    goto finish;
                    
                case '@':
                    scanf("%c%c",&nC,&c);
                    for(i=0;i<nC-49;i++){
                        printf("%c",c);
                    }
                    
                default:
                    printf("%c",c);
                    break;
            }
        }
        finish:
        c=0;
    }
    return 0;
}
