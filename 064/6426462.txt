#include<stdio.h>

int main(){
    int w, h, s, t;
    
    for(;;){
        scanf("%d %d", &h, &w);
        if(h == 0 || w == 0)break;
        for(s = 1;s <= h;s++){
            for(t = 1;t <= w;t++){
                printf("#");
            }
            printf("\n");
        }
    }
    return 0;
}
