#include <iostream>

using namespace std;
int main(){
    
    int count[26] = {};
    string Ward;
    while(true){
        getline(cin, Ward);
        if(Ward == "") break;
        for(int i=0; i < Ward.size(); ++i){
            if(Ward[i] >= 'a' && Ward[i] <= 'z') count[Ward[i]-'a']++;
            else if(Ward[i] >= 'A' && Ward[i] <= 'Z') count[Ward[i]-'A']++;
        }
    }
         for(int i = 0; i < 26; i++){
           cout << (char)('a'+ i) << " : " << count[i] << endl;
         }
          
}
