    cin>>n;
    for(int i = 1; i <= n; i++) {
        int u, k; cin>>u>>k;
        while(k--) {
            int v; cin>>v;
            e[u].push_back(v);
        }
    }
    queue<int> q;
    q.push(1);
    v[1] = 1;
    d[1] = 0;
    while(q.size()) {
        int x = q.front(); q.pop();
        for(auto y : e[x]) {
            if(!v[y]) {
                v[y] = 1;
                d[y] = d[x] + 1;
                q.push(y);
            }
        }
    }
    for(int i = 1; i <= n; i++) {
        cout<<i<<" "<<(v[i]?d[i]:-1)<<endl;
    }
