#include<iostream>
using namespace std;
int main(){
    int n, cnt, ans = 0;
    cin>>n>>cnt;
    for (int i = 0; i < cnt; i++) {
        int a;
        cin>>a;
        ans += a;
    }
    int sub;
    sub =  ans / (n + 1);
    if(ans % (n + 1) != 0){
        sub++;
    }
    cout<<sub<<endl;
    return(0);
}
