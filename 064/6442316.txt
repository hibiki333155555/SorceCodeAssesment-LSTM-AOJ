#include <iostream>
using namespace std;

int main(){
    int H, W;
    int count;
    
    while(true){
    cin >> H >> W;
    if(H == 0 && W == 0) break;
        for(int i = 0; i < H; i++){
            if(i % 2 == 0) count = 0;
            else count = 1;
            for(int j = 0; j < W; j++){
                if(count % 2 == 0) cout << '#';
                else if(count % 2 == 1) cout << '.';
                count++;
            }
            cout << endl;
        }
    cout << endl;
    }  

    return 0;
}
