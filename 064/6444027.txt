#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;
#define drep(i, cc, n) for (ll i = (cc); i <= (n); ++i)
#define rep(i, n) drep(i, 0, n - 1)
#define all(a) (a).begin(), (a).end()
#define pb push_back
#define fi first
#define se second

const ll MOD = 1000000007;
const ll MOD2 = 998244353;
const ll INF = 1LL << 60;
const ll N_MAX = 2e5;

int main(){
    ll t;
    cin >> t;

    vector<ll> n(t);
    ll n_max = 0;
    rep(i, t){
        cin >> n[i];
        n_max = max(n_max, n[i]);
    }

    vector<bool> isprime(n_max+1, true);
    vector<ll> phi(n_max+1);
    for(ll i=2; i<=n_max; i++) phi[i] = i;
    for(ll i=2; i<=n_max+1; i++){
        if(!isprime[i]) continue;
        phi[i] = (phi[i]/i)*(i-1);
        for(ll j=i+i; j<=n_max; j+=i){
            isprime[j] = false;
            phi[j] = (phi[j]/i)*(i-1);
        }
    }

    vector<ll> f(n_max+1);
    f[1] = 2;
    for(ll i=2; i<=n_max; i++) f[i] = f[i-1] + phi[i];
    rep(i, t) cout << f[n[i]] << endl;
}
