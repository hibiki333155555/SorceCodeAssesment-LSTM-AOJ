#include <stdio.h>

int main(void){

    while(1){
        int a, b;
        char op;
        scanf("%d %c %d", &a, &op, &b);
        int t;
        

        if(op == '?') return 0;
        switch(op){
            case '+': t = a + b; break;
            case '-': t = a - b; break;
            case '*': t = a * b; break;
            case '/': t = a / b; break;
        };
        printf("%d\n", t);
    }

    return 0;
}
