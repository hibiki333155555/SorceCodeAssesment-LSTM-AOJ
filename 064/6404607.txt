while True:
    try:
        s = input()
    except EOFError:
        break

Alfa = [chr(i) for i in range(97,123)]
Answer =[0]*26
s = s.lower()
for ch in s:
    num = ord(ch) - ord('a')
    if num>=0 and num<=26:
        Answer[num] += 1
    else:
        pass

for i in range(26):
    print(f'{Alfa[i]} : {Answer[i]}')
