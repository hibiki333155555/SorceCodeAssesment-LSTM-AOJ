N, K = map(int, input().split())
A = list(map(int, input().split()))

maxi = {1:5, 2:6, 3:6, 4:6, 5:6, 6:5}
used = [False]*N

sum = 0
bool = True
for i in range(N):
    if bool:
        if A[i]!=1 and A[i]!=6:
            sum += 6
            used[i] = True
            K -= 1
            if K==0:
                for j in range(N):
                    if used[j]==False:
                        sum += A[j]
                bool = False
    else:
        sum += A[i]
print(sum)
