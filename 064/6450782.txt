void solve(){
    ll a,b;cin>>a>>b;
    if(a>b)swap(a,b);
    if(a==0&&b==0){
        cout<<1<<endl;
        return;
    }
    bool f=0;
    ll cnt=0;
    for(int j=31;j>=0;j--){
        if(f){
            if(((1<<j)&a)==((1<<j)&b)){
                cnt++;
            }else{
                break;
            }
        }else{
            if((1<<j)&b){
                f=1;
                if((1<<j)&a)cnt++;
                else break;
            }
        }
    }
    cout<<cnt<<endl;
}

    solve();
