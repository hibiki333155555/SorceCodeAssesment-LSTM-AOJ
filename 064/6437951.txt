#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int n;
  int score_T=0; 
  int score_H=0;
  string T, H;
  cin >> n;
  
  for (int i=0; i<n; i++){
    cin >> T >> H;
    
    if (T > H){
      score_T += 3;
    }
    
    else if (T == H){
      score_H += 1;
      score_T += 1;
    }
    
    else if (T < H){
      score_H += 3;
    }
    
  }
  
  cout << score_T << " " << score_H << endl;
  
  
}
