#include <iostream>
#include <math.h>
using namespace std;
int main(){
    int a,b;
    char c;
    cin>>a>>b>>c;
    for(int i=1;i<=b;i++){
        for(int j=1;j<=a;j++){
            if(i==b/2+1&&j==a/2+1)cout<<c;
            else if(i==1||i==b){
                if(j==1||j==a)cout<<"+";
                else cout<<"-";
            }else if(j==1||j==a)cout<<"|";
            else cout<<".";
        }cout<<"\n";
    }
    return(0);
}
