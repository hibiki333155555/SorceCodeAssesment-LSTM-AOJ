#include <iostream>
using namespace std;

int main(){
    int n, q, A[10001], B[501], sum = 0;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> A[i];
    }
    cin >> q;
    for(int i = 0; i < q; i++){
        cin >> B[i];
        for(int j = 0; j < n; j++){
            if(B[i] == A[j]){
                sum++; continue;
            }
        }
    }
    cout << sum << endl;
}
