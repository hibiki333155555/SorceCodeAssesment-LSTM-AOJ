#include<stdio.h>
#include<vector>
#include<algorithm>
#include<iostream>
using namespace std;
          
int main(){
  int n;
  scanf("%d",&n);
  vector<pair<int,int> > v(n);
  for(int i = 0; i < n; i++){
    int row,column;
    scanf("%d%d",&row,&column);
    v.at(i) = make_pair(row,column);
  }
  vector<vector<int> > dp(n,vector<int>(n));
  for(int l = 0; l < n; l++){
    for(int i = 0; i < n-l; i++){
      if(l == 0){
        dp.at(i).at(i+l) = 0;
      }
      else{
        int min_v = 100000000;
        for(int k = i; k < i+l; k++){
          int a = v.at(i).first;
          int b = v.at(k).second;
          int c = v.at(i+l).second;
          min_v = min(min_v,dp.at(i).at(k) + dp.at(k+1).at(i+l) + a*b*c);
        }
        dp.at(i).at(i+l) = min_v;
      }
    }
  }
  printf("%d\n",dp.at(0).at(n-1));
      
}
