#include <iostream>
using namespace std;

int main()
{
   int n;
   cin >> n;
   int a;
   int cnt = 0, maxcnt = 0;
   for (int i = 0; i < n; i++)
   {
      cin >> a;
      if (a == 1)
      {
         ++cnt;
         maxcnt = max(maxcnt, cnt);
      }
      else
         cnt = 0;
   }
   cout << maxcnt + 1 << endl;
   return 0;
}
