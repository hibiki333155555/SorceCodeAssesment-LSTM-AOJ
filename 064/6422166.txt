#include<stdio.h>
#include<stdlib.h>
int binarySearch(int,int,int);
int na,nb;
int *a,*b;
int main(){
    int cnt=0;
    scanf("%d",&na);
    a=(int *)malloc(sizeof(int)*na);
    for(int i=0;i<na;i++)scanf("%d",a+i);
    for(int i=1;i<na;i++){
        int v=a[i];
        int j=i-1;
        while(j>=0&&a[j]>v){
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=v;
    }
    scanf("%d",&nb);
    b=(int *)malloc(sizeof(int)*nb);
    for(int i=0;i<nb;i++)scanf("%d",b+i);
    for(int i=0;i<nb;i++)cnt+=binarySearch(b[i],0,na);
    printf("%d\n",cnt);
    return 0;
}
int binarySearch(int tar,int rig,int lef){
    int mid=(rig+lef)/2;
    if(rig-lef<=1)return 0;
    if(a[mid]==tar)return 1;
    else if(tar<a[mid])return binarySearch(tar,rig,mid);
    else return binarySearch(tar,mid,lef);
}
