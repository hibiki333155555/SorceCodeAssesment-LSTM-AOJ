import java.util.Scanner;

// 長方形クラス
class Rectangle {
    private int height;
    private int width;

    public Rectangle(int width, int height) {
        this.width = width;
        this.height = height;
    }

    public int getWidth() {
        return width;
    }

    public int getHeight() {
        return height;
    }

    // 円が四角形の中に含まれるか判定する
    public boolean isRectangleInCircle(Circle circle) {
        // 値を取得
        int x = circle.getX();
        int y = circle.getY();
        int r = circle.getRadius();

        // 円が四角形の中に含まれるか判定するフラグ
        boolean flag = true;
        
        // x軸の範囲内か
        if ((x - r) < 0 || (x + r) > width) {
            flag = false;
        }

        // y軸の範囲内か
        if ((y - r) < 0 || (y + r) > height) {
            flag = false;
        }
        return (flag);
    }
}

class Circle {
    private int x;
    private int y;
    private int radius;

    public Circle(int x, int y, int radius) {
        this.x = x;
        this.y = y;
        this.radius = radius;
    }

    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }

    public int getRadius() {
        return radius;
    }

    // 円が四角形の中に含まれるか判定する
    public boolean isRectangleInCircle(Rectangle rectangle) {
        // 値を取得
        int width = rectangle.getWidth();
        int height = rectangle.getHeight();

        // 円が四角形の中に含まれるか判定するフラグ
        boolean flag = true;

        // x軸の範囲内か
        if ((x - radius) < 0 || (x + radius) > width) {
            flag = false;
        }

        // y軸の範囲内か
        if ((y - radius) < 0 || (y + radius) > height) {
            flag = false;
        }
        return (flag);
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int width = sc.nextInt();
        int height = sc.nextInt();
        int x = sc.nextInt();
        int y = sc.nextInt();
        int r = sc.nextInt();

        // クラスを宣言
        Rectangle rectangle = new Rectangle(width, height);
        Circle circle = new Circle(x, y, r);

        if (rectangle.isRectangleInCircle(circle)) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }
}
