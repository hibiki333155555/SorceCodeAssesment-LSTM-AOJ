#include <iostream>
#include <vector>
using namespace std;
int main()
{
    int i,num;
    int n,q;
    int m,x,y;
    cin>>n>>q;
    vector <int> a[n];
    while(q--){
        cin>>m;
        if(m==0){
            cin>>x>>y;
            a[x].push_back(y);
        }
        else if(m==1){
            cin>>x;
            num=a[x].size();//3
            for(i=0; i<num; i++){ // 0 1 2
                cout<<a[x][i]; //1 2 3
                if(i<(num-1)){
                    cout<<" "; //1 2 3
                }
            }
            cout<<endl;
        }
        else{
            cin>>x;
            a[x].clear();
        }
    }
}
