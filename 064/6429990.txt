#include<bits/stdc++.h>
using namespace std;
#define endl "\n"
#define ll long long
#define fix(ans,x) fixed << setprecision(x)<<ans
int main(){
    int n;
    cin>>n;
    int r[n];
    for(int i=0;i<n;i++)cin>>r[i];
    int ans=1,mix=r[n-1];
    for(int i=n-1;i>=0;i--){
        if(mix<r[i]){
            ans++;
            mix=r[i];
        }
    }
    cout<<ans<<endl;
    return 0;
}
