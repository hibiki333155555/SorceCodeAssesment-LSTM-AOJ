#include <bits/stdc++.h>
using namespace std;

using Int = long long;
const char newl = '\n';

template<typename T1,typename T2> inline void chmin(T1 &a,T2 b){if(a>b) a=b;}
template<typename T1,typename T2> inline void chmax(T1 &a,T2 b){if(a<b) a=b;}
template<typename T> void drop(const T &x){cout<<x<<endl;exit(0);}
template<typename T=Int>
vector<T> read(size_t n){
  vector<T> ts(n);
  for(size_t i=0;i<n;i++) cin>>ts[i];
  return ts;
}

//INSERT ABOVE HERE
signed main(){
  cin.tie(0);
  ios::sync_with_stdio(0);

  Int n;
  cin>>n;

  auto rs=read<string>(n);

  Int m=rs.front().size();

  Int ans=0;
  for(Int a=0;a<n;a++){
    Int res=0;
    for(Int b=0;b<n;b++){
      if(a==b) continue;
      auto s=rs[a],t=rs[b];
      Int diff=0;
      for(Int i=0;i<m;i++){
        diff+=s[i]!=t[i];
      }
      res|=diff<=1;
    }
    ans+=res;
  }
  drop(ans);
  return 0;
}
