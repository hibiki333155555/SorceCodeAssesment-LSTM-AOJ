#include<bits/stdc++.h>

using namespace std;

int main()
{
    int q;
    cin >> q;

    set <int> s;

    while(q--){
        int x, l, r, op;

        cin >> op;
        
        if(op == 0){
            cin >> x;

            s.insert(x);

            cout << s.size() << endl;

        }

        else if(op == 1){
            cin >> x;
            auto it = s.find(x);
            
            if(it == s.end()){
                cout << 0 << endl;
            }
            else{
                cout << 1 << endl;
            }

        }

        else if(op == 2){
            cin >> x;
            auto it = s.find(x);

            if(it != s.end()){
                s.erase(it);
            }


        }

        else{
            cin >> l >> r;

            for(auto value: s){
                if(l<=value and value<=r){
                    cout << value << endl;
                }
            }
        }
    
    
    


    }

}
