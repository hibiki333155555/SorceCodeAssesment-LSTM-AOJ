#include<iostream>
using namespace std;
int main(){
    char i;
    int j;
    int n;
    bool cards[4][13];
    cin >> n;
    for(int i=0;i<n;i++){
        cin >> i >> j;
        if(i=='S')cards[0][j]=true;
        if(i=='H')cards[1][j]=true;
        if(i=='C')cards[2][j]=true;
        if(i=='D')cards[3][j]=true;

    }
}
