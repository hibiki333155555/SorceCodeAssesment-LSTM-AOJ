#include "bits/stdc++.h"

#define rep(i,a,b) for(int i=a;i<b;i++)
#define rrep(i,a,b) for(int i=a;i>=b;i--)
#define fore(i,a) for(auto &i:a)
#define all(x) (x).begin(),(x).end()

// #define INF (1LL << 60)
#define INF 1 << 29;

using namespace std;
using ll = long long;
using P = pair<int, int>;
using Pl = pair<ll, ll>;
using T = tuple<int, int, int>;
using Graph = vector<vector<int>>;


// 入力
ll n, W; //荷物の数、許容容量
ll weight[110], value[110]; //重さ、価値

ll dp[10010]; //個数、重さ

int main() {
    cin >> n >> W;
    for (ll i = 0; i < n; ++i) cin >> value[i] >> weight[i];

    // DP初期条件: dp[0][w] = 0
    for (ll w = 0; w <= W; ++w) dp[0] = 0;

    // DPループ
    for (ll i = 0; i <= W; ++i) { //許容容量
        for (ll w = 0; w <= n; ++w) { //荷物の数
            if (i >= weight[w]) dp[i] = max(dp[i - weight[w]] + value[w], dp[i]);
            else dp[i + 1] = dp[i]; //最初のみ使用
        }
    }

    cout << dp[W] << endl; //n以下 W以下 の最大値
}
