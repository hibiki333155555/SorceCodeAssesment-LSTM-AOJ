#include <bits/stdc++.h>
using namespace std;

int main(){
    int H,R;
    cin>>H>>R;
    if(H<0){
        if(H+R==0){
            cout<<0<<"\n";
        }else if(H+R<0){
            cout<<-1<<"\n";
        }else{
            cout<<1<<"\n";
        }
    }else{
        cout<<1<<"\n";
    }
    return(0);
}
