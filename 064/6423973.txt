//char '0-9': 48-57, 'A-M,N-Z': 65-77,78-90,
//     'a-m,n-z': 97-109,110-122

int main(){
    string str;
    stack<ll> s;
    ll a, b;
    while(cin >> str){
        if(str == "+"){
            b = s.top();
            s.pop();
            a = s.top();
            s.pop();
            s.push(a+b);
        }
        else if( str == "-"){
            b = s.top();
            s.pop();
            a = s.top();
            s.pop();
            s.push(a-b);
        }
        else if( str == "*"){
            b = s.top();
            s.pop();
            a = s.top();
            s.pop();
            s.push(a*b);
        }
        else{
            s.push(stoi(str));
        }
    }
    coutll(s.top());
}
