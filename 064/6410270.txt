#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main(){
	cin.tie(nullptr);
	ios::sync_with_stdio(false);
	int n,m; cin >> n >> m;
	vector<int> a(n),b(m);
	set<int> st;
	for(int i=0;i<n;i++){
		cin >> a[i];
		st.insert(a[i]);
	}
	for(int i=0;i<m;i++){
		cin >> b[i];
	}
	sort(b.begin(), b.end());
	for(int i=0;i<m;i++){
		if(st.find(b[i])!=st.end()){
			printf("%d\n",b[i]);
			st.erase(b[i]);
		}
	}
}
