#include <stdio.h>
 int main(void){
    int a,b,i,j;
    
    while(1){
    	scanf("%d %d",&a,&b);
    	if(a==0 && b==0)break;
    	
        for(j=0;j<a;j++){
            for(i=0;i<b;i++){
                printf("#");
            }
            printf("\n");
        }
    	printf("\n");
    }
    return 0;
}
