#include <iostream>
#include <string>
using namespace std;

int Min(int a, int b, int c) {
    int min = a; 
    if (b < min) min = b;
    if (c < min) min = c;
    return min;
}

int EditDistance(string a, string b) {
    int arr[a.size()+1][b.size()+1];
    for (size_t i = 0; i < a.size()+1; i++) {
        for (size_t j = 0; j < b.size()+1; j++) {
            arr[i][j] = 0;
        }
    }
    for (size_t i = 1; i < a.size()+1; i++) arr[i][0] = i;
    for (size_t j = 1; j < b.size()+1; j++) arr[0][j] = j;
    for (size_t i = 1; i < a.size()+1; i++) {
        for (size_t j = 1; j < b.size()+1; j++) {
            int subscost = 0;
            if (a.at(i-1) != b.at(j-1)) subscost = 1;
            arr[i][j] = Min(arr[i-1][j]+1,arr[i][j-1]+1,arr[i-1][j-1]+subscost);
        }
    }
    return arr[a.size()][b.size()];
}

int main() {
    string a;
    string b;
    cin >> a >> b;
    int distance = EditDistance(a, b);
    cout << distance << endl;
    return 0;
}
