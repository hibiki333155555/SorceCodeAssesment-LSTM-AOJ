/*-- 入力 --*/
//main("1 2 9 81 9")
main(require("fs").readFileSync("/dev/stdin", "utf8"));

/*-- 計算&出力 --*/
function main(raw) {
    // 入力処理
    const args = raw.split("\n");
    const input = args[0].split(' ').map(Number);
    // 計算
    const input_sorted = input.sort(function(a,b){
        return a-b;
    });
    // 表示
    const output = input_sorted.reduce((accu,curr) => {
       return `${accu} ${curr} `;
    },"");
    console.log(`${output}`);
}
