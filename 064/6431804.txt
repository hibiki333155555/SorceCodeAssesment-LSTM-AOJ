import java.util.Scanner;

public class Main{
	public static void main(String[] args) {
		Scanner scan=new Scanner(System.in);
		
		int p=0;//置き換え変数
		
		while(true) {//x,yがどちらも0ではない
			int x=scan.nextInt();
			int y=scan.nextInt();
			
			if(x==0||y==0) {
				break;
			} 
				if(x>y) {
					p=x;//置き換え変数にxを保存
					x=y;//yにxを代入
					y=p;//yに置き換え変数を代入
				}
				System.out.println(x+" "+y);
		}
	}
}
