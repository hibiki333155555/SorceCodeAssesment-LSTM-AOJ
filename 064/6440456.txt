#include<iostream>
#include<vector>
#include<cmath>
using namespace std;
const int N = 8;
vector<pair<int,int>> Q;

bool set(vector<pair<int,int>>& S, int q){
    if(S.empty()) return false;

    for(auto [r,c]: S){
        vector<pair<int,int>> T = S;
        for(auto it=T.begin(); it!=T.end(); ){
            int R = it->first, C = it->second;
            if(R==r || C==c || abs(R-r)==abs(C-c)) it = T.erase(it);
            else ++it;
        }
        if(q==1){
            Q.push_back({r,c});
            return true;
        }
        else{
            if(set(T, q-1)){
                Q.push_back({r,c});
                return true;
            }
            else continue;
        }
    }
    return false;
}

int main(){
    int k; cin >> k;
    vector<pair<int,int>> S;
    for(int i=0; i<N; ++i){
        for(int j=0; j<N; ++j){
            S.push_back({i,j});
        }
    }
    
    for(int i=0; i<k; ++i){
        int r, c; cin >> r >> c;
        Q.push_back({r,c});
        for(auto it=S.begin(); it!=S.end(); ){
            int R = it->first, C = it->second;
            if(R==r || C==c || abs(R-r)==abs(C-c)) it = S.erase(it);
            else ++it;
        }
    }
    
    set(S, N-k);
    vector<vector<char>> B(N, vector<char>(N,'.'));
    for(auto [r,c]: Q) B[r][c] = 'Q';
    for(int i=0; i<N; ++i){
        for(int j=0; j<N; ++j){
            cout << B[i][j];
        }
        cout << endl;
    }
    return 0;
}
