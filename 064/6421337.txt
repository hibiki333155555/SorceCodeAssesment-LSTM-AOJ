//int N, M, K, T, H, W, L, R;
long long int N, M, K, T, H, W, L, R;

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);

	cin >> N;
	vector<int>v(N);
	for (auto& i : v)cin >> i;
	cin >> K;
	deque<pair<int, int>>Q;
	while (K--) {
		cin >> M;
		if (M == 1) {
			cin >> L >> R;
			L--, R--;
			Q.push_back({ L,R });
		}
		else if (M == 2) {
			Q.pop_back();
		}
		else {
			cin >> M;
			vector<int>nx(N);
			for (int i = 0; i < N; i++) {
				nx[i] = i;
			}
			for (auto i : Q) {
				swap(nx[i.first], nx[i.second]);
			}
			for (int b = 0; b < 30; b++) {
				if (M >> b & 1) {
					vector<int>w(N);
					for (int i = 0; i < N; i++) {
						w[i] = v[nx[i]];
					}
					v = w;
				}
				vector<int>w(N);
				for (int i = 0; i < N; i++) {
					w[i] = nx[nx[i]];
				}
				nx = w;
			}
		}
	}
	for (auto i : v)cout << i << " ";
	cout << endl;
}
