using System;
using System.Collections;
using System.Collections.Specialized;
using System.Linq;

namespace AOJ_ITP1_6_B
{
    internal class Program
    {
        public static void Main()
        {
            var row = int.Parse(Console.ReadLine());
            var cards = new OrderedDictionary();
            foreach (var suit in new[] { "S", "H", "C", "D" })
                foreach (var number in Enumerable.Range(1, 13))
                {
                    cards.Add(suit + " " + number, false);
                }

            for (int i = 0; i < row; i++)
            {
                cards[Console.ReadLine()] = true;
            }

            foreach (var card in cards.Cast<DictionaryEntry>().Where(c => (bool)c.Value == false))
            {
                Console.WriteLine(card.Key);
            }
        }
    }
}
