#include <bits/stdc++.h>
using namespace std;

int main(){
    while(true){
        int H,W;
        cin >> H >> W;
        if(H==0&&W==0){
            cout << endl;
            return 0;
        }
        for(int i = 0;i < W;i++) cout << '#';
        cout << endl;
        for(int i = 0;i < H-2;i++){
            cout << '#';
            for(int j = 0;j < W-2;j++) cout << '.';
            cout << '#' << endl;
        }
        for(int i = 0;i < W;i++) cout << '#';
        cout << endl << endl;
    }
}
