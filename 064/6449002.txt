struct node{
    int v, w;
    bool operator < (const node b) {
        return v * b.w > b.v * w;
    }
}a[maxn];

int n, W;
double ans;

signed main() {
    #ifdef moyi_qwq
        freopen("D:/source file/intxt/in.txt","r",stdin);

    cin>>n>>W;
    for(int i = 1; i <= n; i++) {
        cin>>a[i].v>>a[i].w;
    }
    sort(a+1,a+1+n);
    for(int i = 1; i <= n; i++) {
        if(W >= a[i].w) {
            ans += a[i].v;
            W -= a[i].w;
        }
        else {
            ans += 1.0*W/a[i].w*a[i].v;
            W = 0;
        }
        if(!W) break;
    }
    printf("%.10lf\n",ans);
    
    
    //cerr<<"Time : "<<1000*((double)clock())/(double)CLOCKS_PER_SEC<<"ms";
