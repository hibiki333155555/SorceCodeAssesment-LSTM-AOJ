#include <iostream>
#include <vector>
#include <tuple>

using namespace std;

int main()
{
	int n; cin >> n;
	vector<tuple<int, int, int, int>> S(n);
	for (auto& [x1, y1, x2, y2] : S) cin >> x1 >> y1 >> x2 >> y2;
	vector<tuple<int, int, int>> YX, XY;
	for (const auto& [x1, y1, x2, y2] : S) if (x1 == x2) {
		YX.push_back({x1, y1, y2});
	} else {
		XY.push_back({y1, x1, x2});
	}
	int c = 0;
	for (auto [y, x1, x2] : XY) {
		for (auto [x, y1, y2] : YX) if (x1 <= x && x <= x2 && y1 <= y && y <= y2) ++c;
	}
	cout << c << endl;
	return 0;
}
