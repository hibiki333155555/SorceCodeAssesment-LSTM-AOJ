using System;
using System.Linq;
class program
{
    static void Main(string[] args)
    {
        int k = int.Parse(Console.ReadLine());
        int[] x = Console.ReadLine().Split().Select(int.Parse).Toarray;
        long sum = 0;
        for(int i = 0; i < k; i++)
        {
            sum = sum + x[i];
        }
        Array.Sort(x);
        Console.WriteLine(x[0] + " " + x[k - 1] + " " + sum);
    }
}
