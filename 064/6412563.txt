#include<bits/stdc++.h>
using namespace std;
 int main()
{

    int sum_r=0;
    int sum_c = 0;
    vector<vector<int>> v(n, vector<int> (m));
    vector<vector<int>> v1(n+1, vector<int> (m+1));
    for(int i = 0; i< n; i++){
        for(int j = 0; j< m; j++){
           cin >> a;
           v[i][j] = a;
        }
    }
   
    for(int i = 0; i< n+1; i++){
    sum_r =0;
        for(int j = 0; j< m+1; j++){
            if(i == n){
            v1[i][j] =0;
            }
            else{
            if(j == m){
            v1[i][j] = sum_r;
            }
            else{
            sum_r = sum_r + v[i][j];
            v1[i][j] = v[i][j];
            }
            }
    }
    }
   
    for(int j = 0; j< m+1; j++){
    sum_c =0;
        for(int i = 0; i< n+1; i++){
            if(i == n){
            v1[i][j] =sum_c;
            }
            else{
            sum_c = sum_c + v1[i][j];
            }
    }
    }
   
     for(int i = 0; i< n+1; i++){
        for(int j = 0; j< m+1; j++){
        if(j == m){
        cout << v1[i][j];
        }

        cout << v1[i][j] << " ";
        }
         
        }

    }
   
    return 0;
}
