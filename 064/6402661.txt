#include <bits/stdc++.h>
using namespace std;

int main(){
    while(EOF){
        string op;
        int a, b;
        cin >> a >> op >> b;
        if(op == "?") break;
        if(op == "+") cout << a + b << endl;
        else if(op == "-") cout << a - b << endl;
        else if(op == "*") cout << a * b << endl;
        else if(op == "/") cout << a / b << endl;
    }
    return 0;
}
