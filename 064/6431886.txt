#include <iostream>
#include <math.h>
#include <vector>
#include <math.h>
#include <string>

using namespace std;
int main() {
    //ios::sync_with_stdio(false);
  //  cin.tie(nullptr);
    
    int N, Q;
    cin >> N;
    vector<int>A(N);
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    cin >> Q;
    vector<int>M(Q);
    for (int i = 0; i < Q; i++) {
        cin >> M[i];
    }

    int N2jou=pow(2,N);//bit全探索するにはN桁ぶんの二進数を作らねばならない。
    vector<int>Keisan(N2jou);

    int NowTashizan = 0;
    bool OK = false;
    for (int i = 0; i < N2jou; i++) {
        NowTashizan = 0;
        for (int nowketa = 0; nowketa < N; nowketa++) {
            if (i & (1 << nowketa)) {
                NowTashizan += A[nowketa];
            }
        }
        Keisan[i] = NowTashizan;
    }
    NowTashizan = 0;
    for (int m = 0; m < Q; m++) {
        for (int i = 0; i <= N2jou; i++) {
            if (Keisan[i] == M[m]) {
                OK = true;
                break;
            }
        }
        if (OK) {
            cout << "yes" << endl;
        }
        else {
            cout << "no" << endl;
        }
    }
    return 0;
}
