#include <bits/stdc++.h>
using namespace std;

int main() {
    while(true) {
        int N,K;
        cin >> N >> K;
        if(N == 0 && K == 0) {
            return 0;
        }
        int l = 0,r = 1001001001;
        while (l+1 < r) {
            int mid = (l+r)/2;
            long long sum = 0,cnt = 1;
            for(long long i = 1; i <= mid; i *= 10) {
                long long j = min((long long)mid,i*10-1);
                sum += (j-i+1)*cnt;
                cnt++;
            }
            if(sum < N) {
                l = mid;
            }
            else {
                r = mid;
            }
        }
        long long sum = 0,cnt = 1;
        for(long long i = 1; i <= l; i *= 10) {
            long long j = min((long long)l,i*10-1);
            sum += (j-i+1)*cnt;
            cnt++;
        }
        string ans = "";
        for(int i = l+1; ; i++) {
            ans += to_string(i);
            if(ans.size() >= K+(N-sum-1)) {
                cout << ans.substr(N-sum-1,K) << endl;
                break;
            }
        }
    }
}
