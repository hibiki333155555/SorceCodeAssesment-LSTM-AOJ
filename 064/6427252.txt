// G - Sum of Numbers

#include <iostream>
#include <string>
using namespace std;

int main() {
    int n;
    int sum =0;
    int count=0;
    int m;
    while (cin >> n)
    {
        count = to_string(n).length();
        if (n!=0) {
            for (int i=0; i<count; i++) {
                sum = sum + (n%10);
                n = n/10;
            }
        }
        else {
            break;
        }
        cout << sum << endl;
        sum =0;
        count =0;
    }
}
