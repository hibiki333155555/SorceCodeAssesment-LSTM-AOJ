#include<bits/stdc++.h>
using namespace std;
int main()
{
   long long int a[100][100],b[100][100],mul[100][100],c,d,e,k;
cin>>c>>d>>e;
for(int i=0;i<c;i++)
 {
	for(int j=0;j<d;j++)
	{
		cin>>a[i][j];
	}
 }
 for(int i=0;i<d;i++)
 {
 	for(int j=0;j<e;j++)
 	{
 		cin>>b[i][j];
 	}
 }
 for(int i=0;i<c;i++)
 {
 	for(int j=0;j<e;j++)
 	{
 		mul[i][j]=0;
 		for(int k=0;k<d;k++)
 		{
 			mul[i][j]+=a[i][k]*b[k][j];
 		}
 	}
 }
 for(int i=0;i<c;i++)
 {
   for(int j=0;j<e;j++)
   {
   	cout<<mul[i][j];
   	if(j!=e-1)
   	cout<<" ";
   	
  
   }
   cout<<endl;
 }
 return 0;
}
