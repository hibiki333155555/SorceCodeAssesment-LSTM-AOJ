#include <iostream>

using namespace std;

int main () {
    int n, count[4][3][10] = {{{}}};
    int b, f, r, v;
    cin >> n;
    
    for (int i = 0; i < n; i++) {
        cin >> b >> f >> r >> v;
        
        count[b - 1][f - 1][r - 1] += v;
    }
    
    for(int a = 0; a < 4; a++) {
        for(int b = 0; b < 3; b++) {
            for(int c = 0; c < 10; c++) {
                cout << " " << count[b][f][c]; 
            }
            cout << endl;
        }
        if(a < 3) cout << "####################" << endl;
    }
}
