#include <iostream>
#include <vector>
using namespace std;

int N;
int arr[10];

int main() {
  cin >> N;
  while (N--) {
    for (int i = 0; i < 10; ++i)
      cin >> arr[i];

    vector<int> left, right;
    bool f = true;
    for (int i = 0; i < 10; ++i) {
      if (left.size() == 0 || arr[i] > left.back())
        left.push_back(arr[i]);
      else if (right.size() == 0 || arr[i] > right.back())
        right.push_back(arr[i]);
      else {
        f = false;
        break;
      }
    }
    if (f)
      cout << "YES\n";
    else
      cout << "NO\n";
  }
}
