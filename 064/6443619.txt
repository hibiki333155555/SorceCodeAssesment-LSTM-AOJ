#include<iostream>
#include<string>
using namespace std;
int main(){
    int judge[2001];
    for(int i=0; i<2001; i++){
        judge[i]=0;
    }
    int N;
    cin >> N;
    int A[N];
    for(int i=0; i<N; i++){
        cin >> A[i];
        judge[A[i]]++;
    }
    int min=101;
    for(int i=0; i<2001; i++){
        if(judge[i]<min && judge[i]!=0){
            min=judge[i];
        }
    }
    for(int i=0; i<2001; i++){
        if(judge[i]==min){
            cout << i << endl;
            return 0;
        }
    }
}
