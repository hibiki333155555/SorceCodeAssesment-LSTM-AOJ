#include <iostream>
#include <climits>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >>n;
    int minN = INT_MAX, maxN=INT_MIN;
    long long sum=0;
    for (int i=0; i<n; i++) {
        int x; cin>>x;
        minN = min(minN,x);
        maxN = max(maxN,x);
        sum+=x;
    }
    cout <<minN<<" "<< maxN<<" "<< sum<< endl;
}
