#include <iostream>
#include <list>

using namespace std;

bool canLoad(int *weights, int weights_size, int trucks_size, int P)
{
  int trucks_i = 0;
  int weight_i;
  int trucks_sum = 0;
  for (weight_i = 0; weight_i < weights_size; weight_i++)
  {
    trucks_sum += weights[weight_i];

    if (trucks_sum > P)
    {
      trucks_sum = weights[weight_i];
      trucks_i++;

      if (trucks_i > trucks_size - 1) break;
    }
    else if (trucks_sum == P)
    {
      trucks_sum = 0;
      trucks_i++;
      if (trucks_i > trucks_size - 1) break;
    }
  }
  
  return (trucks_i == trucks_size - 1 && weight_i == weights_size - 1) ? true : false;
}

int main(int argc, char const *argv[])
{
  int n, k;
  cin >> n >> k;

  int *weights = new int[n];
  int sum = 0;

  for (int i = 0; i < n; i++)
  {
    cin >> weights[i];
    sum += weights[i];
  }
  
  int P = sum;

  int left = 0;
  int right = sum;

  while (left <= right)
  {
    int mid = (left + right) / 2;
    if (canLoad(weights, n, k, mid))
    {
      P = mid;
      right = mid - 1;
    }
    else left = mid + 1;
  }
  
  cout << P << endl;

  return 0;
}
