#include <stdio.h>
int main(void){
    int i, n, a,  max = -1000000, min = 1000000;
    long int sum = 0;
    scanf("%d", &n);
    for(i = 0; i < n; i++){
        scanf("%d", &a);
        if(a < min ){
            min = a;
        }
        if(a > max){
            max = a;
        }
        sum = sum + a;
    }
    printf("%d %d %ld\n", min, max, sum);
    return 0;
}
