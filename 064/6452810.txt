#include <bits/stdc++.h>
using namespace std;
int main(){
  int N;
  cin >> N;
  vector<string> report(N);
  for (int i = 0; i < N; i++){
    cin >> report[i];
  }
  int ans = 0;
  for (int i = 0; i < N; i++){
    bool ok = false;
    for (int j = 0; j < N; j++){
      if (j != i){
        int cnt = 0;
        for (int k = 0; k < report[i].size(); k++){
          if (report[i][k] != report[j][k]){
            cnt++;
          }
        }
        if (cnt <= 1){
          ok = true;
        }
      }
    }
    if (ok){
      ans++;
    }
  }
  cout << ans << endl;
}
