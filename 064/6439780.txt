#include <iostream>
#include <string>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string line;
    char c;
    getline(cin, line);
    int alphCnt[26];
    for(int i = 0; i< 26; i++){
        alphCnt[i] = 0;
    }
    for (int i = 0; i < line.length(); i++)
    {
        if (isalpha(line[i]))
        {
            alphCnt[tolower(line[i]) - 'a']++;
        }
        else
        {
            continue;
        }
    }

    for (int i = 0; i < 26; i++)
    {
        cout << (char)('a' + i) << " : " << alphCnt[i] << endl;
    }

    return 0;
}
