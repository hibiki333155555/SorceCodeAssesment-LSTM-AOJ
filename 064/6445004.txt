#include <bits/stdc++.h>

int main() {
    int n; std::cin >> n;
    std::vector<int> S(n); for (auto &e : S) std::cin >> e;
    int q; std::cin >> q;

    int ans = 0;
    for (int i = 0; i < q; i++) {
        int t; std::cin >> t;
        int left = -1, right = n;
        while (abs(right - left) > 1) {
            int mid = (left + right) / 2;
            if (S[mid] >= t) {
                right = mid;
            } else {
                left = mid;
            }
        }
        if (S[right] == t) ans++;
    }
    std::cout << ans << std::endl;
}
