ll mod_pow(ll x, ll expo, ll mod) {
    /*
        return (x  ** expor ) % mod
    */

    ll res = 1;

    while (expo > 0) {
        if (expo & 1) {
            res = (res * x) % mod;
        }
        x = (x * x) % mod;

        expo = expo >> 1;
    }
    return res;
}
