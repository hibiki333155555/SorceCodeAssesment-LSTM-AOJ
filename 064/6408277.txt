#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void _min(vector<long long>v, a, b)
{
    long long ret = 1000000001;
    for(int i = a; i < b; i++){
        ret = min(ret, v.at(i));
    }
    return ret;
}

void _max(vector<long long>v, a, b)
{
    long long ret = -1000000001;
    for(int i = a; i < b; i++){
        ret = max(ret, v.at(i));
    }
    return ret;
}

int main(){
    long long n, e;
    vector<long long> v;
    cin>>n;
    while(n--){
        cin>>e;
        v.push_back(e);
    }
    cin>>n;
    long long op, a, b;
    while(n--){
        cin>>op>>a>>b;
        switch(op){
            case 0:
            cout << _min(v, a, b) << endl;
            break;
            case 1
            cout << _max(v, a, b) << endl;
        }
    }
}
