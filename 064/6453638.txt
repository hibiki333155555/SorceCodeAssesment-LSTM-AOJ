#include<iostream>
#include<vector>
using namespace std;
using ll = long long;

int main(){
    int N, Q; cin >> N >> Q;
    vector<ll> s(N+1,0);
    for(int i=1; i<=N; ++i){
        ll a; cin >> a;
        s[i] = s[i-1] + a;
    }

    while(Q--){
        ll ans = 0;
        ll x; cin >> x;
        int r = 0;
        for(int l=0; l<=N; ++l){
            while(r <= N && s[r]-s[l] <= x) r++;
            ans += r - l - 1;
            if(r == l) r++;
        }
        cout << ans << endl;
    }
    return 0;
}
