#include <iostream>
using namespace std;

int main()
{
   int n, a, b, c, d;
   cin >> n >> a >> b >> c >> d;
   int cost_x = 0, cost_y = 0;
   int pen = 0;
   do
   {
      pen += a;
      cost_x += b;
   } while (pen < n);

   pen = 0;
   do
   {
      pen += c;
      cost_y += d;
   } while (pen < n);
   
   cout << min(cost_x, cost_y) << endl;
   return 0;
}
