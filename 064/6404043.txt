int convert(char x)
{
  if (x >= 'A' and x <= 'Z')
    return x - 'A';
  if (x >= 'a' and x <= 'z')
    return x - 'a' + 26;
  if (x == '+')
    return 62;
  if (x == '/')
    return 63;
  return x - '0' + 52;
}

vector<vector<int>> spin(vector<vector<int>> now)
{
  vector<vector<int>> ans = now;
  REP(i, now.size())
  {
    REP(q, now[i].size())
    {
      ans[q][now.size() - i - 1] = now[i][q];
    }
  }
  return ans;
}

            checks[t].second %= mod2;

        {
          ans += (checks == moves);
        }
        else

          S.second *= 2;
          S.first %= mod1;
          S.second %= mod2;
        }
      }
    }
  }
  cout << ans << endl;
}
