import java.math.BigDecimal;
import java.math.MathContext;
import java.text.DecimalFormat;
import java.util.Scanner;

public class Main {
	
	@SuppressWarnings("resource")
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);				
		BigDecimal r = sc.nextBigDecimal();
		
		String format = "0.######";
		DecimalFormat decimalFormat = new DecimalFormat(format);
		
		BigDecimal r2 = new BigDecimal(3.141592);
		BigDecimal r3 = r.add(r);
		BigDecimal bd = r.multiply(r);
		BigDecimal bd1 = bd.multiply(r2 , MathContext.DECIMAL64);
		BigDecimal bd2 = r3.multiply(r2 , MathContext.DECIMAL64);
		
		System.out.println(decimalFormat.format(bd1) + " " + decimalFormat.format(bd2));
    }
}
