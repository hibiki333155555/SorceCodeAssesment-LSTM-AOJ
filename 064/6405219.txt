#include <bits/stdc++.h>
using namespace std;
const double PI =acos(-1);
const long double EPS = 1e-14;

int main(){
  long N, cnt = 0;
  cin >> N;
  vector<long> A(N);
  for(long i = 0; i < N; i++){
    cin >> A[i];
  }
  for(long i = 0; i < N; i++){
    for(long j = N - 1; i < j; j--){
      if(A[j - 1] < A[j]){
        continue;
      }else{
        long tmp = A[j - 1];
        A[j - 1] = A[j];
        A[j] = tmp;
        cnt++;
      }
    }
  }
  for(long i = 0; i < A.size(); i++){
    if(i == A.size() - 1){
      cout << A[i] << endl;
    }else{
      cout << A[i] << " ";
    }
  }
  cout << cnt << endl;
  return 0;
}
