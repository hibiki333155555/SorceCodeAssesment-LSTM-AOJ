from collections import deque

ops = {"insert ": deque.appendleft,
       "delete ": lambda deq, x: deq.remove(x) if x in deq else None, #キー x を持つ要素があればそのうち最初の要素を連結リストから削除する。
       "deleteF": deque.popleft, #リストの先頭の要素を削除する。
       "deleteL": deque.pop} #リストの末尾の要素を削除する。

deq = deque()
for _ in range(int(input())):
  op = input()
  if op[6] == " ":
    ops[op[:7]](deq, op[7:])
  else:
    ops[op[:7]](deq)
print(*deq)
