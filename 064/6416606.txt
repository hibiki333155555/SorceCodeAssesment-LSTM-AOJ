#include <iostream>
#include <cmath>
#include <cstdlib>
#include <stdio.h>
#include <vector>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)


int main() {
    int n;
    cin >> n;
    vector<int>r(n);
    rep(i,n) cin >> r[i];
    int minv = r[0];
    int maxv = r[1]-minv;
    for(int i=1;i<n;i++) {
        maxv = max(maxv,r[i]-minv);
        minv = min(minv,r[i]);
    }
    cout << maxv << endl;
    return 0;
}
