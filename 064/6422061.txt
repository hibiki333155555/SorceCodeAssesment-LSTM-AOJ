#include <stdio.h>

int main(void){

    int n;
    scanf("%d", n);

    int arr[4][13];
    int i, j;
    for(i = 0; i < 4; i++) for(j = 0; j < 13; j++) arr[i][j] = 0; // initialize

    char sym;
    int num;
    i = n;
    while(i--){
        scanf("%c %d", &sym, &num);
        if(sym == 'S') arr[0][num] = 1;
        if(sym == 'H') arr[1][num] = 1;
        if(sym == 'C') arr[2][num] = 1;
        if(sym == 'D') arr[3][num] = 1;
    }

    int counts[4];
    char syms[4] = {'S', 'H', 'C', 'D'};
    for(i = 0; i < 4; i++) counts[i] = 0; // initialize
    for(i = 0; j < 4; j++){
        for(j = 0; j < 13; j++) if(arr[i][j]) counts[i]++;
        printf("%c %d\n", syms[i], counts[i]);
    }



    return 0;
}
