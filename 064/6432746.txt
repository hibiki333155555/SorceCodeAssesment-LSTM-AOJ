#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;
	cin >> n;
	int a[n];
	for(int i=0;i<n;i++){
		cin >> a[i];
	}
	int m;
	cin >> m;
	int b[m];
	for(int i=0;i<m;i++){
		cin >> b[i];
	}
	int count = 0;
	for(int i=0;i<m;i++){
		for(int j=0;j<n;j++){
			if(b[i] == a[j]){
				count ++;
			}
		}
	}
	if(count == m){
		cout << "1" << endl;;
	}
	else{
		cout << "0" << endl;
	}
	return 0;
}
