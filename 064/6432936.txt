#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int z[n];
    int ans[n];
    for(int i=0;i<n;i++){
        cin>>z[i];
        ans[i]=z[i];
    }
    sort(ans,ans+n);
    int sum=0;
    for(int i=0;i<n;i++){
        ans[i]==z[i]?sum++:sum=sum;
    }
    if(sum==n){
        cout<<0<<endl;
        return 0;
    }
    int q;
    cin>>q;
    int x[q],y[q];
    for(int i=0;i<q;i++){
        cin>>x[i]>>y[i];
    }
    for(int i=0;i<q;i++){
        swap(z[x[i]-1],z[y[i]-1]);
        z[x[i]-1]==ans[x[i]-1]?sum++:sum--;
        z[y[i]-1]==ans[y[i]-1]?sum++:sum--;
        cout<<endl;
        if(sum==n){
            cout<<i+1<<endl;
            return 0;
        }
    }
    cout<<-1<<endl;
    return(0);
}
