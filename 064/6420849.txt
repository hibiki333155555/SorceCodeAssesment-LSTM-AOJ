#include <stdio.h>

int main()
{
	char S[200001], ans[200001];
	scanf("%s", S);
	
	int i, j, k, s[200001], head = 0;
	for (i = 0, j = 0, k = 0; S[i] != 0; i++) {
		if (S[i] >= '0' && S[i] <= '9') {
			s[head] = S[i] - '0';
			k += s[head++];
		} else if (S[i] == '(') continue;
		else if (S[i] == ')') k -= s[--head];
		else ans[j++] = ((S[i] - 'a') + k) % 26 + 'a';
	}
	ans[j] = 0;
	printf("%s\n", ans);
	fflush(stdout);
	return 0;
}
