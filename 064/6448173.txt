#define FOR(i, a, b) for (int i = (int)(a); i < (int)(b); ++i)
#define rep(i, n) for (int i = 0; i < (int)(n); ++i)

struct card
{
    char x;
    int y;

    bool operator<(const card &right) const
    {
        return x == right.x ? y < right.y : x < right.x;
    }
};

int main()
{
#ifndef DEBUG
    int n;
    int a[1000];

    cin >> n;
    rep(i, n)
    {
        cin >> a[i];
    }

    int v, j;
    
    rep(i, n - 1) cout << a[i] << " ";
    cout << a[n - 1] << endl;

    FOR(i, 1, n)
    {
        v = a[i];
        j = i - 1;
        while (j >= 0 && a[j] > v)
        {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = v;
        rep(i, n - 1) cout << a[i] << " ";

        cout << a[n - 1] << endl;
    }

#else
    int j;

    }
#endif
}
