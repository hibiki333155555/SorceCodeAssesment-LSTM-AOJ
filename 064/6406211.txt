struct SegmentTree {
private:
  int n;
  vector<ll> node;

public:
  SegmentTree(int sz) {
    n = 1; while(n < sz) n *= 2;
    node.assign(2*n-1, 0);
  }

  void add(int i, ll val) {
    i += n-1;
    node[i] += val;
    while(i>0) {
      i = (i-1)/2;
      node[i] = node[i*2+1] + node[i*2+2];
    }
  }

  ll getsum(int a, int b, int k = 0, int l = 0, int r = -1) {
    if(r < 0) r = n;
    if(b <= l or r <= a) return 0;
    if(a <= l and r <= b) return node[k];
    int m = (l+r)/2;
    ll v1 = getsum(a, b, 2*k+1, l, m);
    ll v2 = getsum(a, b, 2*k+2, m, r);
    return v1 + v2;

  }

};


int main(){
  cin.tie(nullptr);
  ios::sync_with_stdio(false);
 
  int n, q;cin>>n>>q;
  SegmentTree seg(n);
  while(q--) {
    ll c,x,y;cin>>c>>x>>y;
    if(c == 0) seg.add(x, y);
    else cout << seg.getsum(x, y+1) << endl;
  }
}
