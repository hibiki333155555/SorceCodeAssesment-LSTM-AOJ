#include<stdio.h>
int da[12];
void fue(int l,int r,int m){
    int yo[12];
    for(int i=l;i<m;i++) yo[i]=da[i];
    for(int i=m;i<r;i++) yo[m+r-i-1]=da[i];
    int tr=r-1;
    for(int i=l;i<r;i++){
        if(yo[l]>yo[tr]){da[i]=yo[tr];tr--;}
        else{da[i]=yo[l];l++;};
    }
}
void sort(int l,int r){
    if(l+1>=r){return;};
    int meas=(l+r)/2;
    sort(l,meas);
    sort(meas,r);
    fue(l,r,meas);
}
int main(void){
    for(int i=0;i<12;i++) scanf("%d",&da[i]);
    sort(0,12);
    int j=1;
    for(int i=0;i<12;i+=4){
        if(da[i]!=da[i+3]) j=0;
    }
    printf(j?"yes\n":"no\n");
    return 0;
}
