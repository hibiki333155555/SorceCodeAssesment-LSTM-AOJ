#include <iostream>
#include <utility>
#include <algorithm>
using namespace std;

string bin(int x, int digit = 1)
{
    string ret;
    int cnt = 0;
    while (x > 0)
    {
        ret += (char)('0' + x % 2);
        x /= 2;
        cnt++;
    }
    for (; cnt < digit; cnt++)
    {
        ret += '0';
    }
    reverse(ret.begin(), ret.end());
    return ret;
}

int main()
{
    int a, b;
    cin >> a >> b;
    if (b > a)
    {
        swap(a, b);
    }
    string as = bin(a);
    string bs = bin(b, as.size());
    int ans = 0;
    for (int i = 0; i < as.size(); i++)
    {
        if (as[i] != bs[i])
        {
            break;
        }
        ans++;
    }
    cout << ans << endl;
    return 0;
}
