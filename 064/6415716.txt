#include<iostream>
#include<string>
using namespace std;

int main(){

    string str,ans;
    int num,k;

    while(1){
        cin>>str;
        if(str == "_")break;
        cin>>num;
        for(int i=0;i<num;++i){
            cin>>k;
            ans=str.substr(0,k)+str.substr(k);
        }
        cout<<ans<<endl;
    }

    

    return 0;
}
