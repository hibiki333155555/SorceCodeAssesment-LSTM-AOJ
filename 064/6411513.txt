#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main()
{
    int a,b,c;
    scanf("%d %d %d",&a,&b,&c);
    if(a<b && b<c)
    {
        printf("%d %d %d\n",a,b,c);
    }
    else if(b<a && a<c)
    {
        printf("%d %d %d\n",b,a,c);
    }
    else if(c<a && a<b)
    {
        printf("%d %d %d\n",c,a,b);
    }
    else if(b<c && c<a)
    {
        printf("%d %d %d\n",b,c,a);
    }
    else if(c<b && b<a)
    {
        printf("%d %d %d\n",c,b,a);
    }
    else
    {
        printf("%d %d %d\n",a,c,b);
    }
    return 0;
}
