#include<iostream>
using namespace std;
int main(){
    int edges[100]={0};
    for(int i=0;i<12;i++){
        int tmp;
        cin>>tmp;
        edges[tmp-1]++;
    }
    for(int i=0;i<100;i++){
        if((edges[i]%4)){
            cout<<"no"<<endl;
            return 0;
        }
    }
    cout<<"yes"<<endl;
    return 0;
}
