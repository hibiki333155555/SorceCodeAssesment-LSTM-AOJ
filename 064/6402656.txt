import numpy as np
n, m = map(int, input().split())
A = np.array([[0 for i in range(m)] for j in range(n)])
b = np.array([0 for i in range(m)])

for i in range(n):
    A[i] = list(map(int, input().split()))

for i in range(m):
    b[i] = int(input())

m = np.dot(A, b)

for i in range(n):
    print(f"{m[i]}")
