import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int S= sc.nextInt();
        int a=S/3600;
        int b=(S-3600*a)/60;
        int c=(S-3600*a)%60;
        System.out.println(a+":"+":"+b+":"+c);
        
    }
}
