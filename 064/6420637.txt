int main() {
  int n; cin >> n;
  map<int, pair<int, int>> a;
  REP(i, n) REP(j, n) {
    int aij; cin >> aij;
    if (!a.count(aij)) a[aij] = {i, j};
  }
  int q; cin >> q;
  while (q--) {
    int t; cin >> t;
    if (a.count(t)) {
      const auto [i, j] = a[t];
      cout << i + 1 << ' ' << j + 1 << '\n';
    } else {
      cout << "-1\n";
    }
  }
  return 0;
