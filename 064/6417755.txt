#include <iostream>
#include <cmath>
#include <cstdlib>
#include <stdio.h>
#include <vector>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)


int main() {
    int n;
    cin >> n;
    vector<int>a(n);
    rep(i,n) cin >> a[i];
    bool flag = 1;
    int cnt = 0;
    while(flag) {
        flag = 0;
        for(int i=n-1;i>0;i--) {
            if(a[i] < a[i-1]) {
                swap(a[i],a[i-1]);
                flag = 1;
                cnt++;
            }
        }
    }
    rep(i,n) cout << a[i] << " ";
    cout << endl;
    cout << cnt << endl;
    return 0;
}
