#include <iostream>
using namespace std;

int main(){
   
   int num,h,m,s;
   cin >> num;
   
   m = num / 60;
   s = num % 60;
   
   if(m<60)
       cout << 0 << ":" << m << ":" << s << '\n';
       
   else
     h = m / 60;
     m = m % 60;
     cout << h << ":" << m << ":" << s << '\n';
       
   
   
    return 0;
}
