#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string str;
    int q;
    cin >> str >> q;
    for(int i=0;i<q;i++) {
        string ord;
        int a,b;
        cin >> ord >> a >> b;
        if (ord == "replace") {
            string p;
            cin >> p;
            str.replace(a, b-a+1, p);
        } else if (ord == "reverse") {
            reverse(str.begin()+a, str.begin()+b+1);
        } else {
            cout << str.substr(a, b-a+1) << endl;
        }
    }
}
