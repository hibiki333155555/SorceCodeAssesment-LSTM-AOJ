#include <iostream>
using namespace std;

int main() {
    int r, c;
    cin >> r >> c;
    int sum2[100] = {};
    for (int i = 0; i < r; i++) {
        int sum1 = 0;
        for (int j = 0; j < c; j++) {
            int x;
            cin >> x;
            cout << x << " ";
            sum1 += x;
            sum2[j] += x;
            sum2[c] += x;
        }
        cout << sum1 << endl;
    }
    for (int j = 0; j < c; j++) {
        cout << sum2[j] << " ";
    }
    cout << sum2[c] << endl;

    return 0;
}
