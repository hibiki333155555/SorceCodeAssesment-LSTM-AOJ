#include <bits/stdc++.h>
using namespace std;
int main(){
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    int sum = 0;
    priority_queue<tuple<int, int, int>> q;
    for(int i = 0; i < N; i++){
        cin >> A[i];
        sum += A[i];
        if(A[i] == 1 || A[i] == 6) q.push({5 - A[i], 5, i});
        else q.push({6 - A[i], 6, i});
    }
    while(K-- && !q.empty()){
        auto [sa, num, id] = q.top();
        if(sa < 0) break;
        q.pop();
        sum += sa;
        A[id] = num;
        if(num == 5){
            q.push({1, 6, id});
        }
    }
    cout << sum << endl;
}
