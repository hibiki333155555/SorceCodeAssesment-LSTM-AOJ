import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        int W = sc.nextInt();
        int H = sc.nextInt();
        int x = sc.nextInt();
        int y = sc.nextInt();
        int r = sc.nextInt();
        String ans = "No";
        
        if (x>=r && y>=r && x+r<=W && y+r<=H){
                ans = "Yes";
        }
        
        System.out.println(ans);
    }
}
