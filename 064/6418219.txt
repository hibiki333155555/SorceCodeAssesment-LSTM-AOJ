N, K = map(int, input().split())
A = list(map(int, input().split()))

maxi = {1:5, 2:6, 3:6, 4:6, 5:6, 6:5}
used = [False]*N

sum = 0

for i in range(1, 6):
    for j in range(N):
        if used[j]==False:
            used[j] = True
            if K>0 and A[j]==i and A[j]!=6:
                K -= 1
                sum += maxi[A[j]]
            else:
                sum += A[j]

print(sum)
