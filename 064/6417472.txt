#include <bits/stdc++.h>
using namespace std;
int main(){
  int T;
  cin >> T;
  for (int i = 0; i < T; i++){
    int N, M, v, a, b;
    cin >> N >> M >> v >> a >> b;
    if (N > M){
      cout << "Yes" << endl;
    } else {
      vector<bool> used(M, false);
      used[v] = true;
      bool ok = false;
      for (int j = 0; j < N - 1; j++){
        v = (v * a + b) % M;
        if (used[v]){
          ok = true;
        }
        used[v] = true;
      }
      if (ok){
        cout << "Yes" << endl;
      } else {
        cout << "No" << endl;
      }
    }
  }
}
