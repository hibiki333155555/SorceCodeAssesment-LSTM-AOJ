#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
using P = pair<int, int>;

int main(){
    vector<P> cards(6);
    for(int i = 0; i < 6; i++){
        cin >> cards[i].first >> cards[i].second;
    }
    string ans = "no";
    sort(cards.begin(), cards.end());
    do{
        for(int bit = 0; bit < (1 << 6); bit++){
            for(int i = 0; i < 6; i++){
                if((bit >> i) & 1){
                    swap(cards[i].first, cards[i].second);
                }
            }
            bool isok = true;
            for(int i = 0; i < 6; i += 2){
                if(cards[i] != cards[i+1]
                || cards[i].second != cards[(i+2)%6].first){
                    isok = false;
                }
            }
            if(isok) ans = "yes";
            for(int i = 0; i < 6; i++){
                if((bit >> i) & 1){
                    swap(cards[i].first, cards[i].second);
                }
            }
        }
    }while(next_permutation(cards.begin(), cards.end()));
    
    cout << ans << endl;
    return 0;
}
