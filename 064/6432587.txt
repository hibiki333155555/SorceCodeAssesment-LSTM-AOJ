#include <iostream>
#include <string>
using namespace std;
int is_prime(long long n) {
    if (n <= 1) return 0;
    for (long long p = 2; p * p <= n; ++p) {
        if (n % p == 0) return 0;
    }
    return 1;
}
int main() {
    int N;
    cin >> N;
    int  A[202002];
    for (int i = 1; i <= N; i++)cin >> A[i];
    int count = 0;
    for (int i = 1; i <= N; i++)if (is_prime(A[i]) == 1)count++;
    cout << count << endl;
	
    
  
}
