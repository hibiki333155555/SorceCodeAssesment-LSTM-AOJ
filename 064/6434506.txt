#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int INF = 1001001001;
const ll LINF = 1001001001001001001;
const int MOD = 1000000007;

int main(){
    int n;cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;i++) cin>>a.at(i);
    int mi = a.at(0);
    int ans = -INF;
    for(int i=1;i<n;i++) {
        ans = max(ans,a.at(i)-mi);
        mi = min(mi,a.at(i));
    }
    cout << ans << endl;
}
