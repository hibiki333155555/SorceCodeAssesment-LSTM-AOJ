#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int selectsort(int a[], int n) {
    int minj; int i; int j;
    int sw = 0;
    for (i = 0; i < n - 1; i++) {
        minj = i;
        for (j = i; j < n; j++) {
            if (a[j] < a[minj]) {
                minj = j;

            }
            swap(a[i], a[minj]);
            if (i != minj) { sw++; }
        }



    }


    return sw;
}

int main() {

    int a[10000];
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) { cin >> a[i]; }
    int sw = selectsort(a, n);
    for (int i = 0; i < n; i++) {

        if (i > 0)
            printf(" ");
        printf("%d", a[i]);

    }


    printf("\n");
    printf("%d\n", sw);
    return 0;
}
