#include <bits/stdc++.h>
#define ll long long int
using namespace std;

void solve() {
	ll n, q;
	cin >> n >> q;
	
	vector<vector<ll>>  a(n);
	
	for(int i=0; i<q; i++){
	    int x;
	    cin >> x;
	    if(x==0){
	        int y, z;
	        cin >>  y >> z;
	        a[y].push_back(z);
	    }else if(x==1){
	        int d;
	        cin >> d;
	        if(a[d].size()==0){
	            cout << "\n";
	        }else{
	            cout << a[d][0];
	        
	            for(int k=1; k<a[d].size(); k++){
	                cout << " " << a[d][k];
	            
	            }
	        
	            cout << "\n";
	        }
	    }else if(x==2){
	        int e;
	        cin >> e;
	        a[e].clear();
	    }
	}
	
	
	
}

int main(){
    ios_base::sync_with_stdio(0);
	cin.tie(0); cout.tie(0);

	// freopen("input.txt", "r", stdin);
	// freopen("output.txt", "w", stdout);

	int t = 1;
	// cin >> t;
	while (t--) solve();
	
    return 0;
}
