	return vector<decltype(make_v<T>(ts...))>(a, make_v<T>(ts...));
}

template<typename T, typename V>
typename enable_if<is_class<T>::value == 0>::type
fill_v(T& t, const V& v) { t = v; }

template<typename T, typename V>
typename enable_if<is_class<T>::value != 0>::type
fill_v(T& t, const V& v) {
	for (auto& e : t) fill_v(e, v);
}

	}

}

using P = pair<ll, ll>;

int main() {
	cin.tie(nullptr);
	ios::sync_with_stdio(false);
	int N, M;
	cin >> N >> M;
	vi A(N), B(N), C(N);
	REP(i, N) cin >> A[i] >> B[i] >> C[i];
	int Q;
	cin >> Q;
	vector<P> V(Q);
	cin >> V;
	vll ONI(M+2);
	REP(i, Q){
		ONI[V[i].second] += V[i].first;
	}
	so(V);
	priority_queue<P,vector<P>,greater<> > Q1;
	REP(i, N) {
		Q1.emplace(B[i], i);
	}
	
	int ind = 0;
	ll sum = 0;
	int mx = 0;
	int oni = 0;
	ll ans = 0;
	vll L(M+5);
	FOR(cur,1,M+1){
		while(!Q1.empty() && Q1.top().first == cur){
			auto [b,i] = Q1.top();
			Q1.pop();
			sum += A[i];
			L[C[i]+1] -= A[i];
		}
		sum += L[cur];
		if(sum == 0){
			cout << -1 << endl;
			return 0;
		}
		if (sum < ONI[cur]){
			ll drag = ONI[cur] - sum;
			ans += drag;
			L[mx+1] -= drag;
			sum += drag;
		}
	}
	cout << ans << endl;
	return 0;
}
