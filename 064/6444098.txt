#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,m;
    cin >> n >> m;
    vector<vector<int> > A(n,vector<int>(m));
    for(int i = 0;i < n;i++){
        for(int j = 0;j < m;j++) cin >> A[i][j];
    }
    vector<int> b(m);
    for(int i = 0;i < m;i++) cin >> b[i];
    vector<int> ans(n,0);
    for(int i = 0;i < n;i++){
        for(int j = 0;j < m;j++){
            ans[i] += A[i][j]*b[j];
        }
    }
    for(int i = 0;i < n;i++) cout << ans[i] << endl;
}
