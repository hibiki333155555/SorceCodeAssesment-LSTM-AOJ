#include <iostream>
#include <vector>

using namespace std;

void trace(vector<int>& A, int N){
    for(int i = 0; i < N; i++)
    {
        cout << A[i];
        if(i != N - 1) cout << " ";
    }
    cout << endl;
}

void insertionSort(vector<int>& A, int N){
    int v;
    for(int i = 1; i < N; i++){
        v = A[i];
        int j = i - 1;
        while(j>=0 && v < A[j]){
            A[j + 1] = A[j];
            j--;
        }
        A[j + 1] = v;
        trace(A, N);
    }
}

int main()
{
    ios::sync_with_stdio(false);
    int N, i, j;
    cin >> N;
    vector<int> A(N);
    for(i = 0; i < N; i++){
        cin >> A[i];
    }
    
    trace(A, N);
    insertionSort(A, N);
    
    return 0;
}
