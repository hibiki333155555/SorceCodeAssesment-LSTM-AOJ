
import java.util.Scanner;

public class Main {

	public static final int BIG_NUM = 2000000000;
	public static final long HUGE_NUM = 4000000000000000000L;

	@SuppressWarnings("resource")
	public static void main(String[] args) {

		Scanner scanner = new Scanner(System.in);

		final int SIZE = 1000005;
		Queue Q = new Queue(SIZE);

		int N = scanner.nextInt();
		int q = scanner.nextInt();

		int now = 0,rest = N;
		for(int i = 0; i < N; i++) {

			String name = scanner.next();
			int TIME = scanner.nextInt();

			Q.push(new Process(name,TIME));
		}

		StringBuilder sb = new StringBuilder();

		while(rest > 0) {

			Process p = Q.pop();
			int minus = Math.min(p.getTIME(), q);
			now += minus;

			p.setTIME(p.getTIME()-minus);
			if(p.getTIME() == 0) {

				rest--;
				sb.append(p.getName()).append(" ").append(Integer.toString(now));
				continue;
			}
			Q.push(p);
		}

		System.out.println(sb.toString());
	}
}


class Queue{

	Queue(int arg_size){
		this.size = arg_size;
		A = new Process[size];
		head = 0;
		tail = 0;
	}

	public void push(Process p) {

		A[tail] = new Process(p.getName(),p.getTIME());
		tail++;
	}

	public Process pop() {

		Process ret = new Process(A[head].getName(),A[head].getTIME());
		head++;
		return ret;
	}

	private Process A[];
	private int size;
	private int head,tail;
}

class Process{

	Process(String arg_name,int arg_TIME){
		this.name = arg_name;
		this.TIME = arg_TIME;
	}
	private String name;
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public int getTIME() {
		return TIME;
	}
	public void setTIME(int tIME) {
		TIME = tIME;
	}
	private int TIME;
}
