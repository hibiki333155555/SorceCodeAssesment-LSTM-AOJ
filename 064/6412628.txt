#include <bits/stdc++.h>
using namespace std;

int main() {
	int a,m,l;
	cin>>a>>m>>l;
	vector<vector<long long> >v1(a,vector<long long>(m));
	for(int i=0;i<a;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin>>v1[i][j];
		}
	}
	vector<vector<long long> >v2(m,vector<long long>(l));
	for(int i=0;i<m;i++)
	{
		for(int j=0;j<l;j++)
		{
			cin>>v2[i][j];
		}
	}
	vector<vector<long long> >v3(a,vector<long long>(l));
	for(int i=0;i<a;i++)
	{
		for(int j=0;j<l;j++)
		{
			v3[i][j]=0;
			for(int k=0;k<m;k++){
			v3[i][j]=v3[i][j]+(v1[i][k] * v2[k][j]);
			}
		}
		
	}
	for(int i=0;i<a;i++)
	{
			for(int j=0;j<l;j++)
			{
				cout<<v3[i][j];
				if(j!=l-1){
					cout<<" ";
				}
			}
			cout<<"\n";
	}
	
	return 0;
	
}
