#include <bits/stdc++.h>
using namespace std;
#define endl "\n"

int main ()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

	multimap<string,int> m;
    int q, x;
    string s;
	cin>>q;
	while(q--){
        cin >> x;
        if(!x)
        {
            cin >> s >> x;
            m.insert({s, x});
        }
        else if (x==1)
        {
            cin>> s;
            auto range = m.equal_range(s);
            for (auto x = range.first; x != range.second;++x)
                cout << (*x).second << endl;
        }
        else if (x==2)
        {
            cin>> s;
            m.erase(s);
        }
            

        else 
        {
            string L, R;
            cin >> L >> R;
            map<string, int>::iterator itr, ub;
            ub = m.upper_bound(R);
            for (itr = m.lower_bound(L);itr!=ub;++itr)
                cout << (*itr).first << " " << (*itr).second << endl;
        }
	}
    return 0;
