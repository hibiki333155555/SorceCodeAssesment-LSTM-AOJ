#include <iostream>
#include <iomanip>
#include <vector>
#include <cmath>

using namespace std;

pair<int, double> f(const vector<pair<double, double>>& P, int i)
{
	auto [x0, y0] = P[i];
	double maxD = 0;
	for (int j = 0; j < (int) P.size(); ++j) {
		auto [x, y] = P[j];
		double d = (x - x0) * (x - x0) + (y - y0) * (y - y0);
		if (maxD > d) continue;
		maxD = d;
		i = j;
	}
	return {i, maxD};
}

int main()
{
	int n; cin >> n;
	vector<pair<double, double>> P(n); for (auto& [x, y] : P) cin >> x >> y;
	double d = 0;
	for (int i = 0;;) {
		auto [j, t] = f(P, i);
		if (d >= t) break;
		d = t;
		i = j;
	}
	cout << fixed << setprecision(12) << sqrt(d) << endl;
	return 0;
}
