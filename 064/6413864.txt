#include<iostream>
using namespace std;

int main()
{
	int n;
	int b,f,r,v;
	int room[4][3][10]={{},{},{}};

	cin >> n;

	for (int i = 0; i < n; i++)
	{
		cin >> b >> f >> r >> v;

		room[b-1][f-1][r-1]=+v;
    }
		
	for (int i = 0; i < 4 ; i++){
		for (int j = 0; j < 3 ; j++){
            for(int l=0;l< 10 ;l++){
                
                cout<<" "<<room[i][j][l];
            }
        cout<<endl;
        }
    if(i!=3)cout<<"####################"<<endl;
    }
	return 0;

}
