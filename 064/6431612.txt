#include<bits/stdc++.h>
using namespace std;

int sortCount(vector<int>& v){
	int n=v.size();
	int c=0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(v[j]>v[j+1]){
				swap(v[j],v[j+1]);
				c++;
			}
		}
	}
	return c;
}
int main(){

		int n;cin>>n;
		vector<int> v(n,0);
		for(int i=0;i<n;i++){
			int a;cin>>a;
			v[i]=a;
		}
		int r=sortCount(v);
		for(int i=0;i<n;i++){
			cout<<v[i]<<" ";
		}
		cout<<"\n";
		cout<<r;
		
	
}
