#include <bits/stdc++.h>
#define rep(i,n) for(int i=0;i<(n);i++)
using namespace std;
int main(void){
    // Your code here!
    int n,m;
    vector<int> coins(m);
    cin >> n >> m;
    rep(i,m) cin >> coins[i];
    vector<int> dp(n+10005,100100100);
    dp[0] = 0;
    rep(i,n){
        rep(j,m){
            dp[i+coins[j]] = min(dp[i+coins[j]], dp[i] + 1);
        }
    }
    cout << dp[n] << endl;
}
