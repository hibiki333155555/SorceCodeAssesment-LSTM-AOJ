#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (n); i++)

int main()
{
    int n;
    cin >> n;
    vector<int> S(n);
    rep(i, n) cin >> S[i];
    int q;
    cin >> q;
    vector<int> T(q);
    rep(i, q) cin >> T[i];
    // Sは昇順
    int ans = 0;
    for (auto &it : T)
    {
        int right = n - 1;
        int left = 0;
        int middle;
        while (right - left > 0)
        {
            middle = (right + left) / 2;
            if (it < S[middle])
            {
                right = middle;
            }
            if (it > S[middle])
            {
                left = middle + 1;
            }
            if (it == S[middle])
            {
                ans++;
                break;
            }
        }
    }
    cout << ans << endl;
}
