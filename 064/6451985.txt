#include <stdio.h>

int main(){
    int n,i,a[101],j,t,cnt=0;
    
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    
    for(i=0;i<n;i++){
        for(j=n-1;j>i;j--){
            if(a[j]<a[j-1]){
                t=a[j];
                a[j]=a[j-1];
                a[j-1]=t;
                cnt++;
            }
        }
    }
    
    for(i=0;i<n;i++){
        if(i!=0)printf(" ");
        printf("%d",a[i]);
    }
    printf("\n");
    printf("%d\n",cnt);
}
