import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
    	Scanner sc = new Scanner(System.in);
    	
    	int a;
    	int b;
    	a = sc.nextInt();
    	b = sc.nextInt();
    	
    	int d;
    	int r;
    	double f;
    	
    	d = a / b;
    	r = a % b;
    	f = 1.0 * a / b;
    	
    	System.out.println(d + " " + r + " " + String.format("%.8f", f));
    }
}
