#include <iostream>
#include <vector>
using namespace std;
int main(){
  int m,n;cin>>m>>n;
  vector<vector<char>> A(m,vector<char>(n));
  for(int i = 0; m > i; i++){
    for(int j = 0; n > j; j++){
      cin>>A[i][j];
    }
  }
  int ans = m*n;
  for(int i = 1; m-1 >= i; i++){
    int tmp = 0;
    for(int j = 0; m > j; j++){
      for(int k = 0; n > k; k++){
        if(i <= j){
          if(A[j][k] == 'o')tmp++;
        }else{
          if(A[j][k] == 'x')tmp++;
        }
      }
    }
    ans = min(ans,min(tmp,m*n-tmp));
  }
  for(int i = 1; n-1 >= i; i++){
    int tmp = 0;
    for(int j = 0; m > j; j++){
      for(int k = 0; n > k; k++){

        if(i <= k){
          if(A[j][k] == 'o')tmp++;
        }else{
          if(A[j][k] == 'x')tmp++;
        }
      }
    }
    ans = min(ans,min(tmp,m*n-tmp));
  }
  cout << ans << endl;
}
