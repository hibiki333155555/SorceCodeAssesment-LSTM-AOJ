#include <iostream>
using namespace std;

int main()
{
   int n;
   string s;
   cin >> n >> s;
   int ans = 0;
   for (int i = 0; i < n; i++)
   {
      if (i % 2 == 1 && s[i] != 'O')
         ++ans;
      else if (i % 2 == 0 && s[i] != 'I')
         ++ans;
   }
   cout << ans << endl;
   return 0;
}
