#include <iostream>
using namespace std;

int main()
{
    int n, k;
    cin >> n >> k;
    int d[7]{0};
    for(int i = 0; i < n; i++){
        int a;
        cin >> a;
        d[a]++;
    }
    if(k < d[1]){
        d[1] -= k;
        d[5] += k;
        k = 0;
    }
    else{
        d[5] += d[1];
        k -= d[1];
        d[1] = 0;
    }
    for(int i = 2; i < 6; i++){
        if(k < d[i]){
            d[i] -= k;
            d[6] += k;
            k = 0;
        }
        else{
            d[6] += d[i];
            k -= d[i];
            d[i] = 0;
        }
    }
    int ans = 0;
    for(int i = 1; i <= 6; i++) ans += i * d[i];
    cout << ans << endl;
}
