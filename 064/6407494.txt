void solve() {
    ll n;
    cin >> n;
    vector<string> v(n);
    for (auto &x : v) cin >> x;
    ll res = 0;
    for (int i = 0; i < n; i++)
        for (int j = i + 1; j < n; j++) {
            ll cnt = 0;
            for (int k = 0; k < v[i].size(); k++) {
                if (v[i][k] != v[j][k]) cnt++;
            }
            if (cnt <= 1) res++;
        }
    cout << res << endl;
}

    // cin >> t;
    while (t--) solve();
