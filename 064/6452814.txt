#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n; cin>>n;
    vector<vector<int>> b1(3,vector<int>(10,0));
    vector<vector<int>> b2(3,vector<int>(10,0));
    vector<vector<int>> b3(3,vector<int>(10,0));
    vector<vector<int>> b4(3,vector<int>(10,0));
    while(n--) {
        int b,f,r,v; cin>>b>>f>>r>>v;
        if(b==1) {
            b1[--f][--r] += v;
        }
        else if(b==2) {
            b2[--f][--r] += v;
        }
        else if(b==3) {
            b3[--f][--r] += v;
        }
        else {
            b4[--f][--r] += v;
        }
    }
    for(int i=0;i<3;i++) {
        for(int j=0;j<10;j++) cout<<" "<<b1[i][j];
        cout<<endl;
    }
    cout<<"####################"<<endl;
    for(int i=0;i<3;i++) {
        for(int j=0;j<10;j++) cout<<" "<<b2[i][j];
        cout<<endl;
    }
    cout<<"####################"<<endl;
    for(int i=0;i<3;i++) {
        for(int j=0;j<10;j++) cout<<" "<<b3[i][j];
        cout<<endl;
    }
    cout<<"####################"<<endl;
    for(int i=0;i<3;i++) {
        for(int j=0;j<10;j++) cout<<" "<<b4[i][j];
        cout<<endl;
    }
    return 0;
}
