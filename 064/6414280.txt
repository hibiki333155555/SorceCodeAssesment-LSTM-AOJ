import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        double r = sc.nextDouble();
        
        double area = r*r*Math.PI; 
        double len = 2*r*Math.PI;
        
        System.out.printf("%.6f %.6f\n",area,len);
    }
}
