#include <iostream>
#include <string>
#include <iomanip>
#include <cmath>
using namespace std;
int main( )
{
   int H,W;
  string sq = "#";
 
  cin >> H >> W;
  while ( H != 0){
  for (int i = 0;i < H; i++){
      for (int j = 0; j < W; j++){
        if(j == 0 || j == W-1) cout << sq;
        else if(i == 0 || i == H - 1){
            cout << sq;
        }else if (i > 0 && i < H - 1){
            cout << '.';
        }
      }
      cout << endl;
  } 
  cout << endl;
  cin >> H >> W;
 
  }
  
}
