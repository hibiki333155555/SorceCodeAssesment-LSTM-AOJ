(inputs=>{
    let n = Number(inputs[0]);
    let r = [];
    for(let i = 0 ; i < n ; i++){
        r[i] = Number(inputs[1+i]);
    }
    let min=r[0],max=r[1]-r[0];
    for(let i = 1 ; i < n ; i++){
        max=Math.max(max,r[i]-min);
        min=Math.min(min,r[i]);
    }
    console.log(max);
})(require("fs").readFileSync("/dev/stdin","utf-8").trim("\n").split("\n"));
