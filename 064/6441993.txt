#include <iostream>
using namespace std;

int main(){
    int a, b;
    int ans;
    char op;
    cin >> a >> op >> b;
    
    for(int i = 0; i < 1;){
        if(op == '?') break;
        switch(op){
            case '+': 
                ans = a + b;
                break;
            case '-':
                ans = a - b;
                break;
            case '*':
                ans = a * b;
                break;
            case '/':
                ans = a / b;
                break;
        }
        cout << ans << endl;
        
        return 0;
    }
}
