#include<bits/stdc++.h>

using namespace std;

int main(){
    vector<int> num(100,0);
    int judge = 0;
    for(int i=0;i<12;i++){
        int a;
        cin >> a;
        num[a]++;
        if(4 <= num[a]){
            if(4 == num[a]){
                judge++;
            }
            if(4 < num[a]){
                judge--;
            }
        }
    }
    if(judge == 3){
        cout << "yes" << endl;
    }else{
        cout << "no" << endl;
    }
    return(0);
}
