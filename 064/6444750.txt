#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    int N,sum=0;
    string S;
    cin >> N >> S;
    for(int i=0; i<N; i++){
        if(i%2==0 && S[i]!='I'){
            sum++;
        }
        if(i%2==1 && S[i]!='O'){
            sum++;
        }
    }
    cout << sum << endl;
}
