import java.util.*;

class Solve{
    public void Do(){
        final Scanner sc = new Scanner(System.in);
        int a=sc.nextInt();
        int b=sc.nextInt();
        System.out.println( a*b +" "+ (2*a)+(2*b) );
    }
}
class Main{
    public static void main(final String[] _args){
        new Solve().Do();
    }
}
