#include<stdio.h>
#include<string.h>
int main()
{
	int m,f,r,i=0,j;
	char g[50]= "";
	do	
	{		
		scanf("%d %d %d",&m,&f,&r);
		if ((m==-1)&&(f==-1)&&(r==-1))
			break;
		else if((m==-1)||(f==-1))
			strcpy(g+i,"F");
		else if((m+f)>=80)
			strcpy(g+i,"A");
		else if(((m+f)>=65) && ((m+f)<80))
			strcpy(g+i,"B");
		else if((((m+f)>=50) && ((m+f)<65)) || ((((m+f)>=30) && ((m+f)<50)) && (r>=50)))
			strcpy(g+i,"C");
		else if((((m+f)>=30) && ((m+f)<50)) && (r<50))
			strcpy(g+i,"D");
		else if((m+f)<30)
			strcpy(g+i,"F");
		i++;
	}
	while((m!=-1)||(f!=-1)||(r!=-1));	
	for(j=0;j<=i;j++)
		printf("%c%s",g[j],(j<i)?"\n":"");
	return 0;
}
