#include<iostream>
#include<vector>
#include<algorithm>
#include<map>
#include<set>
#include<cmath>
#include<queue>
#include<stack>

#define rep(i,n) for(int i=0;i<n;i++)

using ll = long long;
using namespace std;

const int INF = (1 << 29);

int main() {
	int n;
	cin >> n;
	vector<int> p(n + 1);
	rep(i, n) cin >> p[i] >> p[i + 1];

	vector<vector<int>> dp(n + 1, vector<int>(n + 1, INF));
	rep(i, n) dp[i][i + 1] = 0;

	for (int j = 2; j <= n; j++) {
		for (int i = 0; i <= n - 1; i++) {
			for (int k = i + 1; k < i+j; k++) {
				if (i + j > n) continue;
				dp[i][i + j] = min(dp[i][i + j], dp[i][k] + dp[k][i + j] + p[i] * p[k] * p[i + j]);
			}
		}
	}

	cout << dp[0][n] << endl;

	return 0;
}
