#include <iostream>

using namespace std;

int main() {
    int r, c, sum, tableSum = 0;
    
    cin >> r >> c;
    
    int table[r + 1][c + 1] = {};
    
    for(int i = 0; i < r; i++) {
        sum = 0;
        for(int j = 0; j < c; j++) {
            cin >> r >> c;
            sum += table[i][j];
            table[r][j] += table[i][j];
        }
        table[i][c] = sum;
        tableSum += sum;
    }
    table[r][c] = tableSum;

    for(int i = 0; i < r + 1; i++) {
        for(int j = 0; j < c + 1; j++) {
            cout << " " << table[i][j];
        }
        cout << endl;
    }
}
