/* function */
template<class T>
T GCD(T a, T b) {
	if (b != 0) return GCD(b, a % b);
	return a;
}
template<class T>
T LCM(T a, T b) {
	return a / GCD(a, b) * b;
}

const int dy[4] = { 0, 1, -1, 0 };
const int dx[4] = { 1, 0, 0, -1 };

int main() {
    ll sum=0;
    string s,ans;
    cin >> s;
    stack<ll> V;
    for(auto c:s){
        if('0'<=c&&c<='9'){
            sum+=(c-'0');
            V.push(c-'0');
        }else if(c=='('){
            continue;
        }else if(c==')'){
            sum-=V.top();
            V.pop();
        }else{
            ans+=(c-'a'+sum)%26+'a';
        }
    }
    cout << ans << endl;
}
