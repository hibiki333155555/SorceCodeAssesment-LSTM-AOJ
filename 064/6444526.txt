#include <iostream>

using namespace std;

int main () {
    int n, m, l;
    
    cin >> n >> m >> l;
    
    int A[n][m] = {};
    int B[m][l] = {};
    int C[n][l] = {};
    
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cin >> A[i][j];
        }
    }
    
    for(int i = 0; i < m; i++) {
        for(int j = 0; j < l; j++) {
            cin >> B[i][j];
        }
    }
    
    for(int i = 0; i < n; i++) {
        for(int k = 0; k < l; k++) {
            for(int j = 0; j < m; j++) {
                C[i][k] += A[i][j] * B[j][k];
            }
        }
    }
    
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < l - 1; j++) {
            cout << C[i][j] << " ";
             
        }
        cout << C[i][l - 1] << endl;
    }
    
}
