#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include <string.h>

char *gets(char *s);

int main()
{
	char card[2][210], buf[100];
	int id0, id1;
	int i, m, h, len;

	while (1) {
		gets(card[0]); len = strlen(card[0]); card[1][len] = 0;
		if (card[0][0] == '-') break;
		gets(buf);	m = atoi(buf);
		for (i = 0, id0 = 0; i < m; i++) {
			gets(buf); h = atoi(buf);
			if (h > 0 && h < len) {
				id1 = !id0;
				memcpy(card[id1], card[id0] + h, len - h);
				memcpy(card[id1] + len - h, card[id0], h);
				id0 = id1;
			}
		}
		puts(card[id0]);
	}
	return 0;
}
