using System;
using System.Collections.Generic;
using System.Linq;

namespace AOJ_ALDS1_1_C
{
    internal class Program
    {
        public static void Main()
        {
            var n = int.Parse(Console.ReadLine());
            var numbers = new List<int>();
            for (int i = 0; i < n; i++)
            {
                numbers.Add(int.Parse(Console.ReadLine()));
            }
            Console.WriteLine(numbers.Count(i => IsPrime(i)));
        }

        static bool IsPrime(int number)
        {
            foreach (var p in primes)
            {
                if (number < p * p)
                    break;
                if (number % p == 0)
                    return false;
            }
            return true;
        }
        readonly static int[] primes;
        static Program()
        {
            var list = new LinkedList<int>(Enumerable.Range(2, 10000));
            var current = list.First;
            do
            {
                var node = current.Next;
                while (node != null)
                {
                    var next = node.Next;
                    if (node.Value % current.Value == 0)
                    {
                        list.Remove(node);
                    }
                    node = next;
                }
            } while ((current = current.Next) != null);
            primes = list.ToArray();
        }
    }
}
