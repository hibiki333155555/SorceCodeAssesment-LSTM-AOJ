#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    int m;
    cin>>m;
    int b[m];
    for(int i=0;i<m;i++){
        cin>>b[i];
    }
   sort(a,a+n);
   sort(b,b+m);
    if ( std::includes(a,a+n,b,b+m) )
    std::cout << "1\n";
    else{
        cout<<"0\n";
    }
    return 0;
}
