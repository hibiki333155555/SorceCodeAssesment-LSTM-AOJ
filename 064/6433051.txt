#include<iostream>
using namespace std;
int main(){
    int n;
    cin>>n;
    int a[100],w[100];
    int aCount=0,rightMin=1001,leftMin=1001;
    for(int i=0;i<n;i++){
        cin>>a[i];
        if(a[i]){
            aCount++;
        }
    }
    for(int i=0;i<n;i++){
        cin>>w[i];
        if(a[i]){
            rightMin=min(w[i],rightMin);
        }else{
            leftMin=min(w[i],leftMin);
        }
    }
    if(aCount==n||aCount==0)
        cout<<0<<endl;
    else
        cout<<rightMin+leftMin<<endl;
    
}
