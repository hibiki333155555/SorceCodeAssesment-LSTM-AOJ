#include<bits/stdc++.h>
using namespace std;
void swap4(int &a, int &b, int &c, int &d) {
	int t = a;
	a = b;
	b = c;
	c = d;
	d = t;
}
void f(vector<int> &a, char c) {
	switch(c) {
		case 'W': swap4(a[0],a[2],a[5],a[3]); break;
		case 'S': swap4(a[0],a[4],a[5],a[1]); break;
		case 'N': swap4(a[0],a[1],a[5],a[4]); break;
		case 'E': swap4(a[0],a[3],a[5],a[2]); break;
	}
}
int main() {
	vector<int> a(6), b(6);
	for (int i = 0; i < 6; i++) cin>>a[i];
	for (int i = 0; i < 6; i++) cin>>b[i];
	for (int i = 0; i < 10000; i++) {
		if (a == b) {
			cout<<"Yes"<<endl;
			return 0;
		} 
		if (rand()%2 == 0) f(a, 'N');
		else f(a, 'E');
	}
	cout<<"No"<<endl;
	
	
	return 0;
}
