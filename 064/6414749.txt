int gcd(int a, int b)
{
    if(a < b)
        a ^= b ^= a ^= b;

    while(b != 0)
    {
        int tmp = b;
        b = a%b;
        a = tmp;
    }
    return a;
}

int main()
{
    
    int a, b;
    cin >> a >> b;
    cout << gcd(a, b) << endl;
    return 0;
}
