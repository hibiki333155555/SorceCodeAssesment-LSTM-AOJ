using System;

class Program
{
    static void Main()
    {
        int []seito = new int[30];
        for (int i = 0; i < 28; i++)
        {
            seito[int.Parse(Console.ReadLine()) - 1] = 1;
        }
        for(int i = 0; i < 30; i++)
        {
            if(seito[i] == 0)
            {
                Console.WriteLine(i + 1);
            }
        }
    }
}
