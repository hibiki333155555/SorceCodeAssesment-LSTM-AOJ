#include "iostream"
#include "stdio.h"
#include "math.h"
using namespace std;

typedef struct Point{
    double x;
    double y;
}Point;
void Kehe(int m,Point left,Point right){
    if(m==0){
        return;
    }
    else{
        Point s,t,u;
        s.x = (2*left.x+right.x)/3;
        s.y = (2*left.y+right.y)/3;
        t.x = (left.x+2*right.x)/3;
        t.y = (left.y+2*right.y)/3;
        u.x = (t.x-s.x)* cos(M_PI/3)-(t.y-s.y)* sin(M_PI/3) + s.x;
        u.y = (t.x-s.x)* sin(M_PI/3)+(t.y-s.y)* cos(M_PI/3) + s.y;
        Kehe(m-1,left,s);
        printf("%0.8f %0.8f\n",s.x,s.y);
        Kehe(m-1,s,u);
        printf("%0.8f %0.8f\n",u.x,u.y);
        Kehe(m-1,u,t);
        printf("%0.8f %0.8f\n",t.x,t.y);
        Kehe(m-1,t,right);
    }
    return ;
}

int main(){
    int n;
    Point p1,p2;
    p1.x = 0;
    p1.y = 0;
    p2.x = 100;
    p2.y = 0;
    cin>>n;
    printf("%0.8f %0.8f\n", p1.x,p1.y);

    Kehe(n,p1,p2);
    printf("%0.8f %0.8f\n", p2.x,p2.y);
    return 0;
}
