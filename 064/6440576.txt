int main() {
    while (true) {
        int n, m;
        cin >> n >> m;
        if (n == 0 && m == 0)break;

        vector<vector<int>> p(m, vector<int>(n));
        for (auto &x: p)
            for (auto &y: x) cin >> y;

        int ret = 0;
        for (int i = 0; i < n; i++) {
            int now = 0;
            for (int j = 0; j < m; j++) now += p[j][i];
            ret = max(ret, now);
        }
        
        cout << ret << endl;
    }
    return 0;
}
