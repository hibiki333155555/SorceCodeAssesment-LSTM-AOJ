#include <iostream>
using namespace std;

int main() {
    int x,y, temp;
    while (cin >>x >>y) 
    {
        // cout << endl;
        if ((x!=0 and y!=0) and x>y) {
            cout << endl;
            temp = x;
            x=y;
            y=temp;
            cout << x << " " << y;
        }
        else if ((x!=0 and y!=0) and y>=x)
        {
            cout << x << " " << y;
        }
        else break;
    }
    return 0; 
}
