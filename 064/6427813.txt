#include<stdio.h>

int main() {
    int a, result[10000];
    for (int i = 0; i<= 10000; i++) {
        scanf("%d", &a);
        if (a != 0) {
            result[i] = a;
        } else {
            break;
        }
    }
    for (int i = 0; i < result[i]; i++) {
        printf("Case %d: %d\n", i+1, result[i]);
    }
    return 0;
}
