#include<bits/stdc++.h>
using namespace std;
int main()
{
	int c1,c2,r1;
	cin>>r1>>c1>>c2;
	vector<vector<int>>v1(r1,vector<int>(c1));
	for(int i=0;i<r1;i++)
	{
		for(int j=0;j<c1;j++)
		{
			cin>>v1[i][j];
		}
	}
	vector<vector<int>>v2(c1,vector<int>(c2));
	for(int i=0;i<c1;i++)
	{
		for(int j=0;j<c2;j++)
		{
			cin>>v2[i][j];
		}
	}
	if(c1==c1)
	{
		vector<vector<int>>v3(r1,vector<int>(c2));
		for(int i=0;i<r1;i++)
		{
			for(int j=0;j<c2;j++)
			{
				v3[i][j]=0;
				for(int k=0;k<c1;k++)
				{
					v3[i][j]=v3[i][j]+(v1[i][k]*v2[k][j]);
				}
			}
		}
		for(int i=0;i<r1;i++)
		{
			for(int j=0;j<c2;j++)
			{
				
				if(j==c2-1)
				{
					cout<<v3[i][j];
				}
				else
				{
					cout<<v3[i][j]<<" ";
				}
			}
			cout<<"\n";
		}
	}
	return 0;
}
