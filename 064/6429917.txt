#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
    int n;
    cin>>n;
    int p[n],k[n];
    for(int i=0;i<n;i++){
        cin>>k[i]>>p[i];
    }
    for(int i=0;i<n;i++){
        if(p[i]>k[i]){
            cout<<k[i]<<endl;
        }else{
            if(k[i]%p[i]==0) cout<<p[i]<<endl;
            else cout<<(k[i]%p[i])<<endl;;
        }
    }
    return 0;
}
