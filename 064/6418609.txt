#include <bits/stdc++.h>
using namespace std;

int main(){
        int flag=1, n=0, hoge=0, time=0; cin>>n;
        int A[n+10]; for(int i=0;i<n;i++) cin>>A[i];
        while(flag){
                flag=0;
                for(int i=n-1;i>0;i--){
                        if(A[i]<A[i-1]){
                                hoge=A[i];
                                A[i]=A[i-1];
                                A[i-1]=hoge;
                                flag=1;
                                time++;
                        }
                }
        }
        for(int i=0;i<n;i++){
                if(i!=n-1) cout<<A[i]<<" ";
                else cout<<A[i]<<endl;
        }
        cout<<time<<endl;
        return 0;;
}
