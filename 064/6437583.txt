#include<stdio.h>
#define MAX 100

int bubble_sort(int A[], int n){

  int cnt = 0;

  for(int i = n - 1; i > 0; i-- ){
    for(int j = 0; j < i; j++ ){
      if(A[j] > A[j + 1]){
        int temp = A[j];
        A[j] = A[j + 1];
        A[j + 1] = temp;
        cnt++;
      }
    }
  }

  return cnt;
}

int main()
{
  int n;
  int A[MAX];
  scanf("%d", &n);
  for(int i = 0; i < n; i++ ) scanf("%d", &A[i]);
  int ans = bubble_sort(A, n);
  for(int i = 0; i < n; i++ ){
    if(i) printf(" ");
    printf("%d", A[i]);
  }
  printf("\n");
  printf("%d\n", ans);
  return 0;
}
