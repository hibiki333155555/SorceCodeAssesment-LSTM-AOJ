#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

int main()
{
   map<int, int> mp;
   for (int i = 0; i < 28; ++i)
   {
      int n;
      cin >> n;
      mp[n]++;
   }

   for (int i = 1; i <= 30; i++)
   {
      if (mp.count(i) == 0)
         cout << i << endl;
   }

   return 0;
}
