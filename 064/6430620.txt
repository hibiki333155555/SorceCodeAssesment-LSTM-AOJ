#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    string s;
    cin >> s;
    vector<bool> v(5);
    for(int i = 0; i < 5; i++){
        v[i] = false;
    }
    for(int i = 0; i < n; i++){
        if(s[i] == 'A'){
            v[0] = true;
            continue;
        }
        if(s[i] == 'B'){
            v[1] = true;
            continue;
        }
        if(s[i] == 'C'){
            v[2] = true;
            continue;
        }
        if(s[i] == 'D'){
            v[3] = true;
            continue;
        }
        if(s[i] == 'E'){
            v[4] = true;
            continue;
        }
    }
    int t = 0;
    for(int i = 0; i < 5; i++){
        if(v[i] == true){
            t++;
        }
    }
    if(t >= 3){
        cout << "Yes" << endl;
        return 0;
    }
    cout << "No" << endl;
    return 0;
}
