#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    string str;
    cin >> str;
    int q;
    cin >> q;
    for(int i = 0;i < q; i++)
    {
        int a,b;
        string s;
        cin >> s >> a >> b;
        if(s == "print")
        {
            str = str.substr(a,b-a+1);
        }
        else if(s == "replace")
        {
            string p;
            cin >> p;
            str = str.replace(a , b-a+1 , p);
        }
        else if(s == "reverse")
        {
            string t;
            t = str.substr(a,b-a+1);
            reverse(t.begin(),t.end());
            str = str.replace(a, b-a+1,t);

        }

    }
    cout << str << endl;
    return 0;
}
