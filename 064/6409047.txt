#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include<string.h>

int main()
{
	char s[1200];
    char c;
    int i=0;
    while(1){
        scanf("%c",&s[i]);
        if(s[i]=='\n')
        break;

        if(s[i]>='a'&&s[i]<='z'){
        s[i]+='A'-'a';
        }
        else if(s[i]>='A'&&s[i]<='Z'){
        s[i]-='A'-'a';
        }
        i++;
    }
    for(int g=0; g<i; g++){
        printf("%c",s[g]);
    }
    printf("\n");
	return 0;
}
