#include <iostream>
using namespace std;
 
int main(){
    int x,y,t;
    cin >> x >> y;

    for(int i = 0; i < 3000; i++){
        if(x > y){
            t = x;
            x = y;
            y = t;
        }
        if(x  == 0 && y == 0)break;
        cout << x << " " << y << endl;
        cin >> x >> y;
    }
    return 0;
}
