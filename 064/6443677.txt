#include<iostream>
#include<string>
using namespace std;
int main(){
    int N,M;
    cin >> N >> M;
    int Np=N+1;
    int X[Np][Np];
    for(int i=1; i<=N; i++){
        for(int j=1; j<=N; j++){
            if(i==1){
                X[i][j]=j;
            }
            else{
                X[i][j]=0;
            }
        }
    }
    int a,b; 
    for(int m=0; m<M; m++){
        cin >> a >> b; 
        for(int i=1; i<=N; i++){
            for(int j=1; j<=N; j++){
                if(X[i][j]==a){
                    X[i][j]=0;
                    for(int k=1; k<=N; k++){
                        if(X[k][b]==0){
                            X[k][b]=a;
                            break;
                        }
                    }
                    i=N;
                    break;
                }
            }
        }
    }
    for(int k=1; k<=N; k++){
        for(int i=1; i<=N; i++){
            for(int j=1; j<=N; j++){
                if(X[i][j]==k){
                    cout << j << endl;
                    i=N;
                    break;
                }
            }
        }
    }
}
