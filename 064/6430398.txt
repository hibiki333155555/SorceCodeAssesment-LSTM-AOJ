#include <bits/stdc++.h>
using namespace std;
int bublesort(vector<int>&);
int  bubblesort(vector<int> &a)
{
	//int count=0;
	int n=a.size();
	int count=0;
	for(int i=0;i<n-1;i++)
	{
	 for(int j=0;j<n-i-1;j++)
	 {
	 	if(a[j]>a[j+1])
	 	{	
	 		swap(a[j],a[j+1]);
	 		count=count+1;
	 		//cout<<count<<"\n";
	 	}
	 }
	}
	//cout<<"Count :"<<count<<"\n";
	return count;
}
int main() {
	int n;
	int c;
	cin>>n;
	vector<int>a(n);
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	} 
	c=bubblesort(a);
	for(int i=0;i<n;i++)
	{
		cout<<a[i]<<" ";
	}
	cout<<"\n";
	//n=bubblesort(a);
	cout<<c;
	return 0;
}
