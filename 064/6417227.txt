#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <iterator>
#include <numeric>
#include <string>
#include <vector>
using namespace std;


int main()
{
    
    int n,val;
    cin >> n >> val;

    int mn=val, res=-2e9;
    istream_iterator<int> isi(cin), eof;

    for_each(isi, eof, [&mn, &res](int val){
        res = res > val-mn  ? res : val-mn;
        mn = mn < val ? mn : val;
    });

    cout << res << endl;
    return 0;
}
