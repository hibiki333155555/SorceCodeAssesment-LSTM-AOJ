int selectsort(int a[],int n){
    int minj;int i;int j;
    int sw=0;
    for(i=0;i<n-1;i++){
        minj=i;
        for(j=i;j<n;j++){
            if(a[j]<a[minj]){
                minj=j;
            
            }
            int t=a[i];a[i]=a[minj];a[minj]=t;
          if(i!=minj){  sw++;}
        }
        
        
        
    }
    
    
    return sw;
}
