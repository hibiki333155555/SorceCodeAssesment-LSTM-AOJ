#define rep(i, a, n) for (int i = (a); i < (n); ++i)
#define bit(n, k) ((n >> k) & 1)

using Vector = pair<long, long>;

class Program {
private:

    bool max(int &a, const int b) {
        if (a < b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    Vector p0, p1, p2;
    int q;

public:

    Program() {
    }

    void input() {
        {
            auto& [x, y] = p0;
            cin >> x >> y;
        }

        {
            auto& [x, y] = p1;
            cin >> x >> y;
        }

        cin >> q;
    }

    long norm(const Vector &a, const Vector &b) {
        return a.first * b.first + a.second * b.second;
    }

    long cross(const pair<float, float> &a, const pair<float, float> &b) {
        return a.first * b.second - a.second * b.first;
    }

    pair<float, float> projection(const Vector &a, const Vector &b) {
        auto [x, y] = b;

        if (x == 0 && y == 0) {
            return make_pair(0.0, 0.0);
        } else if (x == 0) {
            return make_pair(0.0, 1.0 * a.second);
        } else if (y == 0) {
            return make_pair(1.0 * a.first, 0.0);
        } else {
            return make_pair(
                    1.0 * norm(a, b) / norm(b, b) * b.first,
                    1.0 * norm(a, b) / norm(b, b) * b.second);
        }
    }

    pair<float, float> rejection(const Vector &a, const Vector &b) {
        auto [x, y] = projection(a, b);
        return make_pair(a.first - x, a.second - y);
    }

    // a: p1-p0
    // b: p2-p0
    void solve() {
        input();

        // ベクトルの始点を原点とする
        Vector b = make_pair(p1.first - p0.first, p1.second - p0.second);

        rep(i, 0, q) {
            auto& [x, y] = p2;
            cin >> x >> y;

            Vector a = make_pair(x - p0.first, y - p0.second);

            long z = cross(b, a);

            if (z > 0) {
                cout << "COUNTER_CLOCKWISE" << endl;
            } else if (z < 0) {
                cout << "CLOCKWISE" << endl;
            } else {
                assert(z == 0);

                // cout << a.first << " " << a.second << endl;
                // cout << b.first << " " << b.second << endl;

                if (a.first > 0 && b.first < 0 || a.first < 0 && b.first > 0 ||
                        a.second > 0 && b.second < 0 || a.second < 0 && b.second > 0) {
                    cout << "ONLINE_BACK" << endl;
                } else if (norm(a, a) > norm(b, b)) {
                    cout << "ONLINE_FRONT" << endl;
                } else {
                    cout << "ON_SEGMENT" << endl;
                }
            }

        }

        // printf("%.8f\n", 0.00000001);
    }
};

int main() {

#ifdef DEV
    std::ifstream in("input");
    cin.rdbuf(in.rdbuf());

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();
#endif

    return 0;
}
