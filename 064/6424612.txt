int main(){
    //SHCD
    vector<vector<int>> cards(4,vector<int>(13,0));
    int n ;
    cin >> n;
    int a; string s;
    rep(i,0,n){
        cin >> s >> a;
        if (s == "S") cards[0][a-1] = 1;
        else if (s == "H") cards[1][a-1] = 1;
        else if (s == "C") cards[2][a-1] = 1;
        else cards[3][a-1] = 1;
    }
    rep(i,0,4)rep(j,0,13){
        if (cards[i][j] == 0){
            if (i == 0) s = "S";
            else if(i==1) s = "H";
            else if (i == 2) s = "C";
            else s = "D";
            cout << s << " " << j + 1 << endl;
