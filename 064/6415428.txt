#include "iostream"
using namespace std;
int A[40];
bool examine(int output,int i){
    if(output==0)return true;
    else if(i==-1)return false;
    else return examine(output-A[i],i-1)|| examine(output,i-1);
}
int main(){

    int n;
    cin>>n;
    for (int i = 0; i < n; ++i) {
        cin>>A[i];
    }
    int q;
    cin>>q;
    for (int i = 0; i < q; ++i) {
        int x;
        cin>>x;
        if(examine(x,n-1))cout<<"yes"<<endl;
        else cout<<"no"<<endl;
    }
    return 0;
}
