#include <cstdio>

const int MAX_n = 100;
const int MIN_a = 0;
const int MAX_a = 1000;


int main(void){
    int num;
    int a[num];
    int cov;

    scanf("%d", &num);
    if(num>MAX_n){
        return 0;
    }

    for(int i=0; i<num; i++){
        scanf("%d", &cov);
        if(cov<MIN_a || cov>=MAX_a){
            return 0;
        }
        a[i] = cov;
        printf("%d\n", &cov);
    }
    for(int i=num-1; i>0; i--){
        printf("%d ", a[i]);
    }
    return 0;
}
