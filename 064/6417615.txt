template <class T>
T gcd(T a, T b) {
    if (b == 0)
        return a;
    else
        return gcd(b, a % b);
}

template <class T>
T lcm(T a, T b) {
    a /= gcd(a, b);
    return a * b;
}

template <class T>
T Pow(T x, i64 n) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = res * x;
        x = x * x;
        n >>= 1;
    }
    return res;
}

template <class T>
T mod_pow(T x, i64 n, i64 mod) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = (res * x) % mod;
        x = (x * x) % mod;
        n >>= 1;
    }
    return res;
}

namespace ebi {

void main_() {
    int h,w,n;
    std::cin >> h >> w >> n;
    int sx,sy,gx,gy;
    std::cin >> sx >> sy >> gx >> gy;
    sx--; sy--;
    gx--; gy--;
    std::vector grid(h+1, std::vector<int>(w+1, 0));
    rep(i,0,n) {
        int x,y,k;
        std::cin >> x >> y >> k;
        x--; y--;
        assert(k <= 500);
        rep(i,0,k+1) {
            grid[x-k+i][y-i]++;
            grid[x-k+i][y+i+1]--;
            grid[x+k-i][y-i]++;
            grid[x+k-i][y+i+1]--;
        }
    }
    rep(i,0,h) {
        rep(j,0,w) {
            grid[i][j+1] += grid[i][j];
        }
    }
    std::queue<std::pair<int,int>> que;
    que.push({sx, sy});
    std::vector seen(h, std::vector<int>(w, -1));
    seen[sx][sy] = 1;
    while(!que.empty()) {
        auto [x, y] = que.front();
        que.pop();
        rep(i,0,4) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(0 <= nx && nx < h && 0 <= ny && ny < w && seen[nx][ny] < 0 && grid[nx][ny] == 0) {
                que.push({nx, ny});
                seen[nx][ny] = 1;
            }
        }
    }
    if(seen[gx][gy] > 0) {
        std::cout << "Yes\n";
    }
    else {
        std::cout << "No\n";
    }
}

}  // namespace ebi

int main() {
    std::cout << std::fixed << std::setprecision(15);
    std::cin.tie(nullptr);
    std::ios::sync_with_stdio(false);
    int t = 1;
    // std::cin >> t;
    while (t--) {
        ebi::main_();
    }
    return 0;
}
