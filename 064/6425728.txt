#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;
static const int INFINITY = INT_MAX;

int main() {
    int n, m;
    cin >> n >> m;

    int T[n + 1], Coins[m];
    for (int i = 0; i <= n; ++i) T[i] = INFINITY;
    for (int i = 0; i < m; ++i) cin >> Coins[i];

    T[0] = 0;

    for (int i = 0; i < m; ++i) {
        for (int j = Coins[i]; j <= n; ++j) {
            T[j] = min(T[j], T[j - Coins[i]] + 1);
        }
    }

    cout << T[n] << endl;
}
