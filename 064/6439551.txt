#include <bits/stdc++.h>
using namespace std;

int main() {
  int arr[10000][2];
  int i = 0;
  while (1) {
    cin >> arr[i][0] >> arr[i][1];
    if (arr[i][0] == 0 && arr[i][1] == 0) break;
    i++;
  }
  for (int l = 0; l < i; l++) {
    for (int k = 0; k < 300; k++) {
      if (k == arr[l][0]) break;
      if (k == 0) {
        for (int m = 0; m < 300; m++) {
          if (m == arr[l][1]) break;
          printf("#");
        }
        printf("\n");
      }
      if (0 < k && k < arr[l][0] - 1) {
        for (int j = 0; j < 300; j++) {
          if (j == arr[l][1]) break;
          if (j == 0) printf("#");
          if (0 < j && j < arr[l][1] - 1) printf(".");
          if (j == arr[l][1] - 1) printf("#");
        }
        printf("\n");
      }
      if (k == arr[l][0] - 1) {
        for (int n = 0; n < 300; n++) {
          if (n == arr[l][1]) break;
          printf("#");
        }
        printf("\n");
      }
      
    }
    printf("\n");
  }
  
  return 0;

}
