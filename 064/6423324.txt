#include <iostream>
#include <algorithm>
#include <iomanip>
#include <cmath>
#include <string>
using namespace std;

int main()
{
	int n; cin >> n;
	string a; string b;
	int totalA = 0; int totalB = 0;
	int l;
	for (int i = 0; i <= n - 1; i++)
	{
		cin >> a >> b;
		if (a == b) 
		{
			totalA++;
			totalB++;
			continue;
		}
		l = min(a.length(), b.length());
		for (int j = 0; j <= l; j++)
		{
			if (a[j] > b[j])
			{
				totalA += 3;
				break;
			}
			else if (a[j] < b[j])
			{
				totalB += 3;
				break;
			}
			else if (a[j] = b[j] && i != n - 1)
			{
				continue;
			}
		}
	}
	cout << totalA << ' ' << totalB << endl;
}
