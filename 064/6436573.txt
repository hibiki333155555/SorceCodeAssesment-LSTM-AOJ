#include <iostream>
#include <vector>


int main()
{
    std::vector<int> R;
    
    int N = 0;
    std::cin >> N;

    R.resize(N);

    for(int i = 0; i < N; ++i)
    {
        std::cin >> R[i];
    }

    int maxim = -999999;
    for(int i = 0; i < N; ++i)
    {
        for(int j = i+1; j < N; ++j)
        {
            int candidate = R[j] - R[i];
            if(candidate > maxim)
            {
                maxim = candidate;
            }
        }
    }

    std::cout << maxim << std::endl;
}
