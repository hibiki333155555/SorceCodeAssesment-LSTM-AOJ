#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() {
   string t;

   while(true) {
      cin >> t;

      for(int i=0; i<t.size()-1; i++) {
         if(islower(t[i])) {
            cout << toupper(t[i]);
         }else if(isupper(t[i])) {
            cout << tolower(t[i]);
         }
      }
      if(t.back() == ',') cout << ", ";
      else if(t.back() == '.'){
         cout << ".\n";
         break;
      }
   }
}
