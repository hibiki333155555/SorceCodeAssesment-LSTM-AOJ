#include <iostream>
#include <complex>

using namespace std;

complex<double> in()
{
	double x, y;
	cin >> x >> y;
	return {x, y};
}

bool f(complex<double> p1, complex<double> p2, complex<double> p3)
{
	if (arg((p3 - p2) / (p2 - p1)) < 0) {
		cout << 0 << endl;
		return true;
	}
	return false;
}

int main()
{
	int n; cin >> n;
	complex<double> p0 = in(), p1 = p0, p2 = in(), p3;
	for (; n > 2; --n, p1 = p2, p2 = p3) {
		p3 = in();
		if (f(p1, p2, p3)) return 0;
	}
	if (!f(p1, p2, p0)) cout << 1 << endl;
	return 0;
}
