#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef long double LD;

#define rep(i,n) for(LL i=0;i<(n);i++)

int main(){
    LL h_s, w_s;

    cin >> h_s >> w_s;

    LL h, w;

    cin >> h >> w;

    LL need_h = h_s/h;
    LL need_w = w_s/w;

    if(h_s%h != 0){
        need_h++;
    }

    if(w_s%w != 0){
        need_w++;
    }

    cout << need_h * need_w << endl;

    return 0;
}
