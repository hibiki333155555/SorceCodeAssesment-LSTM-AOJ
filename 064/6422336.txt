#include <bits/stdc++.h>
using namespace std;

int main(){
  string s,p;
  cin >> s >> p;
  
  for(int i=0; i<s.size(); i++){
    if(s.at(i) == p.at(0)){
      bool f=true;
      for(int j=1; j<p.size(); j++){
        if(s.at((i+j)%s.size()) != p.at(j)){
          f=false;
          break;
        }
      }
      if(f){
        cout << "Yes" << endl;
        return 0;
      }
    }
  }
  
  cout << "No" << endl;
    
}
