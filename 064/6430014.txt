#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
    int n,c;
    cin>>n>>c;
    int ck[c],s=0;
    for(int i=0;i<c;i++){
        cin>>ck[i];
        s+=ck[i];
    }
    n++;
    if(s%n!=0){
        cout<<s/n+1<<endl;
    }else{
        cout<<s/n<<endl;
    }
    return 0;
}
