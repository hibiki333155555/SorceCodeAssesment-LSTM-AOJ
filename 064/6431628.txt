#include <bits/stdc++.h>
using namespace std;
int main() {
	int t,n;
	int count;	
		cin >> n;
		vector<int> v(n);
		count = 0;
		for(int i = 0; i<n; i++){
			cin >> v[i];
		}
		for(int i =0; i < n-1; i++){
			for(int j = 0; j< n-1-i; j++){
				if(v[j] > v[j+1]){
					count++;
					swap(v[j], v[j+1]);
					
				}
			}
		}
	for(int i = 0; i<n; i++){
		if(i != n-1){
			cout << v[i] << " ";
		}
		else{
			cout << v[i]<< "\n";
		}
		}
		cout << count << "\n";
	return 0;
}//bro
