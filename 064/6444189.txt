using namespace std;

char c;
struct dice{
	int data;
	int dire;
};
struct dice lis[7];

int main()
{	
	int i; 
	 
	for(i=1;i<=6;i++)
	{
		cin>>lis[i].data;
		lis[i].dire=i;   //初始均为如图所示 1上2前3右4左5后6下 
	}
	
	while((c=getchar())!=EOF)
	{
		if(c=='S')
		{
			for(i=1;i<=6;i++)//依次改变6个面的方位值 下同理 
			{
				if(lis[i].dire==1)
					lis[i].dire=2;
				else if(lis[i].dire==2)
					lis[i].dire=6;
				else if(lis[i].dire==6)
					lis[i].dire=5;
				else if(lis[i].dire==5)
					lis[i].dire=1;	
			}	
		} 
		else if(c=='N')
		{
			for(i=1;i<=6;i++)
			{
				if(lis[i].dire==1)
					lis[i].dire=5;
				else if(lis[i].dire==2)
					lis[i].dire=1;
				else if(lis[i].dire==6)
					lis[i].dire=2;
				else if(lis[i].dire==5)
					lis[i].dire=6;	
			}	
		} 
		else if(c=='E')
		{
			for(i=1;i<=6;i++)
			{
				if(lis[i].dire==1)
					lis[i].dire=3;
				else if(lis[i].dire==3)
					lis[i].dire=6;
				else if(lis[i].dire==6)
					lis[i].dire=4;
				else if(lis[i].dire==4)
					lis[i].dire=1;	
			}	
		} 
		else if(c=='W')
		{
			for(i=1;i<=6;i++)
			{
				if(lis[i].dire==1)
					lis[i].dire=4;
				else if(lis[i].dire==3)
					lis[i].dire=1;
				else if(lis[i].dire==6)
					lis[i].dire=3;
				else if(lis[i].dire==4)
					lis[i].dire=6;	
			}	
		} 
		
	}
	
	for(i=1;i<=6;i++) 
		if(lis[i].dire==1)//输出上面的数据 
			cout<<lis[i].data<<endl;
	
return 0;
}
