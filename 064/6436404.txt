#include<bits/stdc++.h>
using namespace std;

int A[100];

int fib(int n){
    if(n == 0 || n == 1){
        return 1;
    }
    else{
        if(A[n] == -1){
            A[n] = fib(n - 1) + fib(n - 2);
        }
        return A[n];
    }
}

int main(){
    int n;
    for(int i = 0; i < 100; i++){
        A[i] = -1;
    }
    cin >> n;
    cout << fib(n) << endl;
}
