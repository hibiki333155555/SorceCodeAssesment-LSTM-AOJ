using ll=long long;

ll xx_abs(ll x){
  if(x<0){
    return -x;
  }
  return x;
}

int main(){
  ll is,js,ks;
  cin>>is>>js>>ks;
  ll ig,jg,kg;
  cin>>is>>js>>kg;
  ll u=xx_abs(is-ig)+xx_abs(js-jg)+xx_abs(ks-kg);
  if(u%3==0){
    cout<<"Yes"<<endl;
  }else{
    cout<<"No"<<endl;
  }
}
