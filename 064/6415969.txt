#include<bits/stdc++.h>
using namespace std;

int main(){

    string str,mg,p;
    int num,a,b;

    cin>>str;
    cin>>num;

    for(int i=0;i<num;++i){

        cin>>mg;
        if(mg=="replace"){
            cin>>a>>b>>p;
            str.replace(a,b-a+1,p);
        }
        else if(mg=="reverse"){
            cin>>a>>b;
            reverse(str.begin()+a,str.begin()+b+1);
        }
        else if(mg=="print"){
            cin>>a>>b;
            cout<<str.substr(a,a-b+1)<<endl;
        }
    }
    

    return 0;
}
