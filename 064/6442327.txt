#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int n,m;
	int i,j,k;
	int a[101][101];
	int b[101],c[101];
	
	cin>>n>>m;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>a[i][j];
	for(i=1;i<=m;i++)
		cin>>b[i];
	
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			c[i]+=a[i][j]*b[j];
		
	for(i=1;i<=n;i++)
		cout<<c[i]<<endl;
	
return 0;
}
