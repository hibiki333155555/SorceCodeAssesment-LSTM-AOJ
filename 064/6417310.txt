#define rep(i,N) for(int i = 0; i < N; i++)
#define ll long long


int main(){
    int N, W;
    cin >> N >> W;

    vector<pair<int, int>> goods(N);
    rep(i, N){
        cin >> goods[i].first >> goods[i].second;
    }

    vector<vector<int>> dp(N+1, vector<int>(W+1, -1));
    dp[0][0] = 0;

    rep(i, N){
        rep(j, W + 1){
            if(dp[i][j] != -1){
                //goods[i]を選ぶ右dp[i+1][j+w]にdp[i][j] + goods[i]を代入するかどうか決める
                if(j + goods[i].second <= W && dp[i][j] + goods[i].first > dp[i+1][j + goods[i].second]){
                    dp[i+1][j + goods[i].second] = dp[i][j] + goods[i].first;
                }
                if(dp[i][j] > dp[i+1][j]){
                    dp[i+1][j] = dp[i][j];
                }
            }
        }
    }
    int ans = -1;
    rep(i, W + 1){
        if(dp[N][i] > ans)ans = dp[N][i];
    }
    cout << ans << "\n";
    // cout << "\n";
