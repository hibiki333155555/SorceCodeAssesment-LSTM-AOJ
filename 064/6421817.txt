import java.util.Scanner;

public class Main{
            public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int high = sc.nextInt();
        int width = sc.nextInt();
        int size = high * width;
        int length = high * 2 + width * 2;
        System.out.println(size + " " + length);
    }
}
