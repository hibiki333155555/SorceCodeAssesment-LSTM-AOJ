int main(void){
    int sx,sy,sz;
    int gx,gy,gz;
    cin >> sx >> sy >> sz;
    cin >> gx >> gy >> gz;
    int ssum=sx+sy+sz;
    int gsum=gx+gy+gz;
    if(abs(ssum-gsum)%2){
        if(ssum%2==0){
            sx++,sy++,sz--;
        }
        else{
            sx--,sy--,sz++;
        }
    }
    ssum=sx+sy+sz;
    gsum=gx+gy+gz;
    bool ans=true;
    if(ssum!=gsum) ans=false;
    if(abs(sx-gx)%2) ans=false;
    if(abs(sy-gy)%2) ans=false;
    if(abs(sz-gz)%2) ans=false;
    if(ans) cout << "Yes" << endl;
    else cout << "No" << endl;
}
