#include <stack>
#include<bits/stdc++.h>
using namespace std;

int main(){
    stack<int> s;
    while(true){
        string a;
        int b,c;
        cin >> a;
        if(a == ""){
            cout << s.top() << endl;
            break;
        }else if(a == "+"){
            b = s.top();
            s.pop();
            c = s.top();
            s.pop();
            s.push(b+c);
        }else if(a == "-"){
            b = s.top();
            s.pop();
            c = s.top();
            s.pop();
            s.push(b-c);
        }else if(a == "*"){
            b = s.top();
            s.pop();
            c = s.top();
            s.pop();
            s.push(b*c);
        }else{
            s.push(std::stoi(a));
        }
    }
}
