#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n;
	cin>>n;
	vector<int>a(n);
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int temp=0;
	int c=0;
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;j<n-1-i;j++)
		{
			if(a[j]>a[j+1])
			{
			swap(a[j],a[j+1]);
				c++;
			}
		}
	}
	for(int i=0;i<n;i++)
	{
		if(i==n-1)
		cout<<a[i]<<"\n";
		else
		cout<<a[i]<<" ";
	}
	cout<<c;
	cout<<"\n";
}
