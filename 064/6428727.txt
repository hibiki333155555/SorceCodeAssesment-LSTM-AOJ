#include <iostream>
#include <string>
using namespace std;

string IntToString(int x) {
	if (x == 0) {
		return "0";
	}
	string str = "";
	while (x >= 1) {
		if (x % 2 == 0) str += "0";
		if (x % 2 == 1) str += "1";
		x /= 2;
	}
	reverse(str.begin(), str.end());
	return str;
}

int main() {
	// 入力
	long long A, B;
	cin >> A >> B;

	// 文字列の比較
	string SA = IntToString(A);
	string SB = IntToString(B);
	while (SA.size() < SB.size()) SA = "0" + SA;
	while (SA.size() > SB.size()) SB = "0" + SB;

	// 出力
	int Answer = 0;
	for (int i = 0; i < (int)SA.size(); i++) {
		if (SA[i] != SB[i]) break;
		Answer += 1;
	}
	cout << Answer << endl;
	return 0;
}
