void print(vector<int>v){
	for(int i=0;i<v.size();i++){
		cout<<v[i]<<" ";
	}
	cout<<"\n";
}
	void InsertionSort(vector<int>&arr){
		int n=arr.size(),j,t;
		for(int i=1;i<=n;i++){
			print(arr);
			j=i-1;
			t=arr[i];
			while(j>=0 && t<arr[j]){
				arr[j+1]=arr[j];
				j--;
				
			}
			arr[j+1]=t;

		}
	}

int main() {
	int n;
	cin>>n;
	vector<int>v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
		
	}
	InsertionSort(v);

}
