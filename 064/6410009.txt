using System;
using System.Linq;

namespace OnlineTest.WBTrees.AOJ
{
	class Set_ITP2_4_D
	{
		static int[] Read() => Array.ConvertAll(Console.ReadLine().Split(), int.Parse);
		static void Main() => Console.WriteLine(Solve());
		static object Solve()
		{
			Console.ReadLine();
			return string.Join(" ", Read().Distinct());
		}
	}
}
