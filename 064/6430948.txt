#include<bits/stdc++.h>
using namespace std;
void print(vector<int>v){
	for(int i=0;i<v.size();i++)
	{
		cout<<v[i];
		if(i!=v.size()-1)
		cout << " ";
	}
	cout<<"\n";
}
void Insertionsort(vector<int>&arr)
{
	int n = arr.size(),j,temp;
	print(arr);
	for(int i=1;i<n;i++)
	{
		j = i-1;
		temp = arr[i];
		while(j>=0 &&  temp< arr[j])
		{
			arr[j+1] = arr[j];
			j--;
		}
		arr[j+1] = temp;
		print(arr);
	
    }
    
}
int main()
{

			
int n;
cin>>n;
vector<int>v(n);
for(int i=0; i<n;i++)
{
cin >> v[i];
}
Insertionsort(v);
}
