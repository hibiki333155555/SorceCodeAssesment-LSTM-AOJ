#include <stdio.h>
#include <string.h>
int main(){
  int i,n = 0;
  char W[10], T[1000];

  scanf("%s", W);

	while(1){
    scanf("%s",T);
    if(strcmp(T,"END_OF_TEXT")) break;
      for(i=0;T[i]!='\0';i++)
        if(T[i]<91) T[i] += 32;
    if( strcmp(W,T) ) n++;
  }

  printf("%d\n",n);

  return 0;
}
