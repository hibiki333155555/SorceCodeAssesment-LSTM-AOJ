#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

map<ll, ll> primeFactrize(ll N){
    map<ll, ll> res;

    for(ll i=2; i*i<=N; ++i){
        if(N%i != 0) continue;

        ll ex = 0;
        while(N%i == 0){
            ex++;
            N /= i;
        }

        res[i] = ex;
    }

    if(N != 1) res[N] = 1;

    return res;
}

int main(){
    ll n;
    cin >> n;

    auto pf = primeFactrize(n);
    vector<ll> ans;
    for(auto p : pf){
        for(ll i=0; i<p.second; ++i) ans.push_back(p.first);
    }

    cout << n << ":";
    rep(i, ans.size()) cout << " " << ans[i];
    cout << endl;
    
    return 0;
}
