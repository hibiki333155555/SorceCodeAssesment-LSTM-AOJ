int main() {
  int k;
  cin >> k;
  vector<int> n(k);
  rep(i,k) cin >> n[i];
  vector<vector<ll>> a(k,vector<ll>(0));
  rep(i,k){
    rep(j,n[i]){
      ll f;
      cin >> f;
      for (ll ii:a[i]){
        if (f>(f^ii)) f=(f^ii);
      }
      if (f!=0) a[i].push_back(f);
    }
  }
  vector<ll> p(62,1);
  rep(i,61) p[i+1]=p[i]*2;
  vector<vector<ll>> base(p[k],vector<ll>(0));
  __int128 ans=0;
  for (int i=1;i<p[k];i++){
    int f=i;
    int count=0;
    rep(i,k){
      if ((f&p[i])!=0) count++;
    }
    for (int j=k-1;j>=0;j--){
      if ((f&p[j])!=0){
        f=(f^p[j]);
        if (f==0){
          base[i]=a[j];
          ans+=p[base[i].size()];
          // cout << i << " " << base[i].size() << endl;
          break;
        }
        vector<ll> temp;
        vector<ll> temp2;
        // if (i==14) cout << a[j].size() << " " << base[f].size() << endl;
        // if (i==13) cout << base[f][0] << endl;
        rep(ii,a[j].size()){
          ll y=0;
          ll now=a[j][ii];
          rep(jj,base[f].size()){
            if (now>(now^base[f][jj])){
              now=(now^base[f][jj]);
              y=(y^base[f][jj]);
            }
          }
          ll to=a[j][ii];
          rep(jj,temp.size()){
            if (now>(now^temp[jj])){
              now=(now^temp[jj]);
              to=(to^temp2[jj]);
            }
          }
          if (now==0){
            base[i].push_back(to);
          }
          else {
            temp.push_back(now);
            temp2.push_back(a[j][ii]);
          }
          // if (i==14) cout << base[i].size() << " " << temp.size() << " " << base[f].size() << " " << temp3.size() << endl;
        }
        if (count%2==1) ans+=p[base[i].size()];
        else ans-=p[base[i].size()];
        vector<ll> temp3;
        for (ll jj:base[i]){
          rep(kk,temp3.size()){
            if (jj>(jj^temp3[kk])) jj=(jj^temp3[kk]);
          }
          if (jj!=0) temp3.push_back(jj);
        }
        // cout << base[i].size() << " " << temp3.size() << endl;
        base[i]=temp3;
        // cout << i << " " << base[i].size() << endl;
        break;
      }
    }
  }
  cout << (ll)ans << endl;
}
