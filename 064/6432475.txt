#include <stdio.h>
#include <stdlib.h>

int A[200000];
int size;

void maxheapfy(int i){
	int left, right, largest, tmp;
	left = 2*i + 1; right = 2*(i+1);
	largest = i;
	if(left < size/2 && A[left] > A[i]) largest = left;
	if(right < size/2 && A[right] > A[largest]) largest = right;

	if(largest != i){
		tmp = A[i];
		A[i] = A[largest];
		A[largest] = tmp;
		maxheapfy(largest);
	}
}

void heapsort(){
	//buildmaxheap
	for(int i = size/2; i >= 0; i--){
		maxheapfy(i);
	}
	//sort
}
	

int main(){
	int n;
	scanf("%d", &n);
	for(int i = 0; i < n; i++){
		scanf("%d", &A[i]);
		size++;
	}
	heapsort();

	while(size--){
		printf("%d ", A[size]);
	}
	puts("");
}
