#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    int N;
    string S;
    cin >> N >> S;
    bool j1=false,j2=false, j3=false;
    int count=0;
    for(int i=count; i<N; i++){
        if(S[i]=='I'){
            j1=true;
            count=i;
            break;
        }
    }
    for(int i=count; i<N; i++){
        if(S[i]=='O'){
            j2=true;
            count=i;
            break;
        }
    }
    for(int i=count; i<N; i++){
        if(S[i]=='I'){
            j3=true;
            count=i;
            break;
        }
    }
    if(j1==true && j2==true && j3==true){
        cout << "Yes" << endl;
    }
    else{
        cout << "No" << endl;
    }
}
