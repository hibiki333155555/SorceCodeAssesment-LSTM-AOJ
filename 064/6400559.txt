#include<stdio.h>

int main()
{
	int number[50] = { 0, };
	double data[2][50] = { 0, };//[0]=weight,[1]=height
	int j = 0;

	while (EOF != scanf("%d,%lf,%lf", &number[j], &data[0][j], &data[1][j]))
	{
		j++;//count data
	}

	for (int k = 0; k < j; k++)
	{
		if ((data[0][k] / (data[1][k] * data[1][k]) )>= 25.0)//judge25
		{
			printf("%d\n", number[k]);
		}
	}


}
