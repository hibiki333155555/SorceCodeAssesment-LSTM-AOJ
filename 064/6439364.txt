#include <bits/stdc++.h>
using namespace std;

int main() {
  int arr[3000][2];
  int i = 0;
  int t;

  while (1) {
    cin >> arr[i][0] >> arr[i][1];
    if (arr[i][0] == 0 && arr[i][1] == 0) break;
    if (arr[i][0] > arr[i][1]) {
      t = arr[i][0];
      arr[i][0] = arr[i][1];
      arr[i][1] = t; 
    }
    i++;
  }

  for (int j = 0; j < i; j++) {
    cout << arr[j][0] << " " << arr[j][1] << endl;
  }
  return 0;
}
