#include<iostream>
#include<algorithm>
#include<cstdio>
#include<string>
#include<climits>
#include<vector>
#define check cout<<"check"<<endl
#define rep(a,b,c)   for(int a = b ; a < c ;a++)
using namespace std;
#define INF 1e9 + 1
using REF = vector< vector<int> > ;

int main(){
    int n,w;
    cin>>n>>w;
    vector< int > W(n);
    vector< int > V(n);
    int MAX = 0;
    rep(i,0,n){
        cin>>V[i]>>W[i];
        MAX += V[i];
    }
    MAX += 1;
    REF dp(n + 1,vector<int> (MAX,w + 1));
    rep(i,0,n + 1)dp[i][0] = 0;
    rep(i,0,n)
    {
        rep(j,0,MAX)
        {
            dp[i + 1][j] = dp[i][j];
            if(j - V[i] < 0)continue;
            int a = dp[i][j - V[i]] + W[i];
            if(a < dp[i + 1][j])dp[i + 1][j] = a;
        }
    }
    int ans ;
    rep(i,0,MAX){
        if( dp[n][i] <= w )ans = i;
    }
    cout<<ans<<endl;



    return 0;


}
