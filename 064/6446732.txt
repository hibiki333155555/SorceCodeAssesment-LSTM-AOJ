#include<stdio.h>
#include<ctype.h>

int main() {
  char s[90001] = {0};
  unsigned long long int c[26] = {0};
  scanf("%[^.]*s", s);
  for (int i = 0; s[i] != '\0'; i++) {
    if (s[i] == 32 || s[i] == '\t' || s[i] == '\n') {
      continue;
    }
    s[i] = toupper(s[i]);
    c[s[i]-65] += 1;
  }
  for (int i = 0; i < 26; i++) {
    printf("%c : %llu\n", i+97, c[i]);
  }
  return 0;
}
