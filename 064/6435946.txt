#include<iostream>
using namespace std;

int main() {
    int a = 0, b = 0, x = 0, a_buy = 0, b_buy = 0;
    cin >> a >> b >> x;
    
    if ( a < b ) {
        a_buy = x / 1000;

        if ( x % 1000 != 0 ) {
            a_buy += 1;
        }    
    } else if ( b * 2 > a ) {
        a_buy = x / 1000;

        if ( x % 1000 != 0 ) {
            if ( x % 1000 > 500 ) {
                a_buy += 1;
            } else {
                b_buy = 1;
            }
        }
    } else if( b * 2 == a ){
        a_buy = x / 1000;

        if ( x % 1000 != 0 ) {
            if ( x % 1000 > 500 ) {
                b_buy = 2;
            } else {
                b_buy = 1;
            }
        }     
    } else if ( b * 2 < a ) {
        b_buy = x / 500;

        if ( x % 500 != 0 ) {
            b_buy += 1;
        }
    }
    cout << a * a_buy + b * b_buy << endl;

    return 0;
}
