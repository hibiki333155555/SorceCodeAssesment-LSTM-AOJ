#include <iostream>
using namespace std;

int main(){
    int n;
    int x;
    int count;
    while (true){
        cin >> n >> x;
        int count = 0;
        if(n == 0 && x == 0){
            break;
        }
        for(int i = 1;i <= x;i++){
            for(int j = i + 1;j <= x;j++){
                for(int k = j + 1;k <= x;k++){
                    if(i + j + k == x){
                        count++;
                    }
                }
            }
        }
    cout << count << endl;
    }
    return 0;
}
