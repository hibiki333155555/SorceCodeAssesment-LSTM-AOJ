#include <iostream>
#include <deque>

using namespace std;

int main(){
    int n, op, dr, e;
    deque<int> deq;
    cin>>n;
    for(int i = 0; i<n; i++){
        cin>>op;
        if(op==0){
            cin>>dr>>e;
            if(dr==0){
                deq.push_front(e);
            }
            else if(dr==1){
                deq.push_back(e);
            }
        }
        else if(op==1){
            cin>>e;
            cout << deq.at(e) << endl;
        }
        else if(op==2){
            cin >>dr;
            if(dr==0){
                deq.pop_front();
            }
            else if(dr==1){
                deq.pop_back();
            }
        }
    }
    return 0;
}
