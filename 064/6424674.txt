#include <iostream>
using namespace std;

int main()
{
   string s;
   cin >> s;
   for (auto &alpha : s)
   {
      if (alpha == 'D')
         alpha = 'A';
      else if (alpha == 'E')
         alpha = 'B';
      else if (alpha == 'F')
         alpha = 'C';
      else
         alpha -= 3;
   }
   cout << s << endl;

   return 0;
}
