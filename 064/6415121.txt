#include <iostream>
#include<cstring>
#include<stack>
using namespace std;

int main()
{
	string s;
	char temp;
	while((temp=cin.get())!='\n')
	if(temp!=' ')
	s+=temp;
	stack<int>a;
	for(int i=0;i<s.length();i++)
	{
		if(s[i]=='+')
		{
			int l,r;
			r=a.top();
			a.pop();
			l=a.top();
			a.pop();
			a.push(l+r);
		}
		else if(s[i]=='-')
		{
			int l,r;
			r=a.top();
			a.pop();
			l=a.top();
			a.pop();
			a.push(l-r);
		}
		else if(s[i]=='*') 
		{
			int l,r;
			r=a.top();
			a.pop();
			l=a.top();
			a.pop();
			a.push(l*r);
		}
		else if(s[i]=='/') 
		{
			int l,r;
			r=a.top();
			a.pop();
			l=a.top();
			a.pop();
			a.push(l/r);
		}
		else a.push(s[i]-'0');
	}
	cout<<a.top()<<endl;
	//cout<<s<<sizeof(s);
return 0;
}
