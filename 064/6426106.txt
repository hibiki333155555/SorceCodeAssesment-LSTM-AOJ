#include <stdio.h>

int main(void){
    double d,w,h,r;
    int i,n;
    
    while(1){
        scanf("%lf %lf %lf",&d,&w,&h);
        if(d*w*h==0) break;
        
        scanf("%d",&n);
        for(i=0;i<n;i++){
            scanf("%lf",&r);
            //printf("d:%lf,w:%lf,h:%lf,d^2:%lf,1:%lf,2:%lf,3:%lf",d,w,h,4*r*r,d*d+h*h,h*h+w*w,w*w+d*d);
            if((4*r*r>d*d+h*h)||(4*r*r>h*h+w*w)||(4*r*r>w*w+d*d)){
                puts("OK");
            }
            else{
                puts("NA");
            }
        }
    }
    
    return 0;
}
