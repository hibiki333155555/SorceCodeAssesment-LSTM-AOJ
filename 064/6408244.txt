#include <iostream>

using namespace std;

int  n, a[50], b[200];

int solve(int i, int m){
    if(m == 0) return 1;
    if(i == n || m < 0) return 0;
    return solve(i + 1, m) || solve(i + 1, m - a[i]);
}

int main()
{
    int q;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    cin >> q;
    for(int i = 0; i < q; i++){
        cin >> b[i];
    }
    for(int i = 0; i < q; i++){
        if(solve(0, b[i])) cout << "yes\n";
        else cout << "no\n"; 
    }
    return 0;
}
