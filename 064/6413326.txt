#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main(){
	int n,b,f,r,v;
	int x[4][11],y[4][11],z[4][11],w[4][11];

	//配列の初期化
	for(int i = 1;i < 4;i++){
		for(int j = 0;j < 11;j++){
			x[i][j] = 0;
			y[i][j] = 0;
			z[i][j] = 0;
			w[i][j] = 0;
		}
	}

	//n件の入力
	scanf("%d",&n);
	//情報の入力
	for(int i = 0;i < n;i++){
		scanf("%d %d %d %d",&b,&f,&r,&v);
		if (b == 1){
			x[f][r] = v;
		}
		if (b == 2){
			y[f][r] = v;
		}
		if (b == 3){
			z[f][r] = v;
		}
		if (b == 4){
			w[f][r] = v;
		}
	}

	for(int i = 1;i < 4;i++){
		for(int j = 1;j < 11;j++){
			printf(" %d",x[i][j]);
		}
		printf("\n");
	}
	printf("####################\n");

	for(int i = 1;i < 4;i++){
		for(int j = 1;j < 11;j++){
			printf(" %d",y[i][j]);
		}
		printf("\n");
	}
	printf("####################\n");

	for(int i = 1;i < 4;i++){
		for(int j = 1;j < 11;j++){
			printf(" %d",z[i][j]);
		}
		printf("\n");
	}
	printf("####################\n");

	for(int i = 1;i < 4;i++){
		for(int j = 1;j < 11;j++){
			printf(" %d",w[i][j]);
		}
		printf("\n");
	}
	return 0;
}
