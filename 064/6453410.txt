#include<stdio.h>
int main(void){
    int a=1;
    int b=1;
    while(a>0&&b>0){
        scanf("%d",&a);
        scanf("%d",&b);
        for(int i=0;i<a;i++){
            for(int j=0;j<b;j++){
                printf("%c",'#');
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
