#include <bits/stdc++.h>
using namespace std;
int main() {
	   int n,i,j,count=0;
           cin>>n;
           int a[n];
            for(i=0;i<n;i++)
            {
               cin>>a[i];
            }
            for(i=0;i<n;i++)
            {
            	int pos=i;
            	for(j=i;j<n;j++)
            	{
            		if(a[pos]>a[j])
            		pos=j;
            	}
            	if(i!=pos)
            	{
            	 swap(a[i],a[pos]);
            	 count++;
            	}
            }
            	for(i=0;i<n;i++)
            	{
            	           cout<<a[i]<<" ";
            	}
            	cout<<"\n"<<count;
}
