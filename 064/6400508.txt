#include<bits/stdc++.h>
using namespace std;
const int N = 110;
typedef long long ll;
typedef pair<int, int> P;
ll dp[N][2]; 
int main(){
    int l,k;
    while(scanf("%d%d",&l,&k)!=EOF){
        dp[0][0] = 1;
        for(int i = 1; i<=l; i++){
            dp[i][1] = dp[i-1][0];
            dp[i][0] = dp[i-1][1];
            if(i>=k) dp[i][1] = dp[i-1][0]+dp[i-k][0];
        }
        ll ans = 0;
        for(int i = 1; i<=l; i++){
            ans += dp[i][1];
        }
        printf("%lld\n",ans);
    }
    return 0;
}
