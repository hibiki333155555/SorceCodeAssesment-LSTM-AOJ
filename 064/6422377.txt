#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  int p1=0,p2=0;
  for(int i=0; i<n; i++){
    string s,t;
    cin >> s >> t;
    if(s==t){
      p1++;
      p2++;
    }else if(s>t){
      p1+=3;
    }else{
      p2+=3;
    }
  }
  
  cout << p1 << " " << p2 << endl;
  return 0;
    
}
