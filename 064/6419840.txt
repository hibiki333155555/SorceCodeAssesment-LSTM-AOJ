#include <bits/stdc++.h>
using namespace std;
struct dot{
    double x, y;
};

double d2(double x1, double x2){
    
    return (x1-x2)*(x1-x2);
}


int main(){
    dot a1, a2;
    cin>>a1.x>>a1.y>>a2.x>>a2.y;
    cout << fixed << setprecision(10);
    cout<<sqrt(d2(a1.x, a2.x)+d2(a1.y, a2.y))<<endl;
    return 0;
}
