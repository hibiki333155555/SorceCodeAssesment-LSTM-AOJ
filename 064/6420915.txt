#include <bits/stdc++.h>
using namespace std;
int main(){
    string S;
    cin >> S;
    int sz = S.size();
    vector<int> sum(sz + 1);
    stack<int> s;
    for(int i = 0; i < sz; i++){
        if('0' <= S[i] && S[i] <= '9'){
            sum[i] += S[i] - '0';
            s.push(S[i] - '0');
        }
        else if(S[i] == ')'){
            sum[i + 1] -= s.top();
            s.pop();
        }
    }
    string ans;
    for(int i = 0; i < sz; i++){
        sum[i + 1] += sum[i];
        if('a' <= S[i] && S[i] <= 'z'){
            ans += 'a' + (S[i] - 'a' + sum[i] % 26) % 26;
        }
    }
    cout << ans << endl;
}
