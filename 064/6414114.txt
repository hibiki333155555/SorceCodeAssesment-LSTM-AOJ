#include<stdio.h>
#include<stdlib.h>
#define MAX 300


int main()
{
  int H;
  int A[MAX];
  scanf("%d", &H);
  for(int i = 1; i <= H; i++ ) scanf("%d", &A[i]);

  for(int i = 1; i <= H; i++ ){
    printf("node %d: ", i);
    printf("key = %d, ", A[i]);
    if(i / 2 > 0) printf("parent key = %d, ", A[i / 2]);
    if(i * 2 <= H) printf("left key = %d, ", A[i * 2]);
    if(i * 2 + 1 <= H) printf("right key = %d, ", A[i * 2 + 1]);
    printf("\n");
  } 

  return 0;
}
