import java.util.Scanner;

class Main{
    
    static class Cell{
        int key;
        Cell next;
        Cell prev;
        Cell head;

        Cell(int key,Cell next,Cell prev){
            this.key = key;
            this.next = next;
            this.prev = prev;
        }

        Cell insert(int key,Cell cell){
            if(cell.key == -1){
                cell.key = key;
            }else{
                cell.prev = new Cell(key,cell,null);
                cell = cell.prev;
            }
            return cell;    
        }

        Cell delete(int key,Cell cell){
            head = cell;
            while(true){
            if(cell.key == key && cell.prev == null){
                head = cell.next;
                head.prev = null;
                break;
            }else if(cell.key == key && cell.next == null){
               cell.deleteLast(cell);
               break;
            }else if(cell.key == key && cell.next != null){
                Cell temp = cell.next;
                cell = cell.prev;
                temp.prev = cell;
                cell.next = temp;
                break;
            }else if(cell.key != key && cell.next == null){
                break;
            }
            cell = cell.next;
        }
            return head;
    }

        Cell deleteFirst(Cell cell){
            Cell initialize = new Cell(-1,null,null);
            if(cell.next == null){
                cell = initialize;
            }else{
            cell = cell.next;
            cell.prev = null;
            }
            return cell;
        }

        void deleteLast(Cell cell){
            Cell initialize = new Cell(-1,null,null);
            if(cell.next == null && cell.prev == null){
                cell = initialize;
            }else if(cell.next == null && cell.prev != null){
                   cell = cell.prev;
                   cell.next = null;
            }else cell.deleteLast(cell.next);
        }

        void print(Cell cell){
            for(boolean b=true;b==true;){
                System.out.print(cell.key);
                if(cell.next == null){
                    b = false;
                    System.out.println();
                }else System.out.print(" ");
                cell = cell.next;    
            }
        }
    }

    public static void main(String[] args) throws Exception{
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();
        Cell cell = new Cell(-1,null,null);
        for(int i=0;i<n;i++){
            String order = s.next();
                switch(order){
                    case "insert" : cell = cell.insert(s.nextInt(),cell); break;
                    case "delete" : cell = cell.delete(s.nextInt(),cell); break;
                    case "deleteFirst" : cell = cell.deleteFirst(cell); break;
                    case "deleteLast" : cell.deleteLast(cell); break;
                }
            }
            cell.print(cell);
        }
    }
