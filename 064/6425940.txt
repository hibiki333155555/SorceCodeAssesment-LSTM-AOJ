#include <iostream>
#include <cstdio>
using namespace std;
int main(void){
    // Your code here!
    int d;
    scanf("%d",&d);
    int S[d];
    for(int i=0;i<d;i++){
        scanf("%d",&S[i]);
        //printf("%d",S[i]);
    }
    int e;
    scanf("%d",&e);
    int T[e];
    for(int i=0;i<e;i++){
        scanf("%d",&T[i]);

    }
    int C=0;
    for (int i=0;i<e;i++){
        for(int j=0;j<d;j++){
            if(T[i]==S[j]) C++;
        }
    }
    
    printf("%d",C);
