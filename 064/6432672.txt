#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

int main(){
    while(true){
        int n, x;
        cin >> n >> x;

        if(!n && !x) return 0;

        int ans = 0;

        for(int i=1; i<=n; ++i){
            for(int j=i+1; j<=n; ++j){
                for(int k=j+1; k<=n; ++k){
                    if(i+j+k == x) ans++;
                }
            }
        }

        cout << ans << endl;
    }
}
