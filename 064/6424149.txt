int main() {
  int N;
  cin >>N;
  
  vector<int> vec(N+1);
  vec[0]=1;
  vec[1]=1;
  for (int i = 2; i <= N; i++)
  {
    vec[i]=vec[i-1]+vec[i-2];
  }
  
  cout << vec[N] << endl;
