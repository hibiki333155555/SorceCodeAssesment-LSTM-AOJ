#include <iostream>
#include <vector>
#include <stack>

using namespace std;

int main(){
    int n, q;
    cin>>n>>q;
    vector<stack<int>> s(n);
    
    int op, t, x;
    for(int i = 0; i < q; i++){
        cin>>op>>t;
        switch(op){
            case 0:
            cin>>x;
            s.at(t).push(x);
            break;
            case 1:
            if(!s.at(t).empty())cout << s.at(t).top() << endl;
            break;
            case 2:
            if(!.s.at(t).empty())s.at(t).pop();
        }
    }
    return 0;
}
