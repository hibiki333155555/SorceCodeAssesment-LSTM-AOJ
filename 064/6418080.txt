#include <iostream>
using namespace std;

void solve(){
    int L, N; cin >> L >> N;
    for(int i = 0; i < N; i++){
        int a; cin >> a;
        L -= a;
        if(L <= 0){
            cout << i + 1 << '\n';
            return;
        }
    }
    cout << "No\n";
}

int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    solve();
    return 0;
}
