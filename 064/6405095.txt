package パソコン甲子園;
import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);	
		//変数
		int d=4;
		double l;
		double n=d/4;
		//処理
		while(4<=d&&d<=400&&d/4==n) {
			d = sc.nextInt();
			l=d/4;
			if(4*(int)l==d&&l>0&&l<=100) {
				System.out.println(d/4);
			}
		}
	}
}
