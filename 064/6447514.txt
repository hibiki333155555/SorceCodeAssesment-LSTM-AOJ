#include <iostream>
using namespace std;

int main(void){
    int n;
    cin >> n;
    
    int min = 20000000000, now = -20000000000, ans = -20000000000;
    int T = 0;
    
    for (int t = 0; t < n; t++) {
        cin >> T;
        
        now = T - min;
        
        if (ans < now) {
            ans = now;
        }
        
        if(T < min) {
            min = T;
        }
        

    }
    
    cout << ans << endl;
    
}
