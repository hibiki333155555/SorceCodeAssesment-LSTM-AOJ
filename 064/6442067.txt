#include<iostream>
#include<vector>
using namespace std;
int n;
vector<int> A;

bool solve(int a, int b) {
    if (b == 0)return true;
    else if (a >= n)return false;

    return solve(a + 1, b) || solve(a + 1, b - A[a]);
}

int main() {
    cin >> n;
    A.resize(n);
    for (int i = 0; i < n; i++)cin >> A[i];

    int q;  
    cin >> q;
    vector<int> m(q);
    for (int i = 0; i < q; i++)cin >> m[i];

    bool result;
    for (int i = 0; i < q; i++) {
        result = solve(0, m[i]);
        if (result == true)cout << "yes" << endl;
        else cout << "no" << endl;
    }
    return 0;
}
