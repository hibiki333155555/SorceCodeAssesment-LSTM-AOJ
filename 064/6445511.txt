int main()
{	
	int n,i,j;
	int a[200005];
	int ma=-1000000000;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	
	for(i=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[j]-a[i]>ma)
				ma=a[j]-a[i];
		}
	}
	 
	cout<<ma<<endl;
	
return 0;
}
