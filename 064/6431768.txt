#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int n;
  cin >> n;
  
  // int型の2次元配列(3×10要素の)の宣言
  int building_1[3][10];
  int building_2[3][10];
  int building_3[3][10];
  int building_4[3][10];
  
  // 入力の受け取り
  int b, f, r, v;
  
  for (int i=0; i<n; i++){
    cin >> b >> f >> r >> v;
    
    if (b == 1){
      building_1.at(f-1).at(r-1) = v;
    }
    
    else if (b == 2){
      building_2.at(f-1).at(r-1) = v;
    }
    
    else if (b == 3){
      building_3.at(f-1).at(r-1) = v;
    }
    
    else if (b == 4){
      building_4.at(f-1).at(r-1) = v;
    } 
  }
  
  //出力
  for (int i=0; i<3; i++){
    for (int j=0; j<10; j++){
      cout << building_1.at(i).at(j) << " ";
    }
    cout << endl;
  }
  
  for (int i=0; i<20; i++){
    cout << "#";
  }
  
  cout << endl;
  
  for (int i=0; i<3; i++){
    for (int j=0; j<10; j++){
      cout << building_2.at(i).at(j) << " ";
    }
    cout << endl;
  }
  
  for (int i=0; i<20; i++){
    cout << "#";
  }
  
  cout << endl;
  
  for (int i=0; i<3; i++){
    for (int j=0; j<10; j++){
      cout << building_3.at(i).at(j) << " ";
    }
    cout << endl;
  }
  
  for (int i=0; i<20; i++){
    cout << "#";
  } 
  
  cout << endl;
  
  for (int i=0; i<3; i++){
    for (int j=0; j<10; j++){
      cout << building_4.at(i).at(j) << " ";
    }
    cout << endl;
  }
  

  
  
  
}
