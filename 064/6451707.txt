#include <bits/stdc++.h>

using namespace std;

int main() {
  int n;
  scanf("%d", &n);
  int S[n];
  for (int i = 0; I < n; i++) {
    scanf("%d", &S[i]);
  }
  int q;
  scanf("%d", &q);
  int T[q];
  for (int i = 0; i < q; i++) {
    scanf("%d", &T[i]);
  }

  int count = 0;
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < q; j++) {
      if (T[j] == S[i]) {
        count++;
        T[j] = -1;
      }
    }
  }

  cout << count << endl;
  return 0;
}
