#include<bits/stdc++.h>
using namespace std;

int main(){

    while(true){
        int a, b;
        char c;

        scanf("%d %c %d", &a, &c, &b);
        if(c == '?')
            break;

        if(c == '+')
            printf("%d\n", a + b);
        else if(c == '-')
            printf("%d\n", a - b);
        else if(c == '*')
            printf("%d\n", a * b);
        else if(c == '/')
            printf("%d\n", a / b);

    }
    return 0;
}
