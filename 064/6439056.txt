#include <iostream>
using namespace std;

const int N=100;

int main(){
	int count;
	int array[N];
	cin>>count;
	for(int i=0;i<count;i++){
		cin>>array[i];
	}
	for(int i=1;i<count;i++){
		for(int k=0;k<count-1;k++){
		cout<<array[k]<<" ";
		}
		cout<<array[count-1]<<endl;
		int a=array[i];
		int j=i-1;
		for(;j>=0&&a<array[j];j--){
			array[j+1]=array[j];
		}
		array[j+1]=a;
	}
	for(int i;i<count-1;i++){
		cout<<array[i]<<" ";
	}
	cout<<array[count-1]<<endl;
