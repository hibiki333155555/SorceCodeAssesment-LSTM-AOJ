#include <stdio.h>
#define small(a,b) ((a)<(b)?(a):(b))
#define large(a,b) ((a)>(b)?(a):(b))

int main(){
   int a, b, c;
   int first, second, third;
   
   scanf("%d %d %d", &a, &b, &c);

   first = small(small(a, b), c);
   second = large(large(a, b), c);
   third = a+b+c-first-second;

   printf("%d %d %d\n", first, second, third);  
   return 0;
}
