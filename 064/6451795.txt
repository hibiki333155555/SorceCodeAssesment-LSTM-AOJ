#include <iostream>
using namespace std;

int main()
{
    int i, j;
    int H, W;
    
    while(1) {
		cin >> H >> W;
		
		if ( ( H | W ) != 0 ) {
			for ( i = 0; i < H; i++ ) {
				for ( j = 0; j < W; j++ ) {
					if ( ( i + j ) % 2 = 0 ) {
					    cout << "#";
					} else {
					    cout << ".";
					}
				}
				cout << endl;
			}
			cout << endl;
		} else {
			break;
		}
	}
    
    return 0;
}
