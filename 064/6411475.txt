#include <iostream>
#include <set>

using namespace std;

int main(){
    int n, op,e;
    set<int>s;
    cin>>n;
    while(n--){
        cin>>op>>e;
        switch(op){
            case 0:
            s.insert(e);
            cout << s.size() << endl;
            break;
            case 1:
            cout << (s.end() != s.find(e)) << endl;
            break;
            case 2:
            s.erase(e);
        }
    }
}
