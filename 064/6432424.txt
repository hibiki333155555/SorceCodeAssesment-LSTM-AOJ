#include <iostream>
#include <vector>
#include <string>
using namespace std;

void rate(int m,int f,int r) {
	int t = m + f;
	if ((m == -1 or f == -1) or t<30) {
		cout << 'F' << endl;
	}
	else if ( t>= 80) {
		cout << 'A' << endl;
	}
	else if (t >= 65 and t < 80) {
		cout << 'B' << endl;
	}
	else if (t >= 50 and t < 65) {
		cout << 'C' << endl;
	}
	else if (t >= 30 and t < 50 and r<50) {
		cout << 'D' << endl;
	}
	else if (t >= 30 and t < 50 and r >= 50) {
		cout << 'C' << endl;
	}
}

int main() {
	int m, f, r;
	vector <vector<int> > data(50, vector<int>(3));
	for (int i = 0;; ++i) {
		for (int j = 0; j < 3; ++j) {
			cin >> data.at(i).at(j);
			
		}
		if (data.at(i).at(0) == -1 && data.at(i).at(1) == -1 && data.at(i).at(2) == -1) break;
	}
	for (int i = 0;; ++i) {
		if (data.at(i).at(0) == -1 && data.at(i).at(1) == -1 && data.at(i).at(2) == -1) break;
		rate(data.at(i).at(0), data.at(i).at(1), data.at(i).at(2));
	}
	

}
