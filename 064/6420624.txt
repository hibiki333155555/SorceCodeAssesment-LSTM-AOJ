// #pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int n;
    long long m, k, x;
    cin >> n >> m >> k >> x;
    vector<long long> a(n);
    long long ma = 0;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        ma = max(ma, a[i]);
    }
    cout << (ma * k >= m * x ? "Yes" : "No") << '\n';
    return 0;
}
