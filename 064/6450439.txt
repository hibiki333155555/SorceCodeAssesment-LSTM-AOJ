int RetDefiniteNum(string s){
    int ret = 0;
    if(SIZE(s)==1){
        ret += s[0]-'A';
    }else{
        ret += 26*(s[0]-'A');
        ret += s[1]-'A';
    }
    return ret;
}

int main(){
    string c1,c2;
    int r1,r2;
    cin >> c1 >> r1 >> c2 >> r2;
    int tate = r2- r1 + 1;
    int yoko = 0;
    yoko += RetDefiniteNum(c2) - RetDefiniteNum(c1) +1;
    if(r1 == r2) tate = 0;
    if(c1 == c2) yoko = 0;
    cout << tate * yoko << endl;
}
