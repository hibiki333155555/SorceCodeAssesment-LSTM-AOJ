#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	int b,f,r,v;
	int n,i,j,k;
	int hous[5][4][11]={0};
	
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>b>>f>>r>>v;
		hous[b][f][r]+=v;
	}
	
	for(i=1;i<=4;i++)
	{
		for(j=1;j<=3;j++)
		{
			for(k=1;k<=10;k++)
				cout<<" "<<hous[i][j][k];
			
			cout<<endl;
		}
		
		if(i!=4)
			for(j=1;j<=20;j++)
				cout<<"#";
			cout<<endl;
	}
	cout<<endl;
	
return 0;
}
