#include <iostream>
#include <algorithm>
#include "vector"

using namespace std;

struct P
{
    double x, y;
};
int q;

P p0;
vector<P> ls;

double cp2d(P a, P b, P c, P d)
{
    return (b.x - a.x) * (d.y - c.y) - (b.y - a.y) * (d.x - c.x);
}

void cal()
{
    for (int i = 0; i < q; ++i)
    {
        if(cp2d(ls[i], ls[i+1], ls[i+1], ls[i+2])<0)
        {
            cout << 0 << endl;
            return;
        }
    }
    cout << 1 << endl;
}

signed main()
{
    cout.setf(ios::fixed);
    cout.precision(1);
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    cin >> q;
    for (int i = 0; i < q; ++i)
    {
        cin >> p0.x >> p0.y;
        ls.push_back(p0);
    }
    ls.push_back(ls[0]);
    ls.push_back(ls[1]);
    cal();
    return 0;
}
