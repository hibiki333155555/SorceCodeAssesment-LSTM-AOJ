#include <stdio.h>

int main(void){

    int n;
    scanf("%d", &n);

    int cards[4][13];
    int i, j;
    for(i = 0; i < 4; i++) for(j = 0; j < 13; j++) cards[i][j] = 0; // initialize

    char sym;
    int rank;
    i = n;
    while(i--){
        scanf("%c %d", &sym, &rank);
        if(sym == 'S') cards[0][rank - 1] = 1;
        if(sym == 'H') cards[1][rank - 1] = 1;
        if(sym == 'C') cards[2][rank - 1] = 1;
        if(sym == 'D') cards[3][rank - 1] = 1;
    }

    for(i = 0; j < 4; j++){
        for(j = 0; j < 13; j++){
            if(cards[0][j]) printf("S %d\n", j + 1);
            if(!cards[1][j]) printf("H %d\n", j + 1);
            if(!cards[2][j]) printf("C %d\n", j + 1);
            if(!cards[3][j]) printf("D %d\n", j + 1);
        }
    }

    return 0;
}
