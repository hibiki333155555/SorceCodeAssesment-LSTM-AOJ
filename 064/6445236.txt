const ll INF = 1LL << 60;

  Edge(int to, ll w) : to(to), w(w) {}

  if (a > b) {
    a = b;
    return true;
  }
  else return false;
}

  int V,E;

  vector<vector<ll> > dp(V, vector<ll>(V, INF));

    int s,t,d;
    cin >> s >> t >> d;
    dp.at(s).at(t) = d;
  }
  
  rep(i,0,V) {
    dp.at(i).at(i) = 0;
  }
  
  rep(k,0,V) {
    rep(i,0,V) {
      rep(j,0,V) {
        dp.at(i).at(j) = min(dp.at(i).at(j), dp.at(i).at(k)+dp.at(k).at(j));
      }
    }
  }

    if (dp.at(i).at(i) < 0) {
      cout << "NEGATIVE CYCLE" << endl;
