#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int x, y;
  int i;
  
  while(i < 3000){
    cin >> x >> y;
    
    if (!(x==0 && y==0)){
      if (x <= y){
        cout << x << " " << y << endl;
      }
    
      else if (x > y){
        cout << y << " " << x << endl;
      }
    
    }

    else{
      break;
    }
    
  }
  
}
