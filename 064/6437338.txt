#include <iostream>
#include <string>

using namespace std;
int main()
{
    string a;
    cin >> a;

    while(a[0] != '0')
    {
        int x = 0;
        for(int i = 0 ; i < a.size(); i++)
        {
        x += a[i] - '0';
        }
         cout << x << endl;
         cin >> a;
    }
    return 0;
}
