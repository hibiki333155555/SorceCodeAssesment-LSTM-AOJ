def print_num_list(num_list):
    print(' '.join([f'{num}' for num in num_list]))


n = int(input())
num_list = list(map(int,input().split()))

print_num_list(num_list)

for i in range(1,n):
    num = num_list[i]
    j = i-1
    while (j>=0 and num_list[j]>num):
        num_list[j+1] = num_list[j]
        j -= 1
    num_list[j+1] = num
    print_num_list(num_list)
