    if (k<0)return 0;ll ret=1;ll now=n;while(k>0){if(k&1)ret *= now; now *= now; k /= 2; }return ret;
}
ll beki(ll n,ll k,ll md) {
    ll ret =1;ll now=n;now%=md;
    while (k>0) {
        if (k%2==1){ret*=now;ret%=md;}
        now *= now; now %= md; k /= 2;
    }return ret;
}
