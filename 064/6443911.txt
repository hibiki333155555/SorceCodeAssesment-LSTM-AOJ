#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;
#define drep(i, cc, n) for (ll i = (cc); i <= (n); ++i)
#define rep(i, n) drep(i, 0, n - 1)
#define all(a) (a).begin(), (a).end()
#define pb push_back
#define fi first
#define se second

const ll MOD = 1000000007;
const ll MOD2 = 998244353;
const ll INF = 1LL << 60;
const ll N_MAX = 2e5;

int main(){
    ll t;
    cin >> t;

    vector<ll> n(t);
    ll n_max = 0;
    rep(i, t){
        cin >> n[i];
        n_max = max(n_max, n[i]);
    }

    vector<ll> cnt(n_max+1, 1);
    vector<bool> prime(n_max+1, true);
    for(ll i=2; i<=n_max; i++){
        if(!prime[i]) continue;

        ll c = 1;
        for(ll j=1; i*(j+1)<=n_max; j++){
            cnt[i * (j + 1)] += c;
            if(prime[i*(j+1)]) c++;
            prime[i*(j+1)] = false;
        }
    }

    vector<ll> f(n_max+1);
    f[1] = 2;
    for(ll i=2; i<=n_max; i++) f[i] = f[i-1] + (i-cnt[i]);
    rep(i, t) cout << f[n[i]] << endl;
}
