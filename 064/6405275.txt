#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    int x[n]={},y[n]={};
    for (int i=0;i<n;i++) {
        cin >> x[i];
    }
    for (int i=0;i<n;i++) {
        cin >> y[i];
    }
    double ans[3];
    int max = 0;
    cout << fixed;
    for (int j=0;j<3;j++) {
        for (int i=0;i<n;i++) {
           ans[j] += pow(y[i]-x[i], j+1);
           if (max < abs(y[i]-x[i])) {
               max = abs(y[i]-x[i]);
           }
        }
        cout << setprecision(6) << pow(ans[j], 1/(j+1)) << endl;
    }
    cout << max << endl;
}
