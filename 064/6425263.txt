#include<stdio.h>

int main()
{
    int i;
    int n,a,min=1000001,max=-1000001;
    long long sum;
    scanf("%d",&n);
    for(i=0 ; i<n ; i++)
    {
        scanf("%d",&a);
        sum+=a;
        if(min>=a) min=a;
        if(max<=a) max=a;
    }
    printf("%d %d %lld\n",min,max,sum);
    return 0;
}
