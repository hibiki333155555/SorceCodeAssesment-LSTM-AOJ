#include<iostream>
#include<unordered_map>
#include<vector>
using namespace std;
int main(){
    int n,q;
    vector<int> S;
    vector<int> T;
    unordered_map<int,int> mp;
    cin>>n;
    for(int i=0;i<n;i++){
        int temp;
        cin>>temp;
        mp[temp]++;
        S.push_back(temp);
    }
    cin>>q;
    int count=0;
    for(int i=0;i<q;i++){
        int temp;
        cin>>temp;
        if(mp[temp]>0){
            count++;
        }
        T.push_back(temp);
    }
    cout<<count<<endl;

    return 0;
}
