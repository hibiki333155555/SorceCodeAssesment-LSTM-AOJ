#include <stdio.h>

#define MAX 999999

int main(void){
    int input[10000],sorted[10000];
    int i,j,n,min,minX,sum;
    
    while(1){
        sum=0;
        
        scanf("%d",&n);
        if(n==0) break;
        
        for(i=0;i<n;i++){
            scanf("%d",&input[i]);
        }
        
        for(i=0;i<n;i++){
            min=MAX;
            for(j=0;j<n;j++){
                if(min>input[j]){
                    min=input[j];
                    minX=j;
                }
            }
            sorted[i]=min;
            input[minX]=MAX;
        }
        
        for(i=0;i<n;i++){
            sum+=sorted[i]*(n-i-1);
        }
        
        printf("%d\n",sum);
    }
    
    return 0;
}
