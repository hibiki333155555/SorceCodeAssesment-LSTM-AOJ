#define rep(i,n) for(int i=0;i<(int)(n);i++)
#define ALL(v) v.begin(),v.end()
typedef long long ll;
typedef unsigned long long ull;

#include <bits/stdc++.h>
using namespace std;

const ull MOD=1e9+7;

//ハッシュの基数
const ull B1=10007;
const ull B2=1007;

ll modpow(ll x,ll n){
  ll ans=1;
  while(n){
    if(n&1) ans=ans*x%MOD;
    x=x*x%MOD;
    n/=2;
  }
  return ans;
}

int main(){
  ull bi1=modpow(B1,MOD-2);
  ull bi2=modpow(B2,MOD-2);
  
  int n;
  string s;
  cin>>n>>s;
  if(n==1){
    cout<<0<<endl;
    return 0;
  }
  
  int cnt=0;
  ull ah=0,ch=0,bh=0,dh=0;
  ull t1=1,t2=1;
  for(int i=0;i<n/2;i++){
    ah=(ah*B1%MOD+s[i]-'a')%MOD;
    ch=(ch*B2%MOD+s[i]-'a')%MOD;
    if(i==0){
      bh=s[1]-'a',dh=s[1]-'a';
    }
    else{
      bh=((bh-s[i]+'a'+MOD)%MOD*bi1%MOD+(s[2*i]-'a'+(s[2*i+1]-'a')*B1%MOD)*t1%MOD)%MOD;
      dh=((dh-s[i]+'a'+MOD)%MOD*bi2%MOD+(s[2*i]-'a'+(s[2*i+1]-'a')*B2%MOD)*t2%MOD)%MOD;
      t1=t1*B1%MOD,t2=t2*B2%MOD;
    }
    if(ah==bh && ch==dh) cnt++;
  }
  reverse(ALL(s));
  ah=0,ch=0,bh=0,dh=0;
  t1=1,t2=1;
  if(n%2==0) n-=2;
  for(int i=0;i<n/2;i++){
    ah=(ah*B1%MOD+s[i]-'a')%MOD;
    ch=(ch*B2%MOD+s[i]-'a')%MOD;
    if(i==0){
      bh=s[1]-'a',dh=s[1]-'a';
    }
    else{
      bh=((bh-s[i]+'a'+MOD)%MOD*bi1%MOD+(s[2*i]-'a'+(s[2*i+1]-'a')*B1%MOD)*t1%MOD)%MOD;
      dh=((dh-s[i]+'a'+MOD)%MOD*bi2%MOD+(s[2*i]-'a'+(s[2*i+1]-'a')*B2%MOD)*t2%MOD)%MOD;
      t1=t1*B1%MOD,t2=t2*B2%MOD;
    }
    if(ah==bh && ch==dh) cnt++;
  }
  cout<<cnt<<endl;
  
  return 0;
}
