#include <bits/stdc++.h>
using namespace std;
void print(vector<int> v)
{
	for(int i=0;i<v.size();i++)
	{
	    if(i==v.size()-1)
	    {
	        	cout<<v[i];
	    }
	    else
	    {
		cout<<v[i]<<" ";
	    }
	}
	cout<<"\n";
}
void inset(vector<int> &arr)
{
	
int n=arr.size(),temp,j;
for(int i=0;i<n;i++)
{
	j=i-1;
	temp=arr[i];
	while(j>=0 && temp<arr[j])
	{
		arr[j+1]=arr[j];
		j--;
	}
	arr[j+1]=temp;
	print(arr);
	cout<<"\n";
}
	
	
}
int main()
{
  
  
  	int n; cin>>n;
	vector<int>v(n);
	for(int i=0;i<n;i++)
	{
			cin>>v[i];
	}
	inset(v);
	cout<<"\n";//
  
	
	return 0;
}
