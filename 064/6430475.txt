#include<bits/stdc++.h>

using namespace std;

int main()
{
    int n,count=0;
    cin >> n;
    vector<int>v(n);
    for(int i=0;i<n;i++)
    {
        cin >>v[i];
    }
    for(int i=0;i<n-1;i++)
    {   
        for(int j=0;j<n-1-i;j++)
        {
            if (v[j] > v[j+1])
            {
                swap(v[j],v[j+1]);
                count=count+1;
            }
        }
    }
    for(int h=0;h<n;h++)
        {
            if(h<n-1){
                cout << v[h] << " ";
            }
            else{
            cout << v[h];
            }
        }
        cout << "\n";
    cout << count << "\n" ;
    
    
    return 0;
}
