#include <iostream>
using namespace std;
using ll = int64_t;

void solve(){
    ll H[2], W[2];
    for(int i = 0; i < 2; i++) cin >> H[i] >> W[i];
    cout << ((H[0] + H[1] - 1) / H[1]) * ((W[0] + W[1] - 1) / W[1]) << '\n';
}

int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    solve();
    return 0;
}
