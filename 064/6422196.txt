import java.util.Scanner;

class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int a= sc.nextInt();
        int b= sc.nextInt();
        int d=a/b;
        int r=a%b;
        double f=1.0*a/b;
        if(f==(int)f){
            System.out.println(d+" "+r+" "+(int) f);
        }
        else{
            System.out.println(d+""+r+""+f);
        }
    }
}
