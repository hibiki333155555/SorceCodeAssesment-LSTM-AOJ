#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
struct Edge{
    int to,next;
}edge[30010];
int head[30010],dfn[10010],low[10010],cnt,scc_cnt,top,t;
int skt[10010],id[10010];
bool in_skt[10010];
int m,n,root;
void add(int a,int b){
    edge[cnt].to=b;
    edge[cnt].next=head[a];
    head[a]=cnt++;
}
void tarjan(int u){
    dfn[u]=low[u]=++t;
    skt[++top]=u,in_skt[u]=true;
    for(int i=head[u];i>=0;i=edge[i].next){
        int j=edge[i].to;
        if(dfn[j]==0){
            tarjan(j);
            low[u]=min(low[u],low[j]);
        }else if(in_skt[j]==true){
            low[u]=min(low[u],low[j]);
        }
    }if(dfn[u]==low[u]){
        int y;
        ++scc_cnt;
        do{
            y=skt[top--];
            in_skt[y]=false;
            id[y]=scc_cnt;
        }while(y!=u);
    }
}
int main(){
    memset(head,-1,sizeof(head));
    memset(dfn,0,sizeof(dfn));
    cnt=0,top=0,scc_cnt=0,t=0;
    cin>>n>>m;
    int checkroot[10010];
    for(int i=0;i<m;i++){
        int a,b;
        cin>>a>>b;
        add(a,b);
        checkroot[b]=1;
    }
    for(int i=0;i<n;i++)if(checkroot[i]==0)root=i;
    tarjan(root);
    int q;
    cin>>q;
    for(int i=0;i<q;i++){
        int a,b;
        cin>>a>>b;
        cout<<(id[a]==id[b])<<endl;
    }
    return 0;
}
