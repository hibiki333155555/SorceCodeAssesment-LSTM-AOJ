#include<iostream>
#include<cstdio>
using namespace std;
int main() {
    int dice[6];
    for(int n = 0; n < 6; n++) {
        cin >> dice[n];
    }
    int vector[6][6] = {{0, 2, 4, 1, 3, 0}, {3, 0, 0, 5, 0, 2}, {1, 5, 0, 0, 0, 4},
                        {4, 0, 0, 0, 5, 1}, {2, 0, 5, 0, 0, 3}, {0, 3, 1, 4, 2, 0}};
    int count;
    cin >> count;
    for(int n = 0; n < count; n++) {
        int a, b;
        cin >> a >> b;
        for(int num = 0; num < 6; num++) {
            if(dice[num] == a) {
                a = num;
                break;
            }
        }
        for(int num2 = 0; num2 < 6; num2++) {
            if(dice[num2] == b) {
                b = num2;
                break;
            }
        }
        cout << dice[vector[a][b]] << endl;
    }
}
