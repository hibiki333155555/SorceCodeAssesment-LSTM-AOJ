import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int i = 0;
        
        while(true){
            int x = sc.nextInt();
            
            if (x==0){
                break;
                
            } else{
                System.out.println("Case %: %",i+1,x)
                i++;
            }

    }
}
