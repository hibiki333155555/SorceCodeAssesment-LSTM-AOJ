#include<iostream>
#include<iomanip>
#include<math.h>

using namespace std;

#define PI 3.14159265

int main() {
    float a, b, c, C, S, L, h;
    cin >> a >> b >> C;
    S = 0.5 * a * b * sin(PI*C / 180);
    c = sqrt(a * a + b * b -2 * a * b * cos(PI*C / 180));
    L = a + b + c;
    h = 2 * S / a;
    cout << setprecision(8) << fixed << S << endl
         << setprecision(8) << fixed << L << endl
         << setprecision(8) << fixed << h << endl;
    return 0;
}
