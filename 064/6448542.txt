#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;
	cin >> n;
	vector<int> v;
	for (int i = 0; i < n; i++) {
		int a;
		cin >> a;
		v.push_back(a);
	}

	// calc max, min and sum.
	int max = *max_element(v.begin(), v.end());
	int min = *min_element(v.begin(), v.end());
	int sum = 0;
	for (int i = 0; i < n; i++) {
		sum += v[ i ];
	}
	printf("%d %d %d\n", min, max, sum);

	return (0);
}
