#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n;

  char pat;
  int x;
  bool v[14][4];

    for(int j = 0;j < n;j++){
      cin >> pat;
      if(pat == 'S'){
        cin >> x;
        v[0][x] = true;
      }else if(pat == 'H'){
        cin >> x;
        v[1][x] = true;
      }else if(pat == 'C'){
        cin >> x;
        v[2][x] = true;
      }else if(pat == 'D'){
        cin >> x;
        v[3][x] = true;
      }
    }

    for(int i = 0;i < 4;i++){
      for(int j = 1;j < 14;j++){
        if(i == 0 && v[i][j] == false){
          cout << "S " << j << endl;
        }else if(i == 1 && v[i][j] == false){
          cout << "H " << j << endl;
        }else if(i == 2 && v[i][j] == false){
          cout << "C " << j << endl;
        }else if(i == 3 && v[i][j] == false){
          cout << "D " << j << endl;
        }
      }
    }
  return 0;
}
