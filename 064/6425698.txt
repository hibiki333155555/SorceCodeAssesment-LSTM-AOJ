#include <iostream>
#include <string>
#include <algorithm>
#include <cmath>
using namespace std;
int main( )
{
   int A[10009];
   int n = 0;
   cin >> n;
   for(int i = 0; i < n; ++i) cin >> A[i];
   int maxs = A[0];
   long long sums = 0;
   for(int i = n-1; i > 0; i--){
       for(int j = 0; j < i; j++){
           if(A[j] > A[j+1]){
               maxs = A[j];
               A[j] = A[j+1];
               A[j+1] = maxs;
           }
        }
       
    }
    for (int i = 0; i<n; i++)  sums += A[i]; 
   cout << A[0] << " " << A[n-1] << " " << sums << endl;
    
   
}
