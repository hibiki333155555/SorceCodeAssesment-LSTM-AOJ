#include <iostream>

using namespace std;

int main()
{
    while (true) {

        int N, M;
        cin >> N >> M;

        if (N == 0 && M == 0) {
            break;
        }

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                cout << '#';
            }
            cout << endl;
        }
        cout << endl;
    }
}
