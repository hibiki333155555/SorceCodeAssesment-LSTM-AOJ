main(require("fs").readFileSync("/dev/stdin", "utf8"));

/*-- 計算&出力 --*/
function main(raw) {
    // 入力処理
    const args = raw.split("\n").map(Number);
    //const input = args[0].split(' ').map(Number);
    //console.log(args);
    const a = args[0];
    const b = args[1];
    const c = args[2];
    // 計算
    let count = 0;
    for(let num=a; num<=b; num++){
        if(c%num==0){
            count++;
        };
    } 
    // 表示
    console.log(`${count}`);
}
