#include <iostream>
#include <iomanip>
#include <complex>

using namespace std;

complex<double> in()
{
	double x, y;
	cin >> x >> y;
	return {x, y};
}

int main()
{
	complex<double> p0 = in(), p1 = in(), p2;
	p1 -= p0;
	p1 /= abs(p1);
	int q;
	for (cin >> q; q > 0; --q) {
		p2 = in() - p0;
		auto op = p1.real() * p2.imag() - p1.imag() * p2.real();
		auto ip = p1.real() * p2.real() + p1.imag() * p2.imag();
		if (op > 0) {
			cout << "COUNTER_CLOCKWISE" << endl;
		} else if (op < 0) {
			cout << "CLOCKWISE" << endl;
		} else if (ip < 0) {
			cout << "ONLINE_BACK" << endl;
		} else if (abs(p1) <= abs(p2)) {
			cout << "ONLINE_FRONT" << endl;
		} else {
			cout << "ON_SEGMENT" << endl;
		}
	}
	return 0;
}
