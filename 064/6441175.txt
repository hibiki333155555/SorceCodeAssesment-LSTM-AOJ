#include <stdio.h>
int value[100],weight[100];
int N,W;

int max(int a,int b){
    if(a>b){
        return a;
    }else{
        return b;
    }
}

int calcMaxValue(int index,int W){
    if(index>N){
        return 0;
    }
    if (W<weight[index]){
        return calcMaxValue(index+1,W);
    }else{
        return max(calcMaxValue(index+1,W),calcMaxValue(index+1,W-weight[index])+value[index]);
    }
}

int main(void){
    // Your code here!
    scanf("%d %d\n",&N,&W);
    int i,j,maxValue;
    for(i=0;i<N;i++){
        scanf("%d %d\n",&value[i],&weight[i]);
    }
    
    maxValue=calcMaxValue(0,W);
    printf("%d\n",maxValue);
}
