#include <iostream>
#include <cstdio> // printf
#include <stack> // stack
using namespace std;
int main() {
    stack<char> s;
    char a;
    cin >> a;
    s.push(a);
    while(s.top() != EOF){
        int  b = 0;
        if(s.top() == '+'){
            b += (int)s.top();
            s.pop();
            b += (int)s.top();
            s.pop();
            s.push(b);
        }
        else if(s.top() == '-'){
            b -= (int)s.top();
            s.pop();
            b += (int)s.top();
            s.pop();
            s.push((char)b);
        }
        else if(s.top() == '*'){
            b += (int)s.top();
            s.pop();
            b *= (int)s.top();
            s.pop();
            s.push((char)b);
        }
        cin >> a;
        s.push(a);
    }
    cout << s.top() << endl;
}
