#include <stdio.h>
int main(){
	int a,i,j,k,n;
	int cards[4][14];
	char m;

	for(i=0;i<4;i++){
		for (j=1;j<14;j++){
			cards[i][j]=0;
		}
	}
	
	scanf("%d",&a);

	for(k=0;k<a;k++){
		scanf("%c",&m);
		scanf("%c %d",&m,&n);
		if(m=='S'){
			cards[0][n]=1;
		}else if(m=='H'){
			cards[1][n]=1;
		}else if(m=='C'){
			cards[2][n]=1;
		}else{
			cards[3][n]=1;
		}
	}

	for(i=0;i<4;i++){
		for(j=1;j<14;j++){
			if(cards[i][j]==0){
				switch(i){
					case 0:printf("S %d\n",j);
					break;
					case 1:printf("H %d\n",j);
					break;
					case 2:printf("C %d\n",j);
					break;
					case 3:printf("D %d\n",j);
					break;
				}
			}
		}
	}

	return 0;
}
