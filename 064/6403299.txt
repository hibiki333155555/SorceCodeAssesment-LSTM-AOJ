#include <iostream>

using namespace std;

int main()
{
	int n;
	double p;

	cin >> n;

	p = double(n);

	for (int i = 2; i * i <= n; i++) {
		
		if (n % i != 0) continue;
		
		while (!(n % i)) n /= i;

		p *= (1.0 - (1.0 / double(i)));
	}

	if (n > 1) p *= (1.0 - (1.0 / n));

	cout << p << endl;
}
