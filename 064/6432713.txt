#include<iostream>
using namespace std;

bool checkSort(int arr[]){
  for(int i=0;i<arr.length-1;i++){
    if(arr[i]>arr[i+1])
      return false;
  return true;
}

int main(){
  int a[300];
  int n;
  cin>>n;
  for(int i=0;i<n;i++){
    cin>>a[i];
  }
  int q;
  cin>>q;
  for(int i=0;i<q;i++){
    int query1,query2;
    cin>>query1>>query2;
    int tmp=a[query1-1];
    a[query1-1]=a[query2-1];
    a[query2-1]=tmp;
    if(checkSort(a)){
      cout<<i+1<<endl;
      return 0;
    }
  }
  cout<<-1<<endl;
  return 0;
}
