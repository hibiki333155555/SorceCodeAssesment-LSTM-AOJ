#include <iostream>
#include <vector>

using namespace std;

int main(){
    int n, seq;
    cin>>n>>seq;
    
    vector<vector<int>> v(3, vector<int>(0));
    int op, t,x;
    for(int i = 0; i < seq; i++){
        cin>>op;
        switch(op){
        case 0:
            cin>>t>>x;
            v.at(t).push_back(x);
            break;
        case 1:
            cin>>t;
            for(auto& e : v.at(t)){
                cout << e << endl;
            }
            break;
        case 2:
            cin>>t;
            v.at(t).clear();
            break;
        }
    }
    
    return 0;
}
