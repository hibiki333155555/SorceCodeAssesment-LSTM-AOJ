# include <bits/stdc++.h>
using namespace std;

int main() {
    while (true) {
        int n, m;
        cin >> n >> m;
        if (n == 0 && m == 0) break;
        vector<string> s(n), t(m);
        for (int i = 0; i < n; ++i) {
            cin >> s[i];
        }
        for (int j = 0; j < m; ++j) {
            cin >> t[j];
        }
        cout << "Yes\n";
    }
    return 0;
}
