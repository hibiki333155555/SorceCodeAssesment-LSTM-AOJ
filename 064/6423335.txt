import java.util.Scanner;
import java.util.Formatter;

class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int a= sc.nextInt();
        int b= sc.nextInt();
        int d=a/b;
        int r=a%b;
        double f=1.0*a/b;
        System.out.println(String.format("%.8f",d+" "+"%.8f",r+" "+"%.8f",f));
    }
}
