que = []
operator = {'+', '-', '*'}
for s in input().split():
    if s in operator:
        b = que.pop()
        a = que.pop()
        if s == '+':
            c = a + b
        elif s == '-':
            c = a - b
        else:
            c = a*b
        que.append(c)
    else:
        que.append(int(s))
        
print(que[0])
