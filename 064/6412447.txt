#include <bits/stdc++.h>
using namespace std;

int main() {
  vector<int> r(3);
  for (int i = 0; i < 3; i++) {
    int g;
    cin >> g;
    r.at(i) = g;
  }
  sort(r.begin(), r.end());
  
  for (int i = 0; i < 3; i++) {
    cout << r.at(i);
    if (i < 2) cout << " ";
    else cout << endl;
  }

}
