int main() {
    cin.tie(0);ios::sync_with_stdio(false);
    int64 N,M;
    cin >> N >> M;
    vector<vector<pair<int64,int64>>> edge(M);
    for(int i=0;i<N;++i) {
        int64 A,B,C; cin >> A >> B >> C;
        B--,C--;
        edge[B].push_back({C,A});
    }
    int Q; cin >> Q;
    vector<int64> power(M,0LL);
    while(Q--) {
        int64 D,E; cin >> D >> E;
        E--;
        power[E]+=D;
    }
    multiset<pair<int64,int64>> mst;
    int64 sum = 0;
    int64 ans = 0;

    for(int i=0;i<M;++i) {
        for(auto& p:edge[i]) {
            mst.insert(p);
            sum += p.second;
        }
        if(sum < power[i]) {
            corner(mst.empty(), -1);
            int64 diff = power[i] - sum;
            auto p = *mst.rbegin();
            mst.erase(mst.find(p));

            p.second += diff;
            sum += diff;
            ans += diff;

            mst.insert(p);
        }

        while (mst.size() && (*mst.begin()).first <= i) {
            auto p = *mst.begin();
            mst.erase(mst.find(p));
            sum -= p.second;
        }        
    }
    cout << ans << endl;
    return 0;
}
