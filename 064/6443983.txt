int main()
{	
	
	double ang,a,b,C,c,S,L,h;
	
	cin>>a>>b>>C;
	
	ang=C*M_PI/180;
	S=0.5*a*b*sin(ang);
	c=sqrt(a*a+b*b-2*a*b*cos(ang));
	L=a+b+c;
	h=2*S/a;
	
	printf("%.6f\n",S);
	printf("%.6f\n",L);
	printf("%.6f\n",h);
	
	
return 0;
}
