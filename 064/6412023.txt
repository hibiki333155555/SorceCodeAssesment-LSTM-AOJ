#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;

int main(){
    int n;
    while(1){
        cin >> n;
        if(n == 0) break;
        int x[1000], sum = 0;
        double a = 0, b;
        for(int i = 0; i < n; i++){
            cin >> x[i];
            sum += x[i];
        }
        double ave = double(sum) / n;
        for(int i = 0; i < n; i++){
            a += pow(x[i] - ave, 2);
        } 
        b = sqrt(a / n);
        printf("%lf\n", b);
    }
    
    return 0;
}
