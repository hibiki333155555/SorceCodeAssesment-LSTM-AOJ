#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    cout<<fixed<<setprecision(6);
    while(1) {
        int n; cin>>n;
        if(n==0) break;
        vector<int> s(n);
        double mu=0.0,dev=0.0;
        for(int i=0;i<n;i++) {
            cin>>s[i];
            mu+=s[i];
        }
        mu/=n;
        for(int i=0;i<n;i++) {
            dev+=pow(s[i]-mu,2);
        }
        dev/=n;
        dev=sqrt(dev);
        cout<<dev<<endl;
    }
    return 0;
}
