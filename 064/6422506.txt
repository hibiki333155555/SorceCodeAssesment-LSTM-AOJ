let [, ...inputArr] = require(`fs`).readFileSync(`/dev/stdin`, `utf8`).split(`\n`);

function isPrime(x) {
    if (x == 2) return true;
    for (let i=2; i<Math.floor(Math.sqrt(x)) + 1; i++) {
        if (x % i === 0) return false;
    }
    return true;
}

console.log(inputArr.map(isPrime).reduce((sum, curr) => sum + curr, -1));
