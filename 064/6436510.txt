
class Cpu
  def initialize(name, time)
    @name = name
    @time = time
  end

  def process(q)
    if @time >= q
      @time -= q
      [0, q]
    else
      @time = 0
      [q - @time, @time]
    end
  end

  def finish?
    @time == 0
  end

  def print(total_time)
    puts "#{@name} #{total_time}"
  end
end

def resolve(arr, q)
  arr = arr.map { |val| Cpu.new(val[0], val[1]) }
  index = 0
  total_time = 0
  current_q = q
  loop do
    break if arr.all?(&:finish?)

    if current_q == 0
      current_q = q
      total_time += q
    end

    index %= arr.size
    cpu = arr[index]
    current_q, used_q = cpu.process(current_q)

    total_time += used_q

    if cpu.finish?
      cpu.print(total_time + )
    end
    index += 1
  end
end

size, q = gets.split(' ')

arr = []
size.times do
  arr << gets.split(' ')
end

resolve(arr, q)
