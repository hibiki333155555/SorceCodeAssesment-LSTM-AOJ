#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;
const int INF = 1e9;
const ll LINF = 1e18;
const ld pi = 3.141592653589793238;
int main(){
    int N,M,m;cin >> N ;
    vector<int> A(N);
    int sum=0;
    for (int i = 0; i < N; i++)
    {
        if(i==1){
            M=A[i];
            m=A[i];
        }
        cin >> A[i] ;
        sum+=A[i];
        if(A[i]>M)M=A[i];
        if(A[i]<m)m=A[i];
    }
    sort(A.begin(),A.end());
    cout << m << ' ' << M << ' ' << sum << endl;
}
