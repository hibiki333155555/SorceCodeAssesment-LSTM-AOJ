#include <iostream>
using namespace std;

int main()
{
   int x, l, r;
   cin >> x >> l >> r;

   int minvalue = INT32_MAX;
   int ans=0;
   for (int i = l; i <= r; i++)
   {
      if(abs(i-x)<minvalue)
      {
         minvalue=abs(i-x);
         ans=i;
      }
   }
   cout << ans << endl;
   return 0;
}
