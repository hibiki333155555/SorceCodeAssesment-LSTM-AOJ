#include<stdio.h>

int main(){
    int H=1,W=1;
    int j;
    while(1){
    scanf("%d %d", &H, &W);
    if(H==0 && W==0)
        break;
    for(int i=0;i<H;i++){
        for(j=0;j<W-1;j++)
            printf("# ");
        for(j;j<W;j++)
            printf("#");
        printf("\n");
    }
    }
    return 0;
}
