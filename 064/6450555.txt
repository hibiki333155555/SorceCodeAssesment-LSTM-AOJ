#include<bits/stdc++.h>
#include <vector> 
using namespace std;

void printingA(vector<int> A, int n) {
    for (int k=0; k<n ; k++){
        if (k==n-1) {
            cout << A.at(k) << endl; 
        }
        else {
            cout << A.at(k) << " "; 
        }
    }
}

void printingCount(int count) {
    cout << count << endl;
}


void bubbleSort(vector<int>& A, int n, int& count) { 
    for (int i=0; i<n; i++) {
        for (int j=n-1; j>i; j--) {
            if (A.at(j-1)>A.at(j)) {
                int tmp = A.at(j);
                A.at(j) = A.at(j-1);
                A.at(j-1) =tmp;
                count++;
            }
        }
    } 
}

int main () {
    int n;
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
        cin >> A.at(i);
    }
    int count = 0;
    bubbleSort(A, n, count);
    printingA(A, n);
    printingCount(count);
    return 0;
}
