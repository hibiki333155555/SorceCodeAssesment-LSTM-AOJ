#define EPS (1e-10)

ll llMax(ll x, ll y) { return (x >= y) ? x : y; }
ll llMin(ll x, ll y) { return (x <= y) ? x : y; }

int solve(int n, int x) {
  int ret = 0;
  for (int i = 1; i <= n - 2; i++) {
    for (int j = i + 1; j <= n - 1; j++) {
      for (int k = j + 1; k <= n; k++) {
        if (i + j + k == x) ret++;
      }
    }
  }
  return ret;
}

  while (true) {

    if (n == 0 && x == 0) break;

    /* output */

  }

}
