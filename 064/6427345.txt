#include<bits/stdc++.h>
using namespace std;
int main(){
    int n; cin>>n;
    int k[n],p[n];
    for(int i=0;i<n;i++){
        cin>>k[i]>>p[i];
    }
    for(int i=0;i<n;i++){
        if(k[i]%p[i]==0){
            cout<<p[i]<<endl;
        }else if(k[i]>p[i]){
            cout<<k[i]%p[i]<<endl;
        }else{
            cout<<k[i]<<endl;
        }
    }
    return 0;
}
