void solve()
{
    int a0, b0, c0, a, b, c;
    scanf("%d%d%d%d%d%d", &a0, &b0, &c0, &a, &b, &c);
    if ((a0 + b0 + c0) % 2) {
        a0--; b0--; c0++;
    }
    if ((a + b + c) % 2) {
        a--; b--; c++;
    }
    a -= a0;
    b -= b0;
    c -= c0;
    if (a % 2 == 0 && b % 2 == 0 && c % 2 == 0 && a + b + c == 0) {
        printf("Yes\n");
    }
    else {
        printf("No\n");
    }

    return;
}
