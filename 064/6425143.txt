#include<bits/stdc++.h>
using namespace std;
int main(){
    int e[4]; cin>>e[0]>>e[1]>>e[2]>>e[3];
    if(e[0]==e[1]&&e[2]==e[3]){
        cout<<"yes"<<endl;
    }else if(e[0]==e[2]&&e[1]==e[3]){
        cout<<"yes"<<endl;
    }else if(e[0]==e[3]&&e[1]==e[2]){
        cout<<"yes"<<endl;
    }else{
        cout<<"no"<<endl;
    }
    return 0;
}
