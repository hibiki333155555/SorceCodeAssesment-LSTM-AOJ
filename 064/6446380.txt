#include<bits/stdc++.h>
#define N 1401
using namespace std;
int a[N][N],f[N][N];
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            cin>>a[i][j];
        }
    }
    int kp=0;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
                if(a[i][j]==1) f[i][j]=0;
                else f[i][j]=min(f[i-1][j],min(f[i][j-1],f[i-1][j-1]))+1;
            kp=max(kp,f[i][j]);
        }
    }
    cout<<kp*kp<<'\n';
    return 0;
}
