void _out(vll& v){

ll _selection_sort(vll& v){
    ll ret = 0;
    mcfor(i, 0, v.size()){
        auto minj = i;
        mcfor(j, i, v.size()){
            if(v.at(j) < v.at(minj)){
                minj = j;
            }
        }
        if(i != minj){
            swap(v.at(i), v.at(minj));
            ret++;
        }
    }
    return ret;
}
