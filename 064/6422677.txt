#include <iostream>
#include <vector>

using namespace std;

void insertion_sort(int a[],int n){
    int i,j,tmp;
    for(i=1;i<n;i++){
        tmp = a[i];
        j = i - 1;
        while(j >= 0 && a[j] > tmp){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = tmp;
        for(i=0;i<n;i++){
            cout << a[i] << " ";
        }
        cout << endl;
    }
}
int main(){
    int i,k,n;
    cin >> n;
    vector<int> a(n);
    for(i=0;i<n;i++){
        cin >> k;
        a[i] = k;
    }
    insertion_sort(a[]],n);
    return 0;
}
