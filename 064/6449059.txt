int main(){

	cin>>v>>e>>r;
	 vector<vector<Pll> >vec(v+1);
	 rep(i,e){
		 ll s,t,d;

		 vec[s].push_back(make_pair(t,d));
	 }
	 priority_queue<Pll,vector<Pll>,greater<Pll> >que;
	 que.push(make_pair(0,r));
	 vector<ll>f(v,INF);
	 while(!que.empty()){
		 ll t=que.top().first;
		 ll u=que.top().second;
		 que.pop();
		 if(f[u]!=INF)continue;
		 f[u]=t;
		 rep(i,vec[u].size()){
			 que.push(make_pair(vec[u][i].second+t,vec[u][i].first));
		 }
	 }
	 rep(i,v){
		 if(f[i]==INF)cout<<"INF"<<endl;
		 else cout<<f[i]<<endl;
	 }
