template <class T>
void vdeb(const std::vector<T> &da) {
    auto n = da.size();
    for (size_t i = 0; i < n; i++) {
        if (i == n - 1)
            std::cout << da[i];
        else
            std::cout << da[i] << " ";
    }
    std::cout << std::endl;
}
template<class T>
void vdeb(const std::vector<std::vector<T>> &da) {
    auto n = da.size();
    for (size_t i = 0; i < n; i++) {
        std::cout << i << " : ";
        vdeb(da[i]);
    }
    std::cout << std::endl;
}

using namespace std;

const int  By = 200, Sy = 195;
const int ald = By * 333 + Sy * (999 - 333) + 1;
int solve(int y, int m, int d) {
    int g = (y-1) / 3, cnt = 0;
    cnt = g * By + (y - 1 - g) * Sy;
    if(y%3 == 0) {
        cnt += (m-1) * 20;
    } else {
        int h = (m-1) / 2;
        cnt += 19 * h + (m-h-1) * 20;
    }
    cnt += d;
    return ald - cnt;
}
