#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9;
const ll inf = 1LL<<62;

void solve() {
  ll a, b; cin >> a >> b;
  string A = "", B = "";
  while (a > 0) {
    if (a & 1) A += '1';
    else A += '0';
    a /= 2;
  }
  while (b > 0) {
    if (b & 1) B += '1';
    else B += '0';
    b /= 2;
  }
  if (A.size() == 0) A += '0';
  if (B.size() == 0) B += '0';
  while (A.size() < B.size()) A += '0';
  while (A.size() > B.size()) B += '0';
  int ans = 0;
  for (int i=A.size()-1; i>=0; i--) {
    if (A[i] != B[i]) {
      cout << ans << '\n';
      return;
    }
    ans++;
  }
  cout << ans << '\n';
}
