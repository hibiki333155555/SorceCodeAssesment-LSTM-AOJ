// DPL_1_A Coin Changing Problem
/*

Find the minimum number of coins to make change for n cents using coins of denominations d1, d2,.., dm.
The coins can be used any number of times.
*/
#include<stdio.h>
#include <iostream>
//#include <algorithm>
using namespace std;
#if 0
#define LOG(format,...) printf("%d->%s: "format"\n",__LINE__,__func__,##__VA_ARGS__)
#else
#define LOG
#endif


#define MAX_N 50005
#define MAX_M 25

#define min(a,b) a>b?b:a

int N,M,D[MAX_M];
int DP[MAX_M];
int main()
{
    int i,j,k;
    scanf("%d%d",&N,&M);
    for(i = 1; i<=M; i++)
    {
        scanf("%d",&D[i]);
    }

    for(j = 1; j<=N; j++)
    {
        DP[j]=j;
    }
    
    for(i = 2; i<=M; i++)
    {
        for(j = D[i]; j<=N; j++)
        {
            DP[j]=min(DP[j-D[i]]+1,DP[j]);
        }
    }

//    for(i=0; i<=N; i++)
//    {
//        LOG("%d ",DP[i]);
//    }
//    printf("%d\n",DP[N]);
    cout<<DP[N]<<'\n';
    return 0;
}
