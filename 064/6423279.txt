#include<iostream>
using namespace std;

int select_sort(int a[],int n){
    int i,j,minj;
    int count = 0;
    
    for(i=0;i<n;i++){
        minj = i;
        for(j=i;j<n;j++){
            if(a[j] < a[minj]){
                minj = j;
                count++;
            }
            swap(a[i],a[minj]);
        }
    }
    return count;
}

int main(){
    int n,i;
    int count = 0;
    cin >> n;
    int a[n];
    for(i=0;i<n;i++){
        cin >> a[i];
    }
    
    count = select_sort(a,n);
    for(i=0;i<n;i++){
        if(i!=0) cout << " ";
    cout << a[i];
    }
    
    cout << endl;
    cout << count << endl;
}
