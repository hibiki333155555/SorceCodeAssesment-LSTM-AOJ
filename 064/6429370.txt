#define MAX 1000000007
using namespace std;
using ll = long long;
template<class T>
bool chmin(T &a,const T & b){
    if(b<a){
        a=b;
        return 1;
    }
    return 0;
}

template<class T>
bool chmax(T &a,const T &b){
    if(a<b){
        a=b;
        return 1;
    }
    return 0;
}

vector<int> N;
int memo[100005];
bool ng[100005];
int func(int i){
    int cnt=0;
    if(memo[i]!=0){
        return memo[i];
    }
    if(i==0){
        cnt=1;
    }else{
        if(!ng[i-1]){
            cnt+=func(i-1);
            cnt%=MAX;
        }
        if(!ng[i-2] and i>1){
            cnt+=func(i-2);
            cnt%=MAX;
        }
    }
    return memo[i]=cnt;
}

ll dp[35];

int main(void){
    
    while(1){
        int n;
        cin>>n;
        if(n==0)break;
        N.push_back(n);

    }

    size_t size=N.size();
    
    rep(j,size){
        rep(i,35)dp[i]=0;
        dp[0]=1;

        rep(i,N[j]){
            dp[i+1]+=dp[i];
            dp[i+2]+=dp[i];
            dp[i+3]+=dp[i];
        }

        ll ans=dp[N[j]]/10;
        ans/=365;
        if(ans%365!=0){
            ans++;
        }
        cout<<ans<<endl;
    }
    
    return 0;
}
