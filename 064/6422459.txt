#include <iostream>
#include <vector>

using namespace std;
int nx[200010],pre[200010];
int que[200010],l[200010],r[200010];
int imo[200010] = {};
int main(){
    string s; cin >> s;
    int i;
    vector<pair<int,int>> v;
    for(i=0;i<s.size();i++){
        if(s[i]==')'){
            if(s[i - 1]>='a' && s[i - 1]<='z') pre[i] = i - 1;
            else pre[i] = pre[i - 1];
        }
    }
    for(i=s.size() - 1;i>=0;i--){
        if(s[i]=='('){
            if(s[i + 1]>='a' && s[i + 1]<='z') nx[i] = i + 1;
            else nx[i] = nx[i + 1];
        }
    }
    int now = 0;
    for(i=0;i<s.size();i++){
        if(s[i]>='0' && s[i]<='9'){
            v.push_back({s[i] - '0',i + 1});
        }
        if(s[i]==')'){
            auto p = v.back();
            que[now] = p.first; l[now] = p.second; r[now] = i;
            now++;
            v.pop_back();
        }
    }
    for(i=0;i<now;i++){
        (imo[l[i]] += que[i]) %= 26; (imo[r[i] + 1] += 26 - que[i]) %= 26;
    }
    for(i=1;i<=s.size();i++) (imo[i] += imo[i - 1]) %= 26;
    string ans;
    for(i=0;i<s.size();i++){
        if(s[i]>='a' && s[i]<='z'){
            int pos = (s[i] - 'a' + imo[i])%26;
            char c = (char)('a' + pos);
            ans.push_back(c);
        }
    }
    cout << ans << endl;
}
