#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
#include <string>
using namespace std;



int main()
{
    int n, tmp;
    cin >> n;
    int pnt1 = 0, pnt2 = 0;

    string str1, str2;
    while(n--!=0)
    {
        cin >> str1 >> str2;
        tmp = str1.compare(str2);
        if(tmp == 0)
        {
            pnt1++;pnt2++;
        }
        else if(tmp > 0)
            pnt1+=3;
        else
            pnt2+=3;
    }

    cout << pnt1 << " " << pnt2 << endl;

    return 0;
}
