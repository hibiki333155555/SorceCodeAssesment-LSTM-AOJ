#include <iostream>
#include <vector>

using namespace std;

std::vector<int> InputNNumbers(int N)
{
    std::vector<int> A;
    for (int i = 0; i < N; i++)
    {
        int number;
        std::cin >> number;
        A.push_back(number);
    }
    return A;
}

void ShowNNumbers(std::vector<int> A)
{
    int N = A.size();
    for (int i = 0; i < N; i++)
    {
        std::cout << A[i];
        if (i < N - 1)
        {
            std::cout << ' ';
        }
    }
    std::cout << std::endl;
}

void ChangePlace(int &a, int &b)
{
    int tmp;
    tmp = b;
    b = a;
    a = tmp;
}

int BubbleSort(std::vector<int> &A)
{
    int N = A.size();
    int change_count = 0;
    bool reverse_flag = true;
    while (reverse_flag)
    {
        reverse_flag = false;
        for (int i = N - 1; i >= 1; i--)
        {
            if (A[i] < A[i - 1])
            {
                ChangePlace(A[i - 1], A[i]);
                change_count++;
                reverse_flag = true;
            }
        }
    }
    return change_count;
}

int main(void)
{
    int N;
    std::cin >> N;
    std::vector<int> A = InputNNumbers(N);
    int change_count = BubbleSort(A);
    ShowNNumbers(A);
    std::cout << change_count << std::endl;
    return 0;
}
