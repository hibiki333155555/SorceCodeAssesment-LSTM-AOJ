#include <iostream>
#include <string>

using namespace std;
int main()
{
    string a1;
    string a2;
    int n = 0;
    cin >> n;
    int x = 0;
    int y = 0;
    for(int i = 0;i < n; i++)
    {
        cin >> a1 >> a2;
        if(a1 > a2)
        {
            x += 3;
        }
        else if(a1 < a2)
        {
            y += 3;
        }
        else if(a1 == a2)
        {
            x += 1;
            y += 1;
        }
    }
    cout << x << " " << y << endl;
    return 0;
}
