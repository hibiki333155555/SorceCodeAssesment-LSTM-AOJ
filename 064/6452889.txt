#include <iostream>
#include <vector>
using namespace std;

int main() {
    int r,c; cin>>r>>c;
    vector<vector<int>> table(r+1,vector<int>(c+1,0));
    vector<int> row(r,0), column(c,0);
    for(int i=0;i<r;i++) {
        for(int j=0;j<c;j++) {
            cin>>table[i][j];
            row[i] += table[i][j];
            column[j] += table[i][j];
        }
    }
    int sum = 0;
    for(int i=0;i<r;i++) table[i][c] = row[i];
    for(int i=0;i<c;i++) {
        table[r][i] = column[i];
        sum += column[i];
    }
    table[r][c] = sum;
    for(int i=0;i<r+1;i++) {
        for(int j=0;j<c;j++) {
            cout<<table[i][j]<<' ';
        }
        cout<<table[i][c]<<endl;
    }
    return 0;
}
