#include <iostream>
using namespace std;

int main()
{
    int total, average;
    int income[10000];
    
    int N;
    while ( 1 )
    {
        cin >> N;
        if ( !N ) break;
        
        total = 0;
        for ( int i = 0; i < N; i++ )
        {
            cin >> income[i];
            total += income[i];
        }
        
        average = total / N;
        
        int count = 0;
        for ( int i = 0; i < N; i++ )
        {
            if ( income[i] <= average ) count++;
        }
        
        cout << count << endl;
    }
    
    return 0;
}
