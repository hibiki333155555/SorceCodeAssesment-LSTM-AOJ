// D - Standard Deviation

#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std; 

int x[1001];

int main() {
    int n;
    while (cin>>n) {
        if (n==0) break;
        int sum =0;
        for (int i=0; i<n; i++) {
            cin >> x[i];
            sum +=x[i];
        }
        double m = (double)sum/n;
        double sum2=0;

        for (int i=0; i<n; i++) {
            sum2 += (x[i]-m)*(x[i]-m);
        }
        cout << setprecision(8) << fixed << sqrt(sum2/n) << endl;
    }
}
