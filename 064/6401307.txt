#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int n, m;
vector<vector<int> > a;
vector<int> b;

int main(){
  cin >> n >> m;
  for (int i = 0; i < n; i++){
    vector<int> tmp;
    for (int j = 0; j < m; j++){
      int input;
      cin >> input;
      tmp.push_back(input);
    }
    a.push_back(tmp);
  }
  for (int i = 0; i < m ; i++){
    int input;
    cin >> input;
    b.push_back(input);
  }
  for (int i = 0; i < n; i++){
    int result = 0;
    for (int j = 0; j < m; j++){
      result += a[i][j] * b[j];
    }
    cout << result << endl;
  }
}
