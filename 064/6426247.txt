#include<stdio.h>

int main(){
    int n, tem;
    int max = 0, min = 0;
    double sum;
    
    scanf("%d",&n);
    
    for(int i = 0; i < n; i++){
        scanf("%d",&tem);
        sum += tem;
        if(max < tem)max = tem;
        if(min > tem)min = tem;
    }
    printf("%d %d %lf\n", min, max, sum);
    return 0;
}
