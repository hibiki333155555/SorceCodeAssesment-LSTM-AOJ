#include <stdio.h>
int swap(int, int);
int main(void)
{
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);
    int tmp;
    if (a > b) {
        tmp = b;
        b = a;
        a = tmp;
    }
    if (b > c) {
        tmp = c;
        c = b;
        b = tmp;
    }
    if (a > b) {
        tmp = b;
        b = a;
        a = tmp;
    }
    printf("%d %d %d\n", a, b, c);

    return 0;
}
