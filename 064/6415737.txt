#include<iostream>
#include<string>
using namespace std;

int main(){

    string str,ans;
    int num,k;

    while(1){
        cin>>str;
        if(str == "-")break;
        cin>>num;
        for(int i=0,k;i<num;++i){
            cin>>k;
            ans=str.substr(k)+str.substr(0,k);
        }
        cout<<ans<<endl;
    }

    return 0;
}
