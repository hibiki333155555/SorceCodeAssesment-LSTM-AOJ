int parse(string &t) {
    int hour = (t[0] - '0') * 10 + (t[1] - '0');
    int min = (t[2] - '0') * 10 + (t[3] - '0');
    int second = (t[4] - '0') * 10 + (t[5] - '0');

    return 3600 * (hour) + 60 * min + second;
}
