#include<bits/stdc++.h>
using namespace std;
int main(){
    int n, ans = 0,cnt = 0;
    cin >> n;
    vector<int> v(n);
    for(int i = 0; i < n; i++ ) {
        cin >> v[i];
    }
    for (int i = n - 1; i >= 0; i--) {
        ans = max(ans, v[i]);
        if(ans <= v[i]) {
            cnt++;
        }
    }
    cout<<cnt<<endl;
    return(0);
}
