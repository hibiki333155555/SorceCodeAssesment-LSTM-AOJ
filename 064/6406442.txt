#include <stdio.h>
#include <stdio.h>

int main(void){
    int r,c,i,j,sum,a;
    int hyou[101][101];
    
    scanf("%d %d",&r,&c);
    
	if(r==100 || c==100){
		for(i=1;i<=r;i++){
        for(j=1;j<=c;j++){
             scanf("%d",&hyou[i][j]);
        }
    }
	}
	else{
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
             scanf("%d",&hyou[i][j]);
        }
    }
	}
	if(r==100 || c==100){
		a=1;
	}
	else{a=0;}
    for(i=a;i<=r;i++){
        sum=0;
        for(j=a;j<=c;j++){
            sum=sum+hyou[i][j];
        }
        hyou[i][c]=sum;
    }
    
    for(i=a;i<=c;i++){
        sum=0;
        for(j=a;j<=r;j++){
            sum=sum+hyou[j][i];
        }
        hyou[r][i]=sum;
    }
    for(i=a;i<=r;i++){
        for(j=a;j<=c;j++){
            if(j==c){
                printf("%d",hyou[i][j]);
            }
            else{
                printf("%d ",hyou[i][j]);
            }
        }
        printf("\n");
    }
}
