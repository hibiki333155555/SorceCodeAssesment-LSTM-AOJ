#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n;
	cin>>n;
	vector<int> v(n);
	for(int i=0; i<n; i++){
		cin>>v[i];
	
	}
	
	for(int i=0; i<v.size(); i++){
		int key=v[i];
		int j=i-1;
		while(j>=0 && v[j]>key){
			v[j+1] = v[j];
            j--;
		}
		v[j+1]=key;
		for(int i=0; i<v.size(); i++){
			cout<<v[i];
			if(i<n-1){
				cout<<" ";
			}
		}
		cout<<"\n";
	}
	return 0;
}
