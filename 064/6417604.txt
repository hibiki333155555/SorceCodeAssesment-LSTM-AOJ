#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)

int main(){
  int n,k;
  cin>>n>>k;
  vector<int> v;
  int sum=0;
  rep(i,n){
    int a;cin>>a;
    sum+=a;
    if(a==1) v.push_back(1),v.push_back(4);
    else v.push_back(6-a);
  }
  sort(begin(v),end(v));
  reverse(begin(v),end(v));
  rep(i,min(k,(int)v.size())) sum+=v[i];
  cout<<sum<<endl;
}
