#!/bin/python

from collections import Counter

n = int(input())
report = [input() for _ in range(n)]
mod = [998244353, 1000000007]
m = len(report[0])
pow26 = []
for i in range(2):
    pow26.append([])
    for j in range(m+1):
        pow26[i].append(1)
#pow26 = [[1] * (m+1) for i in range(2)]
for i in range(2):
    for j in range(m):
        pow26[i][j+1] = pow26[i][j] * 26 % mod[i]
#hsh = [[[0] * (m+1) for j in range(n)] for i in range(2)]
hsh = []
for i in range(2):
    hsh.append([])
    for j in range(n):
        hsh[i].append([])
        for k in range(m+1):
            hsh[i][j].append(0)
for i in range(2):
    for j in range(n):
        for k in range(m):
            hsh[i][j][k+1] = (hsh[i][j][k] + pow26[i][k] * (ord(report[j][k]) - ord('a'))) % mod[i]
#hsh_cnt = [Counter()] * m
hsh_cnt = []
for i in range(m):
    hsh_cnt.append(Counter())
for i in range(n):
    for j in range(m):
        hsh0 = hsh[0][i][j] * mod[0] + (hsh[0][i][m] - hsh[0][i][j+1] + mod[0]) % mod[0]
        hsh1 = hsh[1][i][j] * mod[1] + (hsh[1][i][m] - hsh[1][i][j+1] + mod[1]) % mod[1]
        hsh_cnt[j][(hsh0, hsh1)] += 1
#cheating = [0] * n
cheating = []
for i in range(n):
    cheating.append(0)
for i in range(n):
    for j in range(m):
        hsh0 = hsh[0][i][j] * mod[0] + (hsh[0][i][m] - hsh[0][i][j+1] + mod[0]) % mod[0]
        hsh1 = hsh[1][i][j] * mod[1] + (hsh[1][i][m] - hsh[1][i][j+1] + mod[1]) % mod[1]
        if hsh_cnt[j][(hsh0, hsh1)] >= 2:
            cheating[i] = 1
ans = 0
for i in range(n):
    ans += cheating[i]
print(ans)
