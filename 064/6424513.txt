#include<bits/stdc++.h>
using namespace std;
int S[1000001];

bool binarySearch(int kye, int n) {
    int left, right, mid;
    left = 0;
    right = n;
    while (left < right)
    {
        mid = (left + right) / 2;
        if (S[mid] == kye) {
            return 1;
        } else if (S[mid] < kye) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    return 0;
}


int main(void) {
    
    int n, m, i, kye, ans = 0;
    
    cin >> n;
    for (i = 0; i < n; i++) 
    scanf("%d", &S[i]);

    cin >> m;
    for (i = 0; i < m; i++) {
        scanf("%d", &kye);
        if (binarySearch(kye, n)) ans++;
    }
    
    cout << ans << endl;
    return 0;

}
