#include <iostream>
using namespace std;

int bubblesort(int a[],int n){
    int s = 0;
    bool flag = true;
    for(int i = 0; flag; i++){
        flag = false;
        for(int j = n-1; j>=i+1; j--){
            if(a[j] < a[j-1]){
                swap(a[j],a[j-1]);
                flag = 1;
                s++;
            }
        }
    }
    return s;
}


int main(void){
    int n,sum;
    cin >> n;
    int a[n];
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    sum = bubblesort(a,n);
    for(int i = 0; i < n; i++){
        if(i > 0){
            cout << " ";
        }
        cout << a[i];
    }
    cout << endl;
    cout << sum <<endl;
}
