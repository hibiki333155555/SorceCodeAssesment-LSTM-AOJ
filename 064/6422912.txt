#include <iostream>


int main() {
	int a, b;
	char op = 'a';

	while (op != '?') {

		std::cin >> a >> op >> b;

		if (op == '+')
			std::cout << a + b << std::endl;
		else if (op == '-')
			std::cout << a - b << std::endl;
		else if (op == '*')
			std::cout << a * b << std::endl;
		else if (op == '/')
			std::cout << a / b << std::endl;
	}

	return 0;
}
