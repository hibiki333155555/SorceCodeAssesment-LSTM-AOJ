#include <stdio.h>
#include <string.h>
int main()
{

    int i,j;
    int count[26];

    for(i=0;i<26;i++)
    {
        count[i]=0;
    }

    char s[120],s1[120];
    //scanf("%[^\n]%*c",s);
    fgets(s,1200,stdin);
    

    for(i=0,j=0;i<strlen(s),j<strlen(s);i++,j++)
    {
        s1[j]=tolower(s[i]);
        if(s1[i]>='a' && s1[i]<='z')
            count[s1[i]-'a']++;
    }

    s1[j]='\0';




    for(i=0;i<26;i++)
    {
        printf("%c : %d\n",'a'+i,count[i]);
    }
    //tanvir

    return 0;
}
