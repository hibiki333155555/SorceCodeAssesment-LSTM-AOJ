n = int(input())
X = list(map(int, input().split()))
Y = list(map(int, input().split()))

def dist(ax):
    sum = 0
    for x, y in zip(X, Y):
        sum += abs(x - y) ** ax
    return sum ** (1 / ax)

def dist_max():
    return max([abs(x - y) for x, y in zip(X, Y)])
