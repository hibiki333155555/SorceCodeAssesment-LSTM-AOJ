#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    while(true){
        cin>>s;
        if(s.length() == 1 && s[0] == '0'){
            break;
        }
        int sum = 0;
        for(int i=0; s.length(); i++){
            sum += (s[i] - '0');
        }
        cout << sum <<"\n";
    }
    return(0);
}
