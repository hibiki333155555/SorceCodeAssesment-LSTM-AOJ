int main(){
    ll n;cin >> n;
    vector<string> g(n);
    rep(i,n)cin >> g[i];
    vl ban(n);
    rep(i,n){
        rep(j,n){
            if(i==j)continue;
            ll diff=0;
            rep(t,g[i].size()){
                if(g[i][t]!=g[j][t])diff++;
            }
            if(diff<=1)ban[i]++;
        }
    }
    ll ans=0;
    rep(i,n)if(ban[i])ans++;
    cout << ans << endl;

}
