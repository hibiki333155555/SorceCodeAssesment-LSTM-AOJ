#include <bits/stdc++.h>
using namespace std;

int main(){
    int w,h,x,y,r;
    cin>>w>>h>>x>>y>>r;
    int top,bottom,right,left;
    top=x+r;
    bottom=x-r;
    right=y+r;
    left=y-r;
    if(bottom>=0&&top<=h&&left>=0&&right<=w){
        cout<<"Yes"<<endl;
    }else{
        cout<<"No"<<endl;
    }
}
