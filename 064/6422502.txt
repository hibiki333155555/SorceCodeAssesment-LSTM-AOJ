#include<stdio.h>
#include<stdlib.h>
int binarySearch(int);
int na,nb;
int *a,*b,*c;
int main(){
    //init
    int cnt=0;
    scanf("%d",&na);
    a=(int *)malloc(sizeof(int)*na);
    for(int i=0;i<na;i++)scanf("%d",a+i);
    scanf("%d",&nb);
    b=(int *)malloc(sizeof(int)*nb);
    c=(int *)malloc(sizeof(int)*nb);
    for(int i=0;i<nb;i++){scanf("%d",b+i);c[i]=0;}
    //process
    for(int i=0;i<nb;i++)c[i]=binarySearch(b[i]);
    for(int i=0;i<nb;i++)if(c[i]!=0)cnt++;
    //output
    printf("%d\n",cnt);
    return 0;
}
int binarySearch(int tar){
    int left=0,right=na;
    while(left<right){
        int mid=(left+right)/2;
        if(a[mid]==tar)return 1;
        else if(tar<a[mid])right=mid;
        else left=mid+1;
    }
    return 0;
}
