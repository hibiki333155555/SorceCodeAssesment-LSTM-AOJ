#include <bits/stdc++.h>
#define rep(i,n) for(int i = 0; i < (n); i++)
using namespace std;
typedef long long ll;

int main(){
    cin.tie(0);
    ios::sync_with_stdio(0);
    
    int T; cin >> T;
    rep(_,T) {
        int N,M,v,a,b; cin >> N >> M >> v >> a >> b;
        int ok = 0;
        v %= M; a %= M; b %= M;
        vector<bool> visited(M, false);
        rep(i,N) {
            if(visited[v]) {
                ok = 1; break;
            }
            visited[v] = true;
            v = (a * v % M + b) % M;
        }
        cout << (ok ? "Yes" : "No") << endl;
    }
}
