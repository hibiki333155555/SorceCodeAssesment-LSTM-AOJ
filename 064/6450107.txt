#include <stdio.h>
#include<iostream>
using namespace std;

int main(){
    int n;
    cin >> n;
    long arr[n];
    
    for(int i { 0 }; i < n; ++i){
        cin >> arr[i];
    }
    
    long max=arr[0], min=arr[0], sum=0;
    
    for(int i { 0 }; i < n; ++i){
        if (max<arr[i]) {
            max = arr[i];
        }
        if (min>arr[i]) {
            min = arr[i];
        }
        sum = sum + arr[i];
    }
    
    
    
    cout << min << " " << max << " " << sum << endl;

    return 0;
}
