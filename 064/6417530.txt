#include <iostream>
#include <vector>
using namespace std;

void solve(){
  int n, m, v, a, b;
  cin >> n >> m >> v >> a >> b;
  if(n > m){
    puts("Yes");
    return;
  }
  vector<bool> exist(m);
  for(int i = 0; i < n; ++i){
    if(exist[v]){
      puts("Yes");
      return;
    }
    exist[v] = true;
    v = (v*a+b)%m;
  }
  puts("No");
}

int main(){
  int t;
  cin >> t;
  while(t--) solve();
}
