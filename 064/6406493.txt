using namespace std;

int main(){
    int a,b,c;
    cin >> a >> b >> c;
    if(a=<b){
        if(b=<c) cout << a << " " << b << " " << c << '\n';
        if(b>=c) cout << a << " " << c << " " << b << '\n';
    }
    if(a>=b){
        if(b=<c) cout << a << " " << b << " " << c << '\n';
        if(b>=c) cout << a << " " << c << " " << b << '\n';
    }
    
    return 0;
    
}
