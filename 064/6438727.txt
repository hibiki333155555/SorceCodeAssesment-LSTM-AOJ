#include <iostream>
#include <vector> // vector
#include <utility> // pair, make_pair
using namespace std;


void trace(vector<int> R, int n){
    for(int i = 0;i < n;i++){
        if(i > 0)cout << " ";
        cout << R.at(i);
    }
    cout << endl;
}

void insertsort(vector<int> R, int n){
    for(int i = 1;i < n;i++){
        for(int j = i - 1;j >= 0;j--){
            if(R[j] > R[j+1])swap(R[j], R[j+1]);
        }
        trace(R, n);
    }
}

int main() {
    int  n;

    cin >> n;
    vector<int> R(n);
    for(int i = 0;i < n;i++)cin >> R.at(i);

    trace(R, n);
    insertsort(R,n);


}
