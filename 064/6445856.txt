#include <iostream>
#include <math.h>
#include <string.h>
using namespace std;

#define INF 0x3f3f3f3f

//n代表要付的钱，m代表货币面额的数量
int n, m;
int money[25], dp[50005];

void solve() {

	memset(dp, INF, sizeof(dp));
	//0元0个硬币
	dp[0] = 0;
	for (int i = 1; i <= m; i++) {
		for (int j = 0; j + money[i] <= n; j++) {
			//前者为不用money[i]的情况，这里代表上一次的值
			//后者为用money[i]的情况
			dp[j + money[i]] = min(dp[j + money[i]], dp[j] + 1);
		}
	}

	cout << dp[n] << endl;
}

int main() {
	
	cin >> n >> m;

	for (int i = 1; i <= m; i++) {
		cin >> money[i];
	}
	solve();
}
