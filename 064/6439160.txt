#include <iostream>
#include <vector>
using namespace std;

int main(){
    int n; cin >> n;
    int nn = n;
    vector<int> facs;
    for(int i = 2; i*i <= n; i++){
        if(n%i == 0){
            facs.emplace_back(i);
            while(n%i == 0) n /= i;
        }
    }
    if(n > 1) facs.emplace_back(n);

    int ans = 0;
    const int m = facs.size();
    for(int bit = 1; bit < (1 << m); bit++){
        int fac = 1;
        int cnt = 0;
        for(int i = 0; i < m; i++){
            if((bit >> i) & 1){
                fac *= facs[i];
                cnt++;
            }
        }
        if(cnt%2 == 0) ans -= nn/fac;
        else ans += nn/fac;
    }

    cout << nn-ans << endl;
    return 0;
}
