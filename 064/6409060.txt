#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include<string.h>

int main()
{
    char s[1001];
    while(1){
        scanf("%s",s);
        if(s[0]=='0')
        break;

        int sum=0;
        int n=strlen(s);
        for(int i=0; i<n; i++){
            sum+=s[i]-'0';
        }
        printf("%d\n",sum);

    }
	return 0;
}
