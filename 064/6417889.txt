#include "bits/stdc++.h"
using namespace std;
using ll=long long;

int main(){
    ll n,k; cin>>n>>k;
    vector<int>a(n),b;
    for(int i=0;i<n;i++)cin>>a[i];
    sort(a.begin(),a.end());
    for(int i=0;i<n;i++){
        if(i<k&&a[i]==1)b.push_back(5);
        else if(i<k)b.push_back(6);
        else b.push_back(a[i]);
    }
    sort(b.begin(),b.end());
    if(n<k){
        for(int i=0;i<k-n;i++){
            if(i==n)break;
            if(b[i]==6)break;
            else b[i]=6;
        }
    }
    ll sum=0;
    for(int i=0;i<n;i++){
        sum+=b[i];
    }
    cout<<sum<<endl;
}
