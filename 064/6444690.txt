#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    int N,M;
    cin >> N >> M;
    int A[N],B[M];
    for(int i=0; i<N; i++){
        cin >> A[i];
    }
    for(int i=0; i<M; i++){
        cin >> B[i];
    }
    vector<int> v;
    bool judge=true;
    for(int i=0; i<N; i++){
        for(int j=0; j<M; j++){
            if(A[i]==B[j]){
                for(int k=0; k<v.size(); k++){
                    if(v[k]==A[i]){
                        judge=false;
                    }
                }
                if(judge==true){
                    v.push_back(A[i]);
                }
            }
            judge=true;
        }
    }
    sort(v.begin(), v.end());
    for(int i=0; i<v.size(); i++){
        cout << v[i] << endl;
    }
}
