using System;
using System.Collections.Generic;
using System.Linq;

namespace AOJ_ITP1_10_D
{
    internal class Program
    {
        public static void Main()
        {
            Console.ReadLine();

            var vecX = Console.ReadLine().Split().Select(x => double.Parse(x));
            var vecY = Console.ReadLine().Split().Select(y => double.Parse(y));
            var D = new MinkowskiDistance(vecX, vecY);
            Console.WriteLine(D.GetDist(1));
            Console.WriteLine(D.GetDist(2));
            Console.WriteLine(D.GetDist(3));
            Console.WriteLine(D.GetChebyshev);
        }

    }

    class MinkowskiDistance
    {
        double[] abs;
        public MinkowskiDistance(IEnumerable<double> vecX, IEnumerable<double> vecY)
            => abs = vecX.Zip(vecY, (x, y) => Math.Abs(x - y)).ToArray();
        public double GetDist(int p) => Math.Pow(abs.Select(d => Math.Pow(d, p)).Sum(), 1.0 / p);
        public double GetChebyshev => abs.Max();
    }
}
