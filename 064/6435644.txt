#include <bits/stdc++.h>
using namespace std;

void bubbleSort(int A[], int N) {
    bool flag = true;
    int count = 0;
    for(int i = 0;flag;i++) {
        flag = false;
        for(int j = N - 1; j >= i - 1;j--) {
            if(A[j] < A[j - 1]) {
                swap(A[j],A[j-1]);
                flag = true;
                count++;
            }
        }
    }
    for(int i = 0;i < N-1;i++) {
        cout << A[i] << " ";
    }
    cout << A[N-1] << endl;
    cout << count  << endl;
}

int main() {
    int n,a[n];cin >> n;
    for(int i = 0;i < n; i++) cin >> a[i];
    bubbleSort(a,n);
}
