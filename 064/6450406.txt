#include <bits/stdc++.h>
#define rep(i,n) for(int i = 0; i < (n); i++)
using namespace std;
typedef long long ll;

int main(){
    cin.tie(0);
    ios::sync_with_stdio(0);
    
    int N; cin >> N;
    vector<string> S(N);
    rep(i,N) cin >> S[i];
    int M = S[0].size();

    vector<int> ans(N, 0);
    rep(i,N) {
        rep(j,N) if(i != j) {
            int cnt = 0;
            rep(k,M) if(S[i][k] != S[j][k]) cnt++;
            if(cnt <= 1) ans[i] = 1;
        }
    }

    cout << accumulate(ans.begin(), ans.end(), 0LL) << endl;
}
