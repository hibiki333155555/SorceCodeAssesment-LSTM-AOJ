#include <iostream>
#include <algorithm>

using namespace std;

int main() {
    const int N = 12;
    int e[N];
    for (int i = 0; i < N; i++) {
        cin >> e[i];
    }
    sort(e, e+N);

    if (e[0]==e[3]&&e[4]==e[7]&&e[8]==e[11]) {
        cout << "yes" << endl;
    } else {
        cout << "no" << endl;
    }
}
