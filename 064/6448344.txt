#include <stdio.h>
#include <string.h>

int main() {
    char num[1001]={0};
    int sum=0;
    
    while(1) {
        scanf("%s",num);
        
        if(num[0] == '0') {
            break;
        }
        
        for(int i=0;i<strlen(num);i++) {
            sum = sum + (num[i]-'0');
        }
        
        printf("%d\n", sum);
        sum = 0;
    }
      
    return 0;
}
