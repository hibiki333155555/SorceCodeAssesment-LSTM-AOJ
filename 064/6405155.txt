import java.util.Scanner;
class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        int y = sc.nextInt();
        int z = sc.nextInt();
        int total = (x + y) * 60;
        int max = z * 60 + 30;
        if(max >= total){
            System.out.println(1);
        }else{
            System.out.println(0);
        }
    }
}
