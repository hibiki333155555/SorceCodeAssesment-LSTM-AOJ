#include <iostream> // cout, endl, cin
#include <vector> // vector
using namespace std;

int main() {
	int n;
	cin >> n;
	vector<int> v(n+1);
	for (int i = 1; i <= n; i++) {
		cin >> v.at(i);
	}
	vector<int> c(n);
	for (int i = 1; i <= n; i++) {
		c.at(v.at(i-1))++;
	}
	for (int i = 1; i < n; i++) {
		c.at(i) = c.at(i) + c.at(i - 1);
	}
	vector<int> b(n + 1);
	for (int i = 1; i <= n; i++) {
		b.at(c.at(v.at(i))) = v.at(i);
		c.at(v.at(i))--;
	}
	for (int i = 1; i <= n; i++) {
		if (i == n) {
			cout << b.at(i) << endl;
		}
		else {
			cout << b.at(i) << " ";
		}
	}
}
