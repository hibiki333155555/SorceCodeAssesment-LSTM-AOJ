#include <iostream>
using namespace std;

int main(){
    int n;
    cin >> n;
    
    int a[n];
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    
    int q;
    cin >> q;
    int k;
    int j;
    bool flg;
    for(int i = 0; i < q; i++){
        cin >> k;
        flg = false;
        for(j = 0; j < n; j++){
            if(k <= a[j]){
                cout << j << endl;
                flg = true;
                break;
            }
        }
        if(!flg){
            cout << n << endl;
        }
    }
}
