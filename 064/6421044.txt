using ll = long long;
using P = pair<ll,ll>;
constexpr ll INF = 1e18;
constexpr ll LLMAX = 9223372036854775807;
constexpr int inf = 1e9;
// constexpr ll mod = 1000000007;
constexpr ll mod = 998244353;
// 右下左上
const int dx[8] = {1, 0, -1, 0,1,1,-1,-1};
const int dy[8] = {0, 1, 0, -1,1,-1,1,-1};
template<class T> inline bool chmax(T& a, T b) { if (a < b) { a = b; return 1; } return 0; }
template<class T> inline bool chmin(T& a, T b) { if (a > b) { a = b; return 1; } return 0; }
ll gcd(ll a,ll b){if(b==0){return a;}else{return gcd(b,a%b);}};
#define eb emplace_back
#define pb pop_back
#define eol endl
// ---------------------------------------------------------------------------


int main(){
  ll H,W;
  cin >> H >> W;
  vector<string> G(H);
  for(int i=0; i<H; i++) cin >> G[i];
  map<P,ll> mp;
  ll M;
  cin >> M;
  for(int i=0; i<M; i++){
    ll a,b,t;
    cin >> a >> b >> t;
    a--;b--;
    mp[P(a,b)] = t;
  }
  ll ans = 0;
  {
    queue<P> que;
    vector<vector<ll>> dp(H,vector<ll>(W,INF));
    for(int i=0; i<H; i++){
      for(int j=0; j<W; j++){
        if(G[i][j] == 'S'){
          que.emplace(i,j);
          dp[i][j] = 0;
        }
      }
    }
    while(que.size()){
      ll y = que.front().first;
      ll x = que.front().second;
      que.pop();
      for(int k=0; k<4; k++){
        ll ny = y + dy[k];
        ll nx = x + dx[k];
        if(ny<0 || ny>=H || nx<0 || nx>=W) continue;
        if(G[ny][nx] == '#') continue;
        if(mp.count(P(ny,nx)) && dp[y][x]+1 < mp[P(ny,nx)]){
          if(chmin(dp[ny][nx],dp[y][x]+1)){
            que.emplace(ny,nx);
          }
        }
        if(!mp.count(P(ny,nx))){
          if(chmin(dp[ny][nx],dp[y][x]+1)){
            que.emplace(ny,nx);
          }
        }
      }
    }
    for(int i=0; i<H; i++){
      for(int j=0; j<W; j++){
        if(G[i][j] == 'V'){
          if(dp[i][j] == INF){
            cout << "No" << endl;
            return 0;
          }else{
            ans += dp[i][j];
          }
        }
      }
    }
  }
  {
    queue<P> que;
    vector<vector<ll>> dp(H,vector<ll>(W,INF));
    for(int i=0; i<H; i++){
      for(int j=0; j<W; j++){
        if(G[i][j] == 'V'){
          que.emplace(i,j);
          dp[i][j] = 0;
        }
      }
    }
    while(que.size()){
      ll y = que.front().first;
      ll x = que.front().second;
      que.pop();
      for(int k=0; k<4; k++){
        ll ny = y + dy[k];
        ll nx = x + dx[k];
        if(ny<0 || ny>=H || nx<0 || nx>=W) continue;
        if(G[ny][nx] == '#') continue;
        if(mp.count(P(ny,nx)) && dp[y][x]+1+ans < mp[P(ny,nx)]){
          if(chmin(dp[ny][nx],dp[y][x]+1)){
            que.emplace(ny,nx);
          }
        }
        if(!mp.count(P(ny,nx))){
          if(chmin(dp[ny][nx],dp[y][x]+1)){
            que.emplace(ny,nx);
          }
        }
      }
    }
    for(int i=0; i<H; i++){
      for(int j=0; j<W; j++){
        if(G[i][j] == 'S'){
          if(dp[i][j] == INF){
            cout << "No" << endl;
            return 0;
          }else{
            ans += dp[i][j];
          }
        }
      }
    }
  }
  cout << "Yes" << endl;
  return 0; 
}
