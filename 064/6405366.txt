#include <stdio.h>

int main(void){
    int H, W;
    int i, j;

    while(1){
        scanf("%d%d", &H, &W);

        if(H == 0 || W == 0) break;

        for(i = 0; i < H; i++){
            for(j = 0; j < W; j++) (i + j) % 2 == 0 ? putchar('#') : putchar('.');
            putchar('\n');
        }
        printf("\n");
    }

    return 0;
}
