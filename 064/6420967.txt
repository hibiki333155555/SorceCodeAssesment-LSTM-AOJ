#include <bits/stdc++.h>
using namespace std;
using ll=long long;
int main(){
  ll n;
  cin>>n;
  vector<vector<ll>> v(n,vector<ll>(n));
  map<ll,vector<ll>> s;
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      cin>>v[i][j];
      s[v[i][j]].push_back(i*10000+j);
    }
  }
  map<ll,ll> seen;
  map<ll,vector<ll>> ans;
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(seen[v[i][j]]==0){
        ll count=1000000000000;
        for(int k=0;k<s[v[i][j]].size();k++){
          count=min(count,s[v[i][j]][k]);
        }
        ans[v[i][j]].push_back(count);
        seen[v[i][j]]=1;
      }
    }
  }
  ll q;
  cin>>q;
  while(q--){
    ll t;
    cin>>t;
    if(ans[t].size()>0){
      cout<<ans[t][0]/10000+1<<" "<<ans[t][0]%10000+1<<endl;
    }
    else{
      cout<<-1<<endl;
    }
  }
}
