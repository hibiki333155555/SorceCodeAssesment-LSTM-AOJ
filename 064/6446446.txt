#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<int> a(n);
  for(int i = 0; i < n; i++) cin >> a.at(i);
  for(int i = 0; i < n; i++){
    if(i != n - 1) cout << a.at(i) << " ";
    else cout << a.at(i) << endl;
  }
  for(int i = 1; i < n; i++){
    int v = a.at(i);
    int j = i - 1;
    while(j >= 0 && a.at(j) > v){
      a.at(j + 1) = a.at(j);
      j--;
    }
    a.at(j + 1) = v;
    for(int j = 0; j < n; j++){
      if(j != n - 1) cout << a.at(j) << " ";
      else cout << a.at(j) << endl;
    }
  }

  return 0;
}
