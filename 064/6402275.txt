#include <stdio.h>
#include <string.h>
#define MAX_N 100010

#define INF 0x7E7E7E7E

#if 0
#define LOG(format, ...) printf("[%d->%s] " format "\r\n",__LINE__, __func__, ##__VA_ARGS__)
#else
#define LOG(format, ...)
#endif

//#define LOGG(format,...) printf(""format,##__VA_ARGS__)

int DP[MAX_N];

int A[MAX_N];
int N;
int main()
{
    int i, j,k;
    scanf("%d",&N);
    for(i=1 ; i<=N ; ++i)
    {
        scanf("%d",&A[i]);
    }
    memset(DP,0x7E,(N+5)*sizeof(int));
    int left,right,mid;
    for(i=1; i<=N; i++)
    {
        left = 1,right =i,mid =1;
        while(left<right)
        {
            mid =(left+right)/2;
            if(DP[mid]>=A[i] && DP[mid-1]<A[i])
            {
                break;
            }
            else if(DP[mid]>A[i])
            {
                right =mid;
            }
            else
            {
                left =mid;
            }
        }
        DP[mid]=A[i];

//        for(j=1 ; j<=i ; j++)

//                break;
//            }

    }

    i =1;
    while(DP[i]!=INF)
    {
        LOG("%d",DP[i]);
        i++;
    }
    LOG("\r\n");
    printf("%d\n",i-1);

    return 0;
}
