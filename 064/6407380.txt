import java.util.Scanner;
class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        int i = 1;
        while(x==0){
            System.out.println("Case " + i + ": " + x);
            x = sc.nextInt();
            i++;
        }
    }
}
