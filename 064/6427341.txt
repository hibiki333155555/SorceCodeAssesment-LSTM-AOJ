int main(){
    int w,h;
    char c; cin>>w>>h>>c;
    char f[h][w];
    rep(i,0,h){
        rep(j,0,w){
            if((i==0&&j==0)||(i==h-1&&j==w-1)||(i==0&&j==w-1)||(i==h-1&&j==0)) f[i][j]='+';
            else if(i==0||i==h-1) f[i][j]='-';
            else if(j==0||j==w-1) f[i][j]='|';
            else if(i==h/2&&j==w/2) f[i][j]=c;

        }
    }
    rep(i,0,h){
        rep(j,0,w){
            cout<<f[i][j];
        }
        cout<<"\n";
    }
    return 0;
}
