#include "bits/stdc++.h"
using namespace std;
int main() {
	string  p, c;
	while (1) {a:
		c = "A";
		cin >> p;
		if (p == "#")
			break;
		for (int i = 0; i < p.size(); i++) {
			if (c == "A") {
				if (p[i] == '1')
					c = "Y";
				else
					c = "X";
			}else if (c == "X") {
				if (p[i] == '1')
					c = "Z";
				else {
					cout << "No" << endl;
					goto a;
				}
			}else if (c == "Y") {
				if (p[i] == '1') {
					cout << "No" << endl;
					goto a;
				}
				else
					c = "X";
			}else if (c == "Z") {
				if (p[i] == '1')
					c = "B";
				else
					c = "W";
			}
			else if (c == "W") {
				if (p[i] == '1')
					c = "Y";
				else
					c = "B";
			}
			else if (c == "B") {
				if (p[i] == '1')
					c = "X";
				else
					c = "Y";
			}
		}
		if (c == "B")
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
	return 0;
}
