#include <bits/stdc++.h>
using namespace std;
int main(){
    long long N, M, K, X;
    cin >> N >> M >> K >> X;
    vector<long long> A(N);
    for(int i = 0; i < N; i++) cin >> A[i];
    sort(A.begin(), A.end());
    if(A[N - 1] * K >= M * X) cout << "Yes" << endl;
    else cout << "No" << endl;
}
