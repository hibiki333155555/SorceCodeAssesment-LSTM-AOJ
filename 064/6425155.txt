#include<bits/stdc++.h>
using namespace std;
int main(){
    int b[3]; cin>>b[0]>>b[1]>>b[2];
    if(b[0]==1){
        cout<<"Close"<<endl;
    }else if(b[1]==1){
        cout<<"Close"<<endl;
    }else if(b[1]+b[0]==2){
        cout<<"Open"<<endl;
    }else if(b[2]==1){
        cout<<"Open"<<endl;
    }else{
        cout<<"Close"<<endl;
    }
    return 0;
}
