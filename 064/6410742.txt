int main(){
	cin.tie(0);
	ios::sync_with_stdio(false);
	cout<<fixed<<setprecision(10);

	auto Ho=in(),Mi=in(),N=in();
	auto G=60*Ho+Mi;
	vll A(N);
	REP(i,N){
		auto a=in(),b=in();
		A[i]=60*a+b;
	}
	auto M=in();
	vll B(M);
	REP(i,M){
		auto a=in(),b=in();
		B[i]=60*a+b;
	}
	
	auto middle = lower_bound(ALL(B),G-10);
	if(middle==B.begin()){
		if((*middle)!=G-10){
			out(0);
			return 0;
		}
	}
	else{
		if((*middle)!=G-10){
			middle--;
		}
	}

	auto first = lower_bound(ALL(A),(*middle)-11);
	if(first==A.begin()){
		if((*first)!=(*middle)-11){
			out(0);
			return 0;
		}
	}
