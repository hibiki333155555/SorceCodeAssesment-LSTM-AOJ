#include<bits/stdc++.h>
#include<vector>
#define ll long long
using namespace std;
int main()
{
    //vector<vector<int>> v;
    ll i,j,n,q,a,b,c;
    cin>>n>>q;
    vector <int> v[n];
    for(i=0;i<q;i++)
    {
        cin>>a;
        if(a==0) { 
            cin>>b>>c;
            v[b].push_back(c); 
            }
            else if(a==1)
            {
                cin>>b;
                ll k = v[b].size();
                for(j=0;j<k;j++) { cout<<v[b][j]<<" "; }
                cout<<endl;
            }
            else
            {
                cin>>b;
                v[b].clear();
            }
    }
    return 0;
}
