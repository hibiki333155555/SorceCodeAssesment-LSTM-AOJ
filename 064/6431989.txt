#include <stdio.h>
void prtlst(int lst[],int n){printf("%d",lst[0]);for(int i=1;i<n;i++)printf(" %d",lst[i]);printf("\n");}
int main(void){ // N個の要素を含む0-オリジンの配列A
    int i,v,j,A[100],N;
    scanf("%d",&N);
    for(i=0;i<N;i++)scanf("%d",&A[i]);
    prtlst(A,N);
    for(i=0;i<N-1;i++){
        v = A[i+1];
        j = i;
        while(j>=0&&A[j]>v){
            A[j+1] = A[j];
            j--;}
        A[j+1] = v;
        prtlst(A,N);
}}
