#include<cstdio>

int main(){
    int n;
    int curp;
    int maxdiff;
    int minip;
    scanf("%d",&n);
    int p1,p2;
    scanf("%d%d",&p1,&p2);
    maxdiff=p2-p1;
    minip=p1<p2?p1:p2;
    for(int i=0;i!=n-2;++i){
        scanf("%d",&curp);
        if(curp-minip>maxdiff){
            maxdiff=curp-minip;
        }
        if(curp<minip){
            minip=curp;
        }
    }
    printf("%d\n",maxdiff);
}
