#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (true)
    {
        string x;
        cin >> x;
        if (x.compare("0") == 0)
        {
            break;
        }
        else
        {
            unsigned int sum = 0;
            for(int i = 0; i < x.length(); i++){
                sum += x[i] - '0';
            }
            cout << sum << endl;
        }
    }

    return 0;
}
