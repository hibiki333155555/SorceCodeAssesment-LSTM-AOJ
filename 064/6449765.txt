#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)
#define Inf 1000000001
int diff(string a,string b){
	int ret = 0;
	rep(i,a.size()){
		if(a[i]!=b[i])ret++;
	}
	return ret;
}

int main(){
	
	int n;
	cin>>n;
	
	vector<string> s(n);
	rep(i,n)cin>>s[i];
	int ans = 0;
	rep(i,n){
		bool f = true;
		rep(j,n){
			if(i==j)continue;
			if(diff(s[i],s[j])<=1)f = false;
		}
		if(!f)ans++;
	}
	cout<<ans<<endl;
	return 0;
}
