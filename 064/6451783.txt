int main() {
    int N, M;
    cin >> N >> M;
    vector<vector<pint>> to(N);
    while (M--) {
        int a, b, c;
        cin >> a >> b >> c;
        --a, --b;
        to[a].emplace_back(b, c);
    }

    vector<int> vs;
    REP(S, 1 << 20) {
        lint x = 1;
        REP(d, 20) if ((S >> d) & 1) x = lint(x) * (d + 1) / __gcd<lint>(x, d + 1);
        vs.push_back(x);
    }
    vs = sort_unique(vs);
    dbg(vs.size());
    dbg(vs.back());

    constexpr int BS = 960;
    vector<bitset<BS>> reach(N);

    vector Next(BS, vector<int>(21));
    REP(i, BS) {
        lint now = vs[i];
        FOR(e, 1, 21) {
            int nxt = arglb(vs, int(now * e / __gcd<lint>(now, e)));
            Next[i][e] = nxt;
        }
    }

    vector<pint> st;
    reach[0].set(0);
    st.emplace_back(0, 0);
    while (st.size()) {
        auto [now, d] = st.back();
        st.pop_back();
        for (auto [nxt, e] : to[now]) {
            if (!reach[nxt][Next[d][e]]) {
                reach[nxt][Next[d][e]] = 1;
                st.emplace_back(nxt, Next[d][e]);
            }
        }
    }
    if (reach.back().none()) {
        puts("IMPOSSIBLE");
    } else {
        int i = reach.back()._Find_first();
        int j = BS - 1;
        while (!reach.back()[j]) --j;
        cout << vs[i] << ' ' << vs[j] << '\n';
    }
}
