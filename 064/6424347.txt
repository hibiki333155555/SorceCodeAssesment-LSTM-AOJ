using System;
using System.Linq;

namespace AOJ_ITP1_10_C
{
    internal class Program
    {
        public static void Main()
        {
            while (Console.ReadLine() != "0")
            {
                var data = Console.ReadLine().Split().Select(d => int.Parse(d)).ToArray();
                var ave = data.Average();
                Console.WriteLine(Math.Sqrt(data.Select(s => Math.Pow(s - ave, 2)).Sum() / data.Length));
            }
        }
    }
}
