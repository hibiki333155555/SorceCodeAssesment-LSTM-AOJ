using namespace std;
using ll=long long;

int main(){
  ll N,K;
  cin>>N>>K;
  vector<ll> A(N);
  for(int i=0;i<N;i++){
    cin>>A[i];
  }
  priority_queue<ll,vector<ll>,greater<ll>> que;
  ll sum=0;
  for(auto i:A){
    que.push(i);
    sum+=i;
  }  
  for(int i=0;i<K;i++){
    int u=que.top();que.pop();
    sum-=u;
    if(u==1){
      que.push(5);
      sum+=5;
    }else{
      que.push(6);
      sum+=6;
    }
  }
  cout<<sum<<endl;
  return 0;
}
