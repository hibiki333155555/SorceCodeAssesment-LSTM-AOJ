#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
const int N = 500;
int n;
vector<int>even,odd;
int b[N];
int a[N];

int main() {
	while(cin>>n && n) {
		even.clear();
		odd.clear();
		for(int i=0;i<n*(n+1)/2;i++) {
			cin>>b[i];
			if(b[i]&1)
				odd.push_back(b[i]);
			else even.push_back(b[i]);
		}
		sort(even.begin(),even.end());
		sort(odd.begin(),odd.end());

		int a0 = (int)sqrt((double)even[0]*even[1]/odd[0]+0.5);
		cout<<a0<<endl;
		for(int i=0;i<even.size();i++) {
			if(i) cout<<" ";
			cout<<b[i] / a0;
		}
		cout<<endl;
	}
}
