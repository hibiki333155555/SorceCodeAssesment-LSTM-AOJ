const fs=require("fs");

function Main(input){
    input=input.split(" ");
    input[0]=parseInt(input[0]);
    input[1]=parseInt(input[1]);
    
    if(input[0]==input[1]){
        console.log("a == b");
    }else if(input[0]<input[1]){
        console.log("a < b");
    }else if(input[0]>input[1]){
        console.log("a > b");
    }
}

Main(fs.readFileSync("/dev/stdin","utf8"));
