#include<bits/stdc++.h>
using namespace std;
int main(){
    while(true){
    string s;
        cin>>s;
        int sum = 0;
        for(int i=0;i< s.size(); i++){
            cout<<s.size()<<"\n";
            sum = sum + (int)(s[i] - '0');
        }
        if(sum == 0){
            break;
        }
        cout << sum <<"\n";
    }
    return(0);
}
