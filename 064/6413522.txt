#include <bits/stdc++.h>

using namespace std;

bool is_prime_number(int x) {
  if (x % 2 == 0 && x != 2) {
    return false;
  }
  for (int i = 3; i <= x / 2; i += 2) {
    if (x % i == 0) return false;
  }
  return true;
}

int main() {
  int N;
  scanf("%d", &N);
  int target;
  int count = 0;
  for (int i = 0; i < N; i++) {
    scanf("%d", &target);
    if (is_prime_number(target)) count++;
  }
  cout << count << endl;
}
