#include<stdio.h>
#include<math.h>

int main(void)
{
    int a[1001];
    int n,i,j,k;
    while(1){
    long long m=0;
    scanf("%d",&n);
    if(n==0)break;
    for(i=0;i<n;i++)
    {
        scanf("%d",&j);
        a[i]=j;
        m+=j;
    }
    m/=n;
    for(i=0;i<n;i++)
    {
        k+=pow(a[i]-m,2);
    }
    k=sqrt(k/n);

    }

}
