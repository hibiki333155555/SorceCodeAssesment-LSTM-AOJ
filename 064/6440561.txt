#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

int main(){
    set<int> s;
    int choice, x;
    int q;
    cin >> q;
    for(int i=0; i<q; i++){
        cin >> choice;
        if(choice == 0){
            cin >> x;
            s.insert(x);
            cout << s.size() << endl;
        }
        else if(choice == 1){
            cin >> x;
            cout << s.count(x) << endl;
        }
        else if(choice == 2){
            cin >> x;
            s.erase(x);
        }
        else if(choice == 3){
            int l, r;
            cin >> l >> r;
            for(set<int>:: iterator it = s.lower_bound(l); it != s.upper_bound(r); it++){
                cout << *it << endl;
            }
        }
    }
    return 0;
}
