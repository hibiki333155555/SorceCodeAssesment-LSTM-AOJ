#include<bits/stdc++.h>
using namespace std;
int main(){
    char c;
    int w, h;
    cin>>w>>h>>c;
    vector<vector<char>> s(w,vector<char>(h,'.'));
    for(int i = 0;i < h;i++){
        s[0][i] = '|';
        s[w-1][i] = '|';
    }
    for(int i = 0;i < w;i++){
        s[i][0] = '-';
        s[i][h-1] = '-';
    }
    s[0][0] = '+';
    s[0][h-1] = '+';
    s[w-1][0] = '+';
    s[w-1][h-1] = '+';
    s[w/2][h/2] = c;
    for(int i=0;i < h;i++){
        for(int j = 0;j < w;j++){
            cout<<s[j][i];
        }
        cout<<endl;
    }
    return 0;
}
