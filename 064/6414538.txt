void solve(){
    ll m,n;cin>>m>>n;
    vector<string>s(m);
    rep(i,m)cin>>s[i];
    ll ans=n*m;
    ll cnt=0;
    ll sum=0;
    rep(i,m){
        rep(j,n){
            if(s[i][j]=='o')sum++;
        }
    }
    rep(i,m-1){
        rep(j,n){
            if(s[i][j]=='o'){
                cnt++;
                sum--;
            }
        }
        ans=min(ans,(i+1)*n-cnt+sum);
        ans=min(ans,cnt+(m-i-1)*n-sum);
    }
    cnt=0,sum=0;
    rep(i,m){
        rep(j,n){
            if(s[i][j]=='o')sum++;
        }
    }
    rep(i,n-1){
        rep(j,m){
            if(s[j][i]=='o'){
                cnt++;
                sum--;
            }
        }
        ans=min(ans,(i+1)*m-cnt+sum);
        ans=min(ans,cnt+(n-i-1)*m-sum);
    }
    cout<<ans<<endl;
}

signed main(){
    cin.tie(0);
    ios::sync_with_stdio(false);
    solve();
