#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    int n,e;
    cin>> n;
    vector<int>v;
    while(n--){
        cin>>e;
        v.push_back(e);
    }
    cin>>n;
    while(n--){
        cin>>e;
        if(v.end() != find(v.begin(), v.end(), e))cout << 1<<endl;
        else cout << 0 <<endl;
    }
    return 0;
}
