#include<stdio.h>
int main(void)
{
    int a,b,c,d,e;
    scanf("%d %d %d %d %d",&a,&b,&c,&d,&e);
    int r,l,u,dow;
    r = c+e;
    l = c-e;
    u = d+e;
    dow= d-e;
    if(r<a && l>0 && u<b && dow>0)
    {
        printf("Yes\n");
    }
    else
    {
        printf("No\n");
    }
    return 0;
}
