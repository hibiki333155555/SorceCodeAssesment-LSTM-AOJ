#include<bits/stdc++.h>

using namespace std;

int x, y, e;
vector<int> g[1000 + 9];
vector<bool> used;
vector<int> p;

bool dfs(int v) {
  if(used[v]) return false;
  used[v] = true;
  for(auto u : g[v]) {
    if(p[u] == -1 || dfs(p[u])) {
      p[u] = v;
      return true;
    }
  }
  return false;
}
int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  
  cin >> x >> y >> e;
  for(int i = 0; i < e; i++) {
    int a, b;
    cin >> a >> b;
    b += x;
    g[a].push_back(b);
    g[b].push_back(a);
  }
  int total_match = 0;
  p.assign(y + x, -1);
  for(int i = 0; i < x; i++) {
    used.assign(x + x, false);
    total_match += dfs(i);
  }
  cout << total_match << '\n';
  
  return 0;
}
