#include <iostream>
#include <vector>
#include <string>
using namespace std;

int calc(int n,int m, vector<vector<int> >&a, vector<int> &b,int i){
	int ans = 0;
	for (int j= 0; j < m; ++j) {
		ans += a.at(i).at(j) * b.at(j);
	}
	return ans;
}

int main(){
	int n, m;
	cin >> n >> m;
	vector<vector<int> >a(n, vector<int>(m));
	vector<int> b(m);
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			cin >> a.at(i).at(j);
		}
	}
	for (int i = 0; i < n; ++i) {
		cin >> b.at(i);
	}
	for (int i = 0; i < n; ++i) {
		cout << calc(n, m, a, b,i) << endl;
	}
}
