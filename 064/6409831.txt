#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define rep(i, n) for (int i = 0; i < (n); i++)

bool func(int i, int w, const vector<int> &a){
  if(i == 0){
    if(w == 0) return true;
    else return false;
  }
  if(func(i - 1, w, a)) return true;
  if(func(i - 1, w - a[i - 1], a)) return true;
  return false; 
}

int main(){
  int n; cin >> n;  
  vector<int> a(n);
  rep(i, n) cin >> a[i];
  int q; cin >> q;
  vector<int> m(q);
  rep(i, q) cin >> m[i];
  rep(i, q){
    if(func(n, m[i], a)) cout << "yes" << endl;
    else cout << "no" << endl;
  }
}
