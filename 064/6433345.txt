#include<bits/stdc++.h>
using namespace std;
int N=12;
vector<int> vec(N);
int main (){
    bool check;
    for (int i=0; i<N; i++){
        cin >> vec.at(i);
    }
    sort(vec.begin(), vec.end());
    for(int i=0; i<N; i+=4){
        check=true;
        if(vec.at(i)!=vec.at(i+3)){
            check=false;
        }
    }
    if(check){
        cout << "yes" <<endl;
    }else{
        cout <<"no"<<endl;
    }
    return 0;
}
