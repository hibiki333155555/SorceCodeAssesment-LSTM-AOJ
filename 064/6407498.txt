#include <iostream>

using namespace std;

const int MAX = -999999999;

int main(int argc, char const *argv[])
{
  int t;

  cin >> t;

  int *nums = new int[t];

  for (int i = 0; i < t; i++)
  {
    cin >> nums[i];
  }

  int profit = MAX;
  for (int i = 0; i < t; i++)
  {
    for (int j = i + 1; j < t; j++)
    {
      if (nums[j] - nums[i] > profit) profit = nums[j] - nums[i];
    }
  }
  
  cout << profit << endl;

  return 0;
}
