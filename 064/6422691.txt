#include<bits/stdc++.h>
using namespace std;
typedef long long lint;

int main(){
  int n;
  cin >> n;
  vector<int> a(n);
  int dp[30][2010]={};
  for(int i=0;i<n;i++)cin >> a[i];
  dp[0][0]=1;
  for(int i=0;i<n;i++){
    for(int j=0;j<=2000;j++){
      if(j >= a[i]){
        dp[i+1][j] += dp[i][j-a[i]];
      }
      dp[i+1][j] += dp[i][j];
    }
  }
  
  int q, m;
  cin >> q;
  for(int i=0;i<q;i++){
    cin >> m;
    if(dp[n][m] > 0){
      cout << "yes" << endl;
    }else{
      cout << "no" << endl;
    }
  }
  return 0;
}
