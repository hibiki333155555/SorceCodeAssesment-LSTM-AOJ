using std::cin;
using std::cout;
using std::endl;

using std::sort;
using std::string;
using std::unique;
using std::find;

#define rep(i, a, n) for (int i = (a); i < (n); ++i)
#define bit(n, k) ((n >> k) & 1)

class Program {
private:

    bool max(int &a, const int b) {
        if (a < b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

public:

    Program() {
    }

    void solve() {
        int n; cin >> n;
        vector<int> a;
        a.resize(n);

        int k = 0, v = 0;
        rep(i, 0, n) {
            cin >> a[i];
            if (max(v, a[i])) {
                k = i;
            }
        }

        v = 0;
        rep(i, 0, k) {
            v += a[i];
        }
        cout << v << endl;

        v = 0;
        rep(i, k+1, n) {
            v += a[i];
        }
        cout << v << endl;
    }
};

int main() {

#ifdef DEV
    std::ifstream in("input");
    cin.rdbuf(in.rdbuf());

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();
#endif

    return 0;
}
