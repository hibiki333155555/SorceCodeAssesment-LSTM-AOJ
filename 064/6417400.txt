#include<stdio.h>
#include<string.h>
#include<ctype.h>

int main(void)
{
    char t[100],w[11];
    int count = 0,i;
    scanf("%s",w);
    while(1)
    {
        scanf("%s",t);
        if(strcmp(t,"END_OF_TEXT")==0)break;
        for(i=0;t!='\0';i++)
        {
            t[i]=tolower(t[i]);
        }
        if(strcmp(t,w))count++;
    }
    printf("%d",count);
    
    return 0;
}
