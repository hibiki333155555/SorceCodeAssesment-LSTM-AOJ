#include <iostream>
#include <string>
using namespace std;
// Sensei
int main()
{
    string word, value;
    int count = 0;
    cin >> value;
    while (cin >> word)
    {
        if (word == "END_OF_TEXT")
        {
            break;
        }
        else
        {
            if (word == value)
            {
                count += 1;
            }
        }
    }
    cout << count << endl;
    return 0;
}
