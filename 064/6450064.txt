#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main(){
    ll A,B;
    cin >> A >> B;
    if(A==B&&A==0){
        cout << 1 << endl;
        return 0;
    }
    string Abit,Bbit;
    int cnt=32;
    ll nibeki=1;
    while(cnt>0){
        cnt--;
        nibeki*=2;
    }
    while(nibeki>0){
        nibeki/=2;
        if(A>=nibeki){
            Abit=Abit+"1";
            A-=nibeki;
        }else{
            Abit=Abit+"0";
        }
        if(B>=nibeki){
            Bbit=Bbit+"1";
            B-=nibeki;
        }else{
            Bbit=Bbit+"0";
        }
    }
    int ans=0;
    bool StartFlag=false;
    for(int i=0;i<=31;i++){
        if(Abit[i]==Bbit[i]&&Abit[i]=='0'&&StartFlag==false) continue;
        if(StartFlag==false&&(Abit[i]=='1'||Bbit[i]=='1')) StartFlag=true;
        if(Abit[i]==Bbit[i]) ans++;
        else break;
    }
    cout << ans << endl;
}
