#include <stdio.h>
#include <math.h>

int main(void)
{
    double xp0, yp0;
    double xp1, yp1;
    double xp2, yp2;
    double a, cos;
    int q;
    int i;
    
    scanf("%lf%lf%lf%lf", &xp0, &yp0, &xp1, &yp1);
    xp1 -= xp0;
    yp1 -= yp0;
    
    scanf("%d", &q);
    for (i = 0; i < q; i++){
        scanf("%lf%lf", &xp2, &yp2);
        
        xp2 -= xp0;
        yp2 -= yp0;
        
        a = sqrt(xp1 * xp1 + yp1 * yp1) * sqrt(xp2 * xp2 + yp2 * yp2);
        if (a == 0){
            printf("ON_SEGMENT\n");
        }
        else {
//            printf("<%lf>", xp1*yp2-yp1*xp2);
//            printf("[%lf]", (xp1 * xp2 + yp1 * yp2));
            cos = (xp1 * xp2 + yp1 * yp2) / a;
            if (cos == 1){
                printf("ONLINE_FRONT\n");
            }
            else if (cos == -1){
                printf("ONLINE_BACK\n");
            }
            else if (xp1 * yp2 - yp1 * xp2 > 0){
                printf("COUNTER_CLOCKWISE\n");
            }
            else {
                printf("CLOCKWISE\n");
            }
        }
    }
    
    return (0);
}
