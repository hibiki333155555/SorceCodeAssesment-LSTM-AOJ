#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;
static const int MMAX = 20;
static const int NMAX = 50000;
static const int INFINITY = INT_MAX;

int main() {
    int n, m, C[MMAX + 1], T[NMAX + 1];
    cin >> n >> m;

    for (int i = 1; i <= m; ++i) {
        cin >> C[i];
    }
    for (int i = 0; i <= NMAX; ++i) {
        T[i] = INFINITY;
    }

    T[0] = 0;

    for (int i = 1; i <= m; ++i) {
        for (int j = 0; j + C[i] <= n; ++j) {
            T[j + C[i]] = min(T[j + C[i]], T[j] + 1);
        }
    }

    cout << T[n] << endl;
}
