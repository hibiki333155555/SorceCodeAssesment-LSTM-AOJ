template<typename T>
ostream &operator<<(ostream &os,const vector<T>&v){
    for(int i=0;i<(int)v.size();i++) os<<v[i]<<(i+1==(int)v.size()?"":" ");
    return os;
}
template<typename T>
istream &operator>>(istream &is,vector<T>&v){
    for(T &x:v)is>>x;
    return is;
}

signed main(){
    int H,W,N;cin>>H>>W>>N;

    vector<string> S(H,string(W,'#'));

    for(int i=0,j=0,dir=0;N;N--){
        S[i][j]='X';
        if(dir==0){// j++
            if(j+1<W and S[i][j+1]=='#');
            else                        dir=(dir+1)%4;
        }
        else if(dir==1){// i++
            if(i+1<H and S[i+1][j]=='#');
            else                         dir=(dir+1)%4;
        }
        else if(dir==2){// j--
            if(j-1>=0 and S[i][j-1]=='#');
            else                          dir=(dir+1)%4;
        }
        else if(dir==3){// i--
            if(i-1>=0 and S[i-1][j]=='#');
            else                          dir=(dir+1)%4;
        }

        if(dir==0) j++;
        if(dir==1) i++;
        if(dir==2) j--;
        if(dir==3) i--;

        if(not (0<=i and i<H and 0<=j and j<W)) break;
    }

    for(auto &w:S) cout<<w<<"\n";
    return 0;
}
