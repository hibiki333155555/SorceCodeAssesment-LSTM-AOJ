#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n;
  vector<int> v(2001);
  int ans;
  for(int i = 0; i < n; i++){
    int a;
    cin >> a;
    if(i == 0){
      ans = a;
    }
    v[a]++;
  }
  int kazu = v[ans];
  for(int i = 1; i <= 2000; i++){
    if(v[i] != 0){
      if(v[i] < kazu){
      ans = i;
      kazu = v[i];
    } else if(v[i] == kazu && i < ans){
        ans = i;
        kazu = v[i];
      }
    }
  }
  cout << ans << endl;
  return 0;
}
