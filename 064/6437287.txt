#include <iostream>
#include <cmath>
using namespace std;

bool isprime(int x){
    if(x < 2) return false;
    if(x == 2) return true;
    if(x % 2 == 0) return false;
    for(int i = 3; i*i <= x; i += 2){
        if(x % i == 0) return false;
    }
    return true;
}
int main(){
    int n,sum = 0,s;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> s;
        if(isprime(s)){
            sum++;
        }
    }
    cout << sum << endl;
}
