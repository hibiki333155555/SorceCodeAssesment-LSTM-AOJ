#define rep(i,n) for(int i=0;i<(int)(n);i++)
#define ALL(v) v.begin(),v.end()
typedef long long ll;
 
#include <bits/stdc++.h>
using namespace std;

int main(){
  ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  ll n,m,k,x;
  cin>>n>>m>>k>>x;
  vector<ll> A(n);
  rep(i,n) cin>>A[i];
  sort(ALL(A));
  if(A[n-1]*k>=m*x) cout<<"Yes"<<endl;
  else cout<<"No"<<endl;
  
  
  
  
  return 0;
}
