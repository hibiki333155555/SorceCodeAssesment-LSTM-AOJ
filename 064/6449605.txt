vector<int> v(int a){
	vector<int> res;
	if(a == 0){
		res.eb(0);
		return res;
	}
	else{
		while(a){
			res.eb(a%2);
			a /= 2;
		}
		return res;
	}
}
int main(){
  cout << fixed << setprecision(15);
  
  int a, b;
  cin >> a >> b;
  vector<int> v1 = v(a);
  vector<int> v2 = v(b);
  int s1 = v1.size(), s2 = v2.size();
  if(s1 <= s2){
  	rep(i, s2-s1) v1.eb(0);
  }
  else{
  	rep(i, s1-s2) v2.eb(0);
  }
  reverse(all(v1));
  reverse(all(v2));
  int ans = 0;
  int n = max(s1, s2);
  rep(i, n){
  	if(v1[i] == v2[i]) ans++;
  	else break;
  }
  cout << ans << endl;
  return 0;
}
