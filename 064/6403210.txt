#define _CRT_SECURE_NO_WARNINGS

#include<iostream>
using namespace std;
#define MAXN 100005
int w[MAXN];
int n, k;

int search(int p) {
	int cnt = 1;
	int v = 0;
	for (int i = 0; i < k; i++) {
		int j = cnt;
		int s = 0;
		for (; j <= n; j++) {
			if (s + w[j] <= p) {
				s += w[j];
				v += s;
			}
			else {
				break;
			}
			cnt++;
		}
	}
	return v;
}

int main() {
	cin >> n >> k;
	int v = 0;
	int sum = 0;
	for (int i = 1; i <= n; i++) {
		scanf("%d", w + i);
		sum += w[i];
	}
	int p = 0;
	for (p = 0; p < sum; p++) {
		v = search(p);
		if (v >= sum) {
			break;
		}
	}
	printf("%d", p);
	return 0;
}
