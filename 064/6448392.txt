#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    string p;
    cin>>p;
    int sum=0;
    while(true){
        cin>>s;
        if(s == "END_OF_TEXT"){
            break;
        }
        if(s == p){
            sum++;
        }
    }
    cout<<sum<<"\n";
    return(0);
}
