'''
forで回す範囲を重複しないようにする
i > j > k になるようにする
'''
while True:
    n, x = map(int, input().split())
    if n == 0 and x == 0:
        break

    count = 0
    for i in range(1, n + 1):
        for j in range(1, i):
            for k in range(1, j):
                sum = i + j + k
                if sum == x:
                    count += 1
                    # print(f"{i} {j} {k}")
    print(f"{count}")
