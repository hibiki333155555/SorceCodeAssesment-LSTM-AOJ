#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,count=0,minIdx;
	cin>>n;
	vector<int> v(n,0);
	for(int i=0;i<n;i++)
	{
		cin>>v[i];
	}
	
	for(int i=0;i<n-1;i++)
	{
		int min=v[i];
		 minIdx=i;
		 bool flag=false;
		for(int j=i;j<n;j++)
		{
			if(v[j]<min)
			{
				min=v[j];
				minIdx=j;
				flag=true;
			}
		}
		swap(v[i],v[minIdx]);
		if(flag==true)
		     count++;
		
	}
	for(int i=0;i<n;i++)
	{
		cout<<v[i]<<" ";
	}
	cout<<endl;
	cout<<count;
	return 0;
}
