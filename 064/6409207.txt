#include <iostream>
using namespace std;

// 배열의 요소를 차례대로 출력
void trace(int A[], int N)
{
    int i;
    for (i = 0; i < N; i++)
    {
        if (i > 0)
            cout << " "; // 요소 사이에 공백을 1개 출력합니다.
        cout << A[i];
    }
    cout << endl;
}

//  삽입 정렬(0-기반 배열)
void insertionSort(int A[], int N)
{
    int i, j, v;
    for (i = 1; i < N; i++)
    {
        v = A[i];
        j = i - 1;
        while (j >= 0 && A[j] > v)
        {
            A[j + 1] = A[j];
            j--;
        }
        A[j + 1] = v;
        trace(A, N);
    }
}

int main()
{
    int i, N;

    cin >> N;
    int *A = new int[N];

    for (i = 0; i < N; i++)
        cin >> A[i];

    trace(A, N);
    insertionSort(A, N);

    return 0;
}
