#include <iostream>
#include <string>

using namespace std;
int main()
{
    string W;
    string T;
    cin >> W;
    for(int i = 0; i < W.size(); i++)
    {
        if(W[i] >= 'A' && W[i] <= 'Z') W[i] += 'a' - 'A';
    }
    int x = 0;
    cin >> T;
    while (T != "END_OF_TEXT");
    {
        for(int j = 0; j < T.size(); j++)
        {
            if(T[j] >= 'A' && T[j] <= 'Z') T[j] += 'a' - 'A';
        }
    if(W == T) x++;
    cin >> T;
    }
    cout << x << endl;


    return 0;
}
