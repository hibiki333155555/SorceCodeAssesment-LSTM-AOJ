#include <bits/stdc++.h>
using namespace std;
int main(){
    int N, K;
    cin >> N >> K;
    vector<int> A(N);
    int sum = 0;
    priority_queue<int> q;
    for(int i = 0; i < N; i++){
        cin >> A[i];
        sum += A[i];
        if(A[i] == 1 || A[i] == 6) q.push(5 - A[i]);
        else q.push(6 - A[i]);
    }
    while(K-- && !q.empty()){
        if(q.top() < 0) break;
        sum += q.top();
        q.pop();
    }
    cout << sum << endl;
}
