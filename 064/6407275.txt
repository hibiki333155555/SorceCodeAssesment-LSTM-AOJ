#include<bits/stdc++.h>
using namespace std;
using ll=long long;
int main(){
  ll n,t;
  cin>>n>>t;
  ll k=sqrtl(n);
  ll q=(n+k-1)/k;
  vector<ll> a(n,0);
  vector<ll> b(q,0);
  while(t--){
    ll s;
    cin>>s;
    if(s==0){
      ll x,y;
      cin>>x>>y;
      x--;
      a[x]+=y;
      b[x/k]+=y;
    }
    if(s==1){
      ll x,y;
      cin>>x>>y;
      x--;
      y--;
      ll ans=0;
      if((ll)(x/k)==(ll)(y/k)){
        for(ll i=x;i<=y;i++){
          ans+=a[i];
        }
        cout<<ans<<endl;
        continue;
      }
      bool g=0;
      for(ll i=x;i%k;i++){
        ans+=a[i];
        x=i;
        g=1;
      }
      if(g) x++;
      for(ll i=y;i%k!=k-1;i--){
        ans+=a[i];
        y=i;
      }
      y++;
      for(ll i=x/k;i<=y/k-1;i++){
        ans+=b[i];
      }
      cout<<ans<<endl;
    }
  }
}
