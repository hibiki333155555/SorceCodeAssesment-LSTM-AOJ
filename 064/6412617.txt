#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,m;cin>>n>>m;
	vector<vector<int>> v(n+1,vector<int>(m+1,0));
	for(int i=0;i<n;i++){
		int s=0;
		for(int j=0;j<m;j++){
			cin>>v[i][j];
			s+=v[i][j];
		}
		v[i][m]=s;
	}
	for(int i=0;i<m+1;i++){
		int s=0;
		for(int j=0;j<n;j++){
			s += v[j][i];
		}
		v[n][i]=s;
	}
	for(int i=0;i<n+1;i++){
		for(int j=0;j<m+1;j++){
			cout<<v[i][j]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
