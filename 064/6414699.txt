#include<stdio.h>

int main(void)
{
    int count[26];
    int num,i;
    char ch;
    while(1)
    {
        scanf("%c",&ch);
        if(ch=='\n')break;
    		else if('a'<=ch&&ch<='z')
    		{
                num=ch-'a';
                count[num]++;
    		}
    }
    for(i=0;i<26;i++)
    {
        printf("%c : %d\n",'a'+i,count[i]);
    }
    return 0;
}
