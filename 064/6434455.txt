#include<iostream>
#include<stdio.h>
using namespace std;

    
void bubbleSort(int N,int A[]){
        bool f=true;
        int cnt=0;
        
        while(f){
            f=false;
            for(int i=N-1;i>0;i--)
                if(A[i-1] > A[i]){
                    swap(A[i-1],A[i]);
                    f=true;
            cnt++;
                }

        }
        
        for(int i=0;i<N;i++){
            if(i!=0)
                printf(" ");
            printf("%d",A[i]);
        }
        
        printf("\n");
        printf("%d\n",cnt);
        
    }
    
    int main(){
        int N;
        int A[100] = {};
        
        scanf("%d",&N);
        for(int i=0;i<N;i++)
            scanf("%d",&A[i]);
        
        bubbleSort(N,A);
        
        return 0;
        
    }
