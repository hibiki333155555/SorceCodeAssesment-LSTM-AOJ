#include <iostream>
#include <vector>
#include <algorithm>

int stime_to_seconds(std::string t) {
    return std::stoi(t.substr(0, 2)) * 3600 + std::stoi(t.substr(3, 2)) * 60 + std::stoi(t.substr(6, 2));
}

int main() {
    while (1) {
        int n; std::cin >> n;
        std::vector<int> S(3600*24);

        if (n == 0) break;

        for (int i = 0; i < n; i++) {
            std::string s, t;
            std::cin >> s >> t;
            S[stime_to_seconds(s)]++;
            S[stime_to_seconds(t)]--;
        }

        for (int i = 0; i < 3600*24; i++) {
            S[i+1] += S[i];
        }

        std::cout << *std::max_element(S.begin(), S.end()) << std::endl;;
    }
}
