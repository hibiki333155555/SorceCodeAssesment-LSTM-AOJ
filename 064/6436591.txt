#include <iostream>
#include <limits>
#include <vector>


int main()
{
    std::vector<int> R;
    
    int N = 0;
    std::cin >> N;

    R.resize(N);

    for(int i = 0; i < N; ++i)
    {
        std::cin >> R[i];
    }

    long long maxim  = -9223372036854775808LL;
    long long minim = R[0];
    for(int i = 1; i < N; ++i)
    {
        long long candidate = R[i] - minim;
        maxim = candidate > maxim ? candidate : maxim;
        minim = R[i] < minim ? R[i] : minim;
    }

    std::cout << maxim << std::endl;
}
