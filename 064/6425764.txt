using namespace std;
bool check(const string &Y,const string &n){
  int cur=0;
for (int i = 0; i < (int)Y.size(); i++)
{
  if(Y[i]==n[cur])cur++;
  if(cur==(int)n.size())return true;
}
return false;
}
int main() {
int Q;
cin >>Q;
for (int q = 0; q < Q; q++)
{
  string X,Y;
  cin >>X>>Y;
  map<string,bool>memo;
  memo[""]=true;
  int i=0;
  for (i = 0; i < (int)X.size(); i++){
     map<string,bool>memo_temp;
    for (auto temp:memo){
      memo_temp[temp.first]=true;
        string n=temp.first+X[i];
        if(check(Y,n)==true){
          memo_temp[n]=true;
        }
    }
    memo=memo_temp;
  }
int ret=0;
for (auto temp:memo){
ret=max(ret,(int)temp.first.size());
}
  
  cout << ret << endl;
}

return 0;
}
