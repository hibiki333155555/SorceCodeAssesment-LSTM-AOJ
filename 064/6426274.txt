#include <iostream>
#include <vector>

using namespace std;
int main()
{
    int n;
    cin >> n;
    
    vector<vector<int > > v(n);
    int q;
    int t;
    int x;
    
    int cmd;
    cin >> cmd;
    for(int i = 0; i < cmd; i++)
    {
        cin >> q;
        switch(q)
        {
            case 0:
                // pushback(t, x)
                cin >> t >> x;
                v[t].push_back(x);
                break;
            case 1:
                // dump(t)
                cin >> t; 
                
                for(int j = 0; j < v[t].size(); j++) 
                {
                    if(j) 
                    {
                        cout << " ";
                    }
                    cout << v[t][j];
                }
                cout << endl;
                break;
            case 2:
                // clear(t)
                cin >> t;
                if(!v[t].empty())
                {
                    v[t].clear();
                }
                break;
        }
    }
    return 0;
}
