#include<stdio.h>

int *Sort(int *data);

int main(void){
    int a,b,c;

    scanf("%d %d %d",&a,&b,&c);

    int nums[3] = {a,b,c};
    int *data_nums = Sort(nums);

    printf("%d %d %d\n",data_nums[0],data_nums[1],data_nums[2]);
}

int *Sort(int *data){
    static int Sort[3];
    int empty;

    if(data[0]>data[2]){
        empty = data[0];
        data[0] = data[2];
        data[2] = empty;
    }
    if(data[1]<data[0]){
        empty = data[1];
        data[1] = data[0];
        data[0] = empty;
    }else if(data[1]>data[2]){
        empty = data[1];
        data[1] = data[2];
        data[2] = empty;
    }

    Sort[0] = data[0];
    Sort[1] = data[1];
    Sort[2] = data[2];

    return Sort;
}
