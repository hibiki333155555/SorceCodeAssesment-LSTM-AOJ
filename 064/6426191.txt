#include<stdio.h>

int main(){
    int max, min, sum, n, tem;
    
    scanf("%d",&n);
    
    for(int i = 0; i < n; i++){
        scanf("%d",&tem);
        if(i = 0)max = min = tem;
        sum += tem;
        if(max < tem)max = tem;
        if(min > tem)min = tem;
    }
    printf("%d %d %d\n", min, max, sum);
    return 0;
}
