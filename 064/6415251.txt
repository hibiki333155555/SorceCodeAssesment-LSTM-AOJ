long long modpow(long long a, long long b, long long m) {
    long long p = 1, q = a;
    for (int i = 0; i < 63; i++) {
        if ((b & (1LL << i)) != 0) {
            p *= q;
            p %= m;
        }
        q *= q;
        q %= m;
    }
    return p;
}

int main() {
  long long N,K;
  cin >> N >> K;
  if(K==1){
    cout << 1 << endl;
    return 0;
  }else if(K>2){
    cout << modpow(2LL,N,MOD) << endl;
  }else{
    cout << (modpow(2LL,N,MOD)-1+MOD)%MOD << endl;
  }
}
