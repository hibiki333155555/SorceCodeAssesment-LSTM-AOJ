#include<stdio.h>
#include<math.h>
#include<iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	int mass[4][13];
	int num;
	char des;
	for (int i = 0; i < 4; ++i) {
		for (int j = 0; j < 13; ++j) {
			mass[i][j] = 0;
		}
	}
	for (int i = 0; i < n; ++i) {
		cin >> des >> num;
		if (des == 'S') {
			mass[0][num - 1] = 1;
		}
		if (des == 'H') {
			mass[1][num - 1] = 1;
		}
		if (des == 'C') {
			mass[2][num - 1] = 1;
		}
		if (des == 'D') {
			mass[3][num - 1] = 1;
		}
	}
	for (int i = 0; i < 4; ++i) {
		for (int j = 0; j < 13; ++j) {
			if (mass[i][j] == 0) {
				if (i == 0) {
					cout << 'S ' << j + 1 << endl;

				}
				if (i == 1) {
					cout << 'H ' << j + 1 << endl;

				}
				if (i == 2) {
					cout << 'C ' << j + 1 << endl;

				}
				if (i == 3) {
					cout << 'D ' << j + 1 << endl;

				}
			}
		}
	}
}
