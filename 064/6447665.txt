#include <iostream>
#include <bits/stdc++.h>
#include <string>
using namespace std;

int main() {
   string t;

   while(true) {
      cin >> t;

      for(int i=0; i<t.size()-1; i++) {
         char txt = t[i];
         if(islower(txt)) {
            cout << toupper(txt);
         }else if(isupper(txt)) {
            cout << tolower(txt);
         }
      }
      if(t.back() == '.'){
         cout << ".\n";
         break;
      }else if(t.back() == ',') cout << ", ";
      else cout << " ";
   }
}
