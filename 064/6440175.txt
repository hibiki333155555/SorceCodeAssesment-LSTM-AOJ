#include <stdio.h>
#include <limits.h>

struct work
{
	char *name;
	int time;
	int divq;
	int extq;
	
}	;

int	ft_isspace(int c)
{
	return (('\t' <= c && c <= '\r') || c == ' ');
}

int	ft_atoi(const char *str)
{
	int			minus;
	long		number;

	number = 0;
	minus = 0;
	while (ft_isspace(*str))
		str++;
	if (*str == '+' || *str == '-')
		minus = (*str++ == '-');
	while ('0' <= *str && *str <= '9')
	{
		number = 10 * number + (*str++ - '0');
		if (number < LONG_MIN || LONG_MAX < number)
		{
			if (minus)
				number = 0;
			else
				number = -1;
			return (number);
		}
	}
	if (minus)
		number = -number;
	return ((int)number);
}

void qw (struct work *a, struct work *res, int n)
{
	int i;
	int j;

	i = -1;
	j = -1;
	while (++i < n)
	{
		if (a[i].divq == 0 && a[i].extq == 0 && ++j >= 0)
			res[j] = a[i];
		else if (a[j].divq == 0)
		{
			a[j].extq = 0;
			a[i + n] = a[i];
		}
		else
		{
			a[i].divq -= 1;
			a[i + n] = a[i];
		}
	}
}

int main(void)
{
	int i;
	int j;
	int g[200] = {-1000000006};
	char c[200][11];
	int n;
	int q;
	int swap;
	struct work a[1000];
	struct work res[1000];

	scanf("%d%d", &n, &q);
	i = -1;
	j = -1;
	while (++i < n)
	{
		scanf("%s", c[i]);
		if ((i % 2) == 0)
			a[j].name = c[i];
		else
		{
			a[j].time = ft_atoi(c[i]);
			a[j].divq = a[j].time / q;
			a[j].extq = a[j].time % q;
			++j;
		}
	}
	qw(a, res, n);
	i = -1;
	while (++i < n)
		printf("%s %d\n", a[i].name, a[i].time);
	return (0);

}
