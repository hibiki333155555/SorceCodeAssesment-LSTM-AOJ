#include<stdio.h>
#include<string.h>

int main(void){
    int i,num,m,count,counter,number;
    char str[201],sub[201];
    
    scanf("%s",str);
    
    counter = strlen(str);
    
    while(str != '-'){
        scanf("%d",m);
        
        for(i = 0;i < m;i++){
            scanf("%d",num);
            
            number = 0;
            
            //シャッフルして変わらない場所をsubに入れる
            for(count = num;count <= counter;count++){
                sub[number] = str[num];
                
                number++;
            }
            
            //変わる場所をsubに入れる
            for(count = 0;count < num;count++){
                sub[number] = str[count];
            }
        }
        
        printf("%s\n",sub);
    }
    
    return 0;
}
