#include <iostream>
using namespace std;

int main(){
    int a, b;
    int ans;
    char op;
    
   for(int i = 0; i < 1;){
       cin >> a >> op >> b;
       if(op == '?')break;
        switch(op){
            case '+': 
                ans = a + b;
                break;
            case '-':
                ans = a - b;
                break;
            case '*':
                ans = a * b;
                break;
            case '/':
                ans = a / b;
                break;
        }
        cout << ans << endl;
    }
    
    return 0;
}
