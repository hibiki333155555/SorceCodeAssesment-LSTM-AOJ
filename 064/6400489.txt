#include <stdio.h>

int main() {
    char String[1200];
    
    gets_s(String);
    
    for(int i=0;String[i]!='\0';++i) {
        if(String[i]>='A' && String[i]<='Z') {
            String[i] += 32;
        }
        else if(String[i]>='a' && String[i]<='z') {
            String[i] -= 32;
        }
    }
    
    printf("%s\n", String);
      
    return 0;
}
