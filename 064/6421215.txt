int main() {
  string s;
  cin >> s;
  int u=0;
  string ans;
  stack<int> h;
  rep(i,s.size()){
      if (s[i]=='(') continue;
      if (s[i]-'0'<=9 && s[i]-'0'>=0){
          u+=s[i]-'0';
          h.push(s[i]-'0');
      }
      else if (s[i]==')'){
          int f=h.top();
          h.pop();
          u-=f;
      }
      else{
          int f=(s[i]-'a')+u;
          f%=26;
          ans.push_back('a'+f);
      }
  }
  cout << ans << endl;
}
