    const int M = 220 * 1010 * 2;
    int BASE = M / 2;
    bitset<M> dp(M);
    dp[BASE] = 1;
    for(int i = 0; i < N; i++) {
        dp = (dp << A[i]) | (dp >> A[i]);
    }

    int ans = 1e9;
    for(int j = 0; j <= M; j++) if(dp[j]) ans = min(ans, abs(j - BASE));
    cout << ans << endl;
}
