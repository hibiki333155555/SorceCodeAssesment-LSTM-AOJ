#include<iostream>
using namespace std;
int a[100005],q,key,sum,n;
bool search(int a[],int n,int key)
{
	int i=0;
	a[n]=key;
	while(a[i]!=key)
	{
		i++;
		if(i==n) return 0;
	}
	if(i!=n) return 1;
	
}
int main()
{
	sum=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	scanf("%d",&q);
	for(int i=0;i<q;i++)
	{
		scanf("%d",&key);
		if(search(a , n ,key))
			sum++;
	}
	cout<<sum;
	return 0;
 }
