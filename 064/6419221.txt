#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(){
    vector<int> vec(9);
    for(auto &it: vec) cin >> it;
    int ans = 0;
    vector<int> pat = {1, 2, 3, 4, 5, 6, 7, 8, 9};
    auto f = [](const vector<int> &v){
        int res = v[0] + v[1]*10+v[2] + v[3]*100+v[4]*10+v[5];
        return res == v[6]*100+v[7]*10+v[8];
    };
    do{
        bool isok = true;
        for(int i = 0; i < 9; i++){
            if(pat[i] != vec[i] && vec[i] != -1) isok = false;
        }
        if(isok && f(pat)) ans++;
    }while(next_permutation(pat.begin(), pat.end()));
    cout << ans << endl;
    return 0;
}
