#include<bits/stdc++.h>
using namespace std;
int main(){
    int x,a,b; cin>>a>>b>>x;
    double a1,b1;
    a1/=10;
    b1/=5;
    int ans1,ans2;
    if(x%1000==0){
        ans1=a*(x/1000);
    }else if(a*(x/1000)+a<a*(x/1000)+b){
        ans1=a*(x/1000)+a;
    }else{
        ans1=a*(x/1000)+b;
    }
    if(x%500==0){
        ans2=b*(x/500);
    }else if(b*(x/500)+a<b*(x/500)+b){
        ans2=b*(x/500)+a;
    }else{
        ans2=b*(x/500)+b;
    }
    cout<<min(ans1,ans2)<<endl;
    return 0;
}
