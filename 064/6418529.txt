#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdio>
#include <cmath>
#include <iomanip>
using namespace std;
const int N=2000,M=1e5+5;
int n,m;
int f[M],a,b,s;
int v[N],w[N];
int main()
{
    cin>>n>>m;
	int cnt=0;
	for(int i=1;i<=n;i++)
	{
		cin>>a>>b>>s;
		int k=1;
		while(k<=s)
		{
			cnt++;
			v[cnt]=b*k;
			w[cnt]=a*k;
			s-=k;
			k*2;
		}
		if(s>0)
		{
			cnt++;
			v[cnt]=b*s;
			w[cnt]=a*s;
		}
	}
	for(int i=1;i<=cnt;i++)
	{
		for(int j=m;j>=v[i];j--)
		{
			f[j]=max(f[j],f[j-v[i]]+w[i]);
		 } 
	}
	cout<<f[m]<<endl;
    return 0;
}
