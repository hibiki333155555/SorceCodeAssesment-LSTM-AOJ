#include <bits/stdc++.h>
using namespace std;


int main() {
    int n, x;
    while(true){
        cin >> n >> x;
        if(x==0 && n==0)break;

        int cou = 0;
        for(int i=1;i<=n-2;++i){
            for(int j=i+1;j<=n-1;++j){
                for(int k=j+1;k<=n;++k){
                    if(i+j+k == x) ++cou;
                }
            }
        }

        cout << cou << endl;
    }
	return 0;
}
