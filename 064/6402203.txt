#include <bits/stdc++.h>
using namespace std;

int main(){
  int N,W;
  cin >> N >> W;
  vector<int> v(N),w(N);
  for(int i=0; i<N; i++) cin >> v.at(i) >> w.at(i);
  
  vector<vector<int>> dp(N+1,vector<int>(W+1));
  for(int i=0; i<N; i++){
    for(int j=0; j<=W; j++){
      if(j<w.at(i)){
        dp.at(i+1).at(j) = dp.at(i).at(j);
      }else{
        int m=dp.at(i).at(j);
        for(int k=1; k*w.at(i)<=j; k++){
          m = max(m, dp.at(i).at(j-k*w.at(i)) + k*v.at(i));
        }
        dp.at(i+1).at(j) = m;
      }
    }
  }
  
  cout << dp.at(N).at(W) << endl;

  return 0;
}
