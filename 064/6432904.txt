#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

vector<vector<int>> G;

int main(){
    int n;
    cin >> n;

    G.resize(n);
    rep(i,n){
        int u, k;
        cin >> u >> k;
        rep(j,k){
            int v;
            cin >> v;
            G[u-1].push_back(v-1);
        }
    }

    vector<int> d(n, -1);
    d[0] = 0;

    queue<int> q;
    q.push(0);

    while(!q.empty()){
        int cu = q.front();
        q.pop();

        for(int to : G[cu]){
            if(d[to] != -1) continue;
            d[to] = d[cu] + 1;
            q.push(to);
        }
    }

    rep(i,n) cout << i+1 << " " << d[i] << endl;

    return 0;
}
