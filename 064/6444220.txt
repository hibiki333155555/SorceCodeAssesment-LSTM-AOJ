n = int(input())
count = 0
apartment = [[[0 for k in range(10)]for j in range(3)]for i in range(4)]

for h in range(n):
    b, f, r, v = map(int, input().split())
    
    if apartment[b-1][f-1][r-1] == 0:
        apartment[b-1][f-1][r-1] = v
    else:
        apartment[b-1][f-1][r-1] += v
    

for i in range(4):
    for j in range(3):
        for k in range(10):
            print(' ',{apartment[i][j][k]},end='')
        print()
    print('#'*20)

print()
