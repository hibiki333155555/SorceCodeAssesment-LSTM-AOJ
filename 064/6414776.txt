#include <bits/stdc++.h>
#include <atcoder/modint>
using namespace std;
using Modint = atcoder::modint998244353;


Modint solve(int N, int K){
    if(K == 1) return 1;
    if(K == 2) return N + 1;
    return Modint(2).pow(N);
}
int main(){
    int N, K;
    cin >> N >> K;
    cout << solve(N, K).val() << endl;
}
