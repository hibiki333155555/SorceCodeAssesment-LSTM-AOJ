#include<bits/stdc++.h>
using namespace std;

int n, u, k, v, cnt=0;
vector<int> d(110, 0), f(110, 0);
vector<bool> seen(110, 0);
vector<vector<int> > g(110);

void solve(int x){
  int x_k = g[x].size();
  cnt++;
  d[x]=cnt;
  seen[x]=1;
  for(int i=0;i<x_k;i++){
    if(seen[g[x][i]])continue;
    solve(g[x][i]);
  }
  cnt++;
  f[x]=cnt;
}

int main(){
  cin >> n;
  for(int i=0;i<n;i++){
    cin >> u >> k;
    u--;
    for(int j=0;j<k;j++){
      cin >> v;
      v--;
      g[u].push_back(v);
    }
  }
  
  for(int i=0;i<n;i++){
    if(seen[i])continue;
    solve(i);
  }
  
  for(int i=0;i<n;i++){
    cout << i+1 << " " << d[i] << " " << f[i] << endl;
  }
  
  return 0;
}
