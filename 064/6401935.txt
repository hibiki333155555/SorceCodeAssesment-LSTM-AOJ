#include<stdio.h>

int main(){
    int A,B,C;
    scanf("%d %d %d",&A,&B,&C);

    if(A>=B && B>=C){
        printf("%d %d %d\n",C,B,A);
    }

    else if(B>=C && C>=A){
        printf("%d %d %d\n",A,C,B);
    }

    else if(C>=B && B>=A){
        printf("%d %d %d\n",A,B,C);
    }

    else if(B>=A && A>=C){
        printf("%d %d %d\n",C,A,B);
    }

    else if(C>=A && A>=B){
        printf("%d %d %d\n",B,A,C);
    }

    else if(A>=C && C>=B){
        printf("%d %d %d",B,C,A);
    }
    
    return 0;
}
