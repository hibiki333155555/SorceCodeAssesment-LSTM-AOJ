#include<bits/stdc++.h>
namespace {
using namespace std;
#define rep(i,n) for(int i = 0; i < (int)(n); i++)
#define rrep(i,n) for(int i = (int)(n) - 1; i >= 0; i--)
#define all(x) begin(x), end(x)
#define rall(x) rbegin(x), rend(x)
template<class T> bool chmax(T& a, const T& b) { if (a < b) { a = b; return true; } else return false; }
template<class T> bool chmin(T& a, const T& b) { if (b < a) { a = b; return true; } else return false; }
using ll = long long;
using P = pair<int,int>;
using VI = vector<int>;
using VVI = vector<VI>;
using VL = vector<ll>;
using VVL = vector<VL>;

} int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
  int n, m;
  cin >> n >> m;
  vector<string> s(n);
  rep(i, n) cin >> s[i];
  int ans = 1001001001;
  int xtot = 0;
  rep(i, n) rep(j, m) xtot += s[i][j] == 'x';
  rep(_, 2) {
    rep(_, 2) {
      int score = xtot;
      rep(j, m - 1) {
        rep(i, n) {
          if (s[i][j] == 'x') score--;
          else score++;
        }
        chmin(ans, score);
      }
      for(auto& si: s) reverse(all(si));
    }
    vector<string> ns(m, string(n, '.'));
    rep(i, n) rep(j, m) ns[j][i] = s[i][j];
    swap(n, m);
    swap(s, ns);
  }
  cout << ans << '\n';
}
