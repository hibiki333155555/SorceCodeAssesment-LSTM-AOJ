    ll modpow(ll x, ll b,ll mod=MOD){
        ll res = 1;
        while(b){
            if(b&1)res = res * x % mod;
            x = x * x % mod;
            b>>=1;
        }
        return res;
    }

    ll modinv(ll x){
        return modpow(x, MOD-2);
    }

using namespace templates;

double func(int n){
    double pw = in();
    struct Plant{
        double vw;
        double pf;
        double vf;
        double th;
        double required(double w){
            return max<double>(0,th-vw*w) / vf * pf;
        }
    };
    vector<Plant> Plants(n);
    rep(i,n){
        Plants[i].vw = in();
        Plants[i].pf = in();
        Plants[i].vf = in();
        Plants[i].th = in();
    }

    double sum = 0;
    double dw = pw;
    foreach(i,Plants)sum += i.required(0);

    foreach(i,Plants)if(0 < i.th){
        dw -= i.vw / i.vf * i.pf;
    }
    using P = pair<double,double>;
    vector<P> events;
    foreach(i,Plants){
        if(i.th > 0){
            if(i.vw > 0){
                events.emplace_back(i.vw / i.vf * i.pf, i.th / i.vw);
            }
        }else{
            if(i.vw < 0){
                events.emplace_back(-i.vw / i.vf * i.pf,i.th / i.vw);
            }
        }
    }
    sort(all(events),lambda(bool,P a,P b){return a.second < b.second;});
    double last = 0;
    double res = sum;
    foreach(e,events){
        double adw = e.first;
        double current = e.second;
        sum += (current - last) * dw;
        chmin(res,sum);
        dw += adw;
        last = current;
    }
    return res;
}

int main(){

    while(true){
        int n = in();
        if(n==0)break;
        println(func(n));
    }

    return 0;
}
