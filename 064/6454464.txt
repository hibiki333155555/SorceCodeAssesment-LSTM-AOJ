#include <bits/stdc++.h>
using namespace std;

int main(){
  string s;
  stack<int> a;
  getline(cin, s);
  for(char x: s){
    if(x != '+' && x != '-' && x != '*' && x != ' '){
      int k = int(x - '0');
      a.push(k);
    }
    else if(x == '+'){
      int s, t;
      s = a.top();
      a.pop();
      t = a.top();
      a.push(s+t);
    }
    else if(x == '-'){
      int s, t;
      s = a.top();
      a.pop();
      t = a.top();
      a.push(t-s);
    }
    else if(x == '*'){
      int s, t;
      s = a.top();
      a.pop();
      t = a.top();
      a.push(s*t);
    }
  }
  cout << a.top() << endl;

  return 0;
}
