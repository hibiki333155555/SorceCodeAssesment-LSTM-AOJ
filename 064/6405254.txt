#include<stdio.h>

int main (void)
{
    int h,w;
    while(1)
    {
        scanf("%d %d",&h,&w);
        if(h==0&&w==0)break;
        for(;h>0;h--)
        {
            for(;w>0;w--)
            {
                printf("#");
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
