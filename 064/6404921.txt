#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
    int n;
    cin>>n;
    vector<ll> vec(n);
    ll sum=0;
    for(int i=0; i<n; i++){
        cin>>vec[i];
        sum+=vec[i];
    }
    sort(vec.begin(),vec.end());
    cout<< vec[0]<< " " << vec[vec.size()-1] << " "<<sum<<endl;
}
