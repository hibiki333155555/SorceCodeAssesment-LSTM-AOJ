#include <iostream>
using namespace std;

int main()
{
   string s;
   cin >> s;
   for (auto &alpha : s)
   {
      if (alpha == 'A')
         alpha = 'D';
      else if (alpha == 'B')
         alpha = 'E';
      else if (alpha == 'C')
         alpha = 'F';
      else
         alpha -= 3;
   }
   cout << s << endl;

   return 0;
}
