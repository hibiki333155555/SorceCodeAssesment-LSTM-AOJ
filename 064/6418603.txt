#include <iostream>
#include<queue>

using namespace std;
int dp[10000];

class pp{
public:
    int total;
    int pos;
};

int main(){
    int n,q;
    cin >> n;
    int A[n];
    for (int i = 0; i < n; i++){
        cin >> A[i];
    }
    
    pp p;
    p.pos = 0;
    p.total = 0;
    
    queue<pp>qu;
    qu.push(p);
    while(!qu.empty()){
        p = qu.front();
        qu.pop();
        
        if ( p.pos + 1 <= n){
            p.pos++;
            //dp[p.total] = 1;
            qu.push(p);
            p.total += A[p.pos - 1];
            dp[p.total] = 1;
            qu.push(p);
        }
    }
    
    cin >> q;
    for (int i = 0; i < q; i++){
        int num;
        cin >> num;
        
        if (dp[num]==1) {
            cout << "yes" << endl;
        }else{
            cout << "no" << endl;
        }
    }
}
