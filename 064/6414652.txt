#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <numeric>
#include <set>
#include <map>
#include <queue>
using namespace std;
using ll = long long;
const ll mod = 998244353;

int main(){
    int n;
    cin >> n;
    vector<int> a(n);
    for(int i=0; i<n; i++){
        cin >> a[i];
    }
    vector<int> b(n, 0);
    for(int i=0; i<n; i++){
        b[min(n-1, i+a[i]-1)]++;
    }
    ll ans = 1;
    ll wait = 0;
    for(int i=n-1; i>=0; i--){
        wait += b[i];
        ans = (ans * wait) % mod;
        wait--;
    }
    cout << ans << endl;
    return 0;
}
