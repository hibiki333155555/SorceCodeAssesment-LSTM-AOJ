#include <bits/stdc++.h>
using namespace std;

void bebbleSort(vector<int> &bs, int n) {
    int flag = 1;
    int sum = 0;
    while (flag) {
        flag = 0;
        for (int i = 1; i < n; i++) {
            if (bs.at(i) < bs.at(i-1)) {
                swap(bs.at(i), bs.at(i-1));
                flag = 1;
                sum ++;
            }
        }
    }
    for (int i = 0; i < n; i++) {
        if (i < n-1) cout << bs.at(i) << " ";
        else cout << bs.at(i) << endl;
    }
    cout << sum << endl;
}

int main() {
    int n;
    cin >> n;
    vector<int> bs(n);

    for (int i = 0; i < n; i++) {
        cin >> bs.at(i);
    }
    bebbleSort(bs, n);
}
