#include <iostream>
#include <string>
using namespace std;

int main(){
    int w, h; cin >> w >> h;
    string s; cin >> s;
    for(int i = 0; i < h; i++){
        if(i == 0 || i == h-1){
            cout << "+";
            for(int j = 1; j < w-1; j++) cout << "-";
            cout << "+";
        }else{
            cout << "|";
            for(int j = 1; j < w-1; j++){
                if(i == h/2 && j == w/2) cout << s;
                else cout << ".";
            }
            cout << "|";
        }
        cout << '\n';
    }
    return 0;
}
