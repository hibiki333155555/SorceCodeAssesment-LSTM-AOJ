#include<iostream>
#include<algorithm>
using namespace std;
static const int m = 200000000;
int main() {
    int r[m], n;
    cin >> n;
    for (int i = 0; i < n; i++)cin >> r[i];
    int minv = r[0];
    int maxv = -20000000;


    for (int j = 1; j < n; j++) {
        maxv = max(maxv, r[j] - minv);
        minv = min(minv, r[j]);


    }
    cout << maxv << endl;


}
