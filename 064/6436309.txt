#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int top; int s[1000];
void push(int x) {
    s[++top] = x;

}
int pop() {
    top--;
    return s[top+1];


}

int main() {
   
    char s[100];
    top = 0;
    int a, b;
    while (scanf("%d", s) != EOF) {
        if (s[0]=='+')
        {
            a = pop();
            b = pop();
            push(a + b);
        }
        else if (s[0] == '-')
        {
            a = pop();
            b = pop();
            push(a - b);
        }
        else if (s[0] == '*')
        {
            a = pop();
            b = pop();
            push(a * b);
        }
        else
        {
            push(int(s));
        }

    }
}
