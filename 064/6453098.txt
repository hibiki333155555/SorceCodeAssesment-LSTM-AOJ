#include <iostream>
#include <vector>
using namespace std;

int main() {
    string s,p; cin>>s; cin>>p;
    s = s + s;
    for(int i=0;i<s.size()-p.size();i++) {
        int j=0;
        while(j<p.size()) {
            if(s[i+j]!=p[j]) break;

            if(j==p.size()-1) {
                cout<<"Yes"<<endl;
                return 0;
            }
            j++;
        }
    }
    cout<<"No"<<endl;
    return 0;
}
