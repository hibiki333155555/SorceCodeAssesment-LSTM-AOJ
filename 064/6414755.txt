ll jyo(ll x,ll y,ll z){
  ll H=y; //ここから
       ll a=1,b=(x%z+z)%z,c=1;
       while(H>0){
         a*=2;
         if(H%a!=0){
           H-=a/2;
           c*=b;
           c%=z;
         }
        b*=b;
         b%=z;
      } //ここまで
 return c;
}

void solve(){
	int N;
	cin>>N;
	map<tuple<int,int,int>,int> m;
	vector<tuple<int,int,int>> order;
	int ind=0;
	rep(i,N+1) rep(j,N+1){
		int a=i,b=j;
		int c=N-a-b; 
		if(c<0) continue;
		m[{a,b,c}]=ind;
		order.push_back({a,b,c});
		ind++;
	}
	vector<vector<ll>> base(ind,vector<ll>(ind+1));
	ll R=jyo(N*(N-1),mod-2,mod);
	rep(i,ind){
		ll a,b,c;
		tie(a,b,c)=order[i];
		if(c==N){
			base[0][0]=1;
			continue;
		}
		base[i][ind]=-1;
		base[i][i]=-1;
		base[i][m[{a-2,b+2,c}]]+=(R*a*(a-1))%mod;
		base[i][m[{a,b-2,c+2}]]+=(R*b*(b-1))%mod;
		base[i][m[{a+2,b,c-2}]]+=(R*c*(c-1))%mod;
		base[i][m[{a-1,b,c+1}]]+=(R*a*b*2)%mod;
		base[i][m[{a+1,b-1,c}]]+=(R*b*c*2)%mod;
		base[i][m[{a,b+1,c-1}]]+=(R*a*c*2)%mod;
	}
	rep(i,ind){
		rep(j,ind+1) base[i][j]%=mod;//,cout<<base[i][j]<<" ";
		//cout<<endl;
	}
	rep(i,ind){
		for(int k=i;k<ind;k++){
			base[k][i]=(base[k][i]%mod+mod)%mod;
			if(base[k][i]!=0){
				swap(base[i],base[k]);
				swap(order[k],order[i]);
				m[{order[k]}]=k;
				m[{order[i]}]=i;
				break;
			}
		}
		ll r=jyo(base[i][i],mod-2,mod);
		rep(j,ind+1){
			base[i][j]=(base[i][j]*r)%mod;
		}
		rep(j,ind){
			if(i==j) continue;
			ll B=base[j][i]%mod;
			rep(k,ind+1){
				base[j][k]=(base[j][k]-(base[i][k]*B)%mod)%mod;
			}
		}
	}
	/*rep(i,ind){
		rep(j,ind+1) cout<<base[i][j]<<" ";
		cout<<endl;
	}*/
	int x=m[{N,0,0}];
	ll ans=base[x][ind];
	cout<<(ans%mod+mod)%mod<<"\n";
}
