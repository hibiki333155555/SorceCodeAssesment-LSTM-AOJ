let [n, inputArr] = require(`fs`).readFileSync(`/dev/stdin`, `utf8`).split(`\n`);
let arr = inputArr.split(` `).map(Number);
let swap = 0;
for (let i=0; i<+n; i++) {
    let minIndex = i;
    for (let j=i; j<+n; j++) {
        if(arr[minIndex] < arr[j]) minIndex = j;
        [arr[j], arr[minIndex]] = [arr[minIndex], arr[j]];
        swap++;
    }
}

console.log(`${arr.join(` `)}\n${swap}`);
