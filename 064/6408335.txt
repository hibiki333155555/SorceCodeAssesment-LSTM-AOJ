#include<iostream>
#include<vector>
using namespace std;
int main(){
	int N,M,Q;
	vector< int > b,a;
	int q[1001];
	cin>>N>>M>>Q;
	for(int i=0;i<M;i++){
		int y;
		cin>>y;
		a.push_back(y);
	}
	for(int i=0;i<Q;i++)cin>>q[i];
	for(int i=0;i<N;i++){
		b.push_back(i);
		//b[i].second=1;
	}
	int j=0;int m;
	for(int i=0;i<M;i++){
		m=N-i;
		if(a[i]%2){
			if(j>a[i])j=j-a[i];
			else j=m+j-a[i];
		}
		else {
			j=(j+a[i])%m;
		}cout<<j<<endl;
		b.erase(b.begin()+j);
		
	}
	for(int i=0;i<Q;i++)
	for(int j=0;j<m;j++){
		if(q[i]==b[j]){
			cout<<"1"<<endl;
			break;
		}
		else if(j+1==m)cout<<"0"<<endl;
	}
	return 0;
}
