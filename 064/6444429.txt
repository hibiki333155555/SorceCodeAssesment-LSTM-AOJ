#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int R[200009];
    for (int i = 0; i < n; i++)cin >> R[i];
    int minv = R[0];
    int maxv = R[1] - R[0];
    for (int i = 0; i < n-1; i++) {
        if (minv >= R[i])minv = R[i];
        for (int j = i + 1; j < n; j++) {
            if (maxv <= R[j] - minv)maxv = R[j] - minv;
        }
    }
    cout << maxv << endl;
    return 0;
}
