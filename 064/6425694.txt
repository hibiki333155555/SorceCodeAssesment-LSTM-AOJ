#include <bits/stdc++.h>

using namespace std;

int main() {
  int n;
  scanf("%d", &n);

  int A[n];
  for (int i = 0; i < n; i++) {
    scanf("%d", &A[i]);
  }

  bool flag = true;
  int count = 0;
  while (flag) {
    flag = false;
    for (int i = n - 1; i > 0; i--) {
      if (A[i] < A[i - 1]) {
        swap(A[i], A[i - 1]);
        count++;
        flag = true;
      }
    }
  }
  for (int i = 0; i < n - 1; i++) {
    cout << A[i] << ' ';
  }
  cout << A[n - 1] << endl;
  cout << count << endl;
}
