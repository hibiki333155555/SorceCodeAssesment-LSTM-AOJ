#include <bits/stdc++.h>
using namespace std;
const double pi = 3.1415926535897932384626533832795;

int main()
{
	int a, b; 
	string op;
	while (true){
		cin >> a >> op >> b; 
		if (op == "?"){
			break;
		}else if (op == "+"){
			cout << a + b << endl;
		}else if (op == "-"){
			cout << a - b << endl;
		}else if (op == "*"){
			cout << a * b << endl;
		}else if (op == "/"){
			cout << a / b << endl;
		}
	}
}
