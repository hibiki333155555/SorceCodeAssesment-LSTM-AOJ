#define rep(i, a, n) for (int i = (a); i < (n); ++i)
#define bit(n, k) ((n >> k) & 1)

class Program {
private:

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

public:

    Program() {
    }

    void solve() {
        int n, k; cin >> n >> k;
        string s; cin >> s;

        int r = 0;
        rep(i, 0, n-1) {
            if (s[i] == 'R') {
                r++;
            }
        }

        if (r == k) {
            cout << 'W' << endl;
        } else {
            cout << 'R' << endl;
        }
    }
};

int main() {

#ifdef DEV
    std::ifstream in("input");
    cin.rdbuf(in.rdbuf());

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();
#endif

    return 0;
}
