#include <iostream>
#include <cmath>

using namespace std;

bool IsPrimeNumber(int number)
{
    if (number == 2 || number == 5)
    {
        return true;
    }
    else if (number % 2 == 0)
    {
        return false;
    }
    else if (number % 5 == 0)
    {
        return false;
    }
    else
    {
        for (int i = 1; (2 * i + 1) <= sqrt(number); i++)
        {
            if (number % (2 * i + 1) == 0)
            {
                return false;
            }
        }
    }
    return true;
}

int main(void)
{
    int n;
    std::cin >> n;
    int prime_number_count = 0;
    for (int i = 0; i < n; i++)
    {
        int number;
        std::cin >> number;
        if (IsPrimeNumber(number) == true)
        {
            prime_number_count++;
        }
    }
    std::cout << prime_number_count << std::endl;
    return 0;
}
