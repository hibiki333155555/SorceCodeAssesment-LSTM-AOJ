# include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    vector<int> dp;
    for (int i = 0; i < n; ++i) {
        auto it = lower_bound(dp.begin(), dp.end(), a[i]);
        if (it != dp.end()) {
            *it = a[i];
        }
        else {
            dp.push_back(a[i]);
        }
    }
    
    cout << dp.size() << endl;
}
