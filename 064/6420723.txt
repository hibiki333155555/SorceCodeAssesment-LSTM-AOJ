// #pragma GCC target("avx2")
#pragma GCC optimize("O3")
// #pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int n;
    cin >> n;
    unordered_map<int, int> mp;
    for (int i = 0; i < n * n; ++i) {
        int a;
        cin >> a;
        if (!mp.count(a)) {
            mp[a] = i;
        }
    }
    int q;
    cin >> q;
    for (int _ = 0; _ < q; ++_) {
        int x;
        cin >> x;
        if (mp.count(x)) {
            cout << (mp[x] / n + 1) << ' ' << (mp[x] % n + 1) << '\n';
        }
        else {
            cout << -1 << '\n';
        }
    }
    return 0;
}
