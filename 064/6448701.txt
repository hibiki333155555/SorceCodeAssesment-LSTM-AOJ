#include<bits/stdc++.h>
using namespace std;

const int N = 500;
int n;
vector<long long> even;
vector<long long> odd;
long long x;

int main() {
	while(cin>>n && n) {
		even.clear();
		odd.clear();
		for(int i=0;i<n*(n+1)/2;i++) {
			cin>>x;
			if(x&1)
				odd.push_back(x);
			else 
				even.push_back(x);
		}
		sort(even.begin(),even.end());
		sort(odd.begin(),odd.end());

		long long a0 = sqrt((double)even[0]/(double)odd[0]*(double)even[1])+0.5;
		cout<<a0<<endl;
		for(int i=0;i<even.size();i++) {
			if(i) cout<<" ";
			cout<<even[i] / a0;
		}
		cout<<endl;
	}
}
