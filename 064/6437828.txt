#include <stdio.h>

int main(void) {
	int i, j, x, y, num1[3000], num2[3000];
	
	/*入力*/
	for (i = 0; i < 3000; i++) {
		scanf("%d %d", &x, &y);
		if (x == 0 && y == 0) {
			break;
		} else if (x > y) {
			num1[i] = y;
			num2[i] = x;
		} else {
			num1[i] = x;
			num2[i] = y;
		}
	}
	
	/*出力*/
	for (j = 0; j < i; j++) {
		printf("%d %d\n", num1[j], num2[j]);
	}
	
	return 0;
}
