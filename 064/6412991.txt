#include<stdio.h>
#include<string.h>
#include<ctype.h>

int main(){
    
    char W[10],T[1000];
    int i,n;
    
    scanf("%s",&W);
    
    n=0;
    
    while(strcmp(T,"END_OF_TEXT") == 0){
        for(i=0;i<1000;i++){
            if(T[i] >= 'A' && T[i] <= 'Z')T[i]=tolower(T[i]);
        }
        if(strcmp(W,T)==0)n++;
    }
    
    printf("%d\n", n);
    
    return 0;
}
