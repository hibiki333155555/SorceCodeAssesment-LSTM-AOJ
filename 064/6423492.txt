#include <iostream>
#include <vector>
#include <queue>
using namespace std;

int n, m, q, id = 1;
vector<int> ls[100010];
bool vis[100010][100010];

bool bfs(int u, int v)
{
	id++;
	queue<int> q;
	q.push(u);
	while(!q.empty())
	{
		int x = q.front(); q.pop();
		if(x == v) return true;
		if(vis[id][x]) continue;
		vis[id][x] = true;
		for(int i = 0; i < ls[x].size(); i++) q.push(ls[x][i]);
	}
	return false;
}

int main()
{
	cin >> n >> m;
	for(int i = 0; i < m; i++)
	{
		int u, v;
		cin >> u >> v;
		ls[u].push_back(v);
		ls[v].push_back(u);
	}
	cin >> q;
	for(int i = 0; i < q; i++)
	{
		int u, v;
		cin >> u >> v;
		if(bfs(u, v)) cout << "yes" << endl;
		else cout << "no" << endl;
	}
	return 0;
}
