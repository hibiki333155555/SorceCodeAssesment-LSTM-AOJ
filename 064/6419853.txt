#include <bits/stdc++.h>
using namespace std;

int main(){
    double a, b, c;
    cin>>a>>b>>c;
    cout<<fixed<<setprecision(10);
    double S = (a*b*sin(c))/2;
    double L = sqrt(a*a+b*b-2*a*b*cos(c))+a+b;
    double h = 2*S/a;
    cout<<S<<endl;
    cout<<L<<endl;
    cout<<h<<endl;
    return 0;
}
