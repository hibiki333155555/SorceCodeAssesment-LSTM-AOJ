#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef unsigned long long ul;
typedef long double ld;
#define rep(i, n) for (int i = 0; i < n; i++)

int main() {
  ll r0, w0, c, r;
  ll reslut = -1;
  ll count = 0;
  while (cin >> r0 >> w0 >> c >> r, r0 && w0 && c && r) {
    while (reslut < 0) {
      reslut = r0 + count * r - w0 * c;
      count++;
    }
    reslut = -1;
    cout << count - 1 << endl;
    count = 0;
  }
}
