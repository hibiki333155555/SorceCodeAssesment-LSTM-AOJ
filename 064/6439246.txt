#include <iostream>
#include <stdio.h>
#include <cctype>
using namespace std;

int main(void){
    char ch[1200];

    for (int i=0; i<1200; i++) {
        if (isalpha(ch[i])) {
            if (islower(ch[i])) {
                ch[i] = toupper(ch[i]);
            } else {
                ch[i] = tolower(ch[i]);
            }
        }
    }

    for (int i=0; i<1200; i++) cout << ch[i];
    cout << endl;

    return 0;
}
