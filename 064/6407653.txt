#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
#include <string>
using namespace std;

auto f = [](auto& e)->char
{
    if(islower(e))
        return toupper(e);
    else if(isupper(e))
        return tolower(e);
    else
        return e;
};

int main()
{
    string str;
    getline(cin, str);
    vector<char> vec(str.length());
    

    transform(str.begin(), str.end(), vec.begin(), f);
    copy(vec.begin(), vec.end(), ostream_iterator<char>(cout));
    cout << endl;

    return 0;
}
