#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin>> n;
    vector<int> vecS(13);
    vector<int> vecH(13);
    vector<int> vecC(13);
    vector<int> vecD(13);
    for(int i=0; i<n; i++){
        string s;
        int a;
        cin>>s>>a;
        if(s=="S"){
            vecS.at(a-1)=a;
        }
        else if(s=="H"){
            vecH.at(a-1)=a;
        }
        else if(s=="C"){
            vecC.at(a-1)=a;
        }
        else if(s=="D"){
            vecD.at(a-1)=a;
        }
    }
    for(int i=0; i<13; i++){
        if(vecS.at(i)==0){
            cout<< "S" << " " <<i+1<<endl;
        }
    }
    for(int i=0; i<13; i++){
        if(vecH.at(i)==0){
            cout<< "H" << " " <<i+1<<endl;
        }        
    }
    for(int i=0; i<13; i++){
        if(vecC.at(i)==0){
            cout<< "C" << " " <<i+1<<endl;
        }
    }
    for(int i=0; i<13; i++){
        if(vecD.at(i)==0){
            cout<< "D" << " " <<i+1<<endl;
        }
    }
}
