using System;
class program
{
    static void Main()
    {
        var s = int.Parse(Console.ReadLine());
        var min = s[1];
        var max = s[1];
        int sum = 0;
        for(int i = 0; i < s[0]; i++)
        {
            if(min > s[i])
            {
                min = s[i];
            }
            if(max < s[i])
            {
                max = s[i];
            }
            sum = sum + s[i];
        }
        Console.WriteLine(min + " " + max + " " + sum);
    }
}
