#include <bits/stdc++.h>
using namespace std;
int main(){
  int N;
  cin >> N;
  string S;
  cin >> S;
  vector<int> A(N);
  for (int i = 0; i < N; i++){
    cin >> A[i];
  }
  long long ans = 0;
  for (int i = 0; i < N; i++){
    if (S[i] == '('){
      ans += A[i];
      A[i] *= -1;
    }
  }
  priority_queue<int, vector<int>, greater<int>> pq;
  for (int i = 0; i < N; i++){
    pq.push(A[i]);
    if (i % 2 == 0){
      ans += pq.top();
      pq.pop();      
    }
  }
  cout << ans << endl;
}
