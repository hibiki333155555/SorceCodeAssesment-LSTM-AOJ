#include <bits/stdc++.h>
using namespace std;

int main() {

	while (1) {
		char c = '#';
		int h, w;
		cin >> h >> w;

		if (h == 0 && w == 0) {
			break;
		}

		for (int i = 0; i < h; i++) {
			if (i % 2 == 0) {
				c = '#';
			} else if (i % 2 == 1) {
				c = '.';
			}
			for (int j = 0; j < w; j++) {
				cout << c;
				if (c == '#') {
					c = '.';
				} else {
					c = '#';
				}
			}
			putchar('\n');
		}
		putchar('\n');
	}

	return (0);
}
