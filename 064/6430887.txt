#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,z;
    cin>>n;
    vector<int> a;
    for(int i=0;i<n;i++){
        cin>>z;
        a.push_back(z);
    }
    for(int i=0;i<n-1;i++){
        cout<<a[i]<<" ";
    }
    cout<<a[n-1]<<endl;
    for(int i=1;i<n;i++){
        int j=i-1;
        int temp=a[i];
        while(j>=0 && a[j]>temp){
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=temp;
        for(int i=0;i<n-1;i++){
            cout<<a[i]<<" ";
        }
        cout<<a[n]<<endl;
    }
}
