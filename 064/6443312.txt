#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int check(int x, int n) {
   if(++x <= n) return false;
}

int main() {
   int n, i=1, flag=true;
   cin >> n;

   while(flag) {
      int x = i;

      if(x % 3 ==0) {
         cout << " " << i;
         flag = check(x, n);
      }

      while(x) {
         if(x % 10 == 3) {
            cout << " " << i;
            flag = check(x, n);
         }
         x /= 10;
      }
      flag = check(x, n);
      i++;
   }

}
