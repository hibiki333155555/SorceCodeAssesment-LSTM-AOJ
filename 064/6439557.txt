#include<iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int i = 0;
    int sum = 0, min = 10000, max = 0;
    for (i; i < n; i++) {
        cin >> n;
        if (n > max) {
            max = n;
        }
        else if (n < min) {
            min = n;
        }
        sum += n;
        
        cout << min << " " << max << " " << sum << endl;

    }
}
