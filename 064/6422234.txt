#include <iostream>

using namespace std;

int main()
{
    int n = 3;
    cin>>n;
    for(int i = 1; i <= n;i++)
    {
        if(i % 3 == 0)
        {
            cout<<" "<<i;
        }
        else{
        for(int x = i;x != 0;x /= 10)
        {
            if(x % 10 == 3)
            {
                cout<<" "<<i;
            }

        }
        }

    }
    return 0;
}
