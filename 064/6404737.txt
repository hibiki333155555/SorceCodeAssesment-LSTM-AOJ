#include<stdio.h>
#include<vector>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<set>
#include<map>
using namespace std;

struct point {double x,y;};

void koch(int n, point a, point b){
  if(n == 0){return;}
  
  point s,t,u;
  s.x = a.x + (b.x - a.x)/3;
  s.y = a.y + (b.y - a.y)/3;
  t.x = a.x + 2*(b.x - a.x)/3;
  t.y = a.y + 2*(b.y - a.y)/3;
  u.x = s.x + (t.x - s.x) * 0.5 - (t.y - s.y) * sqrt(3) * 0.5;
  u.y = s.y + (t.y - s.y) * 0.5 + (t.x - s.x) * sqrt(3) * 0.5;
  
  koch(n-1,a,s);
  printf("%.8f %.8f\n", s.x,s.y);
  koch(n-1,s,u);
  printf("%.8f %.8f\n", u.x,u.y);
  koch(n-1,u,t);
  printf("%.8f %.8f\n", t.x,t.y);
  koch(n-1,t,b);
}  

int main(){
  point a,b;
  a.x = 0;
  a.y = 0;
  b.x = 100;
  b.y = 0;
  
  int n;
  scanf("%d",&n);
  
  printf("%.8f %.8f\n", a.x, a.y);
  koch(n,a,b);
  printf("%.8f %.8f\n", b.x, b.y);
  
}
