#include <bits/stdc++.h>

#define rep(i, n) for(int i = 0; i < (n); i++)
#define all(x) x.begin(), x.end()

using ll = long long;
using namespace std;

int main() {
    int n; cin >> n;
    ll m, k, x; cin >> m >> k >> x;
    vector<ll> a(n); rep(i, n) cin >> a[i];
    sort(all(a));
    reverse(all(a));
    ll dist = 0;
    rep(i, k) {
        dist += a[i];
    }
    cout << (dist >= x * m ? "Yes" : "No") << "\n";
}
