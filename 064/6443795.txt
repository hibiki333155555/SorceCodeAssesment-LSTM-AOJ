#include <iostream>
using namespace std;

void insertionsort(int A[],int n,int g, int &cnt){
    for(int i = g;i < n;i++){
        int v = A[i];
        int j = i - g;
        while(j >= 0 && A[j] > v){
            A[j + g] = A[j];
            j = j - g;
            cnt += 1;
        }
        A[j + g] = v;
    }
}

void shellsort(int A[], int n){
    int cnt = 0;
    int m = 2;
    int G[2] = {A[m], A[m - 1]};
    cout << m << endl;
    cout << G[0] << " " << G[1] << endl;
    for(int i = 0;i < m;i++){
        insertionsort(A, n, G[i], cnt);
    }
    cout << cnt << endl;

}

int main() {
    int  n;
    cin >> n;
    int A[n];
    for(int i = 0;i < n;i++)cin >> A[i];
    int m = 2;
    cout << m << endl;
    shellsort(A, n);
    for(int i = 0;i < n;i++){
        cout << A[i] << endl;
    }
}
