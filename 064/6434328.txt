#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, migi=0, hidari=0;
    cin >> n;
    int mh=1000, hh=1000;
    vector<int> a(n), w(n);
    for(int i = 0; i < n; i++){
        cin >> a[i];
        if(a[i] == 1){
            hidari++;
        } else {
            migi++;
        }
    }
    if(hidari == n || migi == n){
        cout << 0 << endl;
        return 0;
    }
    for(int i = 0; i < n; i++){
        cin >> w[i];
        if(a[i] == 1){
            if(w[i] < mh){
                mh = w[i];
            }
        } else {
            if(w[i] < hh){
                hh = w[i];
            }
        }
    }
    cout << mh+hh << endl;
    return 0;
}
