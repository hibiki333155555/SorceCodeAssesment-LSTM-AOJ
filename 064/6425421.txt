#include <iostream>
#include <algorithm>
#include <iomanip>
#include <climits>
#include <string>
#include <cmath>

using namespace std;

int main()
{
int a,b;
string x;
cin >> a >> x >> b;
if (x == "+")
  {
    cout << a + b << endl;
  }
else if (x == "-")
{
  cout << a - b << endl;
}
else if (x == "*")
{
  cout << a * b << endl;
}
else 
{
  cout << a / b << endl;
}
}
