#include <iostream>
using namespace std;

int main(){
    int a,b;
    char t;
    cin >> a >> t >> b;

    for(int i = 0; i < 10000; i++){
    if(t == '+') cout << a + b << endl;
    if(t == '-') cout << a - b << endl;
    if(t == '*') cout << a * b << endl;
    if(t == '/') cout << a / b << endl;
    if(t == '?') break;
    cin >> a >> t >> b;
    }
}
