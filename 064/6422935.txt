#include <cstdio>
#include <vector>
#include <functional>
using namespace std;

const int mod = 1e9 + 7;

int power(int a, int b) {
  int res = 1;
  while (b) {
    if (b & 1) res = res * 1ll * a % mod;
    a = a * 1ll * a % mod;
    b /= 2; 
  }
  return res;
}

int main() {
  int n, m;
while (scanf("%d%d", &n, &m) == 2) {
  if (!n && !m) break;
  vector<int> vis(n + 1, 0);
  int cnt = 0;
  for (int i = 0; i < m; i++) {
    int u, v;
    scanf("%d%d", &u, &v);
    if (!vis[u]) {
      vis[u] = 1;
      cnt++; 
    }
    if (!vis[v]) {
      vis[v] = 1;
      cnt++; 
    }
  }
  if (cnt == 0) {
    printf("%d\n", power(2, n));
  } else if (cnt == 1) {
    printf("%d\n", (power(2, n - 1) + power(2, n - 1)) % mod);
  } else {
    printf("%d\n", (power(2, n - cnt) + power(2, n - cnt) + 1) % mod);
  }
}
  return 0; 
}
