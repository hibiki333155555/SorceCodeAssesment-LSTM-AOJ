#include <bits/stdc++.h>
#define be(v) (v).begin(),(v).end()
#define pb(q) push_back(q)
#define rep(i, n) for(int i=0;i<n;i++)
#define all(i, v) for(auto& i : v)
typedef long long ll;
using namespace std;
const ll mod=1000000007, INF=(1LL<<60);
#define doublecout(a) cout<<fixed<<setprecision(10)<<a<<endl;

int main() {
    cin.tie(0);
    cout.tie(0);
    ios::sync_with_stdio(false);
    string s;
    cin >> s;
    int n = s.length();
    vector<int> cnt(n + 1, 0);
    stack<int> st;
    rep(i, n) {
        if(s[i] == '(') {
            cnt[i] += st.top();
            continue;
        } else if(s[i] == ')') {
            cnt[i] -= st.top();
            st.pop();
            continue;
        } 
        int num = s[i] - '0';
        if(0 <= num && num <= 9) {
            st.push(num);
        } 
    }
    string t = "";
    int num = 0;
    rep(i, n) {
        num += cnt[i];
        num %= 26;
        num += 26;
        num %= 26;
        int num1 = s[i] - 'a';
        if(0 <= num1 && num1 < 26) {
            t += char('a' + (num + num1) % 26);
        }
    }
    cout << t << endl;
    return 0;
}
