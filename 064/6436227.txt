#include<iostream>
using namespace std;
const int N = 100005;
int f[N];

int getf(int x) {
	if(f[x] != x)
		f[x] = getf(f[x]);
	return f[x];
}

int main() {
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		f[i] = i;
	for(int i=0;i<m;i++) {
		int s,t;
		cin>>s>>t;
		s = getf(s), t = getf(t);
		if(s!=t)
			f[s] = t;
	}
	int q;
	cin>>q;
	while(q--) {
		int s,t;
		cin>>s>>t;
		s = getf(s), t = getf(t);
		cout<<(s == t?"yes":"no")<<endl;
	}
}
