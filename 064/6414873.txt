inline i128 parse(str s) {
    std::reverse(s.begin(), s.end());
    i128 ret= 0;
    bool minus= false;
    if(s.back() == '-') {
        minus= true;
        s.pop_back();
    }
    while(!s.empty()) {
        if('0' <= s.back() && s.back() <= '9') {
            ret= ret * 10 + s.back() - '0';
            s.pop_back();
        } else {
            break;
        }
    }
    std::reverse(s.begin(), s.end());
    if(minus) ret*= -1;
    return ret;
}
inline str to_string(i128 val) {
    str ret= ""s;
    bool minus= false;
    if(val < 0) {
        minus= true;
        val*= -1;
    }
    do {
        char digit= static_cast<char>(48 + val % 10);
        ret+= digit;
        val/= 10;
    } while(val != 0);
    if(minus) ret+= "-"s;
    std::reverse(ret.begin(), ret.end());
    return ret;
}
inline str to_string(u128 val) {
    str ret= ""s;
    do {
        char digit= static_cast<char>(48 + val % 10);
        ret+= digit;
        val/= 10;
    } while(val != 0);
    std::reverse(ret.begin(), ret.end());
    return ret;
}

signed main() {
    setvbuf(stdin, nullptr, _IOFBF, 1 << 17);
    setvbuf(stdout, nullptr, _IOFBF, 1 << 17);
    LL(N);
    STR(S);
    VEC(A, N, ll);
    ll ans= 0;
    auto f= [&](ll &ans, str &s, vl &A) {
        priority_queue<pll, vpll, greater<pll>> pq;
        ll sum= 0;
        REP(i, sz(S)) {
            if(S[i] == '(') {
                sum++;
            } else {
                sum--;
                pq.emplace(A[i], i);
            }
            if(sum < 0) {
                auto [c, p]= pq.top();
                pq.pop();
                ans+= c;
                S[p]= '(';
                sum+= 2;
            }
        }
    };
    REP(2) {
        f(ans, S, A);
        reverse(ALL(S));
        REP(i, sz(S)) {
            if(S[i] == '(') {
                S[i]= ')';
            } else {
                S[i]= '(';
            }
        }
        reverse(ALL(A));
    }
    print(ans);
}
