#include <stdio.h>

int solve(int N, int M, int v, int a, int b)
{
	static int flag[501];
	int i, k;
	for (i = 0; i < M; i++) flag[i] = 0;
	for (i = 0, k = v % M; i < N; i++, k = (k * a + b) % M) {
		if (flag[k] == 0) flag[k] = 1;
		else return 1;
	}
	return 0;
}

int main()
{
	int i, t, T, N, M, v, a, b;
	scanf("%d", &T);
	for (t = 1; t <= T; t++) {
		scanf("%d %d %d %d %d", &N, &M, &v, &a, &b);
		if (solve(N, M, v, a, b) == 0) printf("No\n");
		else printf("Yes\n");
	}
	fflush(stdout);
	return 0;
}
