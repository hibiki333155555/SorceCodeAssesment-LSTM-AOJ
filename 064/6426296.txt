#include<bits/stdc++.h>
using namespace std;
int main()
{
    
    int n,k,min;
    cin>>n>>k;
    int arr[n];
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    for(int i=0;i<n-k+1;i++)
    {
        min=arr[i];
        for(int j=i;j<i+k;j++)
        {
            if(min>arr[j])
            {
                min=arr[j];
            }
        }
        cout<<min<<" ";
    }
    cout<<endl;
    return 0;
}
