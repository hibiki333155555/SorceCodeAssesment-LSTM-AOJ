#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define all(x) x.begin(),x.end()
#define fol(i,s,g) for(int i=(int)(s); i<(int)(g); i++)
#define fix(ans,k) fixed << setprecision(k) << ans

int main(){
    int n,x,y,z,k;
    cin>>n;
    set<int> a,b,c;
    cin>>x;
    fol(i,0,x){
        cin>>k;
        a.insert(k);
    }
    cin>>y;
    fol(i,0,y){
        cin>>k;
        b.insert(k);
    }
    cin>>z;
    fol(i,0,z){
        cin>>k;
        c.insert(k);
    }
    int ans=0;
    fol(i,1,n+1){
        if(!a.count(i)&&c.count(i)) ans++;
        else if(b.count(i)&&c.count(i)) ans++;
    }
    cout<<ans<<'\n';
    return 0;
}
