using namespace std;
using ll = long long;
using P = pair<ll,ll>;
using TP = tuple<ll,ll,ll>;
void init_io(){
  cin.tie(0);
  ios::sync_with_stdio(false);
  cout << fixed << setprecision(18);
}
signed main(){
  init_io();
  ll n;
  cin >> n;
  vector<string> s(n);
  for(int i=0;i<n;i++) {
    cin >> s[i];
  }
  ll ans = 0,m=s[0].size();
  for(int i=0;i<n;i++) {
    bool fusei = false;
    for(int j=0;j<n;j++) {
      if (i==j) continue;
      ll match_c = 0;
      for(int k=0;k<m;k++) {
        if (s[i][k] != s[j][k]) match_c++;
      }
      if (match_c <= 1) {
        fusei = true;
        break;
      }
    }
    if (fusei) ans++;
  }
  cout << ans << endl;
}
