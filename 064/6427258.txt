#include <stdio.h>

int main()
{
	int i, j, k, l, N, ans = 0;
	char rep[101][101];
	scanf("%d", &N);
	for (i = 1; i <= N; i++) scanf("%s", rep[i]);
	for (i = 1; i <= N; i++) {
		for (j = 1; j <= N; j++) {
			if (j == i) continue;
			for (k = 0, l = 0; rep[i][k] != 0; k++) if (rep[i][k] != rep[j][k]) l++;
			if (l <= 1) break;
		}
		if (j <= N) ans++;
	}
	printf("%d\n", ans);
	fflush(stdout);
	return 0;
}
