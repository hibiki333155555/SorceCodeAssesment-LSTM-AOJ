#include <iostream>
#include <iomanip>

int main() {
  int q; std::cin >> q;
  for (int i=0; i<q; i++) {
    double x0, y0, x1, y1, x2, y2, x3, y3;
    std::cin >> x0 >> y0 >> x1 >> y1 >> x2 >> y2 >> x3 >> y3;
    double denom = (x1-x0)*(y3-y2) - (x3-x2)*(y1-y0);
    double numer_x = (y3-y2)*(x2-x0) - (x3-x2)*(y2-y0);
    double numer_y = (y1-y0)*(x2-x0) - (x1-x0)*(y2-y0);
    double ratio_x = numer_x / denom;
    double ratio_y = numer_y / denom;
    std::cout << std::fixed << std::setprecision(9);
    std::cout << (1.0 - ratio_x) * x0 + ratio_x * x1 << " ";
    std::cout << (1.0 - ratio_y) * y0 + ratio_y * y1 << std::endl;
  }
}
