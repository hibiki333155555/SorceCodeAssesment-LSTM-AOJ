#include <bits/stdc++.h>
#include <string>
using namespace std;

int main() {

    int n,t=1;
    cin>>n;
    
    for(int i=1;i<=n;++i){
        if(i%3==0)cout<<" "<<i;
        if(i%10==3)cout<<" "<<i;
    }
    for(int i=1;i<=n;++i){
        if(10*t>i)break;
        else if(i/(10*t)==0)cout<<" "<<i;
        t++;
    }
    
    return 0;
}
