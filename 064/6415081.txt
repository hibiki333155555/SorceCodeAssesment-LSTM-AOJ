#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    int min = 1000001, max = -1000001, sum = 0;
    for (int i=0; i<n; i++) {
        int num;
        cin >> num;
        if (num < min) {
            min = num;
        }
        if (num > max) {
            max = num;
        }
        sum += num;
    }
    cout << min << " " << max << " " << sum << endl;
    return 0;
}
