#include <bits/stdc++.h>

using namespace std;
using ll = long long;

int main() {
   cout << fixed << setprecision(15);

   ll N, W;
   cin >> N >> W;

   vector<ll> v_list, w_list;

   v_list.emplace_back(-1);
   w_list.emplace_back(-1);

   for (int i = 0; i < N; i++) {
      ll v, w;
      cin >> v >> w;

      v_list.emplace_back(v);
      w_list.emplace_back(w);
   }

   vector<vector<ll>> dp(N + 1, vector<ll>(W + 1, 0));

   for (int w = 1; w <= W; w++) {
      if (w >= w_list[1]) {
         dp[1][w] = dp[1][w - w_list[1]] + v_list[1];
      }
   }

   for (int i = 2; i <= N; i++) {
      for (int w = 0; w <= W; w++) {
         dp[i][w] = dp[i - 1][w];

         if (w >= w_list[i]) {
            dp[i][w] = max(dp[i][w], dp[i][w - w_list[i]] + v_list[i]);
         }
      }
   }

   cout << dp[N][W] << endl;

   return 0;
}
