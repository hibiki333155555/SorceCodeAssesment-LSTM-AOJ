const ll maxn = 8e0 +7;
const ll maxm = 2e1 +7;
const ll inf = 0x3f3f3f3f;
const ll mod = 1e9 +7;

int k, ans;
int mp[maxn][maxn], rv[maxn], cv[maxn], posx[maxm], negx[maxm];

void dfs(int l) {
    if(ans) return;
    if(l == 8) {
        ans = 1;
        for(int i = 0; i < 8; i++) {
            for(int j = 0; j < 8; j++) {
                printf("%c",(mp[i][j]==1?'Q':'.'));
                //cout<<(mp[i][j]==1?'Q':'.')<<(j==7?"\n":"");
            }
            printf("\n");
        }
        return ;
    }
    for(int i = 0; i < 8; i++) {
        if(rv[i]) continue;
        for(int j = 0; j < 8; j++) {
            if(cv[j]) continue;
            if(posx[i+j] || negx[i-j]) continue;
            if(ans) return;
            rv[i] = 1;
            cv[j] = 1;
            posx[i+j] = 1;
            negx[i-j] = 1;
            mp[i][j] = 1;
            dfs(l+1);
            rv[i] = 0;
            cv[j] = 0;
            posx[i+j] = 0;
            negx[i-j] = 0;
            mp[i][j] = 0;
        }
    }
}

signed main() {
    #ifdef moyi_qwq

    cin>>k;
    for(int i = 1; i <= k; i++) {
        int r, c;
        cin>>r>>c;
        mp[r][c] = 1;
        cv[c] = 1;
        rv[r] = 1;
        posx[r+c] = 1;
        negx[r-c] = 1;
    }
    dfs(k);
    
    //cerr<<"Time : "<<1000*((double)clock())/(double)CLOCKS_PER_SEC<<"ms";
    return (0);
}
