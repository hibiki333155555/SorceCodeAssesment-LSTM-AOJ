#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
#include <string>
#include <cmath>
#include <iomanip>
using namespace std;

inline double square(double num)
{
    return num*num;
}

int main()
{
    double ax, ay, bx, by;
    cin >> ax >> ay >> bx >> by;
    printf("%.8f\n", sqrt(square(ax-bx) + square(ay-by)));

    return 0;
}
