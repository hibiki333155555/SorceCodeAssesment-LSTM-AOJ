#include <stdio.h>

#include <algorithm>
#define N 10000
#define INF 1 << 30

#include <queue>
using namespace std;

int n;
int from, m, to, cost;
int G[N][N][3];
int counts[N];
int cost_list[N];
int visited[N];

priority_queue<pair<int, int> > PQ;

void dijkstra(int start) {
  cost_list[start] = 0;
  PQ.push(make_pair(0, 0));
  while (!PQ.empty()) {
    pair<int, int> top = PQ.top();
    PQ.pop();
    int min_v = top.second;
    visited[min_v] = 1;

    for (int i = 0; i < counts[min_v]; i++) {
      int next_v = G[min_v][i][0];
      int cost_to_next = G[min_v][i][1];
      if (!visited[next_v]) {
        if (cost_list[next_v] > cost_list[min_v] + cost_to_next) {
          cost_list[next_v] = cost_list[min_v] + cost_to_next;
          int cost = cost_list[next_v] * -1;
          PQ.push({cost, next_v});
        }
      }
    }
  }
}

void create_list() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf("%d %d", &from, &m);
    for (int k = 0; k < m; k++) {
      scanf("%d %d", &to, &cost);
      G[from][counts[from]][0] = to;
      G[from][counts[from]][1] = cost;
      counts[from]++;
    }
  }
}

void init_cost_list() {
  for (int i = 0; i < n; i++) {
    cost_list[i] = INF;
  }
}

void print() {
  for (int i = 0; i < n; i++) {
    printf("%d %d", i, cost_list[i]);
    puts("");
  }
}

int main(void) {
  create_list();
  init_cost_list();
  dijkstra(0);
  print();
}
