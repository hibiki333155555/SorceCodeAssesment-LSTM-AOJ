#include <iostream>
#include <vector>
#define I 0x7fffffff

using namespace std;

//Prim
vector<pair<int, int> > Prim(vector<vector<int> >& mat)
{
    vector<pair<int, int> > choose;
    int temp, n = mat.size(), MinSub, MinVal;
    bool* pd = new bool[n];
    for (int i = 0;i < n;++i)
        pd[i] = 0;
    int i;
    choose.push_back(make_pair(0, 0));
    pd[0] = 1;
    while (true)
    {
        i = -1;
        MinVal = 5000;
        for (int x = 0;x < n;++x)
        {
            if (!pd[x])
            {
                for (int y = 0;y < n;++y)
                {
                    if (pd[y] && mat[x][y] < MinVal)
                    {
                        MinVal = mat[x][y];
                        i = y;
                        MinSub = x;
                    }
                }
            }
        }
        if (i == -1)
            return choose;
        pd[MinSub] = 1;
        choose.push_back(make_pair(MinSub, mat[MinSub][i]));
    }
}

int main()
{
    int n, sum = 0;
    vector<pair<int, int> > key;
    cin >> n;
    vector<vector<int> > val_mat(n, vector<int>(n));
    for (int i = 0;i < n;++i)
        for (int j = 0;j < n;++j)
        {
            cin >> val_mat[i][j];
            if (val_mat[i][j] == -1)
                val_mat[i][j] = I;
        }
    key = Prim(val_mat);
    for (int i = 0;i < n;++i)
        sum += key[i].second;
    cout << sum << endl;
    return 0;
}
