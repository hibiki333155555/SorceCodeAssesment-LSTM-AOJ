#include <iostream>
using namespace std;
int LIS(int arr[], int n) {
    int len = n;
    if (len == 0) {
        return 0;
    }
    int L[len];
    int f[len];
    L[0] = 1;
    for (int i = 0; i < len; i++) f[i] = -1;
    f[1] = arr[0];
    for (int i = 1; i < len; i++) {
        int max = 0;
        for (int j = 0; j < L[i] - 1; j++) {
            if (f[j] > max) max = j;
        }
        L[i] = L[i-1];
        if (arr[i] < f[L[i-1]] && arr[i] > f[L[i]-1]) {
            L[i] = L[i-1];
            f[L[i]] = arr[i];
        }
        else if (arr[i] > f[L[i-1]]) {
            L[i] = L[i-1]+1;
            f[L[i]] = arr[i];
        }
        /*else {
            int max = 0;
            //for (int j = 0; j < L[i] - 1; j++) {
                //if (f[j] > max) max = j;
            //}
            L[i] = L[i-1];
        }*/
        cout << L[i] << " " << f[L[i]] << endl;
    }
    int max_len = L[0];
    for (int i = 1; i < len; i++) {
        if (L[i] > max_len)
            max_len = L[i];
    }
    return max_len;
}

int main(int argc, char *argv[]) {
    int n;
    cin >> n;
    int arr[n];
    for (int i = 0; i < n; i++)
        cin >> arr[i];
    cout << LIS(arr, n) << endl;
    return 0;
}
