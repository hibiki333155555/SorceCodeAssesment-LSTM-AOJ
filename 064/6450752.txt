#include <bits/stdc++.h>
using namespace std;

void selectionSort(vector<int> A,int N){
  int num=0;
  for(int i=0;i<N;i++){
    int minj = i;
    for(int j=i;j<N;j++){
      if(A[j] < A[minj])
        minj = j;
    }
    if(A[i] != A[minj]){
      num++;
      swap(A[i], A[minj]);
      
    }
  }
  for(int a=0;a<N;a++){
    if(a != 0)
      cout << " ";
    cout << A[a];
  }
  cout << "\n" <<num <<  endl;
  return;
}
int main(){
  int N;
  cin >> N;
  vector<int> A(N);
  for(int i =0;i<N;i++)
    cin >> A[i];
  
  selectionSort(A,N);

  return 0;
}
