#include <bits/stdc++.h>

using namespace std;
vector<int> S;

int binary_search_iterator(int left, int right, int item)
{
  int mid = (left + right) / 2;

  if (left >= right)
    return 0;
  if (S[mid] == item)
    return 1;
  if (S[mid] < item)
    left = mid + 1;
  if (S[mid] > item)
    right = mid;

  return binary_search_iterator(left, right, item);
}

int binary_search(int item)
{
  int left = 0;
  int right = S.size();

  return binary_search_iterator(left, right, item);
}

int main()
{
  int n, q;
  cin >> n;
  int s;

  for (int i = 0; i < n; i++)
  {
    cin >> s;
    S.push_back(s);
  }

  cin >> q;
  int ans = 0;

  int t;
  for (int i = 0; i < q; i++)
  {
    cin >> t;
    if (binary_search(t) == 1)
    {
      ans++;
    }
  }

  cout << ans << endl;
  return 0;
}
