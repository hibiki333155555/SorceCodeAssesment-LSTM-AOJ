#include <iostream>
using namespace std;

int main(){
    int N, A[100] = {};
    cin >> N;
    for(int i = 0; i < N; i++) cin >> A[i];
    for(int i = 0; i < N; i++){
        int v = A[i], k = i - 1;
        while(k >= 0 && v < A[k]){
            A[k + 1] = A[k];
            k--;
        }
        A[k + 1] = v;
        
        for(int j = 0; j < N; j++){
            if(j) cout << " ";
            cout << A[j];
        }    
        cout << endl;
    }
    
    
    return 0;
}
