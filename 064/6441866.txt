
import java.util.Scanner

object Main {
  def main(args: Array[String]): Unit = {
    val sc = new Scanner(System.in)

    val word = sc.next()

    var T = ""
    var count = 0

    while (T != "end_of_text") {
      T = sc.next().toLowerCase
      if (word == T) count += 1
    }

    println(count)

  }
}
