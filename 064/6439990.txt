#include <iostream>
#include <cmath>
using namespace std;
int main(void){
    // Your code here!
    int n;cin>>n;
    int u,k,e[100+1][100+1],v;
    for(int i=0;i<101;i++){
        for(int j=0;j<101;j++)e[i][j]=0;
    }
    for(int i=0;i<n;i++){
        cin>>u>>k;
        for(int j=0;j<k;j++) {
            cin>>v;
            e[u][v]=1;
        }
    }
    
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(j-1) cout<<" ";
            cout<<e[i][j];
        }
        cout<<endl;
    }
}
