#include <iostream>
#include <iomanip>
#include <cmath>
#include <complex>

using namespace std;

complex<double> in()
{
	double x, y;
	cin >> x >> y;
	return {x, y};
}

int main()
{
	complex<double> p1 = in(), p2 = in(), dp = p2 - p1;
	dp /= abs(dp);
	int q;
	for (cin >> q; q > 0; --q) {
		complex<double> p = in() - p1, x = p1 + (p.real() * dp.real() + p.imag() * dp.imag()) * dp;
		cout << fixed << setprecision(8) << x.real() << ' ';
		cout << fixed << setprecision(8) << x.imag() << endl;
	}
	return 0;
}
