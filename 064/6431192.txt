import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int S=sc.nextInt();
        int h,m,s;
        
        h=S/3600;
        S-=h*3600;
        m=S/60;
        s=S-m*60;
        
        System.out.println(h+":"+m+":"+s);
    }
}
