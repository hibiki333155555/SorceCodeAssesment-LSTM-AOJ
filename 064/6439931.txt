#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string w, t;
    cin >> w;

    int wordCnt = 0;

    while (true)
    {
        cin >> t;
        if (t.compare("END_OF_TEXT") == 0)
        {
            break;
        }

        if (w.length() == t.length())
        {
            for (int i = 0; i < t.length(); i++)
            {
                if (tolower(w[i]) != tolower(t[i]))
                {
                    break;
                }

                if (i = t.length() - 1)
                {
                    wordCnt++;
                }
            }
        }
    }

    cout << wordCnt << endl;
    return 0;
}
