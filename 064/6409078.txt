#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;
using ll = long long;

int main(){
    int n; cin >> n;
    vector<ll> d(n);
    for(auto &it: d) cin >> it;
    vector<ll> tot(n+1, 0);
    for(int i = 0; i < n; i++) tot[i+1] = tot[i]+d[i];
    map<ll, vector<int>> mem;
    for(int i = 0; i <= n; i++) mem[tot[i]].emplace_back(i);
    int ans = 0;
    for(auto &it: mem){
        auto &vec = it.second;
        if(vec.size() > 1) ans = max(ans, vec.back()-vec.front());
    }
    cout << ans << endl;
    return 0;
}
