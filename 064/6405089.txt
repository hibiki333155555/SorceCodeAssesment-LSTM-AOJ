import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
	Scanner sc=new Scanner(System.in);
	int k=0,m=0,y=0, n=sc.nextInt();
	int[] a=new int[1001];
	for(int i=1;i<n;i++) {
		a[i]=sc.nextInt();
	}
	for(int i=1;i<=n;i++) {
		m=m+a[i]-a[i-1];y=y+a[i-1]-a[i];
		if(m>y) {
			k+=y;
		}else {
			k+=m;
		}
	}
	System.out.println(k*-1);
	sc.close();
	}
}
