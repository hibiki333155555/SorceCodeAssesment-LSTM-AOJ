#include <stdio.h>

int main(void){
    int a, b;
    char o;
    scanf("%d%c%d", &a, &o, &b);
    int t;
    
    if(o == '?') return 0;
    switch(o){
        case '+': t = a + b; break;
        case '-': t = a - b; break;
        case '*': t = a * b; break;
        case '/': t = a / b; break;
    };
    printf("%d", t);
    
    return 0;
}
