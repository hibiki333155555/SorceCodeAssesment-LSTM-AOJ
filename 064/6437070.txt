#include <bits/stdc++.h>
using namespace std;

int main() {

    while (1) {
        int n, k;
        cin >> n >> k;

        if (n == 0 && k == 0) break;

        vector<long long> v(n+1, 0);
        for (int i = 1; i <= n; i++) {
            int a;
            cin >> a;
            v[i] = v[i-1] + a;
        }

        long long result = -20000;
        for (int i = n; i >= k; i--) {
            result = max(result, v[i] - v[i-k]);
        }

        cout << result << endl;
    }

    return 0;
}
