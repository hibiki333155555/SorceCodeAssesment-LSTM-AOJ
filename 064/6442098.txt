int main()
{	
	
	int n,i;
	int a[101];
	
	cin>>n;
	
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(i=n-1;i>=0;i--)
	{
		if(i!=0)
			cout<<a[i]<<" ";
		else
			cout<<a[i];
	}
	
return 0;
}
