#include <iostream>
using namespace std;

int main() {
    int h, w;
    for (;;) {
        cin >> h>>w;
        if (h==0 and w==0) break;
        else{
            for (int i=1; i<=h; i++) {
                for (int j=1; j<=h; j++) {
                    cout << '#';
                }
                cout <<"\n";
            }
        }
    }
}
