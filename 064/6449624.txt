#include<stdio.h>
int main()
{
	int n;
	scanf("%d", &n);
	int i, j, k;
	char re[102][102];
	for (i = 0; i < n; i++)
		scanf("%s", re[i]);
	int ans = 0, cnt, count;
	for (i = 0; i < n; i++)
	{
		count = 0;
		for (j = 0; j < n; j++)
		{
			if (i == j)
				continue;
			cnt = 0;
			for (k = 0; re[i][k] != '\0'; k++)
				if (re[i][k] != re[j][k])
					cnt++;
			if (cnt < 2)
				count++;
		}
		if (count > 0)
			ans++;
	}
	printf("%d\n", ans);
	return 0;
}
