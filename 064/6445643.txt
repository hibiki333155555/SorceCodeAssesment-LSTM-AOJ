#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int main() {
    int count = 0;
    int vector[6][6] = {{0, 2, 4, 1, 3, 0}, {3, 0, 0, 5, 0, 2}, {1, 5, 0, 0, 0, 4},
                        {4, 0, 0, 0, 5, 1}, {2, 0, 5, 0, 0, 3}, {0, 3, 1, 4, 2, 0}};
    string message = "No";
    int face;
    int a, b;
    while(count < 2) {
        int dice[6];
        for(int n = 0; n < 6; n++) {
            cin >> dice[n];
        }
        if(count == 0) {
            face = dice[vector[0][1]];
            a = dice[0];
            b = dice[1];
            count++;
        } else {
            int a = dice[0];
            for(int num = 0; num < 6; num++) {
                if(dice[num] == a) {
                    break;
                }
            }
            int b = dice[1];
            for(int num2 = 0; num2 < 6; num2++) {
                if(dice[num2] == b) {
                    break;
                }
            }
            if(face == dice[vector[a][b]]) {
                message = "Yes";
            }
        }
    }
    cout << message << endl;
}
