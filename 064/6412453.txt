#include <bits/stdc++.h>
using namespace std;

int main() {
  int w, h, x, y, r;
  cin >> w >> h >> x >> y >> r;
  bool check = true;
  if (x-r < 0) check = false;
  if (y-r < 0) check = false;
  if (x+r > w) check = false;
  if (y+r > h) check = false;
  
  if (check) cout << "Yes" <<endl;
  else cout << "No" << endl;

}
