#include <iostream>
#include <vector> // vector
#include <algorithm> // min, max, swap, sort, reverse, lower_bound, upper_bound
using namespace std;



int main() {
    int n;
    cin >> n;
    std::vector<int> vec(n);
    std::vector<int> x(n, -1000000000);
    for (int i = 0; i < n; i++)cin >> vec.at(i);

    for(int i = 0;i < n - 1;i++){
        std::vector<int> a(n, -1000000000);
        for(int j = i + 1;j < n;j++){
            a.at(j) = vec.at(j);
        }
        sort(a.begin(), a.end());
        x.at(i) = a.at(n - 1) - vec.at(i);
    }
sort(x.begin(), x.end());
    cout << x.at(n - 1) << endl;

}
