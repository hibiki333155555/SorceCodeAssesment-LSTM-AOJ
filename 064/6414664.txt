int main() {
  int m, n;
  cin >> m >> n;

  vector<string> s(m);
  cin >> s;

  int o_cnt = 0, x_cnt = 0;
  rep(i, m) rep(j, n) {
    if (s[i][j] == 'o')
      o_cnt++;
    if (s[i][j] == 'x')
      x_cnt++;
  }

  int ans = INF32;

  int o_cnt_row = 0;
  int x_cnt_row = 0;
  rep(i, m - 1) {
    rep(j, n) {
      if (s[i][j] == 'o')
        o_cnt_row++;
      if (s[i][j] == 'x')
        x_cnt_row++;
    }

    int v1 = o_cnt_row + x_cnt - x_cnt_row;
    int v2 = x_cnt_row + o_cnt - o_cnt_row;
    chmin(ans, min(v1, v2));
  }


  int o_cnt_col = 0;
  int x_cnt_col = 0;
  rep(j, n - 1) {
    rep(i, m) {
      if (s[i][j] == 'o')
        o_cnt_col++;
      if (s[i][j] == 'x')
        x_cnt_col++;
    }

    int v1 = o_cnt_col + x_cnt - x_cnt_col;
    int v2 = x_cnt_col + o_cnt - o_cnt_col;
    chmin(ans, min(v1, v2));
  }

  cout << ans << endl;
}
