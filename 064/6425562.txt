#include <iostream>
#include <vector>

using namespace std;

pair<int, int> f(const vector<vector<pair<int, int>>>& C, int n, int p)
{
	pair<int, int> m = {0, n};
	for (auto [to, w] : C[n]) if (to != p) {
		auto [ww, k] = f(C, to, n);
		if (m.first < w + ww) m = {ww + w, k};
	}
	return m;
}

int main()
{
	int n; cin >> n;
	vector<vector<pair<int, int>>> C(n);
	for (int s, t, w; n > 1; --n) {
		cin >> s >> t >> w;
		C[s].push_back({t, w});
		C[t].push_back({s, w});
	}
	auto [w, k] = f(C, 0, 0);
	auto [W, j] = f(C, k, k);
	cout << W << endl;
	return 0;
}
