#include <bits/stdc++.h>
using namespace std;

int main() {   
    int n; cin >> n;
    bool cards[4][13];
    for(int i = 0;i<4;i++) {
        for(int j = 0;j<13;j++) cards[i][j] = false;
    }

    string egara;int rank;
    for(int i = 0;i < n;i++) {
        cin >> egara >> rank;
        if(egara == "S") {
            cards[0][rank - 1] = true;
        }else if(egara == "H") {
            cards[1][rank - 1] = true;
        }else if(egara == "C") {
            cards[2][rank - 1] = true;
        }else {
            cards[3][rank - 1] = true;
        }
    }
    // for(int i = 0;i < 4;i++) {
    //     for(int j = 0;j<13;j++) {
    //         if(cards[i][j]) cout << cards[i][j] << endl;
    //     }
    // }
    for(int i = 0;i < 4;i++) {
        for(int j = 0;j < 13;j++) {
            if(!cards[i][j]) {
                if(i == 0) {
                    cout << "S " << j+1 << endl;               
                }else if(i == 1) {
                    cout << "H " << j+1 << endl;
                }else if(i == 2) {
                    cout << "C " << j+1 << endl;
                }else {
                    cout << "D " << j+1 << endl;
                }
            }
        }
    }
    return 0;
}
