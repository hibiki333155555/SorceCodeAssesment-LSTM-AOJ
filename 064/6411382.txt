#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    int n,e;
    vector<int> v;
    cin>>n;
    while(n--){
        cin>>e;
        v.push_back(e);
    }
    cin>>n;
    while(n--){
        cin>>e;
        auto result = equal_range(v.begin(), v.end(), e);
        cout << distance(result.first, v.begin()) << " " << distance(result.second, v.begin()) << endl;
    }
    return 0;
}
