typedef pair<ll, ll> P;

vector<P> prime_factor_as_vect(ll number) {
    // 素因数分解


    vector<P> ret;
    for (ll i = 2; i * i <= number; i++) {
        cout << i << " " << number << endl;
        ll count = 0;
        while (number % i == 0) {
            count++;
            number = number / i;
        }
        if (count != 0) ret.push_back(P(i, count));
    }
    if (number != 1) ret.push_back(P(number, 1));

    return ret;
}
