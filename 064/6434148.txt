#include<bits/stdc++.h>
using namespace std;
int main(){
    vector<int> e(12);
    for(int i=0;i<12;i++){
        cin >> e.at(i);
    }
    sort(e.begin(),e.end());
    if(e.at(0)==e.at(3)&&e.at(4)==e.at(7)&&e.at(8)==e.at(11)){
        cout << "yes" << endl;
    }else{
        cout << "no" << endl;
    }
    return 0;
}
