    int N,K;
    cin>>N>>K;
    vector<int> cnt(5);
    int sum=0;
    for(int i=0;i<N;i++){
        int A;
        cin>>A;
        sum+=A;
        if(A==1){
            cnt[4]++;
            cnt[1]++;
        }else{
            cnt[6-A]++;
        }
    }
    for(int i=4;i>=1;i--){
        int use=min(K,cnt[i]);
        sum+=i*use;
        K-=use;
    }
    cout<<sum<<endl;
    return 0;
}
