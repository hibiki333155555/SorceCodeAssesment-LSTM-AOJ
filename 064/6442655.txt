#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    while(1){
        cin>>n;
        if(n==0) break;
        vector<int> b(n);
        int sum=0;
        for(int i=0;i<n;i++){
            cin>>b[i];
            if(b[i]>0) sum++;
        }
        sort(b.begin(),b.end(),greater<int>());
        int ans=0;
        while(sum>=3){
            int sum=0;
            int i=0,j=1,l=1;
            while(b[i]==0) i++;
            if(i<n-3){
                while(b[i+j]==0) j++;
                if(i+j<n-2){
                    while(b[i+j+l]==0) l++;
                }else break;
            }else break;
            b[i]--;
            b[i+j]--;
            b[i+j+l]--;
            for(int t=0;t<n;t++){
                if(b[t]>0) sum++;
            }
            ans++;
        }
        cout<<ans<<endl;
    }
    return 0;
}
