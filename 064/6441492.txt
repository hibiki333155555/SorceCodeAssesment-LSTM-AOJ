#!/usr/bin/python3

c1, r1, c2, r2 = input().split()

c_cnt = 0
while True:
    c_cnt += 1
    if c1 == c2: break
    if len(c1) == 1:
        if c1 == 'Z': c1 = 'AA'
        else: c1 = chr(ord(c1) + 1)
    else:
        if c1[1] == 'Z': c1 = chr(ord(c1[0]) + 1) + 'A'
        else: c1 = c1[0] + chr(ord(c1[1]) + 1)
print(c_cnt * (int(r2) - int(r1) + 1))
