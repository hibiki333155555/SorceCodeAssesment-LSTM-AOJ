#include <iostream>
#include <vector>
#include <queue>
using namespace std;

vector<vector <int> > seen; 
int H, W;

void dfs(int h,int w){
    seen[h][w] = 0;

    for(int dh = -1; dh < 1; dh++){
        for(int dw = -1; dw < 1; dw++){
            int nh, nw; 
            nh = h + dh; nw = w + dw;
            if(nh < 0 || nh >= H || nw < 0 || nw >= W){
                continue;
            }
            if(seen[nh][nw]){
                dfs(nh,nw);
            }
        }
    }
}

int main(){
    while(1){
        cin >> W >> H;

        if(W==0 && H == 0){
            break;
        }

        seen.assign(H,vector<int>(W,0));

        for(int h = 0; h < H; h++){
            for(int w = 0; w < W; w++){
                cin >> seen[h][w];
            }
        }

        int cnt = 0;
        for(int h = 0; h < H; h++){
            for(int w = 0 ; w < W; w++){
                if(seen[h][w] == 1){
                    cnt++;
                    dfs(h,w);
                }
            }
        }
        cout << cnt << endl;
    }

}
