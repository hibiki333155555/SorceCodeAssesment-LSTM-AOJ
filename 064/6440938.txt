#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < n; ++i)
typedef long long ll;
using namespace std;

void solve(ll b) {
    for (int i = 1; i <= b; ++i) {
        ll l = i, r = b + 1;
        auto check = [&](ll x) -> bool {
            ll n = x - i + 1;
            ll ret = (i + x) * n / 2;

            return ret <= b;
        };
        while (r - l > 1) {
            ll mid = (l + r) / 2;
            if (check(mid))
                l = mid;
            else
                r = mid;
        }

        ll n = l - i + 1;
        ll ret = (i + l) * n / 2;

        if (ret == b) {
            cout << i << " " << n << "\n";
            return;
        }
    }
}

int main() {
    cin.tie(0);
    ios_base::sync_with_stdio(false);

    ll b;
    while (cin >> b, b)
        solve(b);

    return 0;
}
