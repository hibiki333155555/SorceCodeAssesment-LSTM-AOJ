#include <bits/stdc++.h>
using namespace std;

int main(){
    string str;
    vector<int> vec(26);
    while(1){
    getline(cin,str);
    if(str.size()==0) break;
    for(int i='a'; i<'z'+1;i++){
        for(auto &s:str){
            if('a'<=s && s<='z'){
                if(s==i) vec[i-'a']++;
            }
            else if('A'<=s && s<='Z'){
                s+=32;
                if(s==i) vec[i-'a']++;
            }
        }
    }
    }
    for(int i='a'; i<'z'+1; i++){
    printf("%c : %d\n",i,vec[i-'a']);
    }
}
