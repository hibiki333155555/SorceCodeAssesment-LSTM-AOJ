#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(int)(n);i++)

int gcd(int a,int b){

    if(b==0){
        return a; 
    }
    if(a>b){
        a%=b;
        return gcd(b,a);
    }
    else{
        b%=a;
        return gcd(a,b);
    }
}

int main(){
    int a,b;
    cin>>a>>b;
    // a>b
    if(a<b) swap(a,b);
    cout<<gcd(a,b)<<endl;
    return 0;
}
