#include<iostream>
#include<string>
using namespace std;
int main(){
    int N;
    string S;
    cin >> N >> S;
    for(int i=0; i<N; i++){
        if(S[i]=='J'){
            cout << 'J';
        }
    }
    for(int i=0; i<N; i++){
        if(S[i]=='O'){
            cout << 'O';
        }
    }
    for(int i=0; i<N; i++){
        if(S[i]=='I'){
            cout << 'I';
        }
    }
    cout << endl;
}
