import java.util.Scanner;
class Main
{
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int[] seq = new int[a];
        
        for(int i = 0; i < seq.length; i++){
            seq[i] = sc.nextInt();
        }
        for(int j = a-1; j > 0; j--){
            System.out.print(seq[j] + " ");
        }
        System.out.println(seq[0]);
    }
}
