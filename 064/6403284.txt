#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string str,order,p;
    int q,a,b;
    cin >> str >> q;
    for(int i=0;i<q;i++) {
        cin >> order >> a >> b;
        if (order == "replace") {
            cin >> p;
            str = str.replace(a, b-a+1, p);
        } else if (order == "reverse") {
            reverse(str.begin(), str.end());
        } else if (order == "print") {
            cout << str.substr(a, b-a+1) << endl;
        }
    }
}
