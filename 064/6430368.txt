#include <bits/stdc++.h>
using namespace std;
void bubblesort(vector<int> &p,int n){
	int count=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<n-1-i;j++){
			if(p[j]>p[j+1]){
				swap(p[j],p[j+1]);
				count++;
			}
		}
	}
	for(int i=0;i<n;i++){
		if(i!=n-1){
			cout<<p[i]<<" ";
		}
		else{
			cout<<p[i];
		}
	}
	cout<<"\n"<<count<<endl;
}
int main() {
	int n;
	cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
	}
	bubblesort(v,n);
	return 0;
}
