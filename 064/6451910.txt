#include<iostream>
#include<iomanip>
#include<math.h>

using namespace std;

#define PI 3.14159265

int main() {
    float a, b, c, C, S, L, h;
    cin >> a >> b >> C;
    S = 0.5 * a * b * sin(PI*c / 180);
    c = sqrt(a * a + b * b -2 * a * b * cos(PI*c / 180));
    L = a + b + c;
    h = 2 * S / c;
    cout << setprecision(1) << S << endl
         << setprecision(1) << L << endl
         << setprecision(1) << h << endl;
    return 0;
}
