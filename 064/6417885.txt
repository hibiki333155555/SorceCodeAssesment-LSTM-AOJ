import java.util.Scanner;
class Main
{
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int[] seq = new int[a];
        
        for(int i = a; i > 0; i--){
            System.out.print(seq[i] + " ");
        }
        System.out.println(seq[0]);
    }
}
