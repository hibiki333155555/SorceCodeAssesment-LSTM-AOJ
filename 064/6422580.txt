void solve() {
    ll N, Q;
    cin >> N >> Q;
    
    vector<ll> imos(N);
    rep(i, Q) {
        ll l, k;
        cin >> l >> k;
        l--;
        ll r = l + k;
        imos[l]++;
        if (r < N) {
            imos[r] -= r - l + 1;
        }
        if (r+1 < N) {
            imos[r+1] += r - l;
        }
    }

    rep(_, 2) {
        rep(i, 1, N) {
            imos[i] += imos[i-1];
        }
    }
    print(imos);
}
