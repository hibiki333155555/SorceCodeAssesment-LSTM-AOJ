#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;cin>>n;
	int p=0;
	vector<int>v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
	}
	int m;cin>>m;
	vector<int>u(m);
	for(int i=0;i<m;i++){
		cin>>u[i];
	}
	int cnt=0;
	for(int i=0;i<m;i++){
		for(int j=0;j<n;j++){
			if(u[i]==v[j]){
				cnt+=1;
				break;
			}
		}
	}
	if(cnt==m){
	p=1;
	}
cout<<p;
	return 0;
}
