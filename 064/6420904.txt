#include <bits/stdc++.h>
using namespace std;
using ll=long long;
int main(){
  ll n;
  cin>>n;
  vector<vector<ll>> v(n,vector<ll>(n));
  map<ll,vector<ll>> s;
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      cin>>v[i][j];
      s[v[i][j]].push_back(i*10000+j);
    }
  }
  map<ll,ll> seen;
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(seen[v[i][j]]==0){
        sort(s[v[i][j]].begin(),s[v[i][j]].end());
        seen[v[i][j]]=1;
      }
    }
  }
  ll q;
  cin>>q;
  while(q--){
    ll t;
    cin>>t;
    if(s[t].size()>0){
      cout<<s[t][0]/10000+1<<" "<<s[t][0]%10000+1<<endl;
    }
    else{
      cout<<-1<<endl;
    }
  }
}
