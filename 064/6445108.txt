#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

int htable[128];
bool check[1677216];

int stringToIndex(string str)
{
  int index = 0;
  int p = 1;
  for (int i = 0; i < str.length(); i++)
  {
    index += (htable[str[i]] * p);
    p *= 4;
  }

  return index;
}

int main(int argc, char const *argv[])
{
  htable['A'] = 1;
  htable['C'] = 2;
  htable['G'] = 3;
  htable['T'] = 4;

  for (int i = 0; i < 1677216; i++)
  {
    check[i] = false;
  }
  

  int n;
  cin >> n;

  string com;
  string op;
  for (int i = 0; i < n; i++)
  {
    cin >> com >> op;

    if (com[0] == 'i') check[stringToIndex(op)] = true;

    if (com[0] == 'f')
    {
      if (check[stringToIndex(op)]) cout << "yes" << endl;
      else cout << "no" << endl;
    }
  }
  

  return 0;
}
