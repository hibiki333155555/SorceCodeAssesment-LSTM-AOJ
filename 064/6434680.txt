#include <stdio.h>
int main()
{
    int hotel[4][3][10];
    int n, b, f, r, v;
    
    for(int i=0;i<4;i++){
        for(int j=0;j<3;j++){
            for(int k=0;k<10;k++){
                hotel[i][j][k]=0;
        }
    }
 }
        scanf("&d", &n);
        for(i=0;i<n;i++){
            scanf("%d %d %d %d", &b,&f,&r,&v);
            hotel[b-1][f-1][r-1]+=v;
        }
        for(j=0;j<3;j++){
           for(k=0;k<10;k++){
               printf(" %d", hotel[0][j][k]);
           }
           printf("\n");
        }
        for(i=1;i<4;i++){
            printf("####################\n");
            for(j=0;j<3;j++){
                for(k=0;k<10;k++){
                    printf(" %d", hotel[i][j][k]);
                }
                printf("\n");
            }
        }
        return 0;
}
