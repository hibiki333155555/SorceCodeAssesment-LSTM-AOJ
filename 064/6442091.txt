#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int n,i,x;
	
	cin>>n;
	
	for(i=1;i<=n;i++)
	{
		if(i%3==0)
			cout<<" "<<i;
		else if(i%10==3)
			cout<<" "<<i;
		else
		{
			x=i;
			x/=10;
			while(x!=0)
			{
				if(x%10==3)
				{
					cout<<" "<<i;
					break;
				}
				x/=10;
			}
		}
	
	}
	
return 0;
}
