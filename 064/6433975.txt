#include<bits/stdc++.h>

using namespace std;

int main(){
    vector< pair<int,int> > num(6);
    for(int i=0;i<6;i++){
        int a,b;
        cin >> a >> b;
        if(a > b){
            swap(a,b);
        }
        num.at(i) = make_pair(a,b);
    }
    sort(num.begin(),num.end());
    int judge=0;
    for(int i=0;i<6;i+=2){
        if(num.at(i).first == num.at(i+1).first && num.at(i).second == num.at(i+1).second){
            judge++;
        }
    }
    if(judge == 3){
        cout << "yes" << endl;
    }else{
        cout << "no" << endl;
    }
    return(0);
}
