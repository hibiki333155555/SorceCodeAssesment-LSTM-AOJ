#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int m,f,r,sum;
	
	cin>>m>>f>>r;
	
	while(!(m==-1&&f==-1&&r==-1))
	{
		sum=m+f;
		if(m==-1||f==-1)
			cout<<"F"<<endl;
		else if(sum>=80)
			cout<<"A"<<endl;
		else if(sum>=65&&sum<80)
			cout<<"B"<<endl;
		else if(sum>=50&&sum<65)
			cout<<"C"<<endl;
		else if(sum>=30&&sum<50)
		{
			if(r>=50)
				cout<<"C"<<endl;
			else
				cout<<"D"<<endl;
		}
		else
			cout<<"F"<<endl;
		
		cin>>m>>f>>r;
		
	}
		
return 0;
}
