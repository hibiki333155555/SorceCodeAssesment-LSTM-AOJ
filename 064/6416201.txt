#include <stdio.h>

using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)

int cnt = 0;

bool isPrime(int x) {
    if(x==0||x==1)return false;
    for(int i=2; i*i<=x;i++) {
        if(x%i == 0) return false;
    }
    return true;
}

int main() {
    int n,a;
    cin >> n;
    rep(i,n) {
        cin >> a; 
        if(isPrime(a)) cnt++;
    }

}
