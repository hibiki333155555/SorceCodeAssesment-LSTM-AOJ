//二分累乗

ll rui(ll a,ll b){
    ll res=1;
    ll x=a;
    while(b){
        if(b&1)res=res*x%mod;
        x=x*x%mod;
        b/=2;
    }
    return res;
}

vector<ll>pow(vector<ll>a,ll n,ll k){
    vector<ll>res(n);
    rep(i,n)res[i]=i;
    vector<ll>x=a;
    while(k){
        if(k&1){
            vector<ll>c(n);
            rep(i,n){
                c[i]=res[x[i]];
            }
            res=c;
        }
        vector<ll>c(n);
        rep(i,n){
            c[i]=x[x[i]];
        }
        x=c;
        k/=2;
    }
    return res;
}

void solve(){
    stack<pll>st;
    ll n;cin>>n;
    vector<ll>now(n);
    vector<ll>a(n);
    rep(i,n)now[i]=i;
    rep(i,n)cin>>a[i];
    ll q;cin>>q;
    while(q--){
        ll t;cin>>t;
        if(t==1){
            ll x,y;cin>>x>>y;
            x--;y--;
            st.push(mp(x,y));
            swap(now[x],now[y]);
        }
        if(t==2){
            pll k=st.top();
            st.pop();
            ll x=k.fi,y=k.se;
            swap(now[x],now[y]);
        }
        if(t==3){
            ll k;cin>>k;
            vector<ll>c=pow(now,n,k);
            vector<ll>d(n);
            rep(i,n){
                d[i]=a[c[i]];
            }
            a=d;
        }
    }
    rep(i,n){
        cout<<a[i]<<' ';
    }
    cout<<endl;
}
