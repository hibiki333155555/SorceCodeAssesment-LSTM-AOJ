#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
using namespace std;



int main()
{
    int n, x, sum, tmp;
    while(true)
    {
        sum = 0;
        cin >> n >> x;
        if(!n && !x)
            break;
        
        for(int i = n; i > 0; i--)
        {
            // i + (i-1..)
            tmp = x - i;
            if(tmp >= 2*i-2)
                break;
            
            if(tmp == i)
                sum += (i-1)/2;
            else if(tmp > i)
                sum += (i-1-(tmp-(i-1))+1)/2;
            else
                sum += (tmp-1)/2;
        }
        cout << sum << endl;
    }

    return 0;
}
