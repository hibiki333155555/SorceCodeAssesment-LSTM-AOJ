
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, t;
    cin >> n >> m >> t;
    int ans = 0;
    int tmp = -m;
    int a[n];
    for (int i = 0; i < n; i++){
        cin >> a[i];
    }

    for (int i = 0; i < n; i++){
        ans += max(a[i]-tmp-2*m, 0);
        tmp = a[i];
    }

    ans += max(0, t-tmp-m);
    cout << ans << endl;
}
