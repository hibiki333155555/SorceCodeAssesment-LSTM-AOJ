#include <bits/stdc++.h>
using namespace std;
using ll=long long;
using ld=long double;
ll inf=10010010010010010;
ll mod=998244353;
int main(){
    ll n;
    cin >> n;
    vector<string>r(n);
    for (ll i = 0; i < n; i++)
    {
        cin >> r[i];
    }
    map<ll,ll>m;
    ll ans=0;
    for (ll i = 0; i < n; i++)
    {
        for (ll j = i+1; j < n; j++)
        {
            ll memo=0;
            for ( ll k = 0; k < r[i].size(); k++)
            {
                if (r[i][k]!=r[j][k])
                {
                    memo+=1;
                }
            }
            if (memo<=1)
            {
                if (m[i]==0)
                {
                    ans+=1;
                }
                if (m[j]==0)
                {
                    ans+=1;
                }
                m[i]+=1;
                m[j]+=1;
            }
            
        }
        
    }
    cout << ans << endl;
}
