#include <bits/stdc++.h>
using namespace std;

int main() {
  cin.tie(0);
  ios::sync_with_stdio(false);

  while (true) {
    int n;
    cin >> n;
    if (n == 0) break;

    vector<vector<int>> x(n, vector<int>(4));
    for (int i = 0; i < n; i++) {
      cin >> x[i][2] >> x[i][3] >> x[i][0] >> x[i][1];
      x[i][0] *= -1;
    }

    sort(x.begin(), x.end());

    map<int, int> mp;
    int total = 0;
    vector<int> ans;
    for (int i = 0; i < n; i++) {
      if (total < 10) {
        if (mp[x[i][3]] < 3) {
          total++;
          mp[x[i][3]]++;
          ans.emplace_back(x[i][2]);
        }
      }
      else if (total < 20) {
        if (mp[x[i][3]] < 2) {
          total++;
          mp[x[i][3]]++;
          ans.emplace_back(x[i][2]);
        }
      }
      else if (total < 26) {
        if (mp[x[i][3]] < 1) {
          total++;
          mp[x[i][3]]++;
          ans.emplace_back(x[i][2]);
        }
      }
    }

    for (int i = 0; i < (int) ans.size(); i++) cout << ans.at(i) << '\n';
  }
}
