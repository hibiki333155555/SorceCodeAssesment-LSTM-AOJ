#include<bits/stdc++.h>
using namespace std;

int main(){
    int N;
    cin>>N;
    int r[N];
    for(int i=0;i<N;i++){
        cin>>r[i];
    }
    int ans=1,ma=r[N-1];
    for(int i=N-2;i>=0;i--){
        if(ma<r[i]){
            ans++;
            ma=r[i];
        }
    }
    cout<<ans<<"\n";
    return(0);
}
