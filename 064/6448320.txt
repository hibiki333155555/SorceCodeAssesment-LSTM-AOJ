inline int read()
{
    int x=0, f=1; char ch=getchar();
    while(ch<'0' || ch>'9') {if(ch=='-') f=-1; ch=getchar();}
    while(ch>='0' && ch<='9') {x=x*10+ch-'0'; ch=getchar();}
    return x*f;
}
 
//假设在上餐时，打印支票
int main()
{
    int h,m,ls,le,ds,ms,me;
    m=0;
    h=11,ls=h*60+m;
    h=15,le=h*60+m;
    h=18,ds=h*60+m;
    h=21,ms=h*60+m;
    h=26,me=h*60+m;
    while (1)
    {
        int n=read();
        if (!n) break;
 
        int cnt[4],ok[4];
        memset(cnt,0,sizeof(cnt));
        memset(ok,0,sizeof(ok));
        while (n--)
        {
            int h=read(),m=read(),M=read();
            if (h<8) h+=24;
 
            int len,t;
            if (M>m)
            {
                len=M-m;
                //t=h*60+M;
                t=h*60+m;
            }
            else
            {
                len=M+60-m;
                //t=(h+1)*60+M;
                t=h*60+m;
            }
 
            if (t>=ls && t<le)
            {
                cnt[1]++;
                if (len<=8) ok[1]++;
            }
            else if (t>=ds && t<ms)
            {
                cnt[2]++;
                if (len<=8) ok[2]++;
            }
            else if (t>=ms && t<me)
            {
                cnt[3]++;
                if (len<=8) ok[3]++;
            }
        }
 
        printf("lunch ");
        if (!cnt[1]) puts("no guest");
            else printf("%d\n",ok[1]*100/cnt[1]);
        printf("dinner ");
        if (!cnt[2]) puts("no guest");

        printf("midnight ");
        if (!cnt[3]) puts("no guest");
            else printf("%d\n",ok[3]*100/cnt[3]);
    }
 
    return 0;
}
