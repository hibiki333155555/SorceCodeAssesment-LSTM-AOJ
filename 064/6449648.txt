#include<bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ll a, b; cin >> a >> b;
    if (a == 0 && b == 0) {
        cout << 1 << "\n";
        return 0;
    }
    int mx1 = 0;
    ll msk = a | b;
    while (msk > 0) {
        mx1++;
        msk >>= 1;
    }
    msk = a ^ b;
    int mx2 = 0;
    while (msk > 0) {
        mx2++;
        msk >>= 1;
    }
    int ans = mx1 - mx2;
    cout << ans << "\n";
}
