#include<stdio.h>
#include<stdlib.h>
#define N 45
int fibo(int);
int memo[N];
int main(){
    //init and input
    int n;
    for(int i=0;i<N;i++)memo[i]=0;
    memo[0]=memo[1]=1;
    scanf("%d",&n);
    //process and output
    printf("%d\n",fibo(n));
    return 0;
}
int fibo(int n){
    if(n==0||n==1||memo[n]!=0)return memo[n];
    else return memo[n]=fibo(n-2)+fibo(n-1);
}
