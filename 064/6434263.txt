#include<iostream>
using namespace std;
int h[6], w[6];

bool c(){
    for(int i = 0; i < 6; i++){
        int min = i;
        for(int j = i + 1; j < 6; j++){
            if(h[min] > h[j]){
                min = j;
            }
            else if(h[min] == h[j] && w[min] > w[j]){
                min = j;
            }
        }
        swap(h[i], h[min]);
        swap(w[i], w[min]);
    }
    if((h[0]==h[1] && w[0]==w[1] || h[0]==w[1] && w[0]==h[1]) && (h[2]==h[3] && w[2]==w[3] || h[2]==w[3] && w[2]==h[3]) && (h[4]==h[5] && w[4]==w[5] || h[4]==w[5] && w[4]==h[5])){
        return true;
    }
    else return false;
}

int main(){
    for(int i = 0; i < 6; i++){
        cin >> h[i];
    }
    for(int i = 0; i < 6; i++){
        cin >> w[i];
    }
    if(c()){
        cout << "yes" << endl;
    }
    else{
        cout << "no" << endl;
    }

}
