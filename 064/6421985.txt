#include<stdio.h>
#include<stdlib.h>
int main(){
    int cnt=0;
    int na,nb;
    int *a,*b,*c;
    scanf("%d",&na);
    a=(int *)malloc(sizeof(int)*na);
    for(int i=0;i<na;i++)scanf("%d",a+i);
    scanf("%d",&nb);
    b=(int *)malloc(sizeof(int)*nb);
    c=(int *)malloc(sizeof(int)*nb);
    for(int i=0;i<nb;i++){scanf("%d",b+i);c[i]=0;}
    for(int i=0;i<na;i++){
        for(int j=0;j<nb;j++){
            if(a[i]==b[j])c[j]++;
        }
    }
    for(int i=0;i<nb;i++)if(c[i]!=0)cnt++;
    printf("%d\n",cnt);
    return 0;
}
