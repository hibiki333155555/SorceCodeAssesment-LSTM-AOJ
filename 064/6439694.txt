#include<iostream>
using namespace std;

int main() {
    int n;
    long a, i;
    long max = -1000000, min = 1000000;
    long sum = 0;
    cin >> n;

    for (i = 0; i < n; i++) {
        cin >> a;
        sum += a;
        if (a > max) {
            max = a;
        }
        if (a < min) {
            min = a;
        }

    }
    cout << min << " " << max << " " << sum << endl;
    return 0;
}
