#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    string s,t;
    int ss=0,st=0;
    for(int i=0;i<n;i++){
        cin>>s>>t;
        if(s == t){
            ss++;
            st++;
        }else if(s < t){
            st+= 3;
        }else{
            ss+=3;
        }
    }
    cout<< ss<< " "<<st<<"\n";
    return(0);
}
