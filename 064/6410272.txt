#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    int n; cin >> n;
    string s; cin >> s;
    int a=0;
    string t="IOI";
    for(int i=0;i<n and a<3;i++){
        if(s[i]==t[a]){
            a++;
        }
    }
    if(a==3){
        cout << "Yes" << endl;
    }
    else{
        cout << "No" << endl;
    }
}
