#include<bits/stdc++.h>
using namespace std;
using ll=long long;
int main(){
  ll n,k;
  cin>>n>>k;
  vector<ll> a(n);
  ll count=0;
  ll ans=0;
  for(int i=0;i<n;i++){
    cin>>a[i];
    if(a[i]<5) count++;
    ans+=a[i];
  }
  sort(a.begin(),a.end());
  for(int i=0;i<min(count,k);i++){
    if(a[i]==1){
      ans+=4;
      a[i]=5;
    }
    else{
      ans+=6-a[i];
      a[i]=6;
    }
  }
  sort(a.begin(),a.end());
  for(int i=0;i<min(n,k-count);i++){
    if(a[i]==5) ans+=1;
  }
  cout<<ans<<endl;
}
