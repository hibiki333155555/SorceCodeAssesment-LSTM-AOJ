int32_t main() {
    int n; cin >> n;
    string s; cin >> s;
    string S;
    for (int i=0; i<n-1; i++) {
        S.push_back(s[i]);
        S.push_back('$');
    }
    S.push_back(s[n-1]);
    vector<int> R(S.size());
    int i = 0, j = 0;
    while (i < S.size()) {
    while (i-j >= 0 && i+j < S.size() && S[i-j] == S[i+j]) ++j;
    R[i] = j;
    int k = 1;
    while (i-k >= 0 && k+R[i-k] < j) R[i+k] = R[i-k], ++k;
    i += k; j -= k;
    }
    int N = S.size();
    int ans = 0;
    for (int i=0; i<N; i++) {
        if (i % 2 == 0) continue;
        int l = i + 1;
        int r = N - i;
        int m = min(l, r);
        if (R[i] >= m) ans += 1;
    }
    cout << ans << "\n";
}
