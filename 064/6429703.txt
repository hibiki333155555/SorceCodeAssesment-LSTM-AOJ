#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define all(x) x.begin(),x.end()
#define fol(i,s,g) for(int i=(int)(s); i<(int)(g); i++)
#define fix(ans,k) fixed << setprecision(k) << ans

int main(){

    vector<int> fib(2,1);
    int now_i=1;
    while(fib[now_i] < 1e8){
        now_i++;
        int k = fib[now_i-1]+fib[now_i-2];
        fib.push_back(k);
    }

    int x,y;
    cin>>x>>y;
    int rl,ud;
    x>0 ? rl=1 : rl=3;
    y>0 ? ud=2 : ud=0;
    int xk=0,yk=0,ans=1,i=1;
    while(xk<abs(x)||yk<abs(y)){
        if(i%4 == rl) xk += fib[i];
        if(i%4 == ud) yk += fib[i];
        ans++;
        if(ans == 4) ans = 1;
        i++;
    }
    cout<<ans<<'\n';
    return 0;
}
