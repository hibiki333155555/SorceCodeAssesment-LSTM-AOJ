#include <iostream>
using namespace std;

int main() {
    string w; cin>>w;
    int ans = 0;
    while(1) {
        string s; getline(cin,s);
        if(s=="END_OF_TEXT") break;

        string tmp;
        for(int i=0;i<s.size();i++) {
            if('A'<=s[i]&&s[i]<='Z') s[i]=s[i]-'A'+'a';
            if(s[i]!=' ') tmp+=s[i];
            else {
                if(tmp==w) ans++;
                tmp = "";
            }
        }
    }
    cout<<ans<<endl;
    return 0;
}
