#include <stdio.h>

int main(void){
    int mid,last,re,sekiseki;
	
    while(1){
        scanf("%d %d %d",&mid,&last,&re);
        
        if(mid==-1 && last==-1 && re==-1){break;}
        
        sekiseki=mid+last;

		if(mid==-1 || last==-1){
			printf("F\n");
		}
        else if(sekiseki>=80){
            printf("A\n");
        }
        else if(sekiseki>=65){
            printf("B\n");
        }
        else if(sekiseki>=50){
            printf("C\n");
        }
        else if(sekiseki>=30){
            if(re>=50){
                printf("C\n");
            }
            else{
            printf("D\n");
            }
        }
        else {
            printf("F\n");
        }
    }
    
}
