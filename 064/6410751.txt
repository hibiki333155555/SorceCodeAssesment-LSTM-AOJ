//int N, M, K, T, H, W, L, R;
long long int N, M, K, T, H, W, L, R;


int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);

	cin >> N >> K;
	vector<int>v(7);
	for (int i = 0; i < N; i++) {
		cin >> M;
		v[M]++;
	}
	while (K--) {
		for (int j = 1; j < 6; j++) {
			if (v[j]) {
				if (j == 1) {
					v[5]++;
				}
				else {
					v[6]++;
				}
				v[j]--;
				break;
			}
		}
	}
	int ans = 0;
	for (int i = 1; i <= 6; i++) {
		ans += v[i] * i;
	}
	cout << ans << endl;
}
