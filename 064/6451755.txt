#include <stdio.h>

int main(){
    int i,n,k[200001],min=1000000000,max=-1000000000;
    
    scanf("%d",&n);
    for(i=0;i<n;i++){
    scanf("%d",&k[i]);
    }
    min=k[0];
    for(i=1;i<n;i++){
        if(k[i]-min>max) max=k[i]-min;
        //printf("%d",max);
        if(k[i]<min) min=k[i];
    }
    printf("%d\n",max);
}
