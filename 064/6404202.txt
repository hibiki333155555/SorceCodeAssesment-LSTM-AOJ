let input = require("fs").readFileSync("/dev/stdin", "utf8").split("\n");

input.shift();
input.pop();

let taro = 0, hanako = 0;

for (let str of input) {
    if (str.split(" ")[0] === str.split(" ")[1]) {
        taro++; hanako++;
    } else if (str.split(" ")[0] > str.split(" ")[1]) {
        taro += 3;
    } else { hanako += 3; }
}
console.log(taro + " " + hanako);
