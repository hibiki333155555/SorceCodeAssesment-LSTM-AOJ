main(require("fs").readFileSync("/dev/stdin", "utf8"));

/*-- 計算&出力 --*/
function main(raw) {
    // 入力処理
    const args = raw.split("\n");
    const input = args[0].split(' ').map(Number);
    const a = input[0];
    const b = input[1];
    const c = input[2];
    // 計算
    let count = 0;
    for(let num=a; num<=b; num++){
        if(c%num==0){
            count++;
        };
    } 
    // 表示
    console.log(`${count}`);
}
