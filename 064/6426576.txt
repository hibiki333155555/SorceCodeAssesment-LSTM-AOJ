n = int(input())
a, b = 0, 0
for _ in range(n):
  w1, w2 = input().split()
  n_iter = min([len(w1), len(w2)])
  for i in range(n_iter):
    if ord(w1[i]) > ord(w2[i]):
      a += 3
      break
    elif ord(w1[i]) < ord(w2[i]):
      b += 3
      break
    else:
      if n_iter-1 == i:
        if len(w1) > len(w2): a += 3
        elif len(w1) < len(w2): b += 3
        else:
          a += 1
          b += 1
print(f"{a} {b}")
