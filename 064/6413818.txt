void  mergeSort(int  a[], int  tmpArray[], int  left, int right)
{
    if (left < right)
    {
        int  center = (left + right)/2;
        mergeSort(a, tmpArray, left, center);
        mergeSort(a, tmpArray, center + 1, right);
        merge(a, tmpArray, left, right);
    }
}
