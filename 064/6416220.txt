#include<stdio.h>
int main()    {
    int n;
    char ch1[n];
    char ch2[n];
    int sum;
    int m;
    scanf("%d", &n);
    for ( int i=0; i<n; i++ ) {
        scanf("%d", &m);
        putchar("\n");
     sum += m;
    }
for ( int i=0; i<n-sum; i++ ) {
    ch2[i] = ch1[sum+i]; 
}
for ( int i=n-sum+1; i<n; i++ ) {
    ch2[i] = ch1[i-(n-sum+1)];
}
for ( int i=0; i<n; i++ ) {    
    printf("%c", ch2[i]);
}
