#include <stdio.h>

int max(int,int,int);

int main(void){
    int d,i,x,a[3],m[3],ctr[3],c,p,maxCtr;
    
    while(1){
        scanf("%d %d %d %d %d %d",&a[0],&m[0],&a[1],&m[1],&a[2],&m[2]);
        if(a[0]*a[1]*a[2]*m[0]*m[1]*m[2]==0) break;
        
        x=1;
        
        ctr[0]=ctr[1]=ctr[2]=0;
        for(i=0;i<3;i++){
            while(1){
                x=a[i]*x%m[i];
                ctr[i]++;
                if(x==1) break;
            }
        }
        
        p=1;
        maxCtr=max(ctr[0],ctr[1],ctr[2]);
        
        for(d=2;d<=maxCtr;){
            c=0;
            for(i=0;i<3;i++){
                if(ctr[i]%d==0){
                    c++;
                }
            }
            if(c>=2){
                for(i=0;i<3;i++){
                    if(ctr[i]%d==0){
                        ctr[i]=ctr[i]/d;
                    }
                }
                p*=d;
            }
            else{
                if(d%2!=0){d+=2;}
                else{d+=1;}
            }
            if(ctr[0]*ctr[1]*ctr[2]==1) break;
        }
        printf("%d\n",p*ctr[0]*ctr[1]*ctr[2]);
        
    }
    
    return 0;
}

int max(int a, int b, int c){
    if(a>=b&&a>=c) return a;
    else if(b>=a && b>=c ) return b;
    else return c;
}
