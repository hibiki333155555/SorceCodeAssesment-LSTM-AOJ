import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner scan = new Scanner(System.in);
        
        int n = scan.nextInt();
        int x[] = new int[n];
        int benefit = 0;
        
        for(int i = 0;i < n;i++){
            x[i] = scan.nextInt();
        }
        
        for(int i = 0;i < n-1;i++){
            benefit = x[i+1] - x[i];
        }
        
        System.out.println(benefit);
    }
}
