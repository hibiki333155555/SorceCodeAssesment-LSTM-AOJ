#include <iostream>
#include <vector>
#include <string>

using namespace std;


int main() {
	int r, c;
	cin >> r >> c;
	vector<vector<int> >data(r+1, vector<int>(c+1));

	int temp = 0;
	//入力&横を入れてく
	for (int i = 0; i < r; ++i) {
		for (int j = 0; j < c; ++j) {
			cin >> data.at(i).at(j);
			temp += data.at(i).at(j);
		}
		data.at(i).at(c) = temp;
		temp = 0;
	}
	//縦
	for (int i = 0; i <= c; ++i) {
		for (int j = 0; j < r; ++j) {
			temp += data.at(j).at(i);
		}
		data.at(r).at(i) = temp;
		temp = 0;
	}
	//corner
	for (int i = 0; i < r; ++i) {
		temp += data.at(i).at(c);
	}
	data.at(r).at(c) = temp;

	//output
	for (int i = 0; i <= r; ++i) {
		for (int j = 0; j <= c; ++j) {
			cout << data.at(i).at(j) << " ";
		}
		cout << endl;
	}
}
