#include <iostream>
using namespace std;
int n, q;
int a[10000];

bool solve(int i, int m){
    if(m==0) return true;
    else if(i >= n) return false;
    else return solve(i+1, m-a[i]) || solve(i+1, m);
}
int main(){
    cin >> n;
    for(int i=0; i<n; i++){
        cin >> a[i];
    }
    cin >> q;
    while(q--){
        int x;
        cin >> x;
        if(solve(0, x)) cout << "yes" << endl;
        else cout << "no" << endl;
    }
    return 0;
}
