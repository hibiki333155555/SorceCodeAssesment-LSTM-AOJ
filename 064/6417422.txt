#pragma GCC target("avx2")
#pragma GCC optimize("O3")
#pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9;
const ll inf = 1LL<<62;

void solve() {
  int n, k; cin >> n >> k;
  priority_queue<int, vector<int>, greater<int>> q;
  int ans = 0;
  for (int i=0; i<n; i++) {
    int a; cin >> a;
    if (a == 6) ans += 6;
    else q.push(a);
  }
  for (int i=0; i<k; i++) {
    if (q.size() == 0) break;
    auto now = q.top(); q.pop();
    if (now == 1) {
      q.push(5);
    } else {
      ans += 6;
    }
  }
  while (q.size()) {
    ans += q.top();
    q.pop();
  }
  cout << ans << '\n';
}
