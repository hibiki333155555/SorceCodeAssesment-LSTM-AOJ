#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,T=0,H=0;
    string t,h;
    cin>>n;
    int j=0;
    for(int i=0; i<n; i++){
        cin>>t>>h;
        while(min(t.size(),t.size())>j){
            if(t==h){
                T++;
                H++;
                break;
            }
            else if(t[j]>h[j]){
                T+=3;
                break;
            }
            else if(t[j]<h[j]){
                H+=3;
                break;
            }
        }
    }
    cout<<T<<" "<<H<<endl;
}
