#include<bits/stdc++.h>
using namespace std;

void selectionsort(vector<int>&a){
int n=a.size();
for(int i=0;i<n-1;i++){
int m=a[i];
int mi=i;
for(int j=i+1;j<=n;j++){
if(a[j]<m){
m=a[j];
mi=j;
}
}
cout<<mi<<" ";
swap(a[i],a[mi]);
}
}
int main(){
int n;
vector<int>a(n);
for(int i=0;i<n;i++)
cin>>a[i];
selectionsort(a);
}
