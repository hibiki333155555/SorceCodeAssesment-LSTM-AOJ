#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
        int a,b;
        int d,r;
        double f;
        
        cin >> a >> b;
        
        d = a / b;
        r = a % b;
        f = (double)a / (double)b;
        
        printf("%d %d %.6lf", d, r, f);
        return 0;
}
