#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main(){
    string key, str;
    cin>>key;
    
    int c=0;
    while(true){
        cin>>str;
        if(str == "END_OF_TEXT") break;
        transform(str.begin(), str.end(), str.begin(), ::tolower);
        if(str == key) c++;
    }
    cout << c << endl;
}
