template<class T>
static inline std::vector<T> ndvec(size_t&& n, T val) { return std::vector<T>(n, std::forward<T>(val)); }
template<class... Tail>
static inline auto ndvec(size_t&& n, Tail&&... tail) {
  return std::vector<decltype(ndvec(std::forward<Tail>(tail)...))>(n, ndvec(std::forward<Tail>(tail)...));
}

void solve() {
  i64 N, S;
  cin >> N >> S;
  bool strait = false;
  vector<i64> cnt(4);
  i64 pra = 0;
  rep(i,0,N) {
    vector<i64> A(4);
    rep(j,0,4) {
      cin >> A[j];
      cnt[A[j]]++;
    }
    rep(j,0,2) {
      if(A[j] == 1 && A[j + 2] == 1) {
        strait = true;
        break;
      }
    } rep(j,0,3) {
      if(A[j] == 1 && A[j + 1] == 1) {
        pra++;
        break;
      }
    }
  }

  i64 diff = cnt[2] - cnt[3];
  if(strait) {
    if(pra == 2) {
      if(diff == 2) {
        cout << 1 << " " << 3 << " " << 1 << " " << 3 << endl;
      }
      else if(diff == 0) {
        cout << 1 << " " << 2 << " " << 1 << " " << 3 << endl;
      }
      else if(diff == -2) {
        cout << 1 << " " << 2 << " " << 1 << " " << 2 << endl;
      }
    }
    else {
      if(diff == 1) {
        cout << 1 << " " << 1 << " " << 1 << " " << 3 << endl;
      }
      else {
        cout << 1 << " " << 1 << " " << 1 << " " << 2 << endl;
      }
    }
  }
  else {
    if(pra == 4) {
      if(diff == 4) {
        cout << 3 << " " << 3 << " " << 3 << " " << 3 << endl;
      }
      else if(diff == 2) {
        cout << 3 << " " << 3 << " " << 3 << " " << 2 << endl;
      }
      else if(diff == 0) {
        if(S == 1) {
          cout << 3 << " " << 2 << " " << 3 << " " << 2 << endl;
        }
        else {
          cout << 3 << " " << 3 << " " << 2 << " " << 2 << endl;
        }
      }
      if(diff == -2) {
        cout << 2 << " " << 2 << " " << 2 << " " << 3 << endl;
      }
      else if(diff == -4) {
        cout << 2 << " " << 2 << " " << 2 << " " << 2 << endl;
      }
    }
    else {
      if(diff == 2) {
        cout << 1 << " " << 1 << " " << 3 << " " << 3 << endl;
      }
      else if(diff == 0) {
        cout << 1 << " " << 1 << " " << 2 << " " << 3 << endl;
      }
      else if(diff == -2) {
        cout << 1 << " " << 1 << " " << 2 << " " << 2 << endl;
      }
    }
  }
}

int main() {
  i64 T;
  cin >> T;
  while(T--) {
    solve();
  }
}
