#include <stdio.h>
 int main(void)
 {
     int a0,a1,a2,ans,r;
     
    scanf("%d %d %d",&a0,&a1,&a2);
    if((a1 - a0) && (a2 - a1))
    {
        ans = a0 + (a1 - a0) * 9;
    }
    else
    {
        r = (a2 / a1);
        for(int i = 0; i < 9;i++)
        {
            ans = ans + a0 *r;
        }
    }
    printf("%d\n",ans);
    return 0;
 }
