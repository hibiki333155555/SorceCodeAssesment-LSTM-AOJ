#include <bitset>
#include <iostream>
using namespace std;

int main(){
    bitset<64> b;
    int n;
    cin>>n;
    int op, e;
    while(n--){
        cin>>op;
        switch(op){
            case 0:
            cin>>e;
            cout << b.test(e)<<endl;
            break;
            case 1:
            cin>>e;
            b.set(e);
            break;
            case 2:
            cin>>e;
            b.reset(e);
            break;
            case 3:
            cin>>e;
            b.flip(e);
            break;
            case 4:
            cout << b.all() << endl;
            break;
            case 5:
            cout << b.any() << endl;
            break;
            case 6:
            cout << (b.count() == 0) << endl;
            break;
            case 7:
            cout << b.count() << endl;
            break;
            case 8:
            cout << b.to_ullong() << endl;
        }
    }
}
