#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int main() {
    string str1, str2;
    cin >> str1 >> str2;
    string str3 = str1 + str1;
    long num;
    num = str2.size();
    string ans = "No";
    for(int i = 0; i < num; i++) {
        for(int j = 0; j < num; j++) {
            if(str3[j] != str2[j]) {
                continue;
            } else if( str3[num - 1] == str2[num - 1]) {
                ans = "Yes";
                break;
            }
        }
        if (ans == "Yes") {
            break;
        } else {
            str3.erase(str3[0]);
        }
    }
    cout << ans << endl;
}
