#include<stdio.h>
#include<string.h>
#include <ctype.h>

int main(void){
    int cnt[26]={};
    char moji[1200]; 
    char *p,c;
    int i=0;

    while(gets(moji)!=NULL){
        p=moji;
        while(*p){
            c=*p;
            if(islower(c)){
                cnt[((int)p[i]-97)]++;
            }
            else if(isupper(c)){
                cnt[((int)p[i]-65)]++;
            }
            *p++;
            if((int)p[i]==32){
                *p++;
            }
        }
    }
    for(i=0;i<26;i++){
        printf("%c : %d\n",i+97,cnt[i]);
    }
    return 0;
}
