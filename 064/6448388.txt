#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    cin>>s;
    s += s;
    string p;
    cin>>p;
    bool flag = false;
    int sum=0;
    for(int i=0;i<s.size() - p.size();i++){
        if(s[i] == p[0]){
            sum = 1;
            for(int j=1;j<p.size();j++){
                if(s[i+j] == p[j]){
                   sum++; 
                }
            }
            if(sum==p.size()){
                flag = true;
                break;
            }
        }
    }
    if(flag){
        cout<<"Yes\n";
    }else{
        cout<<"No\n";
    }
    return(0);
}
