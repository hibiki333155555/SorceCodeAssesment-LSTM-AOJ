#include <iostream>
using namespace std;
typedef long long llong;

int n,k;
llong t[100000];

int check(llong p){
    int i = 0;
    for(int j = 0; j < k; j++){
        llong s = 0;
        while(s+t[i] <= p){
            s += t[i];
            i++;
            if(i == n){
                return n;
            }
        }
    }
    return i;
}

int solve(){
    llong left = 0;
    llong right = 100000*10000;
    while(right - left > 1){
        llong mid = (right + left) / 2;
        int v = check(mid);
        if(v >= n){
            right = mid;
        }else{
            left =mid;
        }
    }
    
    return right;
}

int main(void){
    cin >> n >> k;
    for(int i = 0; i < n; i++){
        cin >> t[i];
    }
    
    llong ans = solve();
    cout << ans << endl;
}
