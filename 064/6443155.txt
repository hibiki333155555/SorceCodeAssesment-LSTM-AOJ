#include <stdio.h>
 int main(void)
 {
     int a0,a1,a2,ans;
     
    scanf("%d %d %d",&a0,&a1,&a2);
    if((a1 - a0) && (a2 - a1))
    {
        ans = a0 + (a1 - a0) * 9;
    }
    else
    {
        for(int i = 1; i <= 9;i++)
        {
            ans = a0 * (a1 / a0);
        }
    }
    printf("%d\n",ans);
    return 0;
 }
