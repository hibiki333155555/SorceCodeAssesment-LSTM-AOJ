#include<stdio.h>
int main()   {
    int N;
    scanf("%d", &N);
    int c1[N];
    int c2[N(N+1)/2];
    for (int i=0;i<N;i++)  {
        scanf("%d\n", &c1[N]);
    }
   for (int i=0;i<N;i++) {
       for (int j=i;j<N;j++) {
          for (int k=0; k<i*j;k++) {
              c2[k] = c1[j] - c1[i];
          }
       }
   }
 for (int i=0;i<N(N+1)/2;i++) {
     int max = c2[0];
       if (c2[i]>max)
         max=c2[i];
 }   
 printf("%d", max);   
}
