#include <iostream>
using namespace std;

const int MAX_N = 100;

int main()
{
    int N, M;
    int pos[MAX_N + 1];
    
    cin >> N;
    for ( int i = 1; i <= N; i++ )
    {
        cin >> pos[i];
    }
    
    cin >> M;
    int move_piece;
    for ( int i = 0; i < M; i++ )
    {
        cin >> move_piece;
        if ( pos[move_piece] == 2019 ) continue;

        bool is_movable = true;
        for ( int j = 1; j <= N; j++ )
        {
            if ( j == move_piece ) continue;
            
            if ( pos[move_piece] + 1 == pos[j] )
            {
                is_movable = false;
                break;
            }            
        }
        if ( is_movable ) pos[move_piece]++;
    }
    
    for ( int i = 1; i <= N; i++ )
    {
        cout << pos[i] << endl;
    }
    
    return 0;
}
