import java.util.Scanner;

class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int a= sc.nextInt();
        int b= sc.nextInt();
        int c= sc.nextInt();
        if(c%a==0&&c%b==0){
            System.out.println(2);
        }
        else if(c%a==0||c%b==0){
            System.out.println(1);
        }
        else{
            System.out.println(0);
        }
    }
}
