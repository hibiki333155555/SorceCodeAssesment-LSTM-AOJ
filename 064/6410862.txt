#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <iterator>
#include <numeric>
#include <string>
#include <vector>
using namespace std;

class Dice
{
public:
    Dice(int t, int s, int e, int w, int n, int b) : T(t), S(s), E(e), W(w), N(n), B(b){};
    void goS()
    {
        tmp = B;
        B = S;
        S = T;
        T = N;
        N = tmp;
    }
    void goN()
    {
        tmp = B;
        B = N;
        N = T;
        T = S;
        S = tmp;
    }
    void goE()
    {
        tmp = B;
        B = E;
        E = T;
        T = W;
        W = tmp;
    }
    void goW()
    {
        tmp = B;
        B = W;
        W = T;
        T = E;
        E = tmp;
    }

    int getT() { return T; }
    int getE() {return E;}
    int getB() {return B;}
    int getW() {return W;}
    int getN() {return N;}
    int getS() {return S;}
    int getMin()
    {
        return min({T,B,W,E,N,S});
    }
    int getMax()
    {
        return max({T,B,W,E,N,S});
    }

    void setT(int a) 
    {
        
        if (a == W) goE();
        else if(a == N) goS();
        else if(a == E) goW();
        else if(a == S) goN();
        else if(a == B) {goE();goE();}
        else if(a == T) return;
    }

    void rotate()
    {
        tmp = E;
        E = S;
        S = W;
        W = N;
        N = tmp;
    }

    bool setS(int a){   //CALL setT before setS
        int i = 0;
        for(;S != a&&i<4;i++)
            rotate();
        if (i>=4)
            return false;
        return true;
    }

    bool check(int a)
    {
        if(a == T || a == S || a == E || a == W || a == N || a == B)
            return true;
        return false;
    }

    bool operator==(Dice& dice)
    {
        setT(getMax());
        dice.setT(getMax());
        while(B != dice.getB())
            dice.setT(getMax());
        
        if(dice.setS(S) == false)
            return false;
        if(N==dice.N && W==dice.W && E==dice.E)
            return true;
        return false;
    }

private:
    int T, S, E, W, N, B, tmp;
};

int main()
{
    int a, b, c, d, e, f;
    cin >> a >> b >> c >> d >> e >> f;
    Dice dice1(a, b, c, d, e, f);
    
    cin >> a >> b >> c >> d >> e >> f;
    Dice dice2(a, b, c, d, e, f);

    
    if(dice1 == dice2)
        cout << "Yes" << endl;
    else
        cout << "No" << endl;
    return 0;
}
