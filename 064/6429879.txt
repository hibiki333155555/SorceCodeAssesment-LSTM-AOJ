#include <bits/stdc++.h>
using namespace std;
int main(){
    while(true) {
        int a, ans = 0, cnt = 0;
        cin>>a;
        if(a == 0) return(0);
        for(int i=0;i<a;i++) {
            int b;
            cin>>b;
            ans = max(b,ans);
            if(cnt != 0) cnt++;
        }
        if(ans >= 2){
            cout<<cnt+1<<endl;
        }else{
            cout<<"NA"<<endl;
        }
    }
    return(0);
}
