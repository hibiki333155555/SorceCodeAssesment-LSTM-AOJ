#include<bits/stdc++.h>

using namespace std;

int main(){
    vector< pair<int,int> > num(6);
    for(int i=0;i<6;i++){
        int a,b;
        cin >> a >> b;
        if(a>b){
            swap(a,b);
        }
        num.at(i) = make_pair(a,b);
    }
    sort(num.begin(),num.end());
    int judge = 0;
    for(int i=0;i<6;i+=2){
        if(num.at(i).first == num.at(i+1).first && num.at(i).second == num.at(i+1).second){
            judge++;
        }
    }
    if(judge != 3){
        cout << "no" << endl;
        return(0);
    }
    judge = 0;
    int n1 = num.at(0).first;
    int n2 = num.at(2).first;
    int n3 = num.at(2).second;
    int n4 = num.at(4).second;
    int n5 = num.at(0).second;
    int n6 = num.at(4).first;
    if(n1==n2&&n3==n4&&n5==n6){
        cout << "yes" << endl;
    }else{
        cout << "no" << endl;
    }
    return(0);
}
