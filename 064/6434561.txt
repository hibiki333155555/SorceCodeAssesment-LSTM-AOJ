#include<bits/stdc++.h>
using namespace std;
vector <int> graph[1000];
int seen[500] , check[500];
bool kuhn(int p)
{
     for(int i=0;i<graph[p].size();i++)
    {
        int q = graph[p][i];
        if(seen[q]) continue;
        seen[q]=1;
        if(check[q]==-1||kuhn(check[q]))
        {
            check[q]=p;
            return true;
        }
    }
    return false;
}

int main()
{
    memset( check,-1,sizeof(check));
    int m,n,t;
    cin>>m>>n>>t;
    int u,v,cnt=0;
    for(int i=0;i<t;i++)
     {
         cin>>u>>v;
         graph[u].push_back(v);
     }
    for(int i=0;i<m;i++)
    {
        memset(seen,0,sizeof(seen));
        if(kuhn(i)) cnt++;
    }
    cout<<cnt<<endl;

}
