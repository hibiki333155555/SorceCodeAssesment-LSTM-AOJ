#include <iostream>
#include <vector>
#include <map>
using namespace std;
typedef pair<int, int> P;

int main()
{
	int n;
	cin >> n;
	map<int, vector<P>> mp;
	for(int i = 0; i < n; i++){
		for(int j = 0; j < n; j++){
			int a;
			cin >> a;
			mp[a].push_back(P(i, j));
		}
	}
	int q;
	cin >> q;
	while(q--){
		int t;
		cin >> t;
		if(mp.count(t)) cout << mp[t][0].first + 1 << " " << mp[t][0].second + 1 << endl;
		else cout << -1 << endl;
	}
}
