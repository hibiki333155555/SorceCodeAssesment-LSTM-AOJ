#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main(){
    int n;
    cin>>n;
    
    vector<int> pt(2, 0);
    
    vector<string> str(2), srt(2);
    for(int i = 0; i < n; i++){
        cin>>str.at(0)>>str.at(1);
        if(str.at(0)==str.at(1)){
          pt.at(0)++;
          pt.at(1)++;
          continue;
        }
        srt = str;
        sort(srt.begin(), srt.end());
        if(srt.at(0)==str.at(0)) pt.at(1) += 3;
        else pt.at(0) += 3;
    }
    cout << pt.at(0) << " " << pt.at(1) << endl;
    
    return 0;
}
