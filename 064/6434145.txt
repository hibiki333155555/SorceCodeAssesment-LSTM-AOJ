#include <stdio.h>
#define N 12
int main(void){
    int i,j,e[N],task;
    for(i=0;i<N;i++)
    scanf("%d",&e[i]);
    for(i=0;i<N-1;i++){
        for(j=0;j<N;j++){
            if(e[i]>e[j]){
                task=e[j];
                e[j]=e[i];
                e[i]=task;
            }
        }
    }
    if(e[0]==e[3]&&e[4]==e[7]&&e[8]==e[11]){
        printf("yes\n");
    }else{
        printf("no\n");
    }
    return 0;
}
