n = int(input())
A = [*map(int, input().split())]

#疑似コードのPythonコード化
def partition(A, p, r):
  i = p - 1
  for j in range(p, r):
    if A[j] <= A[r]:
      i += 1
      A[i], A[j] = A[j], A[i]
  A[i+1], A[r] = A[r], A[i+1]
  return i+1

#分割された数列の出力
p = partition(A, 0, n-1)
print(*A[:p], f"[{A[p]}]", *A[p+1:])
