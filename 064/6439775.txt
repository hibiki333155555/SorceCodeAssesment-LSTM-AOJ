import java.util.ArrayList;
import java.util.Scanner;

public class Main {
	public static void main(String arg[]) {
		Main main = new Main();
		main.Run();
	}
	int N ;
	Data ans ;
	
	int minnum;
	ArrayList<Data> list;
	
	public void Run() {
		N = 0;
		ans = new Data(Integer.MAX_VALUE);
		ans.setCount(Integer.MAX_VALUE);
		
		list = new ArrayList<Data>();
		
		Scanner scan = new Scanner(System.in);
		
		N = scan.nextInt();
		
		for(int n = 0;n < N;n++) {
			int num = scan.nextInt();
			Search(num);
		}
		
		Answer();
		
		System.out.println(ans.getNum());
		
		scan.close();
	}
	
	void Search(int num) {
		for (Data data : list) {
			if(data.getNum() == num) {
				data.CountUp();
				return;
			}
		}
		list.add(new Data(num));
	}
	
	void Answer() {
		for (Data data : list) {
			if(ans.getCount() > data.getCount() 
					|| ans.getCount() == data.getCount() && ans.getNum() > data.getNum()) {
				ans = data;
			}
		}
	}
}

class Data{
	private int num ;
	private int count;
	
	public Data(int num) {
		this.setNum(num);
		setCount(1);
	}
	
	public void CountUp() {
		setCount(getCount() + 1);
	}

	public int getCount() {
		return count;
	}

	public void setCount(int count) {
		this.count = count;
	}

	public int getNum() {
		return num;
	}

	public void setNum(int num) {
		this.num = num;
	}
}
