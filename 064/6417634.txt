using namespace std;
using P = pair<int,int>;

using ld = long double;

const ld EPS = 1e-10;
const ll MOD = 1000000007LL;
const ll INF = 1LL << 60;
const double PI = M_PI;

int main(){

  int n;
  ll x;
  string command;
  list<ll> l;
  cin >> n;
  for( int i = 0 ; i < n ; i++ ){
    cin >> command;
    if( command == "insert" ){
      cin >> x;
      l.push_front(x);
    }
    else if( command == "delete" ){
      cin >> x;
      auto itr = find(l.begin(), l.end(), x);
      if( itr != l.end() )
        l.erase(itr);
    }
    else if( command == "deleteFirst" )
      l.pop_front();
    else
      l.pop_back();
  }
  auto itr = l.begin();
  cout << *itr;
  for( itr++ ; itr != l.end() ; itr++ )
    cout << " " << *itr;
  cout << endl;

}
