#include <bits/stdc++.h>
using namespace std;

void print(vector<int>v){
	for(int i=0;i<v.size();i++){
		if(i!=v.size()-1){
		cout<<v[i]<<" ";
		
	}
	else{
		cout<<v[i];
	}
	cout<<"\n";
	}
	
}
void BubbleSort(vector<int>&arr){
	int n=arr.size();
	int count=0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(arr[j+1]<arr[j]){
				swap(arr[j+1],arr[j]);
				count++;
			}
		}
	}
	print(arr);
	cout<<count<<"\n";
} 
int main() {
	int n;
	cin>>n;
	vector<int>v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
		
	}
	BubbleSort(v);
	return 0;
}
