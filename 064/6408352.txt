#include<stdio.h>
int main(){
    int card[4][13];
    int a,b,i,j,k;
    char x;
    
    for(i=0;i<4;i++){
        for(j=0;j<13;j++){
            card[i][j]=0;
        }
    }
    scanf("%d",&a);
    
    for(k=0;k<a;k++){
        scanf("%c %d",&x,&b);
        
        if(x=='S'){
            card[0][b]=1;
        }else if(x=='H'){
            card[1][b]=1;
        }else if(x=='C'){
            card[2][b]=1;
        }else if(x=='D'){
            card[3][b]=1;
        }
    }
       for(j=0;j<13;j++){
           if(card[0][j]==0){
               printf("S %d\n",j);
           }
    } 
       for(j=0;j<13;j++){
           if(card[1][j]==0){
               printf("H %d\n",j);
           }
    } 
       for(j=0;j<13;j++){
           if(card[2][j]==0){
               printf("C %d\n",j);
           }
    } 
       for(j=0;j<13;j++){
           if(card[3][j]==0){
               printf("D %d\n",j);
           }
    } 
}
