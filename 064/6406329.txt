#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(int)(n);i++)

struct Point{
    double x;
    double y;
};

double Koch(int d,Point l,Point r){
    if(d==0){
        return 1;
    }
    Point s,u,t;
    // 内分の式
    s.x=(2*l.x+r.x)/3;
    s.y=(2*l.y+r.y)/3;
    t.x=(l.x+2*r.x)/3;
    t.y=(l.y+2*r.y)/3;
    // ベクトルの回転公式
    double a = M_PI/3;
    u.x=(t.x-s.x)*cos(a)-(t.y-s.y)*sin(a)+s.x;
    u.y=(t.x-s.x)*sin(a)+(t.y-s.y)*cos(a)+s.y;
    Koch(d-1,l,s);
    printf("%.8f %.8f\n",s.x,s.y);
    Koch(d-1,s,u);
    printf("%.8f %.8f\n",u.x,u.y);
    Koch(d-1,u,t);
    printf("%.8f %.8f\n",t.x,t.y);
    Koch(d-1,t,r);
    return 1;
}

int main(){
    int n;
    cin>>n;
    vector<Point> P(pow(4,n));
    Point p1,p2;
    p1.x=0.0; p1.y=0.0;
    p2.x=100.0;   p2.y=0.0;
    printf("%.8f %.8f\n",p1.x,p1.y);
    Koch(n,p1,p2);
    printf("%.8f %.8f\n",p2.x,p2.y);
    return 0;
}
