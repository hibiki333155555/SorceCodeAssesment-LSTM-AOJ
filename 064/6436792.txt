#include <bits/stdc++.h>
#define lol long long
using namespace std;
int main(){
    int n,x,sum=0;
    while(true){
        cin>>n>>x;
        if(n==0&&x==0){
            break;
        }
        sum=0;
        for(int i=1;i<=n;i++){
            for(int j=i+1;j<=n;j++){
                for(int k=j+1;k<=n;k++){
                    if(i+j+k == x){
                        sum++;
                    }else if(i+j+k > x){
                        break;
                    }
                }
            }
        }
        cout<<sum<<"\n";
    }
    return 0;
}
