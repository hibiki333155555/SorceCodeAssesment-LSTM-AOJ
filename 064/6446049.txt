int main() {
  ll n; cin >> n;
  vector<ll> dp(n+6*10, false);
  dp[n] = true;
  for(ll i=n-1; i>=1; --i) {
    ll digitSum = 0;
    ll j = i;
    while(j != 0) {
      digitSum += j % 10;
      j /= 10;
    }
    dp[i] = dp[i+digitSum];
  }
  ll ans = 0;
  for(ll i=1; i<dp.size(); ++i) {
    if(dp[i]) {
			++ans;
    };
  }
  cout << ans << endl;
  return 0;
}
