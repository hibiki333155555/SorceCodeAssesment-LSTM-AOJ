#include<iostream>
#include<string>
using namespace std;
int main() {
    string W;
    cin >> W;
    int n = W.size();
    int count = 0;
    while(1) {
        string T;
        cin >> T;
        if(T == "END_OF_TEXT") {
            break;
        } else if(T == W) {
            count++;
        } else {
            for(int num = 0; num < n; num++) {
                if(T.size() != n) {
                    break;
                } else {
                    if(isupper(W[num])) {
                        T[num] = tolower(T[num]);
                        if(W[num] != T[num]) {
                            break;
                        }
                    } else {
                        T[num] = toupper(T[num]);
                        if(W[num] != T[num]) {
                            break;
                        }
                    }
                }
            }
        }
    }
    cout << count << endl;
}
