#include<iostream>
using namespace std;
int a[10],b[10];
int main(){
    int n;
    cin>>n;
    while(n--) {
        cin >> a[0];
        int p = 0, q = 0;
        bool can = true;
        for (int i = 1; i < 10; i++) {
            int x;
            cin >> x;
            if (x > a[p])
                a[++p] = x;
            else if (q == 0)
                b[q++] = x;
            else if (x > b[q])
                b[++q] = x;
            else {
                can = false;
                break;
            }
        }
        if (can)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
    return 0;
}
