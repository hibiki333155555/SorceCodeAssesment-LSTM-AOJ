#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n; cin>>n;
    vector<int> s(13,0),h(13,0),c(13,0),d(13,0);
    while(n--) {
        char suit; int rank; cin>>suit>>rank;
        if(suit=='S') s[rank-1]=1;
        if(suit=='H') h[rank-1]=1;
        if(suit=='C') c[rank-1]=1;
        if(suit=='D') d[rank-1]=1;
    }
    for(int i=0;i<13;i++) if(s[i]==0) cout<<"S "<<i+1<<endl;
    for(int i=0;i<13;i++) if(h[i]==0) cout<<"H "<<i+1<<endl;
    for(int i=0;i<13;i++) if(c[i]==0) cout<<"C "<<i+1<<endl;
    for(int i=0;i<13;i++) if(d[i]==0) cout<<"D "<<i+1<<endl;
    return 0;
}
