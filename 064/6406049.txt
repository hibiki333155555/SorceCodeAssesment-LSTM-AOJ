#include <bits/stdc++.h>
using namespace std;

int main(){
    int r,c;
    cin>>r>>c;
    int sum1=0;
    vector<vector<int>> vec(r+1,vector<int>(c+1));
    for(int i=0; i<r; i++){
        for(int j=0; j<c; j++){
            cin>>vec.at(i).at(j);
            sum1+=vec.at(i).at(j);
        }
        vec.at(i).at(c)=sum1;
        sum1=0;
    }
    int sum2=0;
    for(int i=0; i<c+1; i++){
        for(int j=0; j<r; j++){
            sum2+=vec.at(j).at(i);
        }
        vec.at(r).at(i)=sum2;
        sum2=0;
    }
    for(int i=0; i<r+1; i++){
        for(int j=0; j<c; j++){
            cout<<vec.at(i).at(j)<<" ";
        }
        cout<< vec.at(i).at(c)<<endl;
    }
}
