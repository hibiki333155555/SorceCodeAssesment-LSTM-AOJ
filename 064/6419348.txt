#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    int ma = -1000000;
    int mi = 1000000;
    long long int sum = 0;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        int a;
        cin >> a;
        sum += a;
        //if (a > max) max = a;
        //else if (a < min) min = a;
        ma = max(ma, a);
        mi = min(mi, a);
    }
    printf("%d %d %lld\n", mi, ma, sum);
}
