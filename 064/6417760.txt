#include <iostream>
using namespace std;

int main()
{
    int x[3], y[3];
    for(int i = 0; i < 3; i++) cin >> x[i];
    for(int i = 0; i < 3; i++) cin >> y[i];
    int c = 0;
    for(int i = 0; i < 3; i++) c += abs(x[i] - y[i]) % 2;
    if(c % 3) cout << "No" << endl;
    else cout << "Yes" << endl;
}
