#include <iostream>
#include <list>
using namespace std;

int main()
{
    int n;
    cin >> n;

    // spade, heart, club, diamond
    bool cards[52];
    for (int i = 0; i < 52; i++)
    {
        cards[i] = false;
    }

    char c;
    int input;
    for (int i = 0; i < n; i++)
    {
        cin >> c >> input;
        int suit;
        switch (c)
        {
        case 'S':
            suit = 0;
            break;

        case 'H':
            suit = 1;
            break;

        case 'C':
            suit = 2;
            break;

        case 'D':
            suit = 3;
            break;

        default:
            suit = -1;
            break;
        }

        cards[suit * 13 + input - 1] = true;
        // cout << "card '" << c << ' ' << input << "' is in the list" << endl;
    }

    for (int i = 0; i < 52; i++)
    {
        if (!cards[i])
        {
            // cout<<"i: " << i <<endl;
            switch (i / 13)
            {
            case 0:
                c = 'S';
                break;

            case 1:
                c = 'H';
                break;
            case 2:
                c = 'C';
                break;

            case 3:
                c = 'D';
                break;
            default:
                break;
            }

            cout << c << ' ' << i % 13 + 1 << endl;
        }
    }

    return 0;
}
