#include <iostream>
#include <cstdio>
#include <string>
#include <vector>

using namespace std;

bool isNumber(string s)
{
  for (int i = 0; i < s.size(); i++)
  {
    if (!isdigit(s[i])) return false;
  }
  
  return true;
}

int main(int argc, char const *argv[])
{
  string s;

  getline(cin, s);

  vector<string> v;
  string temp = "";
  for (int i = 0; i < s.length(); i++)
  {
    if (s[i] == ' ')
    {
      v.push_back(temp);
      temp = "";
    }
    else
    {
      temp += s[i];
    }
  }
  v.push_back(temp);

  vector<int> nums;
  for (int i = 0; i < v.size(); i++)
  {
    if (isNumber(v[i]))
    {
      nums.push_back(stoi(v[i]));
    }
    else
    {
      int a2 = nums.back();
      nums.pop_back();
      int a1 = nums.back();
      nums.pop_back();

      if (v[i] == "+")
      {
        nums.push_back(a1 + a2);
      }
      else if (v[i] == "*")
      {
        nums.push_back(a1 * a2);
      }
      else
      {
        nums.push_back(a1 - a2);
      }
    }
  }
  
  cout << nums[0] << endl;

  return 0;
}
