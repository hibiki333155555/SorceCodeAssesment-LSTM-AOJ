<?php

fscanf(STDIN, '%d', $N);
$tmp = explode(' ', trim(fgets(STDIN)));

for ($i=1; $i < $N; $i++) { 
    $v = $tmp[$i];
    $j = $i-1;

    while($j >= 0 && $tmp[$j] > $v){
        $tmp[$j+1] = $tmp[$j];
        $j--;
    }

    $tmp[$j+1] = $v;

    for ($j=0; $j < $N; $j++) { 
        printf("%d",$tmp[$j]);
        if($j !== $N-1){
            printf(" ");
        }
        else{
            printf("\n");
        }
    }
}
