#include <stdio.h>

int main(){
	int countTable[256] = {0};
	char ch;

	while((ch = getchar()) != EOF){//chに文字が読み込まれている間
				countTable[ch]++;//すべての種類の文字がその文字コードに対応する場所がインクリメントされる
	}//chtableの中で小文字は97~122
	for(int i = 97; i <=122;i++)
	printf("%c : %d\n",i,countTable[i]+countTable[i-32]);//大文字部分も足す

}
