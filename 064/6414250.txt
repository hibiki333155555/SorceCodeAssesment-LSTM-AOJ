import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        int a = sc.nextInt();
        int b = sc.nextInt();
        
        int d = a/b;
        int r = a%b;
        float f = a/double(b);
        
        System.out.printf("&d %d %f\n",d,r,f);
    }
}
