#include<bits/stdc++.h>
using namespace std;
int n,m,cnt;
struct Edge{
    int to,next,now;
}edge[1010];
int head[1010];
int in[1010],vis[1010];
void add(int a,int b){
    edge[cnt].to=b;
    edge[cnt].next=head[a];
    edge[cnt].now=0;
    head[a]=cnt++;
}
int main(){
    cnt=0;
    memset(head,-1,sizeof(head));
    memset(in,0,sizeof(in));
    memset(vis,0,sizeof(vis));
    cin>>n>>m;
    for(int i=0;i<m;i++){
        int a,b;
        cin>>a>>b;
        if(a!=b)add(a,b),in[b]++;
    }
    int cntt=n;
    int neww;
    while(cntt>0){
        neww=0;
        for(int i=0;i<n;i++){
            if(in[i]==0&&vis[i]==0){
                neww++,cntt--,vis[i]=1;
                for(int j=head[i];j>=0;j=edge[j].next){
                    in[edge[j].to]--;
                }
            }
        }
        if(neww==0)break;
    }
    if(cnt==0)cout<<"0"<<endl;
    else cout<<"1"<<endl;
    return 0;
}
