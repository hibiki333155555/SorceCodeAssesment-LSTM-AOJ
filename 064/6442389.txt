#include<bits/stdc++.h>
using namespace std;
long long int so,luythua,RES;
long long int LUYTHUA(long long int so, long long int luythua){
	if(luythua==0){
		return 1;
	}
	RES=LUYTHUA(so,luythua/2);
	if(luythua%2==1){
		return so*RES*RES%1000000007;
	}
	return RES*RES%1000000007;
}
int main(){
	long long int x,y;
	cin>>x>>y;
	cout<<LUYTHUA(x,y)<<'\n';
}
