#include <bits/stdc++.h>

using namespace std;
using ll = long long;

int main() {
   cout << fixed << setprecision(15);

   ll n;
   cin >> n;

   vector<ll> S;

   for (int i = 0; i < n; i++) {
      ll a;
      cin >> a;

      S.emplace_back(a);
   }

   ll q;
   cin >> q;

   ll count = 0;

   for (ll i = 0; i < q; i++) {
      ll v;
      cin >> v;

      auto it = lower_bound(S.begin(), S.end(), v);

      if (*it == v) {
         count++;
      }
   }

   cout << count << endl;

   return 0;
}
