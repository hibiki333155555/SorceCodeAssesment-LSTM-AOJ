#include <iostream>

using namespace std;

bool linearSearch(int target, int *nums, int size)
{
  int i = 0;
  while (target != nums[i])
  {
    i++;
    if (i == size) return false;
  }

  return true;
}

int main(int argc, char const *argv[])
{
  int n;
  cin >> n;
  int *S = new int[n];
  for (int i = 0; i < n; i++)
  {
    cin >> S[i];
  }
  
  int q;
  cin >> q;
  int count = 0;
  for (int i = 0; i < q; i++)
  {
    int num;
    cin >> num;
    if (linearSearch(num, S, n)) count++;
  }

  cout << count << endl;

  return 0;
}
