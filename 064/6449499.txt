template <typename T>
ostream& operator<<(ostream& os, const vector<T>& v) {
    for (int i = 0; i < int(v.size()); i++) {
        os << v[i];
        if (i < int(v.size()) - 1) os << ' ';
    }
    return os;
}

void solve(){
    int A,B;
    cin>>A>>B;
    if(A==0 && B==0){
        cout<<1<<endl;
        return;
    }
    int la,lb;
    rep(i,31){
        if(A>>i&1) la=i;
        if(B>>i&1) lb=i;
    }
    // cout<<la<<" "<<lb<<endl;
    if(la<lb) swap(la,lb);
    swap(A,B);
    int cnt=0;
    for(int i=la;i>=0;i--){
        if((A>>i&1)==(B>>i&1)) cnt++;
        else break;
    }
    cout<<cnt<<endl;
}
