N,W=map(int,input().split())

v_w=[]
for i in range(N):
    v,w=map(int,input().split())
    v_w.append([v,w])


dp=[[0]*(W+1) for i in range(N+1)]

for i in range(N):
    for j in range(W+1):
        if j-v_w[i][1]>=0:
            dp[i+1][j]=max(dp[i][j-v_w[i][1]]+v_w[i][0],dp[i][j])

        else:
            dp[i+1][j]=dp[i][j]

print(dp[N][W])
