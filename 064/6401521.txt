import java.util.*;
public class Main {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int one = 0;
    HashSet<Integer> meow = new HashSet<Integer>();
    int place = 0;
    for(int b = 0; b<2; b++){
      one = sc.nextInt();
      for(int a = 0; a<one; a++){
        place = sc.nextInt();
        meow.add(place);
      }
    }
    List<Integer> list = new ArrayList<Integer>(meow);
    Collections.sort(list);
    for(int x:list){
      System.out.println(x);
    }
  }
}
