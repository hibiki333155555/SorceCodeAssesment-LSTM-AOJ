#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin >> n;
    long long small = LLONG_MAX,big = LLONG_MIN,sum = 0;
    for(int i = 0;i < n;i++){
        long long a;
        cin >> a;
        small = min(small,a);
        big = max(big,a);
        sum += a;
    }
    cout << small << ' ' << big << ' ' << sum << endl;
}
