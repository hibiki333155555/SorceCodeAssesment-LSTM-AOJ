#include<iostream>
#include <vector>
#include <algorithm>

using namespace std;

void _print(vector<int>&v){
    bool first = true;
    for(auto& e:v){
        if(first) first = false;
        else cout <<" ";
        cout << e;
    }
    cout << endl;
    
    return;
}

int main(){
    int n;
    cin>>n;
    vector<int>v;
    for(int i = 1; i<=n;i++){
        v.push_back(i);
    }
    do{
        _print(v);
    }while(next_permutation(v.begin(), v.end()));
    
    return 0;
}
