#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,p=0;
	cin>>n;
	vector<int> a(n);
	for(int i=0; i<n; i++){
		cin>>a[i];
	}
	int l=a.size();
	for(int i=0; i<l; i++){
		for(int j=0; j<l; j++){
			if(a[j-1]>a[j]){
				swap(a[j],a[j-1]);
				p++;
			}
		}
	}
	for(int i=0; i<l; i++){
		cout<<a[i];
		if(i<l-1){
			cout<<" ";
		}
		
	}
	cout<<"\n";
	cout<<p<<"\n";
	return 0;
}
