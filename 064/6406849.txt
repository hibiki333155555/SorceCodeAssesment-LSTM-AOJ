#define per(i,j,k) for(int i=(j);i>=(k);i--)

using namespace std;
const int MAXN=2e5+10;
const double eps=1e-8;
int sgn(double x){
	if(fabs(x)<=eps) return 0;
	else if(x<0) return -1;
	else return 1;
}
struct Node{
    double x,y;Node(){}
    Node(double _x,double _y){x=_x;y=_y;}
    void input(){cin>>x>>y;}
    Node friend operator+(Node a,Node b){return Node(a.x+b.x,a.y+b.y);}
    Node friend operator-(Node a,Node b){return Node(a.x-b.x,a.y-b.y);}
    Node friend operator*(Node a,double b){return Node(a.x*b,a.y*b);}
    double friend operator*(Node a,Node b){return a.x*b.x+a.y*b.y;}
    double friend operator^(Node a,Node b){return a.x*b.y-b.x*a.y;}
    void rot(double a){
        double x1=x*cos(a)-y*sin(a);
        double y1=y*cos(a)+x*sin(a);
        x=x1;y=y1;
    }
};
Node p[MAXN];
double dist(Node a,Node b){return sqrt((a-b)*(a-b));}
Node stk[MAXN];int top;
double cdq(int l,int r){
	if(l==r) return 200*sqrt(2);
	if(l+1==r) return dist(p[l],p[r]);
	int mid=(l+r)>>1;
	double ans=min(cdq(l,mid),cdq(mid+1,r));
	top=0;
	rep(i,l,r) if(fabs(p[mid].x-p[i].x)<=ans) stk[++top]=p[i];
	sort(stk+1,stk+1+top,[&](Node a,Node b){return a.y<b.y;});
	rep(i,1,top) rep(j,i+1,top){
		if(stk[j].y-stk[i].y>=ans) break;
		ans=min(ans,dist(stk[i],stk[j]));
	}return ans;
}
int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);cout.tie(0);
	int n;cin>>n;
	rep(i,1,n) p[i].input();
	sort(p+1,p+1+n,[&](Node a,Node b){return a.x==b.x?a.y<b.y:a.x<b.x;});
	cout<<fixed<<setprecision(11)<<cdq(1,n)<<'\n';
	return 0;
}
