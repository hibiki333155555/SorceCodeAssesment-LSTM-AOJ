#include <iostream>
#include <vector>
#include <string>
using namespace std;
using ll = long long;

ll mygcd(ll x, ll y){
    while(y != 0){
        ll r = x%y;
        x = y; y = r;
    }
    return x;
}

int main(){
    string s; cin >> s;
    ll a = 0;
    ll aketa = 0;
    for(auto &it: s){
        if('0' <= it && it <= '9'){
            a *= 10;
            a += it-'0';
            aketa *= 10;
        }
        if(it == '.') aketa = 1;
    }
    ll b = 0;
    ll bketa = 0;
    for(auto &it: s){
        if('0' <= it && it <= '9'){
            b *= 10;
            b += it-'0';
            bketa *= 10;
        }
        if(it == '.') bketa = 1;
        if(it == '(') break;
    }
    
    // cerr << a << " " << aketa << " " << b << " " << bketa << endl;
    if(aketa != bketa){
        ll c = a-b, d = aketa-bketa;
        ll g = mygcd(c, d);
        c /= g; d /= g;
        cout << c << "/" << d << endl;
    }else{
        ll g = mygcd(a, aketa);
        cout << a/g << "/" << aketa/g << endl;
    }
    return 0;
}
