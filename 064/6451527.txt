#include <stdio.h>
int main(){
    int a,b,c;
    int l,m,h;
    scanf("%d%d%d",&a,&b,&c);
    
    if(a<b && a<c){
        l=a 
        if(b<c) m=b,h=c
        else m=c,h=c
    }
    
    else if(b<a & b<c){
        l=b
        if(a<c) m=a,h=c
        else m=c,h=a
    }
    else { 
        l=a 
        if(a<b) m=a,h=b
    }
    printf("\n");
    printf("%d%d%d",&l,&m,&h);
}
