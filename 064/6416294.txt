#include<iostream>

int main(){
    int n;
    int curp;
    int maxdiff;
    int minip;
    std::cin>>n;
    int p1,p2;
    std::cin>>p1>>p2;
    maxdiff=p2-p1;
    minip=p1<p2?p1:p2;
    for(int i=0;i!=n-2;++i){
        std::cin>>curp;
        if(curp-minip>maxdiff){
            maxdiff=curp-minip;
        }
        if(curp<minip){
            minip=curp;
        }
    }
    std::cout<<maxdiff;
}
