#include <bits/stdc++.h>
#include <vector>
using namespace std;

int main() {
    
    vector<int> a(3);
    for(int i=0; i<3; i++) {
        cin >> a[i];
    }
    sort(a.begin(),a.end());
    for(int i=0; i<3; i++) {
        if(i==2) {
            cout << a[i] << endl;
        }
        else {
            cout << a[i] << " ";
        }
    }
    return 0;
    
}
