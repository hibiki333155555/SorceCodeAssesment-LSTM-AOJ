import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);

        int x;
        int result;

        System.out.println("Input: ");
        x = scan.nextInt();

        if (x >= 100 || x <=1) {
            System.out.println("Not allowed");
        }
		else {

            result = x * x * x;
            System.out.println(result);
        }
    }

}
