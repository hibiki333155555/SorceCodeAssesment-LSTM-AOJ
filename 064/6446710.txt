#include<stdio.h>
#include<ctype.h>

int main() {
  char s[2001];
  long long int c[26] = {0};
  scanf("%[^.]*s", s);
  for (int i = 0; s[i] != '\0'; i++) {
    if (s[i] == 32) {
      continue;
    }
    s[i] = toupper(s[i]);
    c[s[i]-65] += 1;
  }
  for (int i = 0; i < 26; i++) {
    printf("%c : %lld\n", i+65+32, c[i]);
  }
  return 0;
}
