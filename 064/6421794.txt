#include <bits/stdc++.h>
using namespace std;

int main() {
    
    long long H1,W1,H2,W2;
    cin >> H1 >> W1 >> H2 >> W2;
    long long h,w;
    if(H1%H2 == 0) {
        h = H1/H2;
    }
    else {
        h = H1/H2 + 1;
    }
    if(W1%W2 == 0) {
        w = W1/W2;
    }
    else {
        w = W1/W2 + 1;
    }
    cout << h*w << endl;
    return 0;
    
}
