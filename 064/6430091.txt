#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
using namespace std;

int main()
{
   int n, m;
   cin >> n >> m;
   vector<int> a(n);
   for (auto &v : a)
      cin >> v;
   sort(a.begin(), a.end());

   vector<int> b(m);
   for (auto &v : b)
      cin >> v;
   sort(b.begin(), b.end());

   vector<int> result;
   set_intersection(a.begin(), a.end(), b.begin(), b.end(), back_inserter(result));
   set<int> st;
   for (auto &v : result)
      st.insert(v);

   for (auto &v : st)
      cout << v << endl;
   return 0;
}
