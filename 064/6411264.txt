#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    int n,e;
    cin>>n;
    vector<int> a,b;
    while(n--){
        cin>>e;
        a.push_back(e);
    }
    cin>>n;
    while(n--){
        cin>>e;
        b.push_back(e);
    }
    cout << includes(a.begin(), a.end(), b.begin(), b.end()) << endl;
}
