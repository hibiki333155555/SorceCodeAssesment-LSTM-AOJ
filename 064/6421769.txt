#include <stdio.h>

int main(void){
    int n;
    scanf("%d", &n);
    
    int a[n];
    int i;
    for(i = n-1; i >= 0; i-- )
        int x;
        scanf("%d", &x)
        a[n] = x;
    };
    
    int j;
    for(j = 0; j < n; j++){
        printf("%d", a[j]);
        if(j != n -1) printf(" ");
    }
    
    return 0;
}
