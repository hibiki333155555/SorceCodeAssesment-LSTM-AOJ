int sta[200];
int top=0;

int pop()
{
	top--;
	return sta[top+1];
}

void push(int x)
{

	top++;
	sta[top]=x;
}

int cal(int x,int y,char op)
{
	switch(op){
		case '+':return x+y;
		case '-':return y-x;
		case '*':return x*y;
	}	
}

int main()
{	
	int re,operand,a,b,c=0,i;
	char operato;
	char s[100];
	
	while(~scanf("%s",s)) 
	{
		if(s[0]>='0'&&s[0]<='9')
		{
			operand=atoi(s);
			push(operand);
			
		/*	cout<<top<<" ";
			for(i=1;i<=top;i++)
				cout<<sta[i]<<" ";
			cout<<endl;
		*/		
		}
		else if(s[0]==' ');//空格滤掉 
		else 	//字符 
		{
			operato=s[0];
			a=pop();
			b=pop();
			c=cal(a,b,operato);
			push(c);
		}
	}
	
	re=pop();
	cout<<re<<endl;
	
return 0;
}
