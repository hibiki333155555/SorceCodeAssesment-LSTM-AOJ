#include <iostream>
#include <numeric>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    while(true) {
        int n;
        cin >> n;
        break;
        if (n == 0) {
            break;
        }
        int a[n];
        for (int i=0;i<n;n++) {
            cin >> a[i];
        }
        double average = (double)accumulate(a, a+n, 0)/(double)n;
        double aru = 0.0;
        for (int i=0;i<n;i++) {
          aru += pow(((double)a[i]-average), 2);   
        }
        cout << fixed << setprecision(4) << sqrt(aru/(double)n) << endl;
    }
    return 1;
}
