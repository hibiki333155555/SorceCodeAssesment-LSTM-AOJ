#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int main() {
   int n;
   cin >> n;
   int b, f, r, v;

   int apart[4][3][10];
   for(int i=0; i<4; i++) {
      for(int j=0; j<3; j++) {
         for(int k=0; k<10; k++) {
            apart[i][j][k] = 0;
         }
      }
   }

   for(int i=0; i<n; i++) {
      cin >> b >> f >> r >> v;
      apart[b-1][f-1][r-1] += v;
   }

   for(int i=0; i<4; i++) {
      for(int j=0; j<3; j++) {
         for(int k=0; k<10; k++) {
            cout << " " << apart[i][j][k];
         }
         cout << "\n";
      }
      if(i!=3) cout << "####################\n";
   }
}
