r, c = map(int, input().split())
A = [[0 for i in range(c + 1)] for j in range(r + 1)]

for i in range(r):
    lst = list(map(int, input().split()))
    for j in range(c):
        A[i][j] = lst[j]
    
for i in range(r + 1):
    for j in range(c):
        A[i][c] += A[i][j] # 行計
        if i < r:
            A[r][j] += A[i][j] # 列計
            
    print(f'{" ".join(str(n) for n in A[i])}')
