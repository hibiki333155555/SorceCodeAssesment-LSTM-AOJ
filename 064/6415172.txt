#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

const int N = 10005;
int tree[N];
int root;
int n;

int main() {
	cin>>n;
	memset(tree,-1,sizeof(tree));
	for(int i=1;i<=n;i++) {
		cin>>tree[i];
	}
	root = 1;
	for(int i=1;i<=n;i++) {
		cout<<"node "<<i<<": key = "<<tree[i]<<", ";
		if(i != root)
			cout<<"parent key = "<<tree[i/2]<<", ";
		if(tree[i<<1]>=0)
			cout<<"left key = "<<tree[i<<1]<<", ";
		if(tree[(i<<1) + 1]>=0)
			cout<<"right key = "<<tree[(i<<1) + 1]<<", ";
		cout<<endl;
	}
}
