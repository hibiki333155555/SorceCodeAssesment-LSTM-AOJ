#define bit(n, k) ((n >> k) & 1)

using Vector = pair<long, long>;

class Program {
private:

    bool max(int &a, const int b) {
        if (a < b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    Vector p0, p1, p2, p3;
    int q;

public:

    Program() {
    }

    long dot(const Vector &a, const Vector &b) {
        return a.first * b.first + a.second * b.second;
    }

    long cross(const pair<float, float> &a, const pair<float, float> &b) {
        return a.first * b.second - a.second * b.first;
    }

    pair<double, double> projection(const Vector &a, const Vector &b) {
        auto [x, y] = b;

        if (x == 0 && y == 0) {
            return make_pair(0.0, 0.0);
        } else if (x == 0) {
            return make_pair(0.0, 1.0 * a.second);
        } else if (y == 0) {
            return make_pair(1.0 * a.first, 0.0);
        } else {
            return make_pair(
                    1.0 * dot(a, b) / dot(b, b) * x,
                    1.0 * dot(a, b) / dot(b, b) * y);
        }
    }

    pair<double, double> rejection(const Vector &a, const Vector &b) {
        auto [x, y] = projection(a, b);
        return make_pair(x - a.first, y - a.second);
    }

    int counterClockWise(const Vector &a, const Vector &b) {
        long z = cross(a, b);

        if (z > 0) {
            // cout << "COUNTER_CLOCKWISE" << endl;
            return 1;
        }

        if (z < 0) {
            // cout << "CLOCKWISE" << endl;
            return -1;
        }

        assert(z == 0);

        if (dot(a, b) < 0) {
            // cout << "ONLINE_BACK" << endl;
            return 2;
        } else if (dot(a, a) < dot(b, b)) {
            // cout << "ONLINE_FRONT" << endl;
            return -2;
        } else {
            // cout << "ON_SEGMENT" << endl;
            return 0;
        }
    }

    bool intersection(const Vector &a, const Vector &b, const Vector &c, const Vector &d) {
        auto [x0, y0] = a;
        auto [x1, y1] = b;

        auto [x2, y2] = c;
        auto [x3, y3] = d;

        Vector ab = make_pair(x1-x0, y1-y0);
        Vector ac = make_pair(x2-x0, y2-y0);
        Vector ad = make_pair(x3-x0, y3-y0);

        if (counterClockWise(ab, ac) * counterClockWise(ab, ad) > 0) {
            return false;
        }

        Vector cd = make_pair(x3-x2, y3-y2);
        Vector ca = make_pair(x0-x2, y0-y2);
        Vector cb = make_pair(x1-x2, y1-y2);

        if (counterClockWise(cd, ca) * counterClockWise(cd, cb) > 0) {
            return false;
        }

        return true;
    }

    pair<double, double> crossPoint(const Vector &a, const Vector &b, const Vector &c, const Vector &d) {
        auto [x0, y0] = a;
        auto [x1, y1] = b;
        auto [x2, y2] = c;
        auto [x3, y3] = d;

        Vector cd = make_pair(x3-x2, y3-y2);
        Vector ca = make_pair(x0-x2, y0-y2);
        Vector ab = make_pair(x1-x0, y1-y0);

        double w = 1.0 * cross(ca, cd) / cross(cd, ab);

        auto [x, y] = ab;

        return make_pair(x0 + w * x, y0 + w * y);
    }

    void solve() {
        cin >> q;

        rep(i, 0, q) {
            auto& [x0, y0] = p0; cin >> x0 >> y0;
            auto& [x1, y1] = p1; cin >> x1 >> y1;

            auto& [x2, y2] = p2; cin >> x2 >> y2;
            auto& [x3, y3] = p3; cin >> x3 >> y3;

            auto [x, y] = crossPoint(p0, p1, p2, p3);
            printf("%.8f %.8f\n", x, y);
        }

        // printf("%.8f\n", 0.00000001);
    }
};

int main() {

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();

    return 0;
}
