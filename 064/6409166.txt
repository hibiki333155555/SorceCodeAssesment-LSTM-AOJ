#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main(){
    int n, e;
    cin>>n;
    vector<int> v;
    while(n--){
        cin>>e;
        v.push_back(e);
    }
    cin>>n;
    while(n--){
        int a,b,c;
        cin>>a>>b>>c;
        rotate(v.begin()+a, v.begin()+b, v.begin()+c);
    }
    for(int i=0;i<v.size();i++){
        cout << v.at(i) <<( (i == v.size()-1) ? "\n" : " ");
    }
    
    return 0;
}
