#include <bits/stdc++.h>
using namespace std;

int main() {
	int w, h, x, y, r;
	cin >> w >> h >> x >> y >> r;

	int maxX, maxY, minX, minY;
	maxX = x + r;
	minX = x - r;
	maxY = y + r;
	minY = y - r;

	if (maxX <= w && maxY <= h && minX >= 0 && minY >= 0) {
		printf("Yes\n");
	} else {
		printf("No\n");
	}

	return (0);
}
