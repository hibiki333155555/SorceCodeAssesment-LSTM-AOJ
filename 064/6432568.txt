template <typename T>
pair<VI, vector<T>> compress(const vector<T> &a) {
    int n = a.size();
    vector<T> x;
    REP(i, n) x.push_back(a[i]);
    sort(ALL(x)); x.erase(unique(ALL(x)), x.end());
    VI res(n);
    REP(i, n) res[i] = lower_bound(ALL(x), a[i]) - x.begin();
    return make_pair(res, x);
}

template <typename T>
pair<vector<T>, vector<T>> factorial(int n) {
    vector<T> res(n+1), rev(n+1);
    res[0] = 1;
    REP(i, n) res[i+1] = res[i] * (i+1);
    rev[n] = 1 / res[n];
    for(int i = n; i > 0; i--) {
        rev[i-1] = rev[i] * i;
    }
    return make_pair(res, rev);
}

void solve(){
    int k; cin >> k;
    VI n(k); cin >> n;
    VVL _a(k);
    REP(i, k) {
        _a[i].resize(n[i]);
        cin >> _a[i];
    }
    using Basis = set<ll, greater<ll>>;
    vector<Basis> a(k);
    REP(_, k) {
        sort(ALL(_a[_]));
        reverse(ALL(_a[_]));
        for(ll e : _a[_]) {
            for(ll b : a[_]) {
                chmin(e, e ^ b);
            }
            if(e) a[_].insert(e);
        }
    }
    ll ans = 0;
    vector<Basis> basises(1 << k);
    REP(_, 1 << k) {
        if(_ == 0) {
            continue;
        }
        VI indexes;
        REP(i, k) if(_ >> i & 1) indexes.push_back(i);
        if(indexes.size() == 1) {
            basises[_] = a[indexes[0]];
            ans += 1LL << a[indexes[0]].size();
            continue;
        }
        int index = indexes.back();
        set<LP, greater<LP>> mat;
        Basis basis;
        for(ll base : basises[_ ^ (1 << indexes.back())]) mat.emplace(base, 0LL);
        for(ll ai : a[index]) {
            ll ai_b = ai;
            for(auto [b, b_only] : mat) {
                if(chmin(ai, ai ^ b)) ai_b ^= b_only;
            }
            if(ai_b == 0) continue;
            if(ai == 0){
                basis.emplace(ai_b);
            }
            else mat.emplace(ai, ai_b);
        }
        for(ll e : basis) {
            for(ll b : basises[_]) chmin(e, e ^ b);
            if(e) basises[_].insert(e);
        }
        if(__builtin_popcount(_) % 2) {
            ans += 1LL << basises[_].size();
        }
        else ans -= 1LL << basises[_].size();
    }
    print(ans);
}
