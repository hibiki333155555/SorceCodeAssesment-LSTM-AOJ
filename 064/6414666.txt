ll mypow(ll a, ll b, ll m = mod) {
    ll x = 1ll;
    while (b) {
        while (!(b & 1ll)) {
            (a *= a) %= m;
            b >>= 1ll;
        }
        (x *= a) %= m;
        b--;
    }
    return x;
}

vec fact, rfact;

void c3_init() {
    fact.resize(fact_table + 5);
    rfact.resize(fact_table + 5);
    fact[0] = rfact[0] = 1;
    for (ll i = 1; i <= fact_table; i++) {
        fact[i] = (fact[i - 1] * i) % mod;
    }
    rfact[fact_table] = mypow(fact[fact_table], mod - 2, mod);
    for (ll i = fact_table; i >= 1; i--) {
        rfact[i - 1] = rfact[i] * i;
        rfact[i - 1] %= mod;
    }
    return;
}

ll c3(ll n, ll r) {
    if (fact.size() == 0) {
        cout << "c3_init() is not called." << endl;
        exit(1);
    }
    if (n < 0) return 0ll;
    if (r < 0) return 0ll;
    if (n < r) return 0ll;
    return fact[n] * rfact[r] % mod * rfact[n - r] % mod;
}

bool icpc = false;
bool multicase = false;

bool solve() {
    ll n;
    cin >> n;

    vec a(n + 1);

    ll s1 = 0, s2 = 0, s3 = 0, a1 = 0, a2 = 0;
    ll ans = 1ll;

    c3_init();

    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i] == 1 || a[i] == 5 || a[i] == 7) {
            if (a[i] == 1) s1++;
            if (a[i] == 5) s2++;
            if (a[i] == 7) s3++;
        } else if (a[i] == 6) {
            ans *= c3(a1 + a2, a1);
            ans %= mod;
            a1 = 0;
            a2 = 0;
        } else if (a[i] == 2 || a[i] == 4 || a[i] == 8) {
            a1++;
        } else {
            a2++;
        }
    }

    ans *= c3(a1 + a2, a1);
    ans %= mod;

    ans *= c3(n, s1);
    ans %= mod;
    ans *= c3(n - s1, s2);
    ans %= mod;
    ans *= c3(n - s1 - s2, s3);
    ans %= mod;

    p(ans);

    return true;
}
