#include <bits/stdc++.h> 
using namespace std;
#define inf 2147483647

int main(void) {
  int N, v;
  cin >> N;
  vector<int> A(N);
  for (int i = 0; i < N; i++) {
    cin >> A[i];
  }
  for (int i = 0; i < N; i++) {
    cout << A[i];
    if (i != N - 1) {
      cout << ' ';
    } else {
      cout << endl;
    }
  }
  for (int i = 1; i < N; i++) {
    v = A[i];
    int j = i - 1;
    while (j >= 0 && A[j] > v) {
      A[j + 1] = A[j];
      j--;
    }
    A[j + 1] = v;
    for (int j = 0; j < N; j++) {
      cout << A[j];
      if (j != N - 1) {
        cout << ' ';
      } else {
        cout << endl;
      }
    }
  }
}
