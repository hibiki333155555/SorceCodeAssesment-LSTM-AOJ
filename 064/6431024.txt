#include <iostream>

using namespace std;

int main() {
	int r, c;
	cin >> r >> c;
	int a[r + 1][c + 1] = {};
	for (int i = 0; i < r; ++i) {
		for (int j = 0; j < c; ++j) {
			cin >> a[i][j];
		}
	}
	for (int i = 0; i < r; ++i) {
		for (int j = 0; j < c; j++) {
			a[i][c] += a[i][j];
			a[r][j] += a[i][j];
		}
	}
	for (int j = 0; j < c; j++) {
		a[r][c] += a[r][j];
	}
	for (int i = 0; i < r + 1; ++i) {
		for (int j = 0; j < c + 1; j++) {
			cout << a[i][j];
			if (j != c) {
				cout << " ";
			}
		}
		cout << endl;
	}
	return 0;
}
