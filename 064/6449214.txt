const x = require("fs").readFileSync("/dev/stdin","utf-8").split("\n");
const ary = x[1].split(' ').map(Number);
const aryMax = (a,b)=>{return Math.max(a,b)};
const aryMin = (a,b)=>{return Math.min(a,b)};
const arySum = (a,b)=>{return a+b};
console.log(`${ary.reduce(aryMin)} ${ary.reduce(aryMax)} ${ary.reduce(arySum)}`);
