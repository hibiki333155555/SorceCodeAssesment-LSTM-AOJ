#include <stdio.h>
#include <math.h>

int main(void){
    double x1,x2,x3,y1,y2,y3;
    double a,b,c,d,e,f,x,y,r;
    int i,n;
    
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%lf %lf %lf %lf %lf %lf",&x1,&y1,&x2,&y2,&x3,&y3);
        
        a=(x2+x3)/2;
        b=(y2+y3)/2;
        c=(x2-x3)/(y3-y2);
        d=(x1+x2)/2;
        e=(y1+y2)/2;
        f=(x1-x2)/(y2-y1);
        
        if((y3-y2)==0){
            x=a;
            y=f*a-d*f+e;
        }
        else if((y2-y1)==0){
            x=d;
            y=c*d-a*c+b;
        }
        else{
            x=(e-b+a*c-d*f)/(c-f);
            y=c*x-a*c+b;
        }
        r=sqrt(pow(x-x1,2)+pow(y-y1,2));
        printf("%.3lf %.3lf %.3lf\n",x,y,round(r*1000)/1000);
    }
    
    return 0;
}
