#include <iostream>

int main()
{
    int a,b=0;

    while (1) {
        std::cin >> a >> b;
        for (int i = 0; i < a; i++) {
            for (int j = 0; j < b ; j++) {
                if ((i+j) % 2 == 1) {
                    std::cout << ".";
                }
                else {
                    std::cout << "#";
                }
            }
            std::cout << "\n";
        }
        if (a ==0 && b == 0) {
            break;
        }
    }
}
