int sta[200];
int top=0;

int pop()
{
	top--;
	return sta[top+1];
}

void push(int x)
{

	top++;

	sta[top]=x;
}

int cal(int x,int y,char op)
{
	switch(op){
		case '+':return x+y;
		case '-':return y-x;
		case '*':return x*y;
	}	
}

int main()
{	
	int re,operand,a,b,c=0,i;
	char ch,operato;
	
	while((ch=getchar())!='\n') 
	{
		if(ch>='0'&&ch<='9')
		{
			operand=ch-'0';
			push(operand);
			
		/*	cout<<top<<" ";
			for(i=1;i<=top;i++)
				cout<<sta[i]<<" ";
			cout<<endl;
		*/		
		}
		else if(ch==' ');//空格滤掉 
		else 		
		{
			operato=ch;
			a=pop();
			b=pop();
			c=cal(a,b,ch);
			push(c);
		}
	}
	
	re=pop();
	cout<<re<<endl;
	
return 0;
}
