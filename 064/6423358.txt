#include <iostream>
#include <vector>

using namespace std;

int Fib(int n)
{
    std::vector<int> fib(n + 2);
    fib[0] = 1;
    fib[1] = 1;
    int previous_fib = 1;
    for (int i = 2; i <= n; i++)
    {
        // fib[i] = fib[i - 1] + fib[i - 2];
        fib[i] = previous_fib + previous_fib - fib[i - 1];
        previous_fib += fib[i];
    }
    return fib[n];
}

int main(void)
{
    int n;
    std::cin >> n;
    std::cout << Fib(n) << std::endl;
    return 0;
}
