#include<bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<int> ord(8);
  vector<pair<int, int>> con(n);
  
  int x, y;
  for(int i=0;i<n;i++){
    cin >> x >> y;
    con[i] = make_pair(x, y);
  }
  for(int i=0;i<8;i++)ord[i]=i;
  
  do{
    
    bool skp=0;
    for(int i=0;i<n;i++){
      if(ord[con[i].first] != con[i].second){
        skp=1;
        break;
      }
    }
    if(skp)continue;
    
    vector<vector<char>> ans(8, vector<char>(8, '.'));
    for(int i=0;i<8;i++){
      ans[i][ord[i]] = 'Q';
    }
    
    for(int i=0;i<8;i++){
      int cnt0=0, cnt1=0, cnt2=0, cnt3=0;
      for(int j=0;j<=i;j++){
        if(ans[i-j][j]=='Q')cnt0++;
        if(ans[7-i+j][7-j]=='Q')cnt1++;
        if(ans[i-j][7-j]=='Q')cnt2++;
        if(ans[7-i+j][j]=='Q')cnt3++;
      }
      if(cnt0>1 || cnt1>1 || cnt2>1 || cnt3>1){
        skp=1;
        break;
      }
    }
    if(skp)continue;
    
    for(int i=0;i<8;i++){
      for(int j=0;j<8;j++){
        cout << ans[i][j];
      }
      cout << endl;
    }
  }while(next_permutation(ord.begin(), ord.end()));
  
  return 0;
}
