import java.util.Scanner;
import java.util.SortedMap;
import java.util.TreeMap;

public class Main {

	public static final int BIG_NUM = 2000000000;
	public static final long HUGE_NUM = 4000000000000000000L;

	@SuppressWarnings("resource")
	public static void main(String[] args) {

		Scanner scanner = new Scanner(System.in);

		int numQ = scanner.nextInt();
		TreeMap<String,Integer> MAP = new TreeMap<String,Integer>();

		int command,x;
		StringBuilder sb = new StringBuilder();

		for(int loop = 0; loop < numQ; loop++) {

			//System.out.printf("loop;%d\n", loop);

			command = scanner.nextInt();
			String key = scanner.next();
			if(command == 0) {

				x = scanner.nextInt();
				MAP.put(key,x);

			}else if(command == 1){

				if(MAP.get(key) == null) {

					sb.append("0\n");

				}else {

					sb.append(Integer.toString(MAP.get(key))).append("\n");
				}
			}else if(command == 2) {

				MAP.remove(key);

			}else {

				String R = scanner.next();
				SortedMap<String,Integer> work = MAP.subMap(key, R);
				for(String k : work.keySet()) {

					sb.append(k).append(" ").append(Integer.toString(work.get(k))).append("\n");
				}
			}
		}
		System.out.print(sb.toString());
	}
}

class UnionFind{
	UnionFind(int arg_N){
		this.N = arg_N;
		this.boss = new int[N];
		this.height = new int[N];
		this.num_member = new int[N];
	}
	public void init() {
		for(int i = 0; i < N; i++) {

			this.boss[i] = i;
			this.height[i] = 0;
			this.num_member[i] = 0;
		}
	}

	public int get_boss(int id) {
		if(this.boss[id] == id) {
			return id;
		}else {

			return this.boss[id] = get_boss(this.boss[id]);
		}
	}

	public Boolean isSame(int x,int y) {

		return get_boss(x) == get_boss(y);
	}

	public void unite(int x,int y) {

		int boss_x = get_boss(x);
		int boss_y = get_boss(y);

		if(boss_x == boss_y)return;

		if(this.height[boss_x] > this.height[boss_y]){

			this.num_member[boss_x] += this.num_member[boss_y];
			this.boss[boss_y] = boss_x;

		}else if(this.height[boss_x] < this.height[boss_y]){

			this.num_member[boss_y] += this.num_member[boss_x];
			this.boss[boss_x] = boss_y;

		}else{ //height[boss_x] == height[boss_y]

			this.num_member[boss_x] += this.num_member[boss_y];
			this.boss[boss_y] = boss_x;
			this.height[x]++;
		}
	}

	private int N;
	private int boss[],height[],num_member[];
}

class Info{
	Info(int arg_L,int arg_R,int arg_value){
		this.L = arg_L;
		this.R = arg_R;
		this.value = arg_value;
	}
	public int L,R,value;
}

class Edge implements Comparable<Edge>{
	Edge(int arg_from,int arg_to,int arg_cost){
		this.from = arg_from;
		this.to = arg_to;
		this.cost = arg_cost;
	}

	public int compareTo(Edge arg) {

		//辺の重みの昇順
		if(this.cost < arg.cost) {

			return -1;
		}else if(this.cost > arg.cost) {

			return 1;
		}else {

			return 0;
		}
	}

	public int from,to;
	public int cost;
}

class State implements Comparable<State>{
	State(int arg_node,int arg_sum_dist){
		this.node = arg_node;
		this.sum_dist = arg_sum_dist;
	}

	public int compareTo(State arg) {

		//■距離の昇順(PQ)
		if(this.sum_dist < arg.sum_dist) {

			return -1;
		}else if(this.sum_dist > arg.sum_dist) {

			return 1;

		}else {

			return 0;
		}
	}

	public int node;
	public int sum_dist;
}
