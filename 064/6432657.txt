int main() {
	int n; std::cin >> n;
	std::vector<std::vector<std::pair<int, int>>> graph(n);
	for (auto i = 0; i < n - 1; ++i) {
		int u, v; char c;
		std::cin >> u >> v >> c; --u; --v;
		graph[u].emplace_back(v, c - 'a');
		graph[v].emplace_back(u, c - 'a');
	}
	std::vector<int> distance(n, -1); distance[0] = 0;
	std::queue<int>  queue; queue.push(0);
	while (!queue.empty()) {
		const auto current = queue.front(); queue.pop();
		for (const auto [next, c] : graph[current]) {
			if (distance[next] >= 0) continue;
			distance[next] = distance[current] ^ (1 << c);
			queue.push(next);
		}
	}
	std::vector<int> count(1 << 26, 0);
	for (const auto d : distance) {
		count[d] += 1;
	}
	long long int result{ 0 };
	for (auto pattern = 0; pattern < count.size(); ++pattern) {
		const long long int c = count[pattern];
		if (c == 0) continue;
		result += c * (c - 1) / 2;
		for (auto i = 0; i < 26; ++i) {
			if ((pattern >> i) & 1) continue;
			result += c * count[pattern ^ (1 << i)];
		}
	}
	std::cout << result << '\n';
}
