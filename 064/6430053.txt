#include <bits/stdc++.h>
using namespace std;

int main(){
  int q;
  cin >> q;
  vector<int> v;
  for(int i=0; i<q; i++){
    int a;
    cin >> a;
    if(a==0){
      int x;
      cin >> x;
      v.push_back(x);
    }else if(a==1){
      int p;
      cin >> p;
      cout << v.at(p) << endl;
    }else{
      v.pop_back();
    }
  }
  return 0;
}
