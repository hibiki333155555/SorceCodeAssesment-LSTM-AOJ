#define _GLIBCXX_DEBUG
#include<iostream>
#include<stdio.h>
using namespace std;

void rec(int idx, int n, int sum, vector<int>& A, unordered_set<int>& S){
    if(idx == n)
        break;
    S.insert(sum);
    
    rec(idx+1,n,sum,A,S);
    rec(idx+1,n,sum+A[idx],A,S);
};

int main(){
    //入力
    int n,q;
    scanf("%d",n);
    
    vector<int> A(n);
    for(int i=0;i<n;i++)
        scanf("%d",A[i]);
    
    scanf("%d",q);
    vector<int> Q(q);
    
    for(int i=0;i<q;i++)
        scanf("%d",Q[i]);
    
    unordered_set<int>S;
    
    //再起関数を実行
    for(int i=0;i<n;i++)
        rec(i,n,A[i],A,S);
    
    for(auto q:Q){
        if(S.count(q))
            printf("yes\n");
        else
            printg("no\n");
    }
    
    return 0;
    
};
