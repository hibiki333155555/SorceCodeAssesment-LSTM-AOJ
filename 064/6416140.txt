#include <cstdio>

int main(void)
{
  int h = 0, w = 0;
  while (1) {
    scanf("%d %d", &h, &w);
    if (h == 0 && w == 0)
      break;

    for (int i = 0; i < h; i = i + 1) {
      for (int j = 0; j < w; j = j + 1) {
        if ((i + j) % 2 == 0)
          putchar('#');
        else
          putchar('.');
      }
      putchar('\n');
    }
    putchar('\n');
  }
  return 0;
}
