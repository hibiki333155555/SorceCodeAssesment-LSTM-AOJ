import java.util.Arrays;
import java.util.Scanner;

@SuppressWarnings("resource")
public class Main {
    public static void main(String[] args) {
    	int[] array = new int[3];
    	for (int i = 0; i < array.length; i++) {
		    Scanner sc = new Scanner(System.in);
    	    array[i] = sc.nextInt();
    	}
    	Arrays.sort(array);
    	
    	for (int i = 0; i < array.length; i++) {
    		System.out.print(array[i] + " ");
    	}
    }
}
