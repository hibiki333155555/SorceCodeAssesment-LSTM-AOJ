#include <iostream>
#include <map>
#include <utility>
using namespace std;
using ll = int64_t;

void solve(){
    int N; cin >> N;

    map<ll, pair<int, int>> mp;
    for(int i = 0; i < N; i++){
        for(int j = 0; j < N; j++){
            ll a; cin >> a;
            mp.insert(make_pair(a, make_pair(i + 1, j + 1)));
        }
    }

    int Q; cin >> Q;
    while(Q--){
        ll t; cin >> t;
        auto it = mp.find(t);
        if(it != mp.end()){
            cout << (*it).second.first << " " << (*it).second.second << '\n';
        }
        else cout << "-1\n";
    }
}

int main(){
    cin.tie(nullptr);
    ios_base::sync_with_stdio(false);
    solve();
    return 0;
}
