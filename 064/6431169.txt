#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n;cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++)
	{
		cin>>v[i];
	}
	int key,j;
	for(int i=0;i<n;i++)
	{
		j=i-1;
		key=v[i];
		while(j>=0 && key<v[j])
		{
			v[j+1]=v[j];
			j--;
		}
		v[j+1]=key;
		for(int k=0;k<n;k++)
		{
			cout<<v[k]<<" ";
		}
		cout<<"\n";
	}
	
	return 0;
}
