#include <bits/stdc++.h>
using namespace std;
const int INF = 1001001001;
const int MOD = 1000000007;

int main(){
    int n;cin>>n;
    int t;cin>>t;
    int time = 0;
    queue<pair<string,int>> q;
    vector<pair<string,int>> a(n);
    for(int i=0;i<n;i++) {
        string a;int b;cin>>a>>b;
        q.push(make_pair(a,b));
    }
    while(!q.empty()) {
        pair<string,int> p = q.front();
        q.pop();
        if(p.second<=t) {
            cout << p.first << " " << time + p.second << endl;
            time += p.second;
        }
        else {
            p.second -= t;
            time += t;
            q.push(p);
        }
    }
}
