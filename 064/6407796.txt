#include <iostream>
#include <string>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
	int n;
	string a, b;
	cin >> n;
	int aa = 0, bb = 0;

	for(int i = 0; i < n; i++)
	{
		cin >> a >> b;
		if (a > b)
			aa = aa + 3;
		else if (a < b)
			bb = bb + 3;
		else
		{
			aa++;
			bb++;
		}
			
	}
	cout << aa << " " << bb << endl;


}
