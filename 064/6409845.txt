#define REP(i,n) for(int i=0;i<(n);i++)
#define ALL(v) v.begin(),v.end()
#define debug(a) cerr<<#a<<":"<<a<<endl;
#define debug2(a,b) cerr<<"("<<#a<<","<<#b<<"):("<<a<<","<<b<<")"<<endl;
#define debug3(a,b,c) cerr<<"("<<#a<<","<<#b<<","<<#c<<"):("<<a<<","<<b<<","<<c<<")"<<endl;

template<typename T>
istream& operator>>(istream&is,vector<T>&v){
  for(T&p:v)is>>p;
  return is;
}
template<typename T>
ostream& operator<<(ostream&os,const vector<T>&v){
  if(&os==&cerr)os<<"[";
  for(int i=0;i<v.size();i++){
    os<<v[i];
    if(i+1<v.size())os<<(&os==&cerr?",":" ");
  }
  if(&os==&cerr)os<<"]";
  return os;
}

vector<vector<pair<int,int>>> g(123456);
map<int,int> mp;
long long ans;
int now;

void dfs(int idx,int pre){
  REP(i,26)ans+=mp[now^(1<<i)];
  ans+=mp[now];
  mp[now]++;
  for(const auto&[to,c]:g[idx])if(to!=pre){
    now^=1<<c;
    dfs(to,idx);
    now^=1<<c;
  }
}

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n;cin>>n;
  REP(_,n-1){
    int u,v;char c;cin>>u>>v>>c;u--;v--;
    g[u].emplace_back(v,c-'a');
    g[v].emplace_back(u,c-'a');
  }
  dfs(0,-1);
  cout<<ans<<endl;
}
