#include <iostream>
using namespace std;
void trace(int A[],int N){
    for (int i=0;i<N;i++) {
        if (i>0)printf(" ");
        printf("%d",A[i]);
    }
    printf("\n");
}

int BubbleSort(int A[],int N){
    static int count=0;
    for (int i=0;i<N;i++){
        for (int j=N-1;j>=1;j--){
            if( A[j]<A[j-1]){
                count++;
                swap(A[j],A[j-1]);
            }
        }
    }
    return count;
}

int main(void){
    // Your code here!
    int N;
    cin>>N;
    int A[N];
    for(int i=0;i<N;i++)cin>>A[i];
    
    int count=BubbleSort(A,N);
    trace(A,N);
    printf("%d",count);
}
