using namespace std;

int main() {
  int N,Q;
  cin >>N>>Q;
  vector<int> a(N);
  for (int i = 0; i < N; i++) cin >> a[i];
  
  for (int q = 0; q < Q; q++)
  {
    long long x;
    cin >>x;
  long long sum=0;
  int ptr=0;
  long long ret=0;
  for (int i = 0; i < N; i++){
    while (ptr<N&&(sum+a[ptr])<=x){
      sum+=a[ptr];
      ptr++;
    }
    ret+=ptr-i;
    if(ptr==i){
      ptr++;
    }else{
      sum-=a[i];
    }
  }
  cout << ret << endl;
  }
  

return 0;
}
