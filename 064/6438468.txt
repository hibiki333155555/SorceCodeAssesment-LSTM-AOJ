import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int idx = 1;
        while(true){
            int n = scanner.nextInt();
            if(n == 0) break;

            System.out.printf("Case %d: %d\n",idx++, n);
        }
    }
}
