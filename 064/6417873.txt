    vector<int> cnt(m);
    int t = 0;
    for (int i = 0; i < min(n,m+1); i++) {
        if (i == 0) {
            t = v%m;
        }
        else {
            t = (t*a + b) % m;
        }
        cnt[t]++;
        if (cnt[t] == 2) {
            printf("Yes\n"); return;
        }
    }
    printf("No\n");
    return;
}
