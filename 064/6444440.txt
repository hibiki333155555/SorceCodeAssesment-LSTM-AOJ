#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int R[200009];
    for (int i = 0; i < n; i++)cin >> R[i];
    int minv = R[0];
    int maxv = R[1] - R[0];
    for (int i = 0; i < n-1; i++) {
        if (minv >= R[i])minv = R[i];
        if(maxv<=R[i+1]-minv)maxv=R[i+1]-minv;
    }
    cout << maxv << endl;
    return 0;
}
