#include <iostream>
#include <vector>

using namespace std;

int main(){
    int n, op, e;
    cin>>n;
    
    vector<int> v;

    for (int i = 0; i< n; i++){
        cin>>op;
        if(op==0){
            cin>>e;
            v.push_back(e);
        }
        if(op==1){
            cin>>e;
            cout << v.at(e) << endl;
        }
        if(op==2){
            v.pop_back();
        }
    }
    return 0;
}
