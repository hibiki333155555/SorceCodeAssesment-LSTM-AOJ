const ll maxn = 1e3 +7;
const ll maxm = 2e5 +7;
const ll inf = 0x3f3f3f3f;
const ll mod = 1e9 +7;

int n;
int dp[maxn][maxn];
char a[maxn], b[maxn];

int solve() {
    int la = strlen(a), lb = strlen(b);
    for(int i = 0; i <= la; i++) {
        for(int j = 0; j <= lb; j++) {
            dp[i][j] = 0;
        }
    }
    for(int i = 1; i <= la; i++) {
        for(int j = 1; j <= lb; j++) {
            dp[i][j] = max(dp[i][j-1],dp[i-1][j]);
            if(a[i-1] == b[j-1]) {
                dp[i][j] = max(dp[i][j], dp[i-1][j-1] + 1);
            }
        }
    }
    return dp[la][lb];
}
