#include<bits/stdc++.h>
using namespace std;
void print(vector<int>v){
 for(int i=0;i<v.size();i++){
  cout<<v[i]<<" ";
 }
 cout<<"\n";
}
void InsertionSort(vector<int>&a){
 int n=a.size();
 for(int i=1;i<n;i++){
  int j=i-1;
  int temp=a[i];
  while(j>=0&&temp<a[j]){
   a[j+1]=a[j];
   j--;
  }
  a[j+1]=temp;
  print(a);
 }
}

 int n,i;
 cin>>n;
 vector<int>v(n);

}
