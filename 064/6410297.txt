#include<stdio.h>

int main(void)
{
    int card[4][13];
    int a,b,c;
    int i;
    int nom;
    char sout;
    for(a=0;a<4;a++)
    {
        for(b=0;b<13;b++)
        {
            card[a][b]=0;
        }
    }
    scanf("%d",&c);
    for(i=0;i<c;i++)
    {
        scanf("%c",&sout);
        scanf("%c %d",&sout,&nom);
        if(sout=='S')
        {
            card[0][nom-1]=1;
        }
        else if(sout=='H')
        {
            card[1][nom-1]=1;
        }
        else if(sout=='C')
        {
            card[2][nom-1]=1;
        }
        else if(sout=='D')
        {
            card[3][nom-1]=1;
        }
    }
    for(a=0;a<4;a++)
    {
        for(b=0;b<13;b++)
        {
            if(card[a][b]==1)
            {
                if(a==0)
                {
                    printf("S ");
                }
                else if(a==1)
                {
                    printf("H ");
                }
                else if(a==2)
                {
                    printf("C ");
                }
                else if(a==3)
                {
                    printf("D ");
                }
                printf("%d\n",b+1);
            }
        }
    }
    return 0;
}
