#include <iostream>

using namespace std;


int main() {
    int h, w;

    char chs[2] = {'#', '.'};
    
    do {
        cin >> h >> w;
        int idx = 0;
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                cout << chs[idx];
                idx = !idx;
            }
            cout << endl;
            if (w % 2 == 0)
                idx = !idx;
        }


        cout << endl;
    } while (h != 0 && w != 0);


    return 0;
}
