#include<stdio.h>
int main(void){
    int i,x,y;
    for(i=0;i<3000;i++){
        scanf("%d %d",&x,&y);
        if(x<y){
            printf("%d %d\n",x,y);
        }else if(x>y){
            printf("%d %d\n",y,x);
        }else if(x==0 && y==0){
            break;
        }
    }
    return 0;
    
}
