#include <iostream>
using namespace std;

int main(){
    int n;
    cin >> n;
    
    int a[n];
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    
    int q;
    cin >> q;
    int k;
    int j;
    for(int i = 0; i < q; i++){
        cin >> k;
        for(j = 0; j < n; j++){
            if(k > a[j]){
                cout << j + 1 << endl;
                break;
            }
        }
        if(j == n - 1){
            cout << n << endl;
        }
    }
}
