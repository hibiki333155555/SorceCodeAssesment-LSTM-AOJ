#include<stdio.h>
#include<vector>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<set>
#include<map>
using namespace std;

bool func(vector<int>A,int i,int q){
  if(q == 0){return true;}
  if(i == A.size()){return false;}
  return func(A,i+1,q-A.at(i)) || func(A,i+1,q);
}


int main(){
  int n;
  scanf("%d",&n);
  vector<int> A(n);
  for (int i = 0; i < n; i++){scanf("%d",&A.at(i));}
  int q;
  scanf("%d",&q);
  vector<int> Q(q);
  for (int i = 0; i < q; i++){scanf("%d",&Q.at(i));}
  for (int i = 0; i < q; i++){
    if(func(A,0,Q.at(i))){
      printf("%s\n","yes");
    }
    else{
      printf("%s\n","no");
    }
  }
}
