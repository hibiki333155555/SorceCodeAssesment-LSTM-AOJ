#include <stdio.h>

int main(void){
    int r,c,i,j,sum;
    int hyou[101][101];
    
    scanf("%d %d",&r,&c);
    
    for(i=0;i<r;i++){
        for(j=0;j<c;j++){
             scanf("%d",&hyou[i][j]);
        }
    }

    for(i=0;i<=r;i++){
        sum=0;
        for(j=0;j<=c;j++){
            sum=sum+hyou[i][j];
        }
        hyou[i][c]=sum;
    }
    
    for(i=0;i<=c;i++){
        sum=0;
        for(j=0;j<=r;j++){
            sum=sum+hyou[j][i];
        }
        hyou[r][i]=sum;
    }
    for(i=0;i<=r;i++){
        for(j=0;j<=c;j++){
            if(j==c){
                printf("%d",hyou[i][j]);
            }
            else{
                printf("%d",hyou[i][j]);
            }
             
        }
        printf("\n");
    }
}
