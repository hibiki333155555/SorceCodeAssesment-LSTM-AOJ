#include<iostream>
#include<string>
using namespace std;

int main(){

    char ch;
    int counter[26]={};
    
    while(cin>>ch){
        ch =tolower(ch);
        int num = ch - 'a';counter[num]++;
       
    }

    for(int i=0;i<26;++i){
        cout<<char('a'+i)<<" : "<<counter[i]<<endl;
        ++i;
    }
  return 0;
}
