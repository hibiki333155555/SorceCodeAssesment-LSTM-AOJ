#include <bits/stdc++.h>
using namespace std;
int bubblesort(vector <int> &p)
{
	int n=p.size();
	int count=0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(p[j]>p[j+1]){
				swap(p[j],p[j+1]);
				count++;
			}
		}
		
	}
	return count;
}
int main() {
int n;
cin>>n;
vector<int>p(n);
for(int i=0;i<n;i++){
	cin>>p[i];
}
int res=bubblesort(p);
for(int i=0;i<n;i++){
	cout<<p[i];
	if(i!=n-1)
	cout << " ";
}
cout<<endl<<res<<endl;
return 0; //hi
}
