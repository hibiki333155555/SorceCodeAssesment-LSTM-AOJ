#include <iostream>
#include <cmath>
using namespace std;

int main() {
    int n; cin>>n;
    int a_min = 1e6, a_max = 0;
    long int a_sum = 0;
    for(int i=0;i<n;i++) {
        int a; cin>>a;
        if(a_min > a) a_min = a;
        if(a_max < a) a_max = a;
        a_sum += a;
    }
    cout<<a_min<<" "<<a_max<<" "<<a_sum<<endl;
    return 0;
}
