#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)
#define Inf 1000000001

int main(){
	
	int n;
	cin>>n;
	
	map<int,pair<int,int>> mp;
	rep(i,n){
		rep(j,n){
			int a;
			cin>>a;
			if(!mp.count(a)){
				mp[a] = make_pair(i,j);
			}
		}
	}
	int q;
	cin>>q;
	rep(i,q){
		int t;
		cin>>t;
		if(!mp.count(t))cout<<-1<<endl;
		else{
			cout<<mp[t].first+1<<' '<<mp[t].second+1<<endl;
		}
	}
	
	return 0;
}
