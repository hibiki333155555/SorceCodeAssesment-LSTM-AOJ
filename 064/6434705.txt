#include <bits/stdc++.h>
using namespace std;

void bsort(vector<int> &a){
	int n=a.size(),cnt=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n-1-i;j++)
		{
			if(a[j]>a[j+1])
			{
				swap(a[j],a[j+1]);
				cnt++;
			}
		}
	}
	
	for(int i=0;i<n;i++)
	{
		cout<<a[i];
		if(i!=n-1) cout<<" ";
	}
	cout<<endl;
	cout<<cnt<<endl;
}

int main() {

		int n;
		cin>>n;
		vector<int> a(n);
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		bsort(a);
	
	return 0;
}
