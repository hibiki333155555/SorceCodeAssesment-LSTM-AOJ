#include<iostream>
using namespace std;

int main()
{
    int h,k;
    while(1)
    {
        cin>>h>>k;

        if(h==0 && k==0)
        {
            return 0;
        }
        for(int i=0;i<h;i++)
        {
            for(int j=0;j<k;j++)
            {
                cout<<"#";
            }
            cout<<endl;
        }
        cout<<endl;
    }

}
