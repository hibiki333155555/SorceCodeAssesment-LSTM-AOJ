while True:
    h, m, s = map(int, input().split())
    if h == -1 and m == -1 and s == -1:
        break
    t = h * 3600 + m * 60 + s
    res = 2 * 3600 - t
    hm1, s1 = divmod(res, 60)
    h1, m1 = divmod(hm1, 60)
    res *= 3
    hm2, s2 = divmod(res, 60)
    h2, m2 = divmod(hm2, 60)
    print(f"{h1:02}:{m1:02}:{s1:02}")
    print(f"{h2:02}:{m2:02}:{s2:02}")
