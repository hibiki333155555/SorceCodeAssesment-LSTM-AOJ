#include<stdio.h>
#define N 10000
int A[N],n;

int binarySearch(int key){
  int left = 0;
  int right = n;
  int middle;
  
  while(right > left){
    middle = (left + right) / 2;
    if(key == middle) return 1;
    else if(key > middle) left = middle + 1;
    else right = middle;
  }

  return 0;
}

int main(){
  int i;
  scanf("%d",&n);

  for(i=0 ;i<n ;i++){
    scanf("%d",&A[i]);
  }
  
  int m,key,count = 0;
  scanf("%d",&m);

  for(i=0 ;i<m ;i++){
    scanf("%d",&key);
    if(binarySearch(key)) count++;
  }

  printf("%d\n",count);

  return 0;
}
