void solve(){
    ll n;cin>>n;
    map<ll,pll>ma;
    rep(i,n){
        rep(j,n){
            ll x;cin>>x;
            if(ma.find(x)==ma.end()){
                ma[x]=mp(i,j);
            }else{
                pll k=ma[x];
                ma[x]=min(k,mp(i,j));
            }
        }
    }
    ll q;cin>>q;
    while(q--){
        ll t;cin>>t;
        if(ma.find(t)==ma.end()){
            cout<<-1<<endl;
        }else{
            pll k=ma[t];
            cout<<k.fi+1<<' '<<k.se+1<<endl;
        }
    }
}
