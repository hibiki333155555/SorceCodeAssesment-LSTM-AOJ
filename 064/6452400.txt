#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for(int i = 0; i < (n); i++)
#define coutf(f) cout << fixed << setprecision(f)
#define all(v) (v).begin(), (v).end()
#define rall(v) (v).rbegin(), (v).rend()
#define pb push_back
using Graph = vector<vector<int>>;

const ll INF = 1LL << 60;

int main() {
    int n, m;
    cin >> n >> m;
    vector<vector<ll>> dp(n, vector<ll> (n, INF));
    rep (e,m) {
        int a, b;
        ll w;
        cin >> a >> b >> w;
        dp[a][b] = w;
    }
    rep (v,n) dp[v][v] = 0;
    rep (k,n) rep (i,n) rep (j,n) dp[i][j] = min(dp[i][j], dp[i][k]+dp[k][j]);
    rep (v,n) {
        if (dp[v][v] < 0) {
            cout << "NEGATIVE CYCLE" << endl;
            return 0;
        }
    }
    rep (i,n) {
        rep (j,n) {
            if(j) cout << " ";
            if(dp[i][j]==INF)cout<<dp[i][j];
            else cout << "INF";
        }
        cout << endl;
    }
}
