#include<iostream>
#include<algorithm>
using namespace std;
 
const int N = 1005;
int n;
int a[N];
 
void display() {
	for(int i=0;i<n;i++)
		cout<<a[i]<<" ";
	cout<<endl;
}
 
int main() {
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
 
	for(int i=0;i<n;i++) {
		int j;
		for(j=i-1;~j;j--)
			if(a[j]>a[j+1])
				swap(a[j],a[j+1]);
		display();
	}
}
