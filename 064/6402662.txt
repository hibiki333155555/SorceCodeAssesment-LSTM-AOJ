#include<bits/stdc++.h>
using namespace std;
int main(){
    int z ; cin >> z ;
    vector < pair <int, int>> v;
    vector<int> v1 ; vector <int> v2 ;
    for(int i = 0 ; i < z ; i++){
        int a , b ; cin >> a >> b ;
        v1.push_back(a) ;
        v2.push_back(b)  ;
    }
    for (int i=0; i<z; i++){
        v.push_back( make_pair(v1[i],v2[i]) );
    }
    sort(v.begin(),v.end());
     for (int i=0; i<z; i++)
    {
        cout << v[i].first << " " << v[i].second << endl;
    }

return 0;
}
