#include <iostream>
#include <vector>
#include <map>

using namespace std;

int main()
{
	int N, L; cin >> N >> L; --L;
	vector<int> a(N); for (auto& an : a) cin >> an;
	map<int, int> S;
	for (int i = 0; i < L; ++i) ++S[a[i]];
	for (int i = L, j = 0; i < N; ++i, ++j) {
		++S[a[i]];
		if (i > L) cout << ' ';
		cout << S.begin()->first;
		if (--S[a[j]] == 0) S.erase(a[j]);
	}
	cout << endl;
	return 0;
}
