#include <bits/stdc++.h>
using namespace std;

int main(){
    int y,m,d,y2,m2,d2;
    cin>>y>>m>>d;
    cin>>y2>>m2>>d2;
    if(y>=y2&&m>=m2&&d>=d2){
        swap(y,y2);
        swap(m,m2);
        swap(d,d2);
    }
    int ans=y2-y;
    if(m2>m){
        ans++;
    }
    cout<<ans<<"\n";
    return(0);
}
