#include <iostream>
using namespace std;

int gcd(int a,int b){
    int tmp;
    if(a){
        return 1;
    }
    if(a < b){
        tmp = b;
        b = a;
        a = tmp;
    }
    gcd(b , a%b);
    return  b;
}
int main(){
    int a,b;
    cin >> a >> b;
    gcd(a,b);
    cout << b << endl;
}
