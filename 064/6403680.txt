#include<iostream>
#include<cstring>
using namespace std;
char g[110][110];
bool b[110][110];
int n,m;
void dfs(int x,int y)
{
	b[x][y] = 1;
	int xi[4] = {-1,1,0,0};
	int yi[4] = {0,0,-1,1};

	for(int i = 0; i < 4; i++)
	{
		int nx = x+xi[i], ny = y +yi[i];
		if(nx>=0 && nx<n && ny>=0 && ny < m &&b[nx][ny]!=1 && g[x][y] == g[nx][ny])
		{
			dfs(nx,ny);
		}
	}
}
void slove()
{
	int res = 0;
	for(int i = 0; i < n; i++)
		{
			for(int j = 0; j <m; j++ )
				if(b[i][j]!=1)
				{
				dfs(i,j);res++;
				}
		}
		cout << res<<endl;
}

int main()
{
	while(1)
	{
		scanf("%d %d",&n,&m);
		if(m==0 && n==0)
		break;
		memset(g,-1,sizeof g);
		memset(b,-1,sizeof b); 
		for(int i = 0; i < n; i++)
		{
			
				scanf("%s",&g[i]);
		}
		
		slove();
	}

 }
