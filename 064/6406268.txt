#include<iostream>

using namespace std;
int main(void){
    int n,m;
    while(true){
        cin>>n>>m;
        int a[n],b[m];
    
        for(int i=0;i<n;i++)
            cin>>a[i];
        for(int i=0;i<m;i++)
            cin>>b[i];
        if(n==0 && m==0)break;
        int c;
        int cn[10]={0};
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++){
                c=a[i]*b[j];
                    while(c>0){
                    int r=c%10;
                    cn[r]++;
                    c/=10;
                    }
            }
    for(int i=0;i<9;i++)printf("%d ",cn[i]);
    printf("%d\n",cn[9]);
    }
    return 0;
}
