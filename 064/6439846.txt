#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int a[n];
    for(int i=0;i<n;i++) {
        cin >> a[i];
    }
    
    for(int i=1;i<n;i++) {
        int j = i-1;
        int v = a[i];
        while (j >= 0 && a[j] > v) {
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = v;
        for (int z=0;z<n-1;z++) {
            cout << a[z] << " ";
        }
        cout << a[n-1] << endl;
    }
}
