#include <bits/stdc++.h>
using namespace std;
 
int main() {
  vector<int> vec(10000);
 
  // 入力を受け取る
  for (int i = 0; i < 10000; i++) {
    cin >> vec.at(i);
  }
  
  for (int i=0; i < 10000; i++){
    if (vec.at(i) != 0){
      cout << "Case" << " " << i << ":" << " " << vec.at(i) << endl;
    }
    
    else{
      break;
    }
    
  }
  
}
