#include<stdio.h>
int main(){
    
    int m,f,r;
    char x;
    
    while(1){
        scanf("%d %d %d",&m,&f,&r);
        if(m+f+r==-3){
            break;
        }
        if(m == -1 || f == -1){
            x = 'F';
        }else if(m+f >= 80){
            x = 'A';
        }else if(m+f >= 65){
            x = 'B';
        }else if(m+f >= 50){
            x = 'C';
        }else if(m+f >= 30){
            if(r >= 50){
                x = 'C';
        }else{
            x = 'D';
        }else{
            x = 'F';
        }

    printf("%c\n", x);
  }
  
  return 0;
}
