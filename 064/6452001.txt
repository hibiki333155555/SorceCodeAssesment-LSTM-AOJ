#include <stdio.h>

int main(){
    int n,i,j,t,min,a[101],s,cnt=0;
    
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    
    for(i=0;i<n-1;i++){
        min=a[i];
        s=i;
        for(j=i;j<n;j++){
            if(a[j]<min){
                min=a[j];
                s=j;
                cnt++;
            }
        }
        t=a[i];a[i]=a[s];a[s]=t;
    }
    
    for(i=0;i<n;i++){
        if(i!=0)printf(" ");
        printf("%d",a[i]);
    }
    printf("\n");
    printf("%d\n",cnt);
}
