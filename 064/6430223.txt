#include <stdio.h>

int main(void){
    int i,n;
    double xA,yA,rA,xB,yB,rB,d;
    
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%lf %lf %lf %lf %lf %lf",&xA,&yA,&rA,&xB,&yB,&rB);
        d=sqrt(pow(xA-xB,2)+pow(yA-yB,2));
        //printf("rA:%lf rB:%lf d:%lf\n",rA,rB,d);
        
        if(rA+rB<d) puts("0");
        else if(rA+rB==d) puts("1");
        else if(abs(rA-rB)<=d) puts("1");
        else if(rA>rB) puts("2");
        else if(rB<rA) puts("-2");
    }
}
