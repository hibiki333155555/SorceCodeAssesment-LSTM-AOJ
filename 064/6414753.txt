#include <bits/stdc++.h>
using namespace std;

#ifdef _RUTHEN
#include "debug.hpp"
#else
#define show(...) true
#endif

using ll = long long;
#define rep(i, n) for (int i = 0; i < (n); i++)
template <class T> using V = vector<T>;

const ll MOD = 998244353;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int N;
    cin >> N;
    V<int> A(N);
    rep(i, N) cin >> A[i];
    V<int> B(N);
    rep(i, N) B[i] = A[i] + i;
    sort(B.begin(), B.end());
    show(B);
    ll ans = 1;
    rep(i, N) { ans = ans * (B[i] - i) % MOD; }
    cout << ans << '\n';
    return 0;
}
