#include <stdio.h>

int a[200000];

int bin_search(int n, int k){
	int l,m,r;
	r = n; l = 0; m = (r+l)/2;
	while(n){
		if(a[m] == k) return 1;
		if(a[m] < k){
			l = m+1;
			m = (r+l)/2;
		}else{
			r = m - 1;
			m = (r+l)/2;
		}
		n = n>>1;
	}
	return 0;
}

int main(){
	int n, q, k;
	int l, m, r;
	
	scanf("%d", &n);
	for(int i = 0; i < n; i++){
		scanf("%d", &a[i]);
	}
	//query
	scanf("%d", &q);
	//search
	while(q--){
		scanf("%d", &k);
		printf("%d\n", bin_search(n, k));
	}
}
