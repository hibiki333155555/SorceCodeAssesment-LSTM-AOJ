#include <iostream>
using namespace std;
// Ara Ara  sensei   C
int main()
{
    int x, y;
    while (cin >> x >> y)
    {
        if (x > 0 && y > 0)
        {
            for (int a = 0; a < x; a++)
            {
                for (int b = 0; b < y; b++)
                {
                    cout << '#';
                }
                printf("\n\n");
            }
        }
        else
        {
            break;
        }
    }
    return 0;
}
