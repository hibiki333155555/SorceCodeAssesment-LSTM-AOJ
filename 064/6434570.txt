import java.util.*;

public class Main {
	public int gcd(int x, int y) {
		int gcdnum;
		int remain;
		if (x > y) remain = x % y;
		else remain = y % x;
		while (remain != 0) {
			if (x > y) {
				x = x % y;
				remain = y % x;
			} else {
				y = y % x;
				remain = x % y;
			}
		}
		if (x > y) gcdnum = y;
		else gcdnum = x;
		return gcdnum;
	}
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		Main c = new Main();
		int xnum = scan.nextInt();
		int ynum = scan.nextInt();
		System.out.println(c.gcd(xnum, ynum));
		
	}
}
