#include<bits/stdc++.h>
using namespace std;
int search(int arr1[],int arr2[],int n,int m)
{
for(int i=0;i<n;i++)
{
  for(int j=0;j<m;j++)	
   if(arr2[i]==arr1[j])
  	 return 1;
  	else
     return 0;
}

}
int main(){
 
 int n;
 cin>>n;
 int arr1[n];
 for(int i=0;i<n;i++)
{
cin>>arr1[i];
}
int m;
cin>>m;
 int arr2[m];
 for(int j=0;j<m;j++)
{
cin>>arr2[j];
}
int r=search(arr1,arr2,n,m);
cout<<r<<endl;
return 0;
}
