#include<stdio.h>
    
    void trace(int N,int A[]){
        for(int i=0;i<N;i++){
            if(i>0)
                printf(" ");
            printf("%d", A[i]);
        }
        printf("\n");
    }
    
    void insertionSort(int N,int A[]){
        int v,j;
        
        for(int i=1;i<N;i++){
            v=A[i];
            j=i-1;
            
            while(j>-1 && A[j]>v){
                A[j] = A[j+1];
                j--;
            }
            
            A[j+1] = v;
            trace(A);
        }
        
    }
    
    int main(){
        int N;
        int A[100] = {};
        
        scanf("%d",&N);
        for(int i=0;i<N;i++)
            scanf("%d",&A[i]);
        
        trace(N,A);
        
        insertionSort(N,A);
        
        return 0;
        
    }
