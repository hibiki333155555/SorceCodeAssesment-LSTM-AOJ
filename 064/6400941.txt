#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int n;

int main(){
  cin >> n;
  ll sum = 0;
  int max = -1000001;
  int min = 1000001;
  for(int i = 0; i < n ; i++){
    int input;
    cin >> input;
    sum += input;
    if (max < input){
      max = input;
    }
    if (min > input){
      min = input;
    }
  }
  cout << min << " " << max << " " << sum << endl;
}
