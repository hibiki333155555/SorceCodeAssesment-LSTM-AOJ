#include <bits/stdc++.h>
using namespace std;
const int INF = 1001001001;
const int MOD = 1000000007;

int main(){
    int n;cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;i++) cin>>a.at(i);
    int cnt = 0;
    for(int i=0;i<n;i++) {
        int minj = i;
        for(int j=i;j<n;j++) {
            if(a.at(j)<a.at(minj)) minj = j;
        }
        swap(a.at(i),a.at(minj));
        if(i!=minj) cnt++;
    }
    for(int i=0;i<n;i++) {
        cout << a.at(i);
        if(i==n-1) cout << endl;
        else cout << " ";
    }
    cout << cnt << endl;
}
