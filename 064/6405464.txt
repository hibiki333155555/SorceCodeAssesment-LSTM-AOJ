#include<stdio.h>
int main(void)
{
    int a,b,i,j,k;
    for(i = 0 ;i < 1000000000;i++)
    {
        if(i != 0)
        {
            printf("\n");
        }
        scanf("%d %d",&a,&b);
        if(a == 0 && b == 0)
        {
            break;
        }
        for(j = 0;j <a;j++)
        {
            for(k = 0;k<b;k++)
            {
                if(j==0 || j == a-1)
                {
                printf("#");
                }
                if(j !=0 && j != a-1)
                {
                    if(k == 0 || k == b-1)
                    {
                        printf("#");
                    }
                    if(k != 0 && k != b-1)
                    {

                        printf(".");
                    }
                }

            }
            printf("\n");
        }
    }
    printf("\n");
    return 0;
}
