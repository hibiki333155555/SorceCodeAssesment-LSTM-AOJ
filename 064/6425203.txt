#include <bits/stdc++.h>
using namespace std;

int main(){
    int A,B,X;
    cin>>A>>B>>X;
    int sum;
    sum=X/1000;
    int ans;
    ans=min(sum*A,(sum*2)*B);
    if(X%1000!=0){
        if(X%1000<=500){
            ans+=min(A,B);
        }else{
            ans+=min(A,B*2);
        }
    }
    cout<<ans<<"\n";
    return(0);
}
