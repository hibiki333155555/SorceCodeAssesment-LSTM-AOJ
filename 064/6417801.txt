#include <iostream>
#include <cmath>
#include <cstdlib>
#include <stdio.h>
#include <vector>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)


int main() {
    int n;
    cin >> n;
    vector<int>a(n);
    rep(i,n) cin >> a[i];
    int cnt = 0;
    rep(i,n) {
        int minj = i;
        for(int j=i;j<n;j++) {
            if(a[j] < a[minj]) minj = j;
        }
        if(i != minj) cnt++;
        swap(a[i],a[minj]);
    }
    rep(i,n) cout << a[i] << (i==n-1?'\n':' ');
    cout << cnt << endl;
    return 0;
}
