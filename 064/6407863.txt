#include <stdio.h>
 int main(void){
    int a,b,i,j,k;
    
    while(1){
        if((a=0) && (b=0)){
            break;
        }
        else{
            scanf("%d %d",&a,&b);
        for(j=0;j<b;j++){
            for(k=0;k<a;k++){
                printf("#");
            }
            printf("\n");
        }
    }
    return 0;
}
