void solve() {
    INT(n, s);
    VV(int, a, n, 4);
    vc<int> cnt(4, 0);
    int cnt11 = 0, cnt111 = 0;
    vc<int> cnt1(4, 0);
    rep(i, n) {
        int co1 = 0;
        int pos1 = -1;
        rep(j, 4) {
            cnt[a[i][j]] ++;
            if(a[i][j] == 1) {
                ++ co1;
                pos1 = j;
            }
        }
        if(co1 == 1) {
            int nj = (pos1 + 2) % 4;
            rep(j, 4) {
                if(j == nj) continue;
                cnt1[a[i][j]] ++;
            }
        }
        if(co1 == 2) ++ cnt11;
        if(co1 == 3) ++ cnt111;
    }
    int diff = cnt[3] - cnt[2];
    for(int x = 1; x <= n + 1; ++ x) {
        unless((n + 1) % x == 0) continue;
        int y = (n + 1) / x;
        int res1 = 2 * (x + y) - cnt[1];
        if(res1 == 3) {
            unless(x == 1 or y == 1) continue;
            unless(cnt111 == 1) continue;
        }
        if(res1 == 2) {
            if(x == 1 or y == 1) {
                unless(cnt111 == 2) continue;
            } else {
                unless(cnt11 == 3) continue;
            }
        }
        if(res1 == 1) {
            if(x == 1 or y == 1) continue;
            else {
                unless(cnt11 == 4) continue;
            }
        }
        if(res1 < 0 or res1 >= 4) continue;
        int sum = 4 - res1;
        unless((sum - diff) % 2 == 0) continue;
        int res2 = (sum - diff) / 2 + diff;
        int res3 = (sum - diff) / 2;
        if(res2 < 0 or res3 < 0) continue;
        vc<int> ids(4);
        iota(all(ids), 0);
        vc<int> ans = vc<int>{4, 4, 4, 4};
        vc<int> now;
        rep(_, res1) now.eb(1);
        rep(_, res2) now.eb(2);
        rep(_, res3) now.eb(3);
        assert(si(now) == 4);
        dump(now);
        do {
            vc<int> res;
            vc<int> pos1;
            rep(i, 4) {
                res.eb(now[ids[i]]);
                if(now[ids[i]] == 1) {
                    pos1.eb(i);
                }
            }
            int curs = (res[0] == res[2] and res[1] == res[3] ? 1 : 0);
            if(res1 == 2) {
                if(x == 1 or y == 1) {
                    if(abs(pos1[0] - pos1[1]) != 2) continue;
                } else {
                    if(abs(pos1[0] - pos1[1]) != 1) continue;
                }
            }
            if(res1 == 1 and (res2 == 1 or res2 == 2)) {
                if(res[0] != 1) continue;
                int diff1 = cnt1[3] - cnt1[2];
                if(diff1 == 0) {
                    if(res[1] == res[3]) continue;
                } else {
                    assert(abs(diff1) == 2);
                    if(res[1] != res[3]) continue;
                }
            }
            if(curs == s) {
                if(chmin(ans, res)) ans = res;
            }
        } while(next_permutation(all(ids)));
        if(ans[0] != 4) {
            out(ans);
            return;
        }
    }
    assert(false);
}
