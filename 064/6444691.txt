#include <iostream>
#include <utility> // pair, make_pair
#include <queue> // queue, priority_queue
using namespace std;

int main() {
    int n, q;
    cin >> n >> q;
    queue<string> P;
    queue<int> Q;
    for(int i = 0;i < n;i++){
        string a;
        int b;
        cin >> a >> b;
        P.push(a);
        Q.push(b);
    }

    int time = 0;
    while(!Q.empty()){
        if(Q.front() <= q){
            int x;
            x = Q.front();
            time += x;
            cout << P.front() << " " << time  << endl;
            Q.pop();
            P.pop();
        }
        else{
            int x;
            string y;
            x = Q.front();
            y = P.front();
            time += q;
            x -= q;
            Q.push(x);
            Q.pop();
            P.push(y);
            P.pop();
        }
    }

}
