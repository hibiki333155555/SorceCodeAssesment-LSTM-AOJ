#include <stdio.h>

int main(void){
    int n,m,i,j;
    int A[101][101], b[101], c[101];
    
    scanf("%d %d",&n,&m);
    
    for(i=0;i<n;i++){
        for(j=0;j<m;j++){
            scanf("%d",&A[i][j]);
        }
    }
    for(i=0;i<m;i++){
        scanf("%d",&b[i]);
    }
    for(i=0;i<n;i++){
        c[i]=0;
        for(j=0;j<m;j++){
            c[i]=c[i]+A[i][j]*b[j];
        }
    }
    for(i=0;i<n;i++){
        printf("%d\n",c[i]);
    }
    return 0;
}
