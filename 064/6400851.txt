

#include<stdio.h>
int main (){

    int a,b,c;
    int num[3];
    int lar,smol;
    scanf("%d %d %d", &a, &b, &c);
    if(a>=1 && a<=10000 && b>=1 && b<=10000 && c>=1 && c<=10000){
        num[0]=a;
        num[1]=b;
        num[2]=c;
        
        lar=num[1];
        for(int i=0;i<3;i++){
            if(num[i]>=lar)
                lar=num[i];
        }
        
        smol=num[1];
        for(int i=0;i<3;i++){
            if(num[i]<=smol)
                smol=num[i];
        }
        
        printf("%d %d %d\n", smol, a+b+c-smol-lar, lar);
        
    }
    return 0;
}
