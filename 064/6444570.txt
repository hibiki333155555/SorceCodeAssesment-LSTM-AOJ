#include <iostream>
#include <cmath>;
using namespace std;

int main()
{
    int i, n, a;
    long min = 1000000, max = -1000000;
    long long sum;
    
    cin >> n;
    
    for ( i = 1; i <= n; i++ ) {
        cin >> a;
        
        if ( a < min ) {
            min = a;
        }
        
        if ( a > max ) {
            max = a;
        }
        
        sum += a;
    }
    
    cout << min << " " << max << " " << sum << endl;
    return 0;
}
