#include <bits/stdc++.h>
using namespace std;

int calc(string c) {
    if(c.size() == 1) {
        return c[0]-'A';
    }
    else {
        return 26+(c[1]-'A')+26*(c[0]-'A');
    }
}

int main() {
    string c1,c2;
    int r1,r2;
    cin >> c1 >> r1 >> c2 >> r2;
    int a = calc(c1),b = calc(c2);
    cout << (r2-r1+1)*(b-a+1) << endl;
}
