#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>
using namespace std;
using ld = long double;

int main(){
    int n; cin >> n;
    vector<int> a(n);
    for(auto &it: a) cin >> it;

    sort(a.begin(), a.end());
    vector<ld> res;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            if(i == j) continue;
            int p, q;
            for(p = n-1; p >= 0; p--){
                if(p != i && p != j) break;
            }
            for(q = n-1; q >= 0; q--){
                if(q != p && q != i && q != j) break;
            }
            res.emplace_back((ld)(a[p]+a[q])/(a[i]-a[j]));
        }
    }
    sort(res.begin(), res.end());
    cout << fixed << setprecision(10) << res.back() << endl;
    return 0;
}
