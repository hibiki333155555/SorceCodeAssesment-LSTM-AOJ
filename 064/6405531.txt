#include<stdio.h>
int n;

int prim(int x){
  int i;
  
  for(i=2 ;i*i<=x ;i++){
    if(x%i == 0) return 0;
  }
  
  return 1;
}

int main(){
  int n,i,k,count=0;

  scanf("%d",&n);

  for(i=0 ;i<n ;i++){
    scanf("%d",&k);
    //printf("%d %d \n",k,prim(k));
    if(prim(k) == 1) count++;
  }

  printf("%d\n",count);
}
