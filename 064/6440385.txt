#include <iostream>
#include <iomanip>

int main() {
  int q; std::cin >> q;
  for (int i=0; i<q; i++) {
    double x0, y0, x1, y1, x2, y2, x3, y3;
    std::cin >> x0 >> y0 >> x1 >> y1 >> x2 >> y2 >> x3 >> y3;
    double denom = (x1-x0)*(y3-y2) - (x3-x2)*(y1-y0);
    double numer= (y3-y2)*(x2-x0) - (x3-x2)*(y2-y0);
    double ratio = numer / denom;
    std::cout << std::fixed << std::setprecision(9);
    std::cout << (1.0 - ratio) * x0 + ratio * x1 << " ";
    std::cout << (1.0 - ratio) * y0 + ratio * y1 << std::endl;
  }
}
