template<typename T>
istream& operator>>(istream&is,vector<T>&v){
  for(T&p:v)is>>p;
  return is;
}
template<typename T>
ostream& operator<<(ostream&os,const vector<T>&v){
  if(&os==&cerr)os<<"[";
  for(int i=0;i<v.size();i++){
    os<<v[i];
    if(i+1<v.size())os<<(&os==&cerr?",":" ");
  }
  if(&os==&cerr)os<<"]";
  return os;
}

int main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n;cin>>n;
  vector<string> s(n);
  cin>>s;
  int ans=0;
  REP(i,n){
    bool out=false;
    REP(j,n)if(i!=j){
      int cnt=0;
      REP(k,s[i].size())cnt+=s[i][k]!=s[j][k];
      if(cnt<=1)out=true;
    }
    ans+=out;
  }
  cout<<ans<<endl;
}
