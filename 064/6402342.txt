#include <iostream>
#include <cctype>
#include <cstring>
using namespace std;

int main(){
    int alp[26] = {0};
    char x;
    while(cin >> x){
        if(isalpha(x)){
            if(isupper(x)) x = tolower(x);
            alp[x-'a']++;
        }
    }
    for(int i = 0; i < 26; i++){
        cout << char('a' + i) << " : " << alp[i] << endl;
    }
    
    return 0;
}
