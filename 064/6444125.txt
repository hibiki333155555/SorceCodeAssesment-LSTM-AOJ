#include <bits/stdc++.h>
using namespace std;
int f(double a)
{
    if (a==(int)a) return 1;
    return 0;
}
int main()
{
    while (1)
    {
        int a,b,c,flag=0;
        scanf("%d%d%d",&a,&b,&c);
        if (a==0) break;
        for (int i=a;i>=1;i--)
        {
            int p,q,r,s;
            if (a%i==0) p=i,r=1.0*a/p;
            else continue;
            if (b*b-4*p*c*r>=0)
            {
                if (f(1.0*(b-sqrt(b*b-4*p*c*r))/(2*p)))
                {
                    s=1.0*(b-sqrt(b*b-4*p*c*r))/(2*p);
                    if (s==0&&b%r==0) q=1.0*b/r;
                    else if (b%r!=0) break; 
                    else if (c%s==0) q=1.0*c/s;
                    if (p*s+q*r==b) flag=1;
                }
                if (!flag
                    &&f(1.0*(b+sqrt(b*b-4*p*c*r))/(2*p)))
                {
                    s=1.0*(b+sqrt(b*b-4*p*c*r))/(2*p);
                    if (s==0&&b%r==0) q=1.0*b/r;
                    else if (s==0&&b%r!=0) break; 
                    else if (c%s==0) q=1.0*c/s;
                    if (p*s+q*r==b) flag=1;
                }
            }
            else continue;
            if (flag)
            {
                printf("%d %d %d %d\n",p,q,r,s);
                break;
            }
            else continue;
        }
        if (!flag) printf("Impossible\n");
    }
}
