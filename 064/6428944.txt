#include<iostream>
#include<cstdio>
using namespace std;

#define MAXN 2000005
#define swap(i, t)	int k = h[i]; h[i] = h[t]; h[t] = k;
int h[MAXN];
int cnt = 1;

void siftdown(int i){
	int t = 0;
	bool flag = 1;
	while(flag && i * 2 <= cnt){
		if(h[i] < h[i * 2]){
			t = i * 2;
		}
		else{
			t = i;
		}
		if(i * 2 + 1 <= cnt){
			if(h[t] < h[i * 2 + 1]){
				t = i * 2 + 1;
			}
		}
		if(t != i){
			swap(i, t);
			i = t;
		}
		else{
			flag = 0;
		}
	}
}

void insert(int key){
	h[cnt] = key;
	for(int i = cnt/2; i >= 1; i--)
		siftdown(i);	
}

void extract(){
	printf("%d\n", h[1]);
	h[1] = h[cnt - 1];
	cnt--;
	siftdown(1);
}

int main(){
	char s[10];
	while(1){
		scanf("%s", s);
		if(s[0] == 'e' && s[1] == 'n'){
			break;
		}
		if(s[0] == 'i'){
			int key;
			scanf("%d", &key);
			insert(key);
			cnt++;
		}
		else{
			extract();
		}
	}
	return 0;
}
