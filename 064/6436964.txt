#include <iostream>
// #include "comp.h"
using namespace std;

#define FOR(i, a, b) for (int i = (int)(a); i < (int)(b); ++i)
#define rep(i, n) for (int i = 0; i < (int)(n); ++i)

int main()
{
    int W, H, x, y, r;
    cin >> W >> H >> x >> y >> r;

    bool isHit = false;

    if (x + r > W)
        isHit = true;
    if (x - r < 0)
        isHit = true;
    if (y + r > H)
        isHit = true;
    if (y - r < 0)
        isHit = true;

    if (isHit == false) cout << "Yes" << endl;
    else cout << "No" << endl;
}
