int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    ll N;
    cin >> N;
    vector<string> G(N);
    for(int i=0; i<N; i++) cin >> G[i];
    vector<bool> ans(N);
    for(int i=0; i<N; i++){
        for(int j=i+1; j<N; j++){
            ll cnt = 0;
            for(int k=0; k<G[i].size(); k++){
                if(G[i][k] != G[j][k]) cnt++;
            }
            if(cnt <= 1){
                ans[i] = ans[j] = true;
            }
        }
    }
    ll res = 0;
    for(int i=0; i<N; i++) res += ans[i];
    cout <<res << endl;
    return 0;
}
