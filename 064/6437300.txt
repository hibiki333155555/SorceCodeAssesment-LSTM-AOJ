#include <bits/stdc++.h>
using namespace std;
using Graph = vector<vector<int>>;

int main() {
    int n, x;
    while(x!=0 && n!=0){
        cin >> n >> x;

        int cou = 0;
        for(int i=1;i<n-2;++i){
            for(int j=i+1;j<n-1;++j){
                for(int k=j+1;k<=n;++k){
                    if(i+j+k == x) ++cou;
                }
            }
        }

        cout << cou << endl;
    }
	return 0;
}
