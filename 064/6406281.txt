#include <iostream>
using namespace std;
int main(void)
{
    int a[3];
    for(int i = 0; i < 3; i++)
    {
        cin >> a[i];
    }
    int tmp;
    
    for(int i = 0; i < 3; i++)
    {
        for( int j = i+1; j < 3; j++)
        {
            if( a[i] > a[j])
            {// 小さかったら入れ替え
                tmp = a[i];
                a[i] = a[j];
                a[j] = tmp;
            }
        }
    }
    
    for(int i = 0; i < 3; i++)
    {
        cout << a[i] << " ";
    }
    

}
