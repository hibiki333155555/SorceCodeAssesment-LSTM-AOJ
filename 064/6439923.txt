#include <bits/stdc++.h>
#define REP(i, n) for (int i = 0; i < (int)(n); i++)
using namespace std;

int main(void)
{
  iostream::sync_with_stdio(false);
  int n;
  cin >> n;
  vector<string> report(n);
  REP(i, n)
  {
    cin >> report[i];
  }
  int m = report[0].length();
  vector<int> cheating(n);
  REP(i, n)
  {
    if (cheating[i])
      continue;
    REP(j, n)
    {
      if (i == j)
        continue;
      int compares = 0;
      REP(t, m)
      {
        if (report[i][t] != report[j][t])
        {
          compares++;
          if (compares > 1)
            break;
        }
      }
      if (compares <= 1)
      {
        cheating[i] = 1;
        cheating[j] = 1;
        break;
      }
    }
  }
  int ans = 0;
  REP(i, n)
  ans += cheating[i];
  cout << ans << '\n';
}
