#include <iostream>
#include <stack>
#include <stdio.h>
using namespace std;

bool isdigit(char c){
    return c>='0' && c<='9';
}
int main(){
    stack <int> s;
    char c;
    while(cin >> c){
        if(isdigit(c)){
            ungetc(c, stdin);
            int x;
            cin >> x;
            s.push(x);
        } 
        else{
            int a = s.top();
            s.pop();
            int b = s.top();
            s.pop();
            switch(c){
                case '+': s.push(a+b);break;
                case '-': s.push(b-a);break;
                case '*': s.push(a*b);break;
                case '/': s.push(b/a);break;
            }
        }
    }
    cout << s.top() << endl;
    return 0;
}
