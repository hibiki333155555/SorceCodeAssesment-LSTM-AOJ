#include <iostream>
using namespace std;

int binarySearch(int A[], int n, int key){
    int left = 0;
    int right = n;
    while(left <= right){
        int mid = (left + right) / 2;
        if(key == A[mid]) return mid;
        else if(key > A[mid]) left =  mid + 1;
        else right = mid;
    }
    return 0;
}
int main(){
    int A[100000], n, q, sum = 0, key;
    cin >> n;
    for(int i = 0; i < n; i++) cin >> A[i];
    cin >> q;
    for(int i = 0; i < q; i++){
        cin >> key;
        if(binarySearch(A, n, key)) sum++;
    }
    cout << sum << endl;
}
