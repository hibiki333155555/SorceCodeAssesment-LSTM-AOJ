int main(){

    ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    ll n,m;
    cin >> n >> m;

    vector<string> s(n);
    rep(i,0,n){
        cin >> s[i];
    }

    ll X = 0;
    ll O = 0;
    rep(i,0,n){
        rep(j,0,m){
            if (s[i][j] == 'x') X++;
            else O++;
        }
    }

    ll ans = inf;
    
    
    {
        ll A = 0;
        ll B = X;
        rep(i,0,n-1){
            rep(j,0,m){
                if (s[i][j] == 'x') B--;
                else A++;
            }
            ans = min(ans , A+B);
        }
    }

    
    {
        ll A = 0;
        ll B = O;
        rep(i,0,n-1){
            rep(j,0,m){
                if (s[i][j] == 'o') B--;
                else A++;
            }
            ans = min(ans , A+B);
        }
    }

    
    {
        ll A = 0;
        ll B = X;
        rep(j,0,m-1){
            rep(i,0,n){
                if (s[i][j] == 'x') B--;
                else A++;
            }
            ans = min(ans , A+B);
        }
    }

    
    {
        ll A = 0;
        ll B = O;
        
        rep(j,0,m-1){
            rep(i,0,n){
                if (s[i][j] == 'o') B--;
                else A++;
            }
            ans = min(ans , A+B);
        }
    }

    cout << ans << endl;
    

}
