#define MAX 1024

class Stack
{
private:
    int *a;
    int top;

public:
    Stack(int stack_size) : top(0)
    {
        a = new int[stack_size];
    };
    ~Stack()
    {
        delete[] a;
    };

    void push(int x)
    {
        a[top++] = x;
    }

    int pop()
    {
        return a[--top];
    }

    bool isEmpty()
    {
        return top == 0;
    }
};
