#include<iostream>
#include<string>

using namespace std;

int main(){
  string S, S1;
  int m, h;
  while(1) {
    cin >> S;
    if(S == "-") break;
    cin >> m;
    for(int i = 0; i < m; i++) {
      cin >> h;
      S1 = S.substr(0, h);
      S.erase(0, h);
      S += S1;
    }
    cout << S << "\n";
  }
  return 0;
}
