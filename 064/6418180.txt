import io, os
input = io.BytesIO(os.read(0,os.fstat(0).st_size)).readline
T=int(input())
for tests in range(T):
    N,M,v,a,b=map(int,input().split())

    if N>M:
        print("Yes")
        continue

    SET={v%M}
    last=v%M
    for i in range(N):
        if (last*a+b)%M in SET:
            break
        else:
            SET.add((last*a+b)%M)
            last=(last*a+b)%M

    #print(LIST)

    if len(SET)==N:
        print("No")
    else:
        print("Yes")
