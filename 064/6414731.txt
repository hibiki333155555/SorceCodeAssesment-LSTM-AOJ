void solve(){
    ll n;
    string s;
    cin>>n>>s;
    vector<ll>a(n);
    rep(i,n)cin>>a[i];
    priority_queue<ll,vector<ll>,greater<ll> >pq;
    ll now=0;
    ll ans=0;
    rep(i,n){
        if(s[i]=='(')now++;
        else{
            now--;
            pq.push(a[i]);
            if(now<0){
                ll k=pq.top();
                pq.pop();
                ans+=k;
                now+=2;
            }
        }
    }
    reverse(all(s));
    while(!pq.empty())pq.pop();
    reverse(all(a));
    rep(i,n){
        if(s[i]==')')s[i]='(';
        else s[i]=')';
    }
    now=0;
    rep(i,n){
        if(s[i]=='(')now++;
        else{
            now--;
            pq.push(a[i]);
            if(now<0){
                ll k=pq.top();
                pq.pop();
                ans+=k;
                now+=2;
            }
        }
    }
    cout<<ans<<endl;
}

signed main(){
    cin.tie(0);
    ios::sync_with_stdio(false);
    solve();
}
