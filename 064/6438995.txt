struct StronglyConnectedComponents {
  Graph g, rg;
  vector<int> compo, order, used;
  int N;

  StronglyConnectedComponents(Graph inpG) {
    N = inpG.size();
    g.resize(N);
    rg.resize(N);
    compo.reisize(N, -1);
    used.resize(N);
    for(int v = 0; v < N; v++) {
      for(int nv : inpG[v]) {
        g[v].push_back(nv);
        rg[nv].push_back(v);
      }
    }
  }

  int operator[](int k) { return compo[k]; }

  void dfs(int v) {
    if(used[v]) return;
    used[v] = true;
    for(int nv : g[v]) dfs(nv);
    order.push_back(v);
  }

  void rdfs(int v, int group) {
    if(compo[v] != -1) return;
    compo[v] = group;
    for(int nv : rg[v]) rdfs(nv, group);
  }
