#include<stdio.h>
#include<string.h>

int main(){
    
    char W[11],T[10000];
    int i,n=0;
    
    scanf("%s", &W);
    
    while(1){
        scanf("%s", &T);
        if(!strcmp(T,"END_OF_TEXT"))break;
        for(i=0;T[i]!='\0';i++){
            if(T[i] >= 'A' && T[i] <= 'Z')T[i]=T[i]+32;
        }
        if(!strcmp(W,T))n++;
    }
    
    printf("%d\n", n);
    
    return 0;
}
