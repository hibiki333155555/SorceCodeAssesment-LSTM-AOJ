#include <iostream>
#include <cstdio> // printf
#include <stack> // stack
using namespace std;

int main() {
    stack<char> s;
    char a;
    while(scanf("%s", &a) != EOF){
        int  b = 0;
        s.push(a);
        if(s.top() == '+'){
            b += (int)s.top();
            s.pop();
            b += (int)s.top();
            s.pop();
            s.push(b);
        }
        else if(s.top() == '-'){
            b -= (int)s.top();
            s.pop();
            b += (int)s.top();
            s.pop();
            s.push((char)b);
        }
        else if(s.top() == '*'){
            b += (int)s.top();
            s.pop();
            b *= (int)s.top();
            s.pop();
            s.push((char)b);
        }
        
    }
    cout << s.top() << endl;
}
