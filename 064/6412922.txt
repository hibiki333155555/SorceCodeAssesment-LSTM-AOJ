import java.util.Scanner;

class Main{
    
    static class DoublyLinkedNode{
        int key;
        DoublyLinkedNode next,prev;
        DoublyLinkedNode(int x){
            key = x;
        } 

    }

    static class CellList{
        DoublyLinkedNode head = null;

        CellList(){
        }

        void insert(int key){
            if(head == null){
                head = new DoublyLinkedNode(key);
                return;
            }
            DoublyLinkedNode node = new DoublyLinkedNode(key);
            node.next = head;
            head.prev = node;
            head = node;
        }

        void delete(int key){
            if(head == null){
                return;
            }
            if(head.key == key){
                head = head.next;
                return;
            }

            DoublyLinkedNode cur = head;
            while(cur.key != key){
                cur = cur.next;
                if(cur.key != key && cur.next == null){
                    return;
                }
            }
            DoublyLinkedNode prev = cur.prev;
            DoublyLinkedNode next = cur.next;
            if(next == null){
                prev.next = null;
            }else{
                prev.next = next;
                next.prev = prev;
            }
    }

        void deleteFirst(){
            if(head == null){
                return;
            }
            head = head.next;
        }

        void deleteLast(){
            if(head == null){
                return;
            }

            DoublyLinkedNode cur = head;
            while(cur.next != null){
                cur = cur.next;
            }
            DoublyLinkedNode prev = cur.prev;
            prev.next = null;

        }

        void print(){
            DoublyLinkedNode cur = head;
            while(true){
                System.out.print(cur.key);
                if(cur.next == null){
                    System.out.println();
                    break;
                }else System.out.print(" ");
                cur = cur.next;   
            }
        }
    }

    public static void main(String[] args) throws Exception{
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();
        CellList cell = new CellList();
        for(int i=0;i<n;i++){
            String order = s.next();
                switch(order){
                    case "insert" : cell.insert(s.nextInt()); break;
                    case "delete" : cell.delete(s.nextInt()); break;
                    case "deleteFirst" : cell.deleteFirst(); break;
                    case "deleteLast" : cell.deleteLast(); break;
                }
            }
            cell.print();
        }
    }
