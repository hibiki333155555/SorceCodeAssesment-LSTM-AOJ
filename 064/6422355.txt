#include <bits/stdc++.h>
using namespace std;

string to_lower(string s){
  for(int i=0; i<s.size(); i++){
    if('A' <= s.at(i) && s.at(i) <= 'Z'){
      s.at(i) = s.at(i) - ('A'-'a');
    }
  }
  return s;
}

int main(){
  string w;
  cin >> w;
  w = to_lower(w);
  int cnt=0;
  while(true){
    string s;
    cin >> s;
    if(s=="END_OF_TEXT") break;
    s = to_lower(s);
    if(w == s) cnt++;
  }
  
  cout << cnt << endl;
  return 0;    
    
}
