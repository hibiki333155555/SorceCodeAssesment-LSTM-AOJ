#include <iostream>
#include <vector>

using namespace std;


const int mod = 1042567;


int h0(int k) {
	return k % mod;
}

int h1(int k) {
	return 1 + (k % (mod - 1));
}

int h(int k, int i) {
	return (h0(k) + i * h1(k)) % mod;
}

int c2num(char c) {
	if (c == 'A') {
		return 1;
	}
	else if (c == 'C') {
		return 2;
	}
	else if (c == 'G') {
		return 3;
	}
	else {
		return 4;
	}
}

int str2k(string s) {
	int k = 0;
	int p = 1;

	for (char c : s) {
		k += p * c2num(c);
		p *= 5;
	}
	return k;
}

void insert(vector<int>& t_vec, string s) {
	int k = str2k(s);
	int i = 0;

	while (i < mod) {
		int j = h(k, i);

		if (t_vec[j] < 0) {
			t_vec[j] = k;
			return;
		}
		i++;
	}
	return;
}

bool find(vector<int>& t_vec, string s) {
	int k = str2k(s);
	int i = 0;

	while (i < mod) {
		int j = h(k, i);

		if (t_vec[j] == k) {
			return true;
		}

		if (t_vec[j] < 0) {
			return false;
		}

		i++;
	}
	return false;
}


int main(int argc, char** argv) {
#if READ_FROM_FILE
	ifstream ifs("test0.txt");
	cin.rdbuf(ifs.rdbuf());
#endif
	int n = -1;
	vector<int> t_vec(mod, -1);
	string op, s;

	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> op >> s;
		if (op == "insert") {
			insert(t_vec, s);
		}
		else if(find(t_vec, s)){
			cout << "yes" << endl;
		}
		else {
			cout << "no" << endl;
		}
	}

	return 0;
}
