#include <stdio.h>

#include <vector>
#define N 100000
using namespace std;

vector<vector<int> > G(N);
int visited[N];

int ok;

void dfs(int from, int tar) {
  if (ok) return;
  visited[from] = 1;
  for (int i = 0; i < G[from].size(); i++) {
    int next_v = G[from][i];
    if (visited[next_v]) {
      continue;
    }
    if (next_v == tar) {
      ok = 1;
    } else {
      dfs(next_v, tar);
    }
  }
}

int main(void) {
  int n, m;
  scanf("%d %d", &n, &m);
  int a, b;
  for (int i = 0; i < m; i++) {
    scanf("%d %d", &a, &b);
    G[a].push_back(b);
    G[b].push_back(a);
  }
  int q;
  scanf("%d", &q);
  int x, y;
  for (int i = 0; i < q; i++) {
    scanf("%d %d", &x, &y);
    ok = 0;
    dfs(x, y);
    if (ok)
      printf("yes\n");
    else
      printf("no\n");
  }
}
