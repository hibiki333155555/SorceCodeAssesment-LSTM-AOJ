int main()
{	
	
	int n,i,j;
	double x[105],y[105],xy[105];
	double s1=0,s2=0,s3=0;
	double p1,p2,p3,p4=0;
	
	cin>>n;
	for(i=0;i<n;i++)
		cin>>x[i];
	for(i=0;i<n;i++)
	{
		cin>>y[i];
		xy[i]=fabs(x[i]-y[i]);
	}
	
	for(i=0;i<n;i++)
	{
		s1+=xy[i];
		s2+=xy[i]*xy[i];
		s3+=pow(xy[i],3);
		if(xy[i]>p4)
			p4=xy[i];
	}
	
	p1=s1;
	p2=sqrt(s2);
	p3=pow(s3,1.0/3);
	
	
	printf("%.6f\n",p1);
	printf("%.6f\n",p2);
	printf("%.6f\n",p3);
	printf("%.6f\n",p4);
	
return 0;
}
