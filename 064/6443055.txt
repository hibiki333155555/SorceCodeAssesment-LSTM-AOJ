template<class T>
bool chmax(T & a, T b) {
	if (a < b) {
		a = b;
		return true;
	}
	return false;
}

int N, M;
vector<vector<int>> G;

int dp[100001];

int func(int i) {
	if (dp[i] != 0)return dp[i];
	cout << i << endl;
	int res = 0;
	
	rep(nv,N) {
		chmax(res, func(nv) + G[i][nv]);
	}
	return dp[i] = res;
}
