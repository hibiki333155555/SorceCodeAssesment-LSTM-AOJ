#include <bits/stdc++.h>
using namespace std;

int selection_sort(int A[],int N) {
    int count = 0;
    for(int i = 0;i < N;i++) {
        int minj = i;
        for(int j = i;j < N;j++) {
            if(A[j] < A[minj]) {
                minj = j;
            }
            swap(A[i],A[minj]);
            if(i != minj) count++;
        }
    }
    return count;
}

int main() {
    int n;
    cin >> n;
    int a[n];
    for(int i = 0;i < n;i++) cin >> a[i];

    int count = selection_sort(a,n);

    for(int i = 0;i < n;i++) {
        cout << a[i];
        if(i == n - 1) {
            cout << endl;
        }else {
            cout << " ";
        }
    }
    cout << count << endl;
    return 0;
}
