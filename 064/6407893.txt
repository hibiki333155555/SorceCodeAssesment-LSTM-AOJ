#include<iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

int main(){
    int H,W;
    cin << H << W;
    while(true){
        if(H==0 && W==0)break;
        for(int i=1,i<=H,i++){
            if(i/2!=0){
                for(int j=1,j<=W,j++){
                    if(j/2!=0){
                        printf("#");
                    }else{
                        printf(".");
                    }
                } printf(" ");
            }else{
                for(int j=1,j<=W,j++){
                    if(j/2!=0){
                        printf(".");
                    }else{
                        printf("#");
                    }
            } printf(" ");
        }
    }
}
