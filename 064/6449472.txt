#include <iostream>
using namespace std;

int main()
{
    int a, b;
    cin >> a >> b;
	if(a == 0 && b == 0){
		cout << 1 << endl;
		return 0;
	}
    bool f = false;
    int c = 0;
    for(int i = 30; i >= 0; i--){
        if(((a >> i) & 1) || ((b >> i) & 1)) f = true;
        if(f){
            if(((a >> i) & 1) == ((b >> i) & 1)) c++;
            else break;
        }
    }
    cout << c << endl;
}
