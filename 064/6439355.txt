import java.util.Scanner;

public class Main{


	public static void main(String[] args){
		
		Scanner scan = new Scanner(System.in);
		
		double r = scan.nextDouble();
		double ca = Circlearea(r) ;
		double cl = Circllength(r);
		
		System.out.println(String.format("%.6f",ca) + " " + String.format("%.6f",cl));
		
		scan.close();
    }
	
	public static double Circlearea(double r) {
		return r * r * Math.PI;
	}
	
	public static double Circllength(double r) {
		return r * 2 * Math.PI;
	}
    
}
