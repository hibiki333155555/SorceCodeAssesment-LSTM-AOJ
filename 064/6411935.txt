#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;

int main(){
    int a, b, C;
    double S, L, h, c;
    double PI = acos(-1);
    cin >> a >> b >> C;
    S = a * b * sin(PI * C / 180) / 2;
    L = a + b + sqrt(pow(a, 2) + pow(b, 2) - 2 * a * b * cos(PI * C / 180));
    h = b * sin(PI * C / 180);
    
    printf("%lf\n%lf\n%lf", S, L, h);
    return 0;
}
