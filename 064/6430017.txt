#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define all(x) x.begin(),x.end()
#define fol(i,s,g) for(int i=(int)(s); i<(int)(g); i++)
#define fix(ans,k) fixed << setprecision(k) << ans

int main(){
    vector<vector<int> > nori(1001,vector<int>(1001,0));
    
    int x,y,w,h;
    cin>>x>>y>>w>>h;
    fol(i,x,x+w){
        fol(j,y,y+h){
            nori[i][j]++;
        }
    }
    cin>>x>>y>>w>>h;
    fol(i,x,x+w){
        fol(j,y,y+h){
            nori[i][j]++;
        }
    }
    int ans=0;
    fol(i,0,1001){
        fol(j,0,1001){
            if(nori[i][j] == 1){
                ans++;
            }
        }
    }
    cout<<ans<<'\n';
    return 0;
}
