#include <iostream>
#include <limits.h>
#include <vector>

using namespace std;
#define MAX_N  999999+2
#define MAX_M  10000
typedef long long ll;

ll gcd(ll a, ll b){
    while(b != 0){
        ll tmp = b;
        b = a % b;
        a = tmp;
    }
    return a;
}

ll lcm(ll a, ll b){
    return a * b / gcd(a,b);
}

int main(){
    ll a,b;
    while(scanf("%lld %lld",&a,&b)!=EOF){
        cout << gcd(a,b) << " " << lcm(a,b) << endl;
    }
    return 0;
}
