#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

const ll mod = 1000000007;

ll pow(ll m, ll n){
    if(n == 0ll) return 1ll;
    else if(n % 2 == 1ll) return m * pow(m, n-1) % mod;
    else return pow(m, n/2) * pow(m, n/2) % mod;
}

int main(){
    ll m, n;
    cin >> m >> n;

    cout << pow(m, n) << endl;
    return 0;
}
