#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string str,order,p;
    int q,a,b;
    cin >> str >> q;
    for (int i=0;i<q;i++) {
        cin >> order;
        str_temp = str;
        if (order == "replace") {
            cin >> a >> b >> p;
            str.replace(a, b, p);
        } else if (order == "print") {
            cin >> a >> b;
            cout << str.substr(a, b-a+1) << endl;
        } else if (order == "reverse") {
            cin >> a >> b;
            reverse(str.begin(), str.end());
        }
    }
}
