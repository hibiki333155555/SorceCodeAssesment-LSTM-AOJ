#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,c=0; cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++){
	cin>>v[i];
	}
    n=v.size();
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(v[j]>v[j+1])
			{
				swap(v[j],v[j+1]);
				c++;
			}
		}
	}
	for(int i=0;i<v.size();i++){
		cout<<v[i];
		if(i!=v.size()-1)
		    cout<<" ";
	}
	cout<<"\n"<<c<<"\n";
	return 0;
}
