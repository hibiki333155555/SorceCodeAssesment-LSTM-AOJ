import java.util.*;

class Solve{
    private Scanner sc = new Scanner(System.in);
    private void Finish(){
        this.sc.close();
    }
    public void Do(){
        int a = sc.nextInt();
        int b = sc.nextInt();
        System.out.println(a / b);
        System.out.println(a % b);
        System.out.println(String.format("%.6f", (double) a / b));
        Finish();
    }
}
class Main{
    public static void main(final String[] _args){
        new Solve().Do();
    }
}
