#include<iostream>

using namespace std;
#define MAXN 25
#define fr(n)	for(int i = 0; i < n; i++)

int n;
int number[MAXN];

bool solve(int i, int tar){
	if(tar == 0)	return true;
	if(i == n)	return 0;
	return solve(i + 1, tar) || solve(i + 1, tar - number[i]);
}

int main(){
	cin >> n;
	fr(n){
		scanf("%d", number + i);
	}
	int q;
	cin >> q;
	fr(q){
		int target;
		cin >> target;
		bool flag = solve(0, target);
		if(flag)	cout << "yes" << endl;
		else	cout << "no" << endl;
	}
	return 0;
}
