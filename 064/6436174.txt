#include <stdio.h>

int ids[100000];
int G[100000][100000];
int count[100000];

void dfs(int from, int id) {
  ids[from] = id;
  for (int i = 0; i < count[from]; i++) {
    int next_v = G[from][i];
    if (ids[next_v] > 0) {
      continue;
    }
    dfs(next_v, id);
  }
}

void print(int x, int y) {
  if (ids[x] == ids[y]) {
    printf("yes\n");
  } else {
    printf("no\n");
  }
}

int main(void) {
  int n, m;
  scanf("%d %d", &n, &m);

  int a, b;
  for (int i = 0; i < m; i++) {
    scanf("%d %d", &a, &b);
    G[a][count[a]++] = b;
    G[b][count[b]++] = a;
  }
  int id = 0;
  for (int i = 0; i < n; i++) {
    if (ids[i] > 0) continue;
    dfs(i, ++id);
  }
  int q;
  scanf("%d", &q);
  int x, y;
  for (int i = 0; i < q; i++) {
    scanf("%d %d", &x, &y);
    print(x, y);
  }
}
