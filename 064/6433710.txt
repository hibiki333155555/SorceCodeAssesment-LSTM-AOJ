#include<bits/stdc++.h>

using namespace std;

int main(){
    int n;
    cin >> n;
    vector< pair<int,int> > a(n);
    for(int i=0;i<n;i++){
        cin >> a.at(i).first;
    }
    for(int i=0;i<n;i++){
        cin >> a.at(i).second;
    }
    sort(a.begin(),a.end());
    int ansa = 0,ansb = 0;
    for(int i=1;i<n;i++){
        if(a.at(i).first != a.at(i-1).first){
            ansa = a.at(0).second;
            ansb = a.at(i).second;
            break;
        }
    }
    cout << ansa + ansb << endl;
    return(0);
}
