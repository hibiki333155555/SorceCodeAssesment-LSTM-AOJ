import java.util.Scanner;

//クラスを定義する
class Class {
    private int number; // 人数
    private int[] scores;   // 得点

    public Class(int number, int[] scores) {
        this.number = number;
        this.scores = scores;
    }

    public int getNumber() {
        return number;
    }

    public void setNumber(int number) {
        this.number = number;
    }

    public int[] getScores() {
        return scores;
    }

    public void setScores(int[] scores) {
        this.scores = scores;
    }

    // 得点の合計を求める
    public int sumScore() {
        int sum = 0;
        for (int i = 0; i < number; i++) {
            sum += scores[i];
        }
        return sum;
    }

    // 得点の平均を求める
    public double aveScore() {
        double ave = ((double) sumScore()) / ((double) number);
        return ave;
    }

    // 分散を求める
    public double varScore() {
        double ave = aveScore();
        double var = 0.0;

        for (int i = 0; i < number; i++) {
            double tmp = (scores[i] - ave);
            var += tmp * tmp;
        }
        return (var / number);
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        while (true) {
            int n = sc.nextInt();

            // 終了条件
            if (n == 0) {
                break;
            }

            int[] scores = new int[1000];
            for (int i = 0; i < n; i++) {
                scores[i] = sc.nextInt();
            }

            // クラスを宣言する
            Class classA = new Class(n, scores);

            // 標準偏差を求める
            System.out.printf("%.8f\n", Math.sqrt(classA.varScore()));
        }
    }
}
