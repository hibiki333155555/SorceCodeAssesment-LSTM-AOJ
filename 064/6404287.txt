#include<iostream>
using namespace std;


int main(void){
	long int n, N, c = 0, D;
	long int m, j;
	int i, z;
	
	cin >> m;
	long int a[m];
	
for(n = 0; n < m; n++){
	cin >> a[n];
	
}

for(i = 0; i < m; i++){
	for(z = 1 + i; z < m; z++){
		if(a[i] > a[z]){
			i = z;
			continue;
		}
		if(a[i] < a[z])
		{
			N = i;
			c = a[z] - a[i];
			z++;
			i = m;
				for( z; z < m; z++){
				if(a[N] < a[z]){
					D = a[z] - a[N];
					if(c < D){
						c = D;
					}
				} 
				if(a[N] > a[z]){
					N = z;
				}
			}
		}
	}
}
	
	
	if (c == 0){
		N = a[m - 1];
		j = m - 1;
		for(n = m - 1; n > 0; n--){
			if(N > a[n - 1]){
			
			N = a[n - 1];
			j = n - 1;
		}
		} 
		D = a[j - 1];
		for(n = j - 1; n > 0; n--){
			if(D > a[n - 1]){
				D = a[n - 1];
			}
			
		}
		c = N - D;
	}
	
	
	cout <<c << "\n";
	return 0;
}
