
#include <bits/stdc++.h>

using namespace std;
void insertion()
{
    int n;
    cin>>n;
    int temp;
    int j;
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
     for(int i=0;i<n;i++)
    {
        cout<<a[i]<<" ";
    }
    cout<<"\n";
    for(int i=1;i<n;i++)
    {
        temp=a[i];
        j=i-1;
        while(j>=0 && temp<a[j])
        {
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=temp;
        for(int k=0;k<n;k++)
        {
            cout<<a[k];
            if(k!=n-1){
                cout<<" ";
            }
        }
        cout<<"\n";
    }
   
    
}

int main()
{
    insertion();
    cout<<"\n";

    return 0;
}
