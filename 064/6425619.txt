#include <iostream>
#include <string>
#include <algorithm>
#include <cmath>
using namespace std;

int main( )
{
   int W, H, x, y, r;
   cin >> W >> H >> x >> y >> r;
   bool In = true;
   if(x + r > W) In = false;
   if(x - r < 0) In = false;
   if(y + r > H) In = false;
   if(y - r < 0) In = false;

   if (In) cout << "Yes" << endl;
   else cout << "No" << endl;
    
   
}
