#include<iostream>
#include<string>
using namespace std;

int main(){

    string str1,str2;
    int num,a=0,b=0;

    cin>>num;

    for(int i=0;i<num;++i){
        cin>>str1>>str2;

        if(str1>str2)a+=3;
        else if(str1<str2)b+=3;
        else {
            a+=1;
            b+=1;
        }
    }

     cout<<a<<" "<<b<<endl;

    return 0;
}
