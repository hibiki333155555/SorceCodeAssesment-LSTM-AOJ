#include <cstdio>
#include <vector>
#include <functional>
using namespace std;

const int mod = 1e9 + 7;

int power(int a, int b) {
  int res = 1;
  while (b) {
    if (b & 1) res = res * 1ll * a % mod;
    a = a * 1ll * a % mod;
    b /= 2; 
  }
  return res;
}

int main() {
  int n, m;
while (scanf("%d%d", &n, &m) == 2) {
  if (!n && !m) break;
  vector<int> x(m), y(m);
  vector<int> p(n + 1);
  for (int i = 1; i <= n; i++) p[i] = i;
  
  function<int(int)> find = [&](int u) {
    if (p[u] == u) return p[u];
    return p[u] = find(p[u]);
  };
  vector<int> have(n + 1, 0);
  for (int i = 0; i < m; i++) {
    scanf("%d%d", &x[i], &y[i]);
    have[x[i]] = have[y[i]] = 1;
    int u = find(x[i]), v = find(y[i]);
    if (u != v) {
      p[u] = v; 
    }
  }
  vector<int> vis(n + 1, 0);
  int cnt = 0;
  for (int i = 0; i < m; i++) {
    int u = find(x[i]);
    if (!vis[u]) {
      vis[u] = 1;
      cnt++; 
    }
  }
  int left = 0;
  for (int i = 1; i <= n; i++) left += (have[i] == 0);
  int ans = cnt == 0 ? 0 : 1;
  if (ans < 0) ans += mod;
  int add = power(2, cnt) * 1ll * power(2, left) % mod;
  ans = (ans + add) % mod;
  printf("%d\n", ans);
}
  return 0; 
}
