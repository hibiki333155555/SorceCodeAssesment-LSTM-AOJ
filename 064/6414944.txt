#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <numeric>
#include <set>
#include <map>
#include <queue>
#include <utility>
using namespace std;
using ll = long long;
const ll mod = 998244353;

int main(){
    int n;
    string s;
    cin >> n >> s;
    vector<ll> a(n);
    for(int i=0; i<n; i++){
        cin >> a[i];
    }
    ll ans = 0;
    for(int d=0; d<2; d++){
        ll dep = 0;
        priority_queue<pair<ll,ll>> wait;
        for(int i=0; i<n; i++){
            if(s[i] == '('){
                dep++;
            }else{
                dep--;
                wait.push({-a[i], -i});
            }
            if(dep < 0){
                auto [cost, idx] = wait.top();
                cost = -cost; idx = -idx;
                wait.pop();
                ans += cost;
                s[idx] = '(';
                dep += 2;
            }
        }
        reverse(a.begin(), a.end());
        reverse(s.begin(), s.end());
        for(char &c: s){
            if(c == '(') c = ')';
            else c = '(';
        }
    }
    cout << ans << endl;
    return 0;
}
