#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n;

  string a;

  for(int i = 1;i <= n;i++){
    a = to_string(i);
    if(i % 3 == 0){
      cout << " " << i;
    }else{
      for(int j = 0;j < a.size();j++){
        if(a[j] == '3'){
          cout << " " << i;
          break;
        }
      }
    }
  }
  

  return 0;
}
