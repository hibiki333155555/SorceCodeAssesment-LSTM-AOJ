#include <bits/stdc++.h>

using namespace std;

signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int h, w, r, c;
    cin >> h >> w;
    vector<string> lsa, lsb;
    string t;
    for (int i = 0; i < h; ++i)
    {
        cin >> t;
        lsa.push_back(t);
    }
    cin >> r >> c;
    for (int i = 0; i < r; ++i)
    {
        cin >> t;
        lsb.push_back(t);
    }

    for (int i = 0; i < h-r+1; ++i)
    {
        for (int j = 0; j < w && j != -1; ++j)
        {
            j = lsa[i].find(lsb[0], j);
            if (j == -1)
                break;
            else
            {
                bool ans = true;
                for (int k = 1; k < r - 1; ++k)
                {
                    if (lsa[i + k].compare(j, c, lsb[k]))
                    {
                        ans = false;
                        break;
                    }
                }
                if (ans)
                    cout << i << " " << j << endl;
            }
        }
    }
    return 0;
}
