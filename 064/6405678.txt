#include <iostream>
#include <algorithm>
#include <list>

using namespace std;

int main(){
    int n;
    cin>>n;
    
    int op, d, x;
    list<int> l;
    auto c = l.end();
    for(int i=0; i<n;i++){
        cin>>op;
        if(op==0){
            cin>>x;
            c = l.insert(c,x);
        }
        else if(op==1){
            cin>>d;
            if(d>0) while(d--)c++;
            else while(d++)c--;
        }
        else if(op==2){
            c = l.erase(c);
        }
    }
    for(auto& e:l){
        cout << e << endl;
    }
    return 0;
}
