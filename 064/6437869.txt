# ２桁の中に3と３の倍数があるかどうかを見つける関数
def che3(n):
    if n%3==0:
        return True
    elif n%10==3:
        return True
def che32(n):
    if n%10==3:
        return True

n = int(input())
arr = []
for i in range(1,n+1):
    if che3(i):
        print("",i,end="")
    elif che32(i//10):
            print("",i,end="")
    elif che32(i//100):
            print("",i,end="")
    elif che32(i//1000):
