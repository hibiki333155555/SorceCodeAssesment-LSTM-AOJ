#include <iostream>
#include <bitset>

using namespace std;

bool BitOperation1(unsigned int x) {

	bitset<32> b = x;
	cout << b << endl;
	cout << ~b << endl;
	cout << (b << 1) << endl;
	cout << (b >> 1) << endl;

	return true;

}

int main() {

	unsigned int x = 0;

	cin >> x;

	if (BitOperation1(x)) {
		return 0;
	}
	else {
		return -1;
	}

	return 0;
}
