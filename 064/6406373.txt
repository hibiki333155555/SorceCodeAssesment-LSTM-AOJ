#include <iostream>
#include <list>
using namespace std;
int n;
list <int> a;
list <int> :: iterator it = a.begin ();
int main () {
    cin >> n;
    while (n--) {
    	string s;
    	int x;
    	cin >> s;
    	if (s == "insert") {
    		cin >> x;
    		a.push_front (x);
		}
		else if (s == "delete") {
			cin >> x;
			a.remove (x);
		}
		else if (s == "deleteFirst") a.pop_front ();
		else a.pop_back ();
	}
	it = a.begin ();
	cout << *it;
	it++;
	for (;it != a.end ();it++) cout << ' ' << *it;
	cout << endl;
    return 0;
}
