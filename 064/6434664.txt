#include <bits/stdc++.h>
using namespace std;
const int INF = 1001001001;
const int MOD = 1000000007;

int main(){
  int n;cin>>n;
  vector<int> a(n);
  for(int i=0;i<n;i++) cin>>a.at(i);
  int cnt = 0;
  bool flg = true;
  while(flg){
    flg = false;
    for(int i=n-1;i>=1;i--) {
      if(a.at(i)<a.at(i-1)) {
        cnt++;
        swap(a.at(i),a.at(i-1));
        flg = true;
      }
    }
  }
  for(int i=0;i<n;i++) {
    cout << a.at(i);
    if(i==n-1) cout << endl;
    else cout << " ";
  }
  cout << cnt << endl;
}
