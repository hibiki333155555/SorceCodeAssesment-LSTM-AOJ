#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,x;
    int count=0;
    while(1){
        cin>>n>>x;
        if(n==0 && x==0) break;
        for(int i=1; i<n+1;i++){
            for(int j=1; j<n+1;j++){
                for(int k=1; k<n+1;k++){
                if(i+j+k==x && j-i>0 && k-j>0) count++;
                }
            }
        }
        cout<< count <<endl;
        count=0;
    }
}
