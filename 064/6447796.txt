#include <iostream>
#include <bits/stdc++.h>
#include <string>
using namespace std;

int main() {
   string t;
   int c=0;

   while(cin >> t) {
      for(int i=0; i<t.size(); i++) {
         if(islower(t[i])) t[i] = toupper(t[i]);
         else t[i] = tolower(t[i]);
      }
      
      if(c!=0) cout << " ";
      cout << t;
      c++;
   }
   cout << "\n";
}
