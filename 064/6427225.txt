#include <bits/stdc++.h>
using namespace std;

int main(){
    int y,m,d,y2,m2,d2;
    cin>>y>>m>>d;
    cin>>y2>>m2>>d2;
    if(y<y2){
        swap(y,y2);
        swap(m,m2);
        swap(d,d2);
    }else if(y==y2){
        if(d<d2||(d==d2&&m<m2)){
            swap(y,y2);
            swap(m,m2);
            swap(d,d2);
        }
    }
    int ans=y-y2;
    if(m2>m){
        ans++;
    }else if(m==m2&&d2>d){
        ans++;
    }
    //cout<<y<<" "<<m<<" "<<d<<"\n";
    //cout<<y2<<" "<<m2<<" "<<d2<<"\n";
    cout<<ans<<"\n";
    return(0);
}
