#include <bits/stdc++.h>
using namespace std;

void bubbleSort(int A[], int N) {
    bool flag = true;
    int count = 0;
    while(flag) {
        flag = false;
        for(int j = N - 1; j >= 1;j--) {
            if(A[j] < A[j-1]) {
                swap(A[j], A[j-1]);
                flag = true;
                count++;
            }
        }
    }
    for(int i = 0;i<N;i++) cout << A[i] << " ";
    cout << count << endl;
}

int main() {
    int n,a[n];cin >> n;
    for(int i = 0;i < n; i++) cin >> a[i];
    bubbleSort(a,n);
}
