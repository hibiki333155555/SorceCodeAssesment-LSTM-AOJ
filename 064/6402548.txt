#include<bits/stdc++.h>
#define endl "\n"
using namespace std ;

int main(){
    while(1){
  int a  ; cin >> a ;
  double k ;
  char c ; cin >> c ;
  int b ; cin >> b ;
  if(c == '+'){
    k = a + b ;
  }
   if(c =='-'){
    k = abs(a-b) ;
  }
   if(c == '*'){
    k = a * b ;
  }
   if(c == '/'){
    k = a/b ;
   }
    if(c == '?') break ;

    cout << k << endl;
    }

return 0 ;

 }
