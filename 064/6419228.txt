#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <iterator>
#include <numeric>
#include <string>
#include <vector>
using namespace std;


int main()
{
    
    int n;
    cin >> n;
    int *arr1 = new int[n], *arr2 = new int[n];

    for(int i = 0;i<n;i++)
        cin >> arr1[i] >> arr2[i];

    auto vv = vector<vector<int>>(n, vector<int>(n, 0));
    for(int l = 2; l <= n; l++)
    {
        for(int i = 0, j; i <= n-l; i++)
        {
            j = i+l-1;
            vv[i][j] = 1000000000;
            for(int k=i, q; k<j;k++)
            {
                q = vv[i][k]+vv[k+1][j]+arr1[i]*arr2[k]*arr2[j];
                if(q < vv[i][j])
                    vv[i][j] = q;
            }
        }
    }
    cout << vv[0][n-1] << endl;
    return 0;
}
