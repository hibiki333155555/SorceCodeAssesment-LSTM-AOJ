#include <bits/stdc++.h>
using namespace std;

int main() {
    int n; double W; cin >> n >> W;
    vector<double> value(n), weight(n), ratio(n);
    for (int i = 0; i < n; i++) 
        cin >> value[i] >> weight[i], ratio[i] = value[i]/weight[i];
    vector<int> idx(n);
    for (int i = 0; i < n; i++) idx[i] = i;
    sort(idx.begin(), idx.end(), [&](int i, int j) { return ratio[i] > ratio[j]; });
    double ans = 0;
    for (int k = 0; k < n; k++) {
        int i = idx[k];
        if (weight[i] <= W) {
            ans += value[i];
            W -= weight[i];
        } else {
            double frac = W/weight[i]; // weight[i] * x == W, 
            ans += value[i] * frac;
            break;
        }
    }
    cout << fixed << setprecision(12) << ans << "\n";
}
