#include <bits/stdc++.h>
using namespace std;
const int INF = 1001001001;
const int MOD = 1000000007;

int main(){
    int n;cin>>n;
    vector<int> s(n);
    for(int i=0;i<n;i++) cin>>s.at(i);
    int m;cin>>m;
    vector<int> t(m);
    for(int i=0;i<m;i++) cin>>t.at(i);
    int cnt = 0;
    for(int i=0;i<m;i++) {
        for(int j=0;j<n;j++) {
            if(s.at(j)==t.at(i)) {
                cnt++;
                break;
            }
        }
    }
    cout << cnt << endl;
}
