#include<stdio.h>
#include<stdbool.h>
#define MAX 100

int bubble_sort(int A[], int n){

  int cnt = 0, temp;
  bool flag = true;
  for(int i = 0; flag; i++ ){
    flag = false;
    for(int j = n - 1; j >= i + 1; j--){
      if(A[j] < A[j - 1]){
        temp = A[j];
        A[j] = A[j - 1];
        A[j - 1] = temp;
        flag = true;
        cnt++;
      }
    }
  }

  return cnt;
}

int main()
{
  int n;
  int A[MAX];
  scanf("%d", &n);
  for(int i = 0; i < n; i++ ) scanf("%d", &A[i]);
  int ans = bubble_sort(A, n);
  for(int i = 0; i < n; i++ ){
    if(i) printf(" ");
    printf("%d", A[i]);
  }
  printf("\n");
  printf("%d\n", ans);
  return 0;
}
