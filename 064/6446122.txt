def factorization(x):
    i = 2
    ans = []
    while i**2 <= x:
        e = 0
        while x%i == 0:
            e += 1
            x //= i
        if e>0:
            ans.append([i, e])
        i += 1
    if x!=1: ans.append([x, 1])
    return ans

N = int(input())
A = factorization(N)
ans = []
for i,j in A:
	ans += [i]*j
print(str(N)+": "+" ".join(map(str, ans)))
