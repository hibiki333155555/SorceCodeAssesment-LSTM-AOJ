#include <bits/stdc++.h>
using namespace std;
int main(){
    vector<int>y(3),x(3);
    for(int i=0;i<3;i++){
        cin>>y[i];
    }
    for(int i=0;i<3;i++){
        cin>>x[i];
    }
    if(y<x){
        swap(y,x);
    }
    if(make_pair(y[1],y[2])>make_pair(x[1],x[2])){
        cout<<y[0]-x[0]+1<<endl;
    }else{
        cout<<y[0]-x[0]<<endl;
    }
}
