#include <bits/stdc++.h>
using namespace std;
#define rep(i, n) for (int i = 0; i < (int)(n); i++)

int main() {
    int N,M=0,m=0;
    cin>>N;
    vector<int> S(2*N+1,-1);
    int A;
    rep(i,N){
        cin>>A;
        S[A]+=2;
    }
    S[0]=0;
    rep(i,2*N){
        S[i+1]+=S[i];
        M=max(M,S[i+1]);
        m=max(m,-S[i+1]);
    }
    if(M<m)swap(M,m);
    int an=(M-m)/2;
    cout<<max(M-an,an+1)<<endl;
}
