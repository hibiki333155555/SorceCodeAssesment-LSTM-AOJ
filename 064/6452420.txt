#include<stdio.h>

int saikoro(int mu, int mt, int mm, int mh, int mo, int ms, char houi, int maeue){
    int ue;
    
    if(mu == maeue){
        if(houi == 'n') ue = mt;
        else if(houi == 's') ue = mo;
        else if(houi == 'e') ue = mh;
        else if(houi == 'w') ue = mh;
    }
    else if(mt == maeue){
        if(houi == 'n') ue = ms;
        else if(houi == 's') ue = mu;
        else if(houi == 'e') ue = mh;
        else if(houi == 'w') ue = mm;
    }
    else if(mm == maeue){
        if(houi == 'n') ue = mt;
        else if(houi == 's') ue = mo;
        else if(houi == 'e') ue = mu;
        else if(houi == 'w') ue = ms;
    }
    else if(mh == maeue){
        if(houi == 'n') ue = mt;
        else if(houi == 's') ue = mo;
        else if(houi == 'e') ue = ms;
        else if(houi == 'w') ue = mu;
    }
    else if(mo == maeue){
        if(houi == 'n') ue = mu;
        else if(houi == 's') ue = ms;
        else if(houi == 'e') ue = mh;
        else if(houi == 'w') ue = mm;
    }
    else if(ms == maeue){
        if(houi == 'n') ue = mo;
        else if(houi == 's') ue = mt;
        else if(houi == 'e') ue = mh;
        else if(houi == 'w') ue = mm;
    }
    
    return ue;
}

int main(void){
    int a,b,c,d,e,f,i,ue;
    char str[101];
    
    scanf("%d %d %d %d %d %d",&a,&b,&c,&d,&e,&f);
    scanf("%s", str);
    
    ue = a;
    
    for(i = 0;str[i] != '\0';i++){
        ue = saikoro(a,b,c,d,e,f,str[i],ue);
    }
    
    printf("%d\n",ue);
    
    return 0;
}
