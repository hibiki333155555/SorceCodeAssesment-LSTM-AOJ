#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
typedef long long ll;
const int N = 10005;
ll tree[N];
int root;
int n;

int main() {
	cin>>n;
	for(int i=0;i<N;i++)
		tree[i] = -10000000000;
	for(int i=1;i<=n;i++) {
		cin>>tree[i];
	}
	root = 1;
	for(int i=1;i<=n;i++) {
		cout<<"node "<<i<<": key = "<<tree[i]<<", ";
		if(i != root)
			cout<<"parent key = "<<tree[i/2]<<", ";
		if(tree[i<<1] >= -2000000000)
			cout<<"left key = "<<tree[i<<1]<<", ";
		if(tree[(i<<1) + 1]>= -2000000000)
			cout<<"right key = "<<tree[(i<<1) + 1]<<", ";
		cout<<endl;
	}
}
