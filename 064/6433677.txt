#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

// dp[i][w] := i番目まで，重さwを超えないときの価値の最大値
int dp[110][10100];

int main(){
    int n, W;
    cin >> n >> W;

    vector<int> v(n+1), w(n+1);
    for(int i=1; i<=n; ++i) cin >> v[i] >> w[i];

    for(int i=1; i<=n; ++i){
        for(int j = 0; j <= W; ++j){
            // i番目を選ぶ場合
            if(j - w[i] >= 0){ 
                dp[i][j] = max(dp[i][j], dp[i-1][j-w[i]] + v[i]);
            }
            // 選ばない場合
            dp[i][j] = max(dp[i][j], dp[i-1][j]);
        }
    }

    int ans = -1;
    for(int j = 0; j <= W; ++j) ans = max(ans, dp[n][j]);

    cout << ans << endl;
    return 0;
}
