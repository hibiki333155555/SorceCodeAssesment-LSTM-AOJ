#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef long double LD;

#define rep(i,n) for(LL i=0;i<(n);i++)

int main(){
    LL n, x;
    LL ans = 0;

    cin >> n >> x;

    rep(i,n){
        LL m; cin >> m;

        if(m >= x){
            ans++;
        }

    }

    cout << ans << endl;

    return 0;
}
