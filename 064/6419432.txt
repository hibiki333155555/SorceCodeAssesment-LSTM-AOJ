#include <iostream>
#include <string>

using std::cout; using std::cin;
using std::endl;

int main()
{
    std::string buff;
    std::getline(std::cin, buff);

    for (int i = 0;i < buff.length();++i) {
        if ('A' <= buf[i] && buff[i] <= 'Z')
            buff[i] += 0x20;
        else if ('a' <= buff[i] && buff[i] <= 'z')
            buff[i] -= 0x20;
    }

    cout << buff;
    return 0;
}
