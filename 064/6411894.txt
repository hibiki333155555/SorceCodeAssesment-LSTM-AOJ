#include <iostream>
#include <cstdio>

using namespace std;

void printArray(int *nums, int size)
{
  for (int i = 0; i < size; i++)
  {
    if (i != size - 1) printf("%d ", nums[i]);
    else printf("%d\n", nums[i]);
  }
}

int selectionSort(int *nums, int size)
{
  int count = 0;
  for (int i = 0; i < size; i++)
  {
    int minj = i;
    for (int j = i+1; j < size; j++)
    {
      if (nums[j] < nums[minj])
      {
        minj = j;
      }
    }

    if (minj != i)
    {
      count++;
      int temp = nums[i];
      nums[i] = nums[minj];
      nums[minj] = temp;
    }
  }

  return count;
}

int main(int argc, char const *argv[])
{
  int n;

  cin >> n;
  int *nums = new int[n];

  for (int i = 0; i < n; i++)
  {
    cin >> nums[i];
  }

  int count = selectionSort(nums, n);

  printArray(nums, n);
  cout << count << endl;

  return 0;
}
