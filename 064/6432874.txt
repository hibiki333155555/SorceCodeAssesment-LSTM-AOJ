#include <stdio.h>

int graph[101][101];

int main(void) {
  int n;
  scanf("%d", &n);

  for (int i = 0; i < n; i++) {
    int to, m;
    scanf("%d %d", &to, &m);
    for (int k = 0; k < m; k++) {
      int v;
      scanf("%d", &v);
      graph[to - 1][v - 1] = 1;
    }
  }
  for (int i = 0; i < n; i++) {
    for (int k = 0; k < n; k++) {
      printf("%d", graph[i][k]);
      if (k != n - 1) {
        printf(" ");
      }
    }
    printf("\n");
  }
}
