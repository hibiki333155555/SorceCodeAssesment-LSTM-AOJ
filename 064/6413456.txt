#include <iostream>
using namespace std;

int main() {
    int r, c;
    cin >> r >> c;

    for (int i = 0; i < r; i++) {
        int sum = 0;
        for (int j = 0; j < c; j++) {
            int x;
            cin >> x;
            cout << x << " ";
            sum += x;
        }
        cout << sum << endl;
    }

    return 0;
}
