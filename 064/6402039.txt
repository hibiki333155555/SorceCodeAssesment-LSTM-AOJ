//

#include <iostream>
using namespace std;
int main()
{
    int n; cin >> n;
    int t; cin >> t;
	int mask = 0;
	for (int i = 0; i < t; i++)
	{
		int a; cin >> a;
		mask += (1 << a);
	}

	int k = mask;
	// print out all super sets of mask
	while (k < (1 << n))
	{
		// test to see if k and mask have the same bits set
		if ((k & mask) == mask)
		{
			cout << k << ":";
			for (int i = 0; i < n; i++)
			{
				if (k & (1 << i))
					cout << " " << i;
			}
			cout << endl;
		}
		k++;
	}
}
