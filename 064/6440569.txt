#include <iostream>
#include <vector> // vector
#include <algorithm> // min, max, swap, sort, reverse, lower_bound, upper_bound
#include <utility> // pair, make_pair
using namespace std;

static const int MAX = 200000;

void bubblesort(vector<int> R, int n) {
    int tmp = 0;
    for (int i = 0; i < n; i++) {
        int minj = i;
        for (int j = i + 1; j < n; j++) {
            if (R.at(minj) > R.at(j )) {
                minj = j;
            }
        }
        if(minj != i){
            swap(R.at(minj), R.at(i));
            tmp++;
        }
    }
    for (int i = 0; i < n; i++) {
    if(i > 0){
        cout << " ";
    }
    cout << R.at(i);
}
    cout << endl << tmp;
}



int main() {
    int  n;

    cin >> n;
    vector<int> R(n);
    for(int i = 0;i < n;i++)cin >> R.at(i);

    bubblesort(R, n);



}
