#include <bits/stdc++.h>
using namespace std;
int isSubset(long long int arr1[],long long int arr2[],int m, int n)
{
	int i = 0,j=0;
	for (i = 0; i < n; i++) {
		for (j = 0; j < m; j++) {
			if (arr2[i] == arr1[j])
				break;
		}
		if (j == m)
			return 0;
	}
	return 1;
}
int main()
{
	long long int m;
    cin>>m;
    long long int arr1[m];
    for(int i=0;i<m;i++)
   {
     cin>>arr1[i];
   }
   long long int n;
   cin>>n;
   long long int arr2[n];
   for(int i=0;i<n;i++)
   {
    cin>>arr2[i];
   }
   if (isSubset(arr1, arr2, m, n))
		cout<<1;
	else
		cout<<0;
    return 0;
}
