#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#define MAX 100
#define INF 1001001000

int n;
int G[MAX][MAX], d[MAX], p[MAX];
bool visited[MAX];

void prim(){
  int u, mincost, ans = 0;
  for(int i = 0; i < n; i++ ){
    visited[i] = false;
    d[i] = INF;
  }
  p[0] = -1;
  d[0] = 0;
  while(1){
    mincost = INF;
    for(int v = 0; v < n; v++ ){
      if(!visited[v] && d[v] < mincost){
        u = v;
        mincost = d[v];
      }
    }

    if(mincost == INF) break;

    visited[u] = true;
    ans += mincost;

    for(int v = 0; v < n; v++ ){
      if(G[u][v] < d[v] && !visited[v]){
        d[v] = G[u][v];
        p[v] = u;
      }
    }
  }
  printf("%d\n", ans);
}

int main(){
  int a;
  scanf("%d", &n);
  for(int i = 0; i < n; i++ ){
    for(int j = 0; j < n; j++ ){
      scanf("%d", &a);
      if(a == -1) G[i][j] = INF;
      else G[i][j] = a;
    }
  }

  prim();
}
