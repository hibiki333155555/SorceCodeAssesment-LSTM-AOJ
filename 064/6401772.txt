#include <bits/stdc++.h>	
using namespace std;	
typedef long long ll;

int main(){
  int n;
  cin>>n;
  vector<int>a(n);
  for(int i=0;i<n;i++){
    cin>>a[i];
  }
  int ans=0;
  int minj;
  for( int i=0;i<n;i++){
    minj=i;
    for(int j=i;j<n;j++){
      if(a[j]<a[minj])minj=j;
    }
    swap(a[minj],a[i]);
    if(minj!=i)ans++;
  }
    
  for(int i=0;i<n;i++){
    cout<<a[i];
    if(i<n-1){
      cout<<" ";
    }
    else{
      cout<<endl;
    }
  }
  cout<<ans<<endl;
 return 0;
  
}
