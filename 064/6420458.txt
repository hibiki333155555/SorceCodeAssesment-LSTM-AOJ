#include <bits/stdc++.h>
using namespace std;
int main(){
  int L, N;
  cin >> L >> N;
  vector<int> a(N);
  for (int i = 0; i < N; i++){
    cin >> a[i];
  }
  int p = -1;
  for (int i = 0; i < N; i++){
    L -= a[i];
    if (L <= 0){
      p = i + 1;
      break;
    }
  }
  if (p == -1){
    cout << "No" << endl;
  } else {
    cout << p << endl;
  }
}
