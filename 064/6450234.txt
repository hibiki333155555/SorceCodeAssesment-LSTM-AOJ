#include <bits/stdc++.h>
using namespace std;

constexpr int mod = 998244353;
long long dp[2001][2001];

int main() {
    int N,M;
    cin >> N >> M;
    vector<int>A(M);
    for(int i = 0; i < M; i++) {
        cin >> A[i];
    }
    dp[0][0] = 1;
    for(int i = 0; i < M; i++) {
        for(int j = 0; j <= N; j++) {
            dp[i+1][j] += dp[i][j];
            dp[i+1][j] %= mod;
            if(j+A[i]+1 <= N) {
                dp[i+1][j+A[i]+1] += mod-dp[i][j];
                dp[i+1][j+A[i]+1] %= mod;
            }
        }
        for(int j = 0; j < N; j++) {
            dp[i+1][j+1] += dp[i+1][j];
            dp[i+1][j+1] %= mod;
        }
    }
    for(int i = 1; i <= N; i++) {
        dp[M][N] *= i;
        dp[M][N] %= mod;
    }
    cout << dp[M][N] << endl;
}
