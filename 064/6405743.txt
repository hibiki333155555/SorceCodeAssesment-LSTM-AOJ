vector<int> fibos;

int fibonacci(int i) {
  if(i == 0 || i == 1) { return 1; }
  if(fibos[i] == -1) { fibos[i] = fibonacci(i - 1) + fibonacci(i - 2); }
  return fibos[i];
}
