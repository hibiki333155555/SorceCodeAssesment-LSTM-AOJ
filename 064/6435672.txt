#include <bits/stdc++.h>
using namespace std;

void print_array(int a[], int n) {
    for(int i = 0; i < n; i++) {
        cout << a[i];
        if(i == n - 1) {
            cout << endl;
        } else {
            cout << " ";
        }
    }
}

int bubble_sort(int a[], int n) {
    bool flag = true;
    int count = 0;
    while(flag) {
        flag = false;
        for(int i = n - 1; i >= 1; i--) {
            if(a[i] < a[i - 1]) {
                swap(a[i], a[i - 1]);
                flag = true;
                count++;
            }
        }
    }
    return count;
}

int main() {
    int n;
    cin >> n;
    int a[n];
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int count = bubble_sort(a, n);

    for(int i = 0; i < n; i++) {
        cout << a[i];
        if(i == n - 1) {
            cout << endl;
        } else {
            cout << " ";
        }
    }
    cout << count << endl;

    return 0;
}
