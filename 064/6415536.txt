#include<iostream>
#include<string>
using namespace std;

int main(){

    char ch;
    int counter[26];
    
    while(cin>>ch){
        int num = ch - 'a';counter[num]++;
        int nun = ch - 'A';counter[nun]++;
    }

    for(int i=0;i<26;++i){
        cout<<char('a'+i)<<" : "<<counter[i];
        ++i;
    }
  return 0;
}
