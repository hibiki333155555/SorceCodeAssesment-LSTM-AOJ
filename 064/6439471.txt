#include <bits/stdc++.h>
using namespace std;

int main() {
  int arr[10000][2];
  int i = 0;
  while (1) {
    cin >> arr[i][0] >> arr[i][1];
    if (arr[i][0] == 0 && arr[i][1] == 0) break;
    i++;
  }
  for (int l = 0; l < i; l++) {
    for (int k = 0; k < 300; k++) {
    if (k == arr[l][0]) break;
    for (int j = 0; j < 300; j++) {
      if (j == arr[l][1]) break;
      printf("#");
    }
    printf("\n");
    }
    printf("\n");
  }
  
  return 0;

}
