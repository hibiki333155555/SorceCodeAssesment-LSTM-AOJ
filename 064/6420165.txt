import java.util.Scanner;

public class Main {
	
	@SuppressWarnings("resource")
	public static void main(String[] args){
		int x = 0;
		int y = 0;
		Scanner sc = new Scanner(System.in);
				
		for (int i = 0; ; i++) {
			x = sc.nextInt();
			y = sc.nextInt();
			if (x == 0 && y == 0) {
				break;
			} else {
				System.out.println(x + " " + y);
			}			
		}
    }
}
