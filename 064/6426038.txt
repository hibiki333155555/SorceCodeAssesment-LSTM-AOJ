#include <stdio.h>
#include <string.h>

void c(char *str){
    int len=strlen(str);
    int i;
    for(i=0;i<len;i++){
        if('a'<=str[i] && 'z'>=str[i]){
            str[i]-=32;
        }
    }
}

int main(){
    char w[11],t[1000];
    char *start,*re;
    int cnt=0,i=0;
    char end[]="END_OF_TEXT";
    
    scanf("%s",w);

    while(1){

        scanf("%s",t);
        if(strcmp(t,end)==0){
            break;
        }
        start=t;
        do{
            re=strstr(start,w);
            if(re!=NULL){
                re++;
                cnt++;
                start=re;
            }
        }while(re!=NULL);
    }
    printf("%d\n",cnt);
}
