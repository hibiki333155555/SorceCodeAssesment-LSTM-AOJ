#include <math.h>
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
int main(){
    ll a,b;cin>>a>>b;
    double pi=acos(-1);
    ll c;cin>>c;
    double rad=c*pi/180;
    double sinrad=sin(rad);
    double ans=a*b*sinrad/2;
    cout<<fixed<<setprecision(12)<<ans<<endl;
    double ans2;
    double temp=sqrt(a*a+b*b-(2*a*b*cos(rad)));
    ans2=a+b+temp;
    cout<<ans2<<endl;
    double ans3=sinrad*b;
    cout<<ans3<<endl;
}
