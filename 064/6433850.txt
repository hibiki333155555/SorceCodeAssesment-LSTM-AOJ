#include<iostream>
using namespace std;
int l, r;

int main(){
    int n, a[1000];
    cin >> n;
    int l = 1001;
    int r = 1001;
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    for(int i = 0; i < n; i++){
        int w;
        cin >> w;
        switch(a[i]){
            case 0:
            if(w < r){
                r = w;
            }
            break;
            case 1:
            if(w < l){
                l = w;
            }
            break;
        }
    }

    cout << l + r << endl;

    return 0;
}
