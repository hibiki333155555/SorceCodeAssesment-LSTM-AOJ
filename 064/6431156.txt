#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int bubulesort(int a[], int m) {
	bool flag = 0;
	int sw;
	int i = 0;
	int j = 0;
	for  ( i= 0; flag; i++)
	{
		for ( j = m-1; j >= i+1; j--)
		{
			swap(a[i], a[j]);
			sw++;
			flag = 1;

		}
	}

	return sw;

}

int main() {
	int a[100];
	int m;
	cin >> m;
	for (int i = 0; i < m; i++)cin >> a[i];
	int sw = bubulesort(a, m);
	for (int i = 0; i <= m; i++) {
	if(i)
		cout << " ";
		cout << a[i] << endl;
		
	}
  
	cout << sw << endl;
}
