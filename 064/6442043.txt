#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    int N;
    int num[11000];
    
    cin >> N;
    for(int i = 0; i < N; i++){
        cin >> num[i]; 
    }
    
    sort(num, num + N);
    int Min = num[0];
    int Max = num[N - 1];
    int sum;
    for(int i = 0; i < N; i++){
        sum += num[i]
    }
    
    cout << Min << " " << Max << " " << sum << endl;
    
    return 0;
}
