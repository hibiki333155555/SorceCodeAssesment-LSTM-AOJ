#include <stdio.h>
#include <math.h>

#define N_MAX 10000

int prime(int n );

int main(void) {
	// your code goes here
	
	int i,n;
	int count=0;
	int data[N_MAX] = {0};
	
	scanf("%d",&n);
	
	for(i=0;i<n;i++){
		scanf("%d",&data[i]);
	}
	
	for(i=0;i<n;i++){
		if(prime(data[i]) == 1){
			count++;
		}
	}
	printf("%d\n",count);
	
	return 0;
}

int prime(int n ){
	
	int i,ret;
	int k = sqrt(n);
	
	if(n <= 2){
		return 1;
	}
	if(n % 2 == 0){
		return 0;
	}
	ret = 1;
	for(i = 3; i <= k;i+=2){
		if(n % i == 0){
			ret =0;
			break;
		}
	}
	return ret;
}
