#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef unsigned long long int ull;
#define BIG_NUM 2000000000
#define HUGE_NUM 4000000000000000000 //オーバーフローに注意
#define MOD 1000000007
#define EPS 0.000000001


int A[4];


int calc(){

	int ret = 0;

	for(int i = 0; i < 4; i++){

		if(A[i] > 0){

			ret++;
		}
	}
	return ret;
}

void func(){

	while(calc() > 1){

		int mini = BIG_NUM;
		int ind = -1;

		for(int i = 0; i < 4; i++){
			if(A[i] == 0)continue;
			if(A[i] < mini){

				mini = A[i];
				ind = i;
			}
		}

		for(int i = 0; i < 4; i++){
			if(A[i] == 0 || i == ind)continue;

			A[i] -= mini;
		}
	}

	for(int i = 0; i < 4; i++){
		if(A[i] > 0){

			printf("%d\n",A[i]);
			return;
		}
	}
}

int main(){

	while(true){

		scanf("%d %d %d %d",&A[0],&A[1],&A[2],&A[3]);
		if(A[0] == 0 && A[1] == 0 && A[2] == 0 && A[3] == 0)break;


		func();
	}

	return 0;
}
