#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, sum = 0;
    int max = -1000000;
    int min = 1000000;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        int a;
        cin >>a;
        sum += a;
        if (a > max) max = a;
        else if (a < min) min = a;
    }
    printf("%d %d %d", min, max, sum);
}
