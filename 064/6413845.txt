L = []
for i in range(3):
    T = []
    for _ in range(8):
        p, t = input().split()
        T.append((int(p), float(t)))
    T.sort(key=lambda x: x[1], reverse=True)
    print(f"{T[-1][0]} {T[-1][1]:.2f}")
    T.pop()
    print(f"{T[-1][0]} {T[-1][1]:.2f}")
    T.pop()
    L += T
L.sort(key=lambda x: x[1])
print(f"{L[0][0]} {L[0][1]:.2f}")
print(f"{L[1][0]} {L[1][1]:.2f}")
