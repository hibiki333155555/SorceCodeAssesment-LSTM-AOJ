using namespace std;
using ll=long long;

int main(){
  ll N;
  cin>>N;
  unordered_map<ll,vector<pair<ll,ll>>> um;
  unordered_set<ll> us;
  for(int i=0;i<N;i++){
    for(int j=0;j<N;j++){
      ll A;
      cin>>A;
      um[A].push_back(make_pair(i+1,j+1));
      us.insert(A);
    }
  }
  for(ll i:us){
    sort(um[i].begin(),um[i].end());
  }
  ll Q;
  cin>>Q;
  for(int i=0;i<Q;i++){
    ll T;
    cin>>T;
    if(um.find(T)==um.end()){
      cout<<-1<<endl;
    }else{
      cout<<um[T][0].first<<" "<<um[T][0].second<<endl;
    }
  }
}
