#include <bits/stdc++.h>
using namespace std;
const long long MOD = 998244353;
int main(){
  int N, K;
  cin >> N >> K;
  if (K == 1){
    cout << 1 << endl;
  } else if (K == 2){
    cout << N + 1 << endl;
  } else {
    long long ans = 1;
    for (int i = 0; i < N; i++){
      ans *= 2;
      ans %= MOD;
    }
    cout << ans << endl;
  }
}
