#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef long double LD;

#define rep(i,n) for(LL i=0;i<(n);i++)


int main(){
    LL limit = 150;

    while(1){
        LL h, w; cin >> h >> w;

        if(h == 0 && w == 0){
            break;
        }

        LL diagonal = h * h + w * w;
        LL ans_h = h, ans_w = w;
        LL ans_s = limit * limit;
        vector<tuple<LL,LL,LL>> data;
        LL same_h = h, same_w = w;

        for(LL current_h = 1; current_h <= limit; current_h++){

            for(LL current_w = 1; current_w <= limit; current_w++){

                if(current_w <= current_h){
                    continue;
                }
                else if(current_h == h && current_w == w){
                    continue;
                }

                LL current_s = current_h * current_h + current_w * current_w;

                if(diagonal == current_s && same_h < current_h){
                    same_h = current_h;
                    same_w = current_w;
                }
                else if(diagonal < current_s && current_s < ans_s){
                    ans_h = current_h;
                    ans_w = current_w;
                    ans_s = current_s;
                }

            }

        }

        if(ans_h == h && ans_w == w){
            ans_h = same_h;
            ans_w = same_w;
        }
        else if(ans_h != h || ans_w != w){

            if(same_h != h){
                ans_h = same_h;
                ans_w = same_w;
            }

        }

        cout << ans_h << " " << ans_w << endl;

    }

    return 0;
}
