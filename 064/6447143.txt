#include<iostream>
#include<string>
#include<cctype>
using namespace std;

string toUpper(string s) {
  string t = s;
  for(int i = 0; i < s.size(); i++) {
    t[i] = toupper(s[i]);
  }
  return t;
}

int main(){
  string W, T;
  cin >> W;
  W = toUpper(W);
  int sum = 0;
  while(1) {
    cin >> T;
    if(T == "END_OF_TEXT") break;
    T = toUpper(T);
    if(T == W) sum++;
  }
  cout << sum << "\n";
  return 0;
}
