#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const int INF = 1001001001;
const int MOD = 1000000007;

int main(){
    int n;cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;i++) {
      cin>>a.at(i);
      cout << a.at(i);
      if(i==n-1) cout << endl;
      else cout << " ";
    }
    for(int i=1;i<n;i++) {
        int v = a.at(i);
        int j = i-1;
        while(j>=0 && a.at(j)>v) {
            a.at(j+1) = a.at(j);
            j--;
        }
        a.at(j+1) = v;
        for(int k=0;k<n;k++) {
            cout << a.at(k);
            if(k!=n-1) cout << " ";
            else cout << endl;
        }
    }
}
