#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
#include <string>
using namespace std;


int main()
{
    istream_iterator<char> isi(cin), eof;

    int count[200] = {0};

    for_each(isi, eof, 
                [&count](auto& e)
                {isupper(e) ? count[tolower(e)]++ : 
                                islower(e) ? 
                                    count[static_cast<int>(e)]++ : 0;}
                );
    for(int i = 'a'; i <= 'z'; i++)
        cout << static_cast<char>(i) << " " << ": " << count[i] << endl;

    return 0;
}
