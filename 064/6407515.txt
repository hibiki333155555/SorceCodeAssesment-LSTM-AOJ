#include <iostream>
#include <cmath>

using namespace std;

const int MAX = -1000000000;

int main(int argc, char const *argv[])
{
  int t;

  cin >> t;

  int profit = MAX;
  int min;
  cin >> min;
  for (int i = 1; i < t; i++)
  {
    int num;
    cin >> num;
    if (num - min > profit) profit = num - min;
    if (num < min) min = num;
  }
  
  cout << profit << endl;

  return 0;
}
