def is_prime(x):
  if x==2:
    return True
  if x<2 or x%2==0:
    return False
  for i in range(3, int(x**0.5)+1):
    if x%i == 0:
      return False
  else:
    return True

values = [int(input().strip()) for _ in range(int(input().strip()))]
primes = list(filter(is_prime, values))
print(len(primes))
