using namespace std;
//挺复杂 
int main()
{	
	int i,j,k,flag=0;
	string s,p;
	cin>>s;
	cin>>p;
	
	for(i=0;i<s.length();i++)
	{
		
		if(s[i]!=p[0])//基础是能找到p[0] 
			continue;
		else
		{//找到p[0]后进行逐个对比 
			k=i;
			for(j=1;j<p.length();j++)
			{
				if(k==s.length()-1)//s是圆环文本 
					k=0;
				else
					k++;
				
				if(s[k]!=p[j])//有一个不同就g 
					break;
				
			} 
			if(j==p.length())//如果完整对比修改flag 
			{
				flag=1;
				break; 
			} 
		} 
	}
	
	if(flag)
		cout<<"Yes"<<endl;
	else
		cout<<"No"<<endl;
	
	
return 0;
}
