#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> e(12);
    for(int i = 0; i < 12; i++){
        cin >> e[i];
    }
    for(int i = 0; i < 12; i++){
        for(int j = 0; j < 12; j++){
            if(e[i] > e[j]){
                swap(e[i], e[j]);
            }
        }
    }
    if(e[0] == e[3] && e[4] == e[7] && e[8] == e[11]){
        cout << "yes" << endl;
        return 0;
    }
    cout << "no" << endl;
    return 0;
}
