#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
#include <string>
using namespace std;



int main()
{
    string str;
    int n, sum, len, val;
    
    while(true)
    {
        cin >> str;
        if(str == "-")
            break;
        
        cin >> n;
        sum = 0;
        len = str.length();
        for(int i = 0; i < n; i++)
        {
            cin >> val;
            sum += val;
        }
        sum %= len;

        str = str + str;
        cout << str.substr(sum, len) << endl;
    }

    return 0;
}
