_, S = input(), [*map(int, input().split())]
_, T = input(), [*map(int, input().split())]


def binary_search(S, t):
  left, right = 0, len(S)
  while left < right:
    mid = (left + right) // 2
    if S[mid] == t:
      return True
    if S[mid] > t:
      right = mid
    else:
      left = mid + 1
  return False

print(sum(binary_search(S, t) for t in T)) #Tに含まれる整数の中でSに含まれるものの個数を調べる
