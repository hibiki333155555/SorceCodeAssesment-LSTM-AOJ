#include <stdio.h>

int main(void){

    int n;
    char syms[4] = {'S', 'H', 'C', 'D'};
    int cards[4][13];
    int i, j;
    char br, sym;
    int rank;
    for(i = 0; i < 4; i++) for(j = 0; j < 13; j++) cards[i][j] = 0;

    scanf("%d", &n);
    while(n--){
        scanf("%c%c %d", &br, &sym, &rank);
        if(sym == 'S') cards[0][rank - 1] = 1;
        if(sym == 'H') cards[1][rank - 1] = 1;
        if(sym == 'C') cards[2][rank - 1] = 1;
        if(sym == 'D') cards[3][rank - 1] = 1;
    }
    for(i = 0; i < 4; i++){
        for(j = 0; j < 13; j++){
            // printf("%c %.2d: %d\n", syms[i], j + 1, cards[i][j]);
            if(!cards[i][j]) printf("%c %d\n", syms[i], j + 1);
        }
    }

    return 0;
}
