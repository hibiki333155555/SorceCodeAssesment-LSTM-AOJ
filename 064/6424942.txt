int clc(int x, int y){
    int nl, ns;
    if(x >= y) {
        nl = x;
        ns = y;
    }else{
        nl = y;
        ns = x;
    }
    int t;
    while(nl%ns != 0){
        t=nl;
        nl=ns;
        ns=t%ns;
    }
    return ns;

}
