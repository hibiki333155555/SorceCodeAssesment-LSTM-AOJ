#include <iostream>
using namespace std;

int bubblesort(int a[],int n){
    int i,flag,j;
    int count = 0;
    flag = 1;
    for(i=0;flag;i++){
        flag = 0;
        for(j=n-1;j>=i+1;j--){
           if(a[j] < a[j-1]){
               flag = 1;
               swap(a[j],a[j-1]);
               count++;
           }
        }
    }
return count;
}

int main(){
    int i,n;
    int count = 0;
    cin >> n;
    int a[n];
    for(i=0;i<n;i++){
        cin >> a[i];
    }
    count = bubblesort(a,n);
        for(i=0;i<n;i++){
            if(i !=0) cout << " ";
        cout << a[i];
    }
    cout << endl;
    cout << count << endl;
    return 0;
}
