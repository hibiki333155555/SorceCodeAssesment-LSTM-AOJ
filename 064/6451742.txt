#include <iostream>
using namespace std;

int main()
{

    int h, w;
    
    while (true) {
		cin >> h >> w;
		if ((h | w) != 0) {
			for (int i = 0; i < h; i++) {
				for (int j = 0; j < w; j++) {
					cout << "#";
				}
				cout << endl;
			}
			cout << endl;
		} else {
			break;
		}
	}
    
    return 0;
}
