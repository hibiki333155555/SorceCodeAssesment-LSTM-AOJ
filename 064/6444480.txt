#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
    double r;
    double area, circ;
    
    cin >> r;
    
    area = M_PI * r * r;
    circ = 2 * M_PI * r;
    
    cout << fixed << setprecision(6);
    
    cout << area << " " << circ << endl;
    return 0;
}
