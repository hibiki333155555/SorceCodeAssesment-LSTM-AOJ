#include <iostream>
using namespace std;
 
int main() {
	int a,b,mm,mx;
	for(int i=1;i<=3000;i++)
	{
		cin>>a>>b;
		if(a==0 && b==0)
		{
			break;
		}
		else
		{
			if(a>b)
			{
				mm=b;
				mx=a;
			}
			else if(a<b)
			{
				mm=a;
				mx=b;
			}
			else
			{
				mm=a;
				mx=b;
			}
		}
		cout<<mm<<" "<<mx<<endl;
	}
 
 
 
}
