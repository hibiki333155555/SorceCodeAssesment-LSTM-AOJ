#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n,m; cin>>n>>m;
    vector<vector<int>> A(n,vector<int>(m,0));
    vector<int> B(m,0);
    for(int i=0;i<n;i++) for(int j=0;j<m;j++) cin>>A[i][j];
    for(int i=0;i<m;i++) cin>>B[i];
    vector<int> C(n,0);
    for(int i=0;i<n;i++) {
        for(int j=0;j<m;j++) {
            C[i] += A[i][j]*B[j];
        }
        cout<<C[i]<<endl;
    }
    return 0;
}
