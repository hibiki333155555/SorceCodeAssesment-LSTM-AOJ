#include<stdio.h>

int main(void)
{
    int n;
    scanf("%d\n",&n);
    int i = 1;
    do
    {
        int x = i;
        if ( x % 3 == 0 )
        {
            printf(" %d",i);
        }
        else
        {
            while(x)
            {
                if ( x % 10 == 3 )
                {
                    printf(" %d",i);
                }
                x /= 10;
            }
        }
    }while(++i<=n);
    printf("\n");
}
