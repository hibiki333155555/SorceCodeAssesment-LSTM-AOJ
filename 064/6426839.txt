#include <bits/stdc++.h>

using namespace std;

struct process {
  char name[10];
  int time;
};

int main() {
  int n, q;
  scanf("%d %d", &n, &q);
  queue<process> proc;
  for (int i = 0; i < n; i++) {
    process tmp;
    scanf("%s", &tmp.name);
    scanf("%d", &tmp.time);
    proc.push(tmp);
  }
  int timer = 0;
  while (!proc.empty()) {
    process check = proc.front();
    proc.pop();
    if (check.time <= q) {
      timer += check.time;
      cout << check.name << ' ' << timer << endl;
      continue;
    }

    check.time -= q;
    timer += q;
    proc.push(check);
  }
}
