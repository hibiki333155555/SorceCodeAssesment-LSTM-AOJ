#include<stdio.h>

int main (void)
{
    char a[201],b[201];
    int m,h,i,x,y,z;
    while(1)
    {
        scanf("%s %d",a,&m);
        if(a[0]=='-')break;
        for(i=0;i<m;i++)
        {
            scanf("%d",&h);
            for(x=h;a[x]!='\0';x++)
            {
                b[x-h]=a[x];
            }
            for(y=0;y<h;y++)
            {
                b[x-h+y]=a[y];
            }
            for(z=0;b[z]!='\0';z++)
            {
                a[z]=b[z];
            }
        }
        printf("%s",a);
    }
    return 0;
}
