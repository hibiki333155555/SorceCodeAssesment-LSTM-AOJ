#include<iostream>
#include<cstdio>
#include<cctype>
using namespace std;

int main(){
  char str;
  while(1) {
    scanf("%c", &str);
    if(str == '\n') break;
    if(isalpha(str)) {
      if(islower(str)) printf("%c", toupper(str));
      else printf("%c", tolower(str));
    } else cout << str;
  }
  cout << endl;
  return 0;
}
