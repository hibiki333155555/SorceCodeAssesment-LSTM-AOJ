#include<stdio.h>
#include<math.h>

int main(void)
{
    int n,i;
    double d=0;
    int a[101],b[101];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        scanf("%d",&b[i]);
    }
    for(i=0;i<n;i++)
    {
        d+=fabs(a[i]-b[i]);
    }
    printf("%lf",d);
    d=0;
    for(i=0;i<n;i++)
    {
        d+=pow(fabs(a[i]-b[i]),2.0);   
    }
    printf("%lf",sqrt(d));
    d=0;
    for(i=0;i<n;i++)
    {
       d+=pow(fabs(a[i]-b[i]),3.0);    
    }
    printf("%lf",pow(d,1.0/3.0));
    d=a[1]-b[1];
    for(i=1;i<n;i++)
    {
        if(d<a[i]-b[i])
        {
            d=a[i]-b[i];
        }
    }
    printf("%d",d);
    return 0;
}
