void swap(int *x,int *y){
  int temp;
  temp = *x;
  *x = *y;
  *y = temp;
}

void bubble(int *A,int n){
  int flag = 1;
  int j,i = 0;
  
  while(flag){
    flag = 0;
    for(j=n-1 ;j>i ;j--){
      if(A[j-1] > A[j]) {
	swap(&A[j-1],&A[j]);
	count++;
	flag = 1;
      }
    }
  }
  for(i=0 ;i<n ;i++){
    if(i) printf(" ");
    printf("%d",A[i]);
  }

  printf("\n");
  printf("%d\n",count);
}

  int i,A[N];

  for(i=0 ;i<n ;i++) scanf("%d",&A[i]);

  bubble(A,n);

}
