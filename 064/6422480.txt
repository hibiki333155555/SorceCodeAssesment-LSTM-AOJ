#include<bits/stdc++.h>
using namespace std;

bool cek(int x){
	bool prime = true;
	for(int i=2; i*i<=x; i++){
		if(x%i == 0){
			prime = false;
			break;
		}	
	}
	if(x == 1){
		return false;
	} else if(x == 2){
		return true;
	} else if(prime == true){
		return true;
	} else if(prime == false){
		return false;
	}
}
int n; 

int main(){
	while(cin >> n && n!=0){
		for(int i=n; i>=1; i--){
			if(cek(n) && cek(n-2)){
				cout << n-2 << " " << n << endl;
				break;
			} else {
				n--;
			}
		}
	}
	
}
