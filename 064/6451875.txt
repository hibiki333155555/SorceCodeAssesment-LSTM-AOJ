#include<bits/stdc++.h>

using namespace std;

int main()
{
    while(1)
    {
        int n,x,cnt=0;
        cin>>n>>x;
        if(n==0 && x==0) break;
        int a[n+5]={0};
        for(int i=1;i<n;i++)
        {
            for(int j=i+1;j<=n;j++)
            {
                if(x-i-j<n && x-i-j>0 && (a[i]==0 || a[j]==0 || a[x-i-j]==0) && i!=x-i-j && j!=x-i-j)
                {
                    cnt++;
                    a[i]=1;
                    a[j]=1;
                    a[x-i-j]=1;
                }
            }
        }
        cout<<cnt<<endl;
    }
    return 0;
}
