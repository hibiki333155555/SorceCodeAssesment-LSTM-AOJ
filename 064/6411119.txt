#include <iostream>
#include <vector>
using namespace std;

int gcd(int a, int b)
{
    if(a % b == 0){
        return b;
    }else{
        return gcd(b, a % b);
    }
}

int main()
{
    int a, b;
    cin >> a >> b;
    
    if (a < b) std::swap(a,b);
    
    int G = gcd(a, b);
    int A = a / G;
    int B = b / G;
    
    int sol_x = 1;
    int sol_y = 0;
    if (B == 1)
    {
        sol_x = 0;
        sol_y = 1;
    }
    else
    {
        for (; sol_x < 1000; sol_x++)
            if ((A * sol_x) % B == 1)
                break;
        sol_y = -1 * (A * sol_x) / B;
    }    
    
    int x = sol_x;
    int y = sol_y;
    
    int dist = abs(x) + abs(y);

    while (1)
    {
        int candX = x - B;
        int candY = y + A;
        int d = abs(candX) + abs(candY);
        if (dist < d)
        {   
            break;
        }
        else if (dist == d)
        {
            dist = d;
            if (candX >= candY)
            {
                x = candX;
                y = candY;
            }
        }
        else
        {
            x = candX;
            y = candY;
            dist = d;
        }
    }
    while (1)
    {
        int candX = x + B;
        int candY = y - A;
        int d = abs(candX) + abs(candY);
        if (dist < d)
        {
            break;
        }
        else if (dist == d)
        {
            dist = d;
            if (candX >= candY)
            {
                x = candX;
                y = candY;
            }
        }
        else
        {
            x = candX;
            y = candY;
            dist = d;
        }

    }

    if (a > b)
        cout << y << " " << x << endl;
    else
        cout << x << " " << y << endl;

    return 0;
}
