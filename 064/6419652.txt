#include <bits/stdc++.h>

int main()
{
    while ( true )
    {
        std::string str;
        std::cin >> str;
        if (str == "-")
            return 0;
        
        int m;
        std::cin >> m;
        
        std::vector<int> h(m);
        for (int i = 0;i < h.size();++i) {
            std::cin >> h.at(i);
        }
        
        for (int i = 0;i < h.size();++i) {
            std::string first( str.begin(), str.begin()+h.at(i) );
            std::string second( str.begin()+h.at(i), str.end() );
            
            str = second + first;
        }
        
        std::cout << str << std::endl;
    }
}
