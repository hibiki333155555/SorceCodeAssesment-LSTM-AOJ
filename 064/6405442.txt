#include<bits/stdc++.h>

using namespace std;

int main() {
	int q;
	cin >> q;
	vector<int> v;

	for (int i = 0; i < q; i++) {
		int n;
		cin >> n;

		if (n == 0) {
			// pushback x
			int x;
			cin >> x;
			v.push_back(x);
		}
		else if(n == 1) {
			int p;
			cin >> p;

			cout << v.at(p) << '\n';
		}
		else {
			v.pop_back();
		}
	}
}
