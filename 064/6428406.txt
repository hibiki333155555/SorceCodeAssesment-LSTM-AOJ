#include<stdio.h>
int main()
{
    int H,W;

   while(1)
    {
        scanf("%d %d",&H,&W);
        if(H==0 && W==0)
            break;
        for(int y=1; y<=H; y++)
        {
            for(int z=1; z<=W; z++)
                {
                    printf("#");
                }
                printf("\n"); //2004061
        }
        printf("\n");
    }

    return 0;
}
