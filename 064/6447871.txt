#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int main() {
   while(true) {
      long x = 0;
      cin >> x;
      if(x == 0) break;

      long sum = 0;
      while(x) {
         sum += x % 10;
         x /= 10;
      }

      cout << sum << "\n";
   }   
}
