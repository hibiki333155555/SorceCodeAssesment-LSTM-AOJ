#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,k;cin>>n>>k;
    int a[n];
    for(int i=0;i<n;i++) cin>>a[i];
    
    deque<int> mindq;
    
    // loop 1 to k
    for(int i=0;i<k;i++)
    {
        
        while(!mindq.empty() && a[mindq.back()]>a[i])
        {
            mindq.pop_back();
        }
        
        mindq.push_back(i);
    }
    
    
    //loop k to n
    for(int i=k;i<n;i++)
    {
        cout<<a[mindq.front()]<<" ";
        
        while(!mindq.empty() && mindq.front()<=i-k  )
        {
            mindq.pop_front();
        }
       
        while(!mindq.empty() && a[mindq.back()]>a[i])
        {
            mindq.pop_back();
        }
        
        mindq.push_back(i);
    }
    cout<<a[mindq.front()]<<"\n";
}
