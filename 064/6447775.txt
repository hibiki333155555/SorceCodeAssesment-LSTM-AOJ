import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner s=new Scanner(System.in);
        int a=s.nextInt();
        int b=s.nextInt();
        int d=a/b;
        int r=a%b;
        double p = 1.0*a/b;
        String f=String.format("%.8f",p);
        System.out.println(d+" "+r+" "+f);
    }
}
