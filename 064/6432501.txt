#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
typedef long long Int;
typedef long long ll;
const int Inf = 1e9 + 7;
const Int INF = (Int)1e18 + 7;
const double EPS = 1e-10;

#define chmax(x, y) x = max(x, y)

int main() {
  int N, W;
  cin >> N >> W;
  vector<int> v(N), w(N);
  for (int i = 0; i < N; i++) cin >> v[i] >> w[i];
  vector<int> dp(W+1, 0);
  for (int j = 0; j < W; j++)
    for (int i = 0; i < N; i++)
      if (j+w[i] <= W) chmax(dp[j+w[i]], dp[j]+v[i]);
  cout << dp[W] << endl;
  return 0;
}
