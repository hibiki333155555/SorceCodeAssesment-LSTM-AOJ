#include <stdio.h>
#include <string.h>

int main(){
    int q,a,b,i,j,k=0;
    char str[1001],op[10],p[1000],t;
    
    scanf("%s",str);
    scanf("%d",&q);
    
    for(i=0;i<q;i++){
        scanf("%s %d %d",op,&a,&b);
        
        if(strcmp(op,"print")==0){
            for(j=a;j<=b;j++){
                printf("%c",str[j]);
            }
            printf("\n");
        }
        if(strcmp(op,"reverse")==0){
            for(j=a;j<=b;j++){
                t=str[a];
                str[a]=str[b];
                str[b]=t;
            }
        }
        if(strcmp(op,"replace")==0){
            scanf("%s",p);
            for(j=a;j<=b;j++){
                str[j]=p[k];
                k++;
            }
        }
    }
}
