#include<stdio.h>
#include<stdbool.h>
#define MAX 100
#define INF 100100100

int n;
int G[MAX][MAX], d[MAX], p[MAX];
bool visited[MAX];

void dijkstra(){
  for(int i = 0; i < n; i++ ){
    visited[i] = false;
    d[i] = INF;
  }
  p[0] = -1;
  d[0] = 0;
  int mincost, u;
  while(1){
    mincost = INF;
    for(int i = 0; i < n; i++ ){
      if(!visited[i] && d[i] < mincost){
        mincost = d[i];
        u = i;
      }
    }

    if(mincost == INF) break;

    visited[u] = true;

    for(int i = 0; i < n; i++ ){
      if(G[u][i] < INF && d[u] + G[u][i] < d[i] && !visited[i]){
        d[i] = d[u] + G[u][i];
        p[i] = u;
      }
    }
  }
} 

int main()
{
  int u, k, v, c;
  scanf("%d", &n);
  for(int i = 0; i < n; i++ ){
    for(int j = 0; j < n; j++ ){
      G[i][j] = INF;
    }
  }

  for(int i = 0; i < n; i++ ){
    scanf("%d %d", &u, &k);
    for(int j = 0; j < k; j++ ){
      scanf("%d %d", &v, &c);
      G[u][v] = c;
    }
  }

  dijkstra();

  for(int i = 0; i < n; i++ ){
    printf("%d %d\n", i, d[i]);
  }

  return 0;
}
