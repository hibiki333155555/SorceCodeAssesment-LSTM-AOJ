#include <iostream>
using namespace std;

int main() {
    int n, x;
    while (cin >> n >> x) 
    {
        int ans = 0;
        if (n == 0 and x == 0) break;
        for (int i=1; i<=n; i++) {
            for (int j=1; j<=n; j++) {
                if (j <= i) {
                    continue;
                }
                else {
                    int c = x-(i+j);
                    if (c>j and c<=n) {
                        ans++;
                    }
                }
            }
        }
        cout << ans << endl;
        ans = 0;
    }
    return 0;
}
