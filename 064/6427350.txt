import math
while True:
    n = int(input())
    if n == 0: break
    s_list = list(map(int, input().split()))
    mean = sum(s_list) / n
    dev_list = [s - mean for s in s_list]
    std = math.sqrt(sum([dev ** 2 for dev in dev_list]) / n)
    print(f"{std:.8f}")
