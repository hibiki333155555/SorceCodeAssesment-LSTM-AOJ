///operators overloading
void operator+=(pii& a,const pii&b){
    a.fi+=b.fi;
    a.se+=b.se;
}
pii operator+(const pii& a,const pii&b){
    return {a.fi+b.fi,a.se+b.se};
}

void operator-=(pii& a,const pii&b){
    a.fi-=b.fi;
    a.se-=b.se;
}
pii operator-(const pii& a,const pii&b){
    return {a.fi-b.fi,a.se-b.se};
}

//https://judge.u-aizu.ac.jp/onlinejudge/description.jsp?id=DSL_1_A

int n;

int find(int a,vi & p,vi & sz){
    for(;p[a]!=a;a=p[a]=p[p[a]]);
    return a;
}

void unite(int a,int b,vi & p,vi & sz){
    a = find(a,p,sz);
    b = find(b,p,sz);
    //enssure sz[a] >= sz[b]
    if(sz[a]<sz[b])
        swap(a,b);
    p[b]=a;
    sz[a]+=sz[b];
}

int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);
    int q;
    vi p,sz;
    cin >> n >> q;
    for(int i = 0 ; i < n ; i ++){
        p.pb(i);
        sz.pb(1);
    }
    int c,x,y;
    while(q--){
        cin >> c >> x >> y;
        if(c)
            cout << (find(x,p,sz)==find(y,p,sz)) << '\n';
        else
            unite(x,y,p,sz);
    }
    return 0;
}
