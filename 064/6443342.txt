main(require("fs").readFileSync("/dev/stdin", "utf8"));

/*-- 入力処理&出力 --*/
function main(raw) {
    // 入力処理
    const args = raw.split("\n");
    const input = args[0].split(' ').map(Number);
    // 計算
    const h = Math.floor(input/(60*60));
    const m = Math.floor((input-h*60*60)/60);
    const s = input-h*60*60-m*60;
    // 表示
    console.log(`${h}:${m}:${s}`);
}
