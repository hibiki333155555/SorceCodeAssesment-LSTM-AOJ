template<typename T> ostream& operator<<(ostream& os,const vector<T>& v1){
    int n=v1.size();
    for(int i=0;i<n;i++){
        if(i)os<<" ";
        os<<v1[i];
    }
    return os;
}
template<typename T1,typename T2> ostream& operator<<(ostream& os,const pair<T1,T2>& p){
    os<<p.first<<" "<<p.second;
    return os;
}
template<typename T> istream& operator>>(istream& is,vector<T>& v1){
    int n=v1.size();

    return is;
}

}

namespace sol{
    vector<vector<char>> v1;
    
    void rotate(){
        int n,m;
        n=v1.size(),m=v1[0].size();
        vector<vector<char>> va(m,vector<char>(n));
        int i,j;
        for(i=0;i<n;i++){
            for(j=0;j<m;j++){
                va[m-j-1][i]=v1[i][j];
            }
        }
        v1=va;
    }

    void solve(){
        int n,m;
        int i,j,k;
        int a,b,c;
        cin>>n>>m;
        string sa;
        v1.assign(n,vector<char>(m));
        for(i=0;i<n;i++){
            cin>>sa;
            for(j=0;j<m;j++){
                if(sa[j]=='o')v1[i][j]=1;
            }
        }
        int s=n*m;
        for(i=0;i<4;i++){
            rotate();
            n=v1.size(),m=v1[0].size();
            a=0;
            for(j=0;j<n;j++){
                for(k=0;k<m;k++){
                    if(v1[j][k])a++;
                }
            }
            b=0;
            for(j=0;j<n;j++){
                if(j)s=min(s,m*j-b+a-b);
                for(k=0;k<m;k++){
                    if(v1[j][k])b++;
                }
            }
        }
        cout<<s<<endl;
    }
}

int main(){
    cin.tie(0);
    ios::sync_with_stdio(false);
    sol::solve();
}
