#include <bits/stdc++.h>
using namespace std;
int main() {


    int t[4][13];
    for (int i=0; i<4; i++) {
        for (int j=0; j<13; j++) {
            t[i][j] = 0;
        }
    }


    int n;
    cin >> n;
    for (int i=0; i<n; i++) {
        char ch;
        int num;
        cin >> ch >> num;
        if (ch == 'S') {
            t[0][num - 1]++;
        } else if (ch == 'H') {
            t[1][num - 1]++;
        } else if (ch == 'C') {
            t[2][num - 1]++;
        } else if (ch == 'D') {
            t[3][num - 1]++;
        }
    }

    for (int i=0; i<4; i++) {
        for (int j=0; j<13; j++) {
            if (t[i][j] == 0) {
                if (i == 0) {
                    cout << "S " << j + 1 << endl;
                } else if (i == 1) {
                    cout << "H " << j + 1 << endl;
                } else if (i == 2) {
                    cout << "C " << j + 1 << endl;
                } else if (i == 3) {
                    cout << "D " << j + 1 << endl;
                }
            }
        }
    }
    return 0;
}
