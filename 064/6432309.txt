n,m,l=map(int,input().split())
A=[]
B=[]
for i in range(n):
    A.append(list(map(int,input().split())))
for i in range(m):
    B.append(list(map(int,input().split())))
for i in range(n):
    for j in range(l):
        sum_=0
        for k in range(m):
            sum_+=A[i][k]*B[k][j]
        print(f'{sum_}',end=' ')
    print('\n')
