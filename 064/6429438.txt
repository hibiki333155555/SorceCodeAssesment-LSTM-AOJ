
#include<stdio.h>

void trace(int A[], int N) {
	int i;
	for (i = 0; i < N; i++) {
		if (i>0)
		printf(" ");
		printf("%d", A[i]);
	}
	printf("\n");

}
void insertsort(int A[],int N) {
	int i, j,v;
	for (int i = 1; i < N; i++)
	{
		 v = A[i];
		 j = i - 1;
		 while (j>=0&&A[j]>v)
		 {
			 A[j + 1] = A[j];
			 j--;
		 }
		 A[j + 1] = v;
		 trace(A, N);
	}


}


int main() {
	int n ;
	int a[110];
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	trace(a, n);
	insertsort(a, n);


  

}
