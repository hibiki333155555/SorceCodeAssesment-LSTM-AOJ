#include<stdio.h>
#include<string.h>

int main(){
    char n[1201],a[26];
    fgets(n,1201,stdin);
    int i,j=0,l=strlen(n);
    for (i=0;i<26;i++){
        a[i]=0;
    }
    for (i=0;i<l;i++){
        if(n[i]>='a' && n[i]<='z'){
            a[(int)n[i]-97]+=1;
        }
        else if(n[i]>='A' && n[i]<='Z'){
            a[(int)n[i]-65]+=1;
        }
    }
    for (i=0;i<26;i++){
        printf("%c : %d\n",i+97,a[i]);
    }

    return 0;
}
