#include <iostream>
#include <vector>

using namespace std;

void _dump(vector<int> v){
    for(int i = 0; i < v.size(); i++){
        cout << v.at(i) << (( i != v.size() -1 ) ? " " : "\n");
    }
}

int main(){
    int n, q;
    cin>>n>>q;
    vector<vector<int>> v(n, vector<int>(0));
    
    int op, t, x;
    for(int i = 0; i < q; i++){
        cin>>op>>t;
        switch(op){
            case 0:
            cin>>x;
            v.at(t).push_back(x);
            break;
            case 1:
            _dump(v.at(t));
            break;
            case 2:
            v.at(t).clear();
        }
    }
    return 0;
}
