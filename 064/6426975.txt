#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[4][3][10] = {{{}}};
    for(int x = 0;x < n; x++)
    {
        int b,f,r,v;
        cin>>b>>f>>r>>v;
        arr[b-1][f-1][r-1] += v;
    }
    for(int i = 0;i < 4;i++)
    {
        for(int j = 0 ;j < 3;j++)
        {
            for(int k = 0;k < 10;k++)
            {
                cout<<" "<<arr[i][j][k];
            }
            cout<<endl;
        }
        cout<<"####################"<<endl;
    }


    return 0;
}
