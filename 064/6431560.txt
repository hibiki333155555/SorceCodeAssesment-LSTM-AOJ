void insertion_sort(vector<int>v){
    
    int n=v.size();

    for(int i=1;i<n;i++){
        
        int temp=v[i];
        int j=i-1;
        while(j>=0 && temp<v[j]){
            v[j+1]=v[j];
                j--;
        }
        
        v[j+1]=temp;
        // cout<<j+1<<" ";
		for(int i=0;i<n;i++){
    	   cout<<v[i];
    	   if(i!=n-1)
    	   cout<<" ";
		}
    cout<<"\n";
      
    }
    // for(int i=0;i<n;i++)
    //     cout<<v[i]<<" ";
    cout<<"\n";
}
