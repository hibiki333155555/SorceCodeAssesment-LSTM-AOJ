typedef long long ll;

int main(void){

	int i;

	cin.tie(0);  ios_base::sync_with_stdio(false);

	int A,B;
	cin >> A >> B;

	vector<int> v,u;
	int n=0,m=0;
	while(1){
		v.push_back(A%2);
		A/=2;
		n++;
		if(A==0) break;
	}

	while(1){
		u.push_back(B%2);
		B/=2;
		m++;
		if(B==0) break;
	}

	if(n>m) for(int i=0;i<n-m;i++) u.push_back(0);
	if(m>n) for(int i=0;i<m-n;i++) v.push_back(0);

	reverse(v.begin(),v.end());
	reverse(u.begin(),u.end());

	int ans=0;
	n=max(n,m);
	for(i=0;i<n;i++){
		if(v[i]==u[i]) ans++;
		else break;
	}

	cout << ans << endl;

	return 0;
}
