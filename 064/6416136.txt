#include<stdio.h>
#include<string.h>

int main(){
    
    int h,s,i,j,k;
    char target[100],sub1[100],sub2[100];
    
    while(1){
        scanf("%s", &target);
        if(target == '-')break;
        s=strlen(target);
    
        scanf("%d", &h);
        int m[h];
    
        for(i=0;i<h;i++){
            scanf("%d", &m[i]);
            for(j=0;j<m[i];j++){
                sub1[j]=target[j];
            }
        
            for(j=0,k=m[i];k<s;j++,k++){
                sub2[j]=target[k];
            }
        }
        
        printf("%s\n", strstr(sub1,sub2));
    }
    
    return 0;
    
}
