#include <cstdio>

int main(void)
{
  int n, x;
  scanf("%d", &n);

  for (int i = 1; i <= n; i = i + 1) {
    x = i;
    if (x % 3 == 0) {
      printf(" %d", i);
      continue;
    }
    for (; x != 0; x = x / 10) {
      if (x % 10 == 3) {
        printf(" %d", i);
        break;
      }
    }
  }
  printf("\n");
  return 0;
}
