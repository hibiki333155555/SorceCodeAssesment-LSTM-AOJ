#pragma GCC optimize("O3")
// #pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    string s;
    cin >> s;
    stack<int> st;
    st.push(0);
    string t = "";
    for (auto& c : s) {
        if ('0' <= c && c <= '9') {
            int x = st.top();
            int y = c - '0';
            st.push((x + y) % 26);
        }
        else if (c == '(') {
            // pass
        }
        else if (c == ')') {
            st.pop();
        }
        else {
            t += (char)((c - 'a' + st.top()) % 26 + 'a');
        }
    }
    cout << t << '\n';
    return 0;
}
