bool is_valid(int ig, int jg, int kg) {
  bool flg1 = (ig + jg + kg) == 0;
  bool flg2 = abs(ig) % 2 == 0;
  bool flg3 = abs(jg) % 2 == 0;
  bool flg4 = abs(kg) % 2 == 0;
  bool flg5 = (ig + jg + kg) % 2 == 0;

  return flg1 and flg2 and flg3 and flg4 and flg5;
}


int main() {
  int is, js, ks;
  cin >> is >> js >> ks;

  int ig, jg, kg;
  cin >> ig >> jg >> kg;

  ig -= is;
  jg -= js;
  kg -= ks;

  is = js = ks = 0;

  if (is_valid(ig, jg, kg)) {
    cout << "Yes" << newl;
    return 0;
  }

  ig -= 1;
  jg += 1;
  kg -= 1;

  if (is_valid(ig, jg, kg)) {
    cout << "Yes" << newl;
    return 0;
  }

  cout << "No" << endl;
}
