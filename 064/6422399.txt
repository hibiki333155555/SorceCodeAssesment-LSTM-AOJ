#include <bits/stdc++.h>
using namespace std;

int main(){
  double a,b,c,d,S,L,h;
  cin >> a >> b >> d;
  d = 2*M_PI*d/360;
  S = a*b*sin(d)/2;
  L = a+b+sqrt( a*a+b*b-2*a*b*cos(d) );
  h = 2*S/a;
  cout << fixed << setprecision(15);
  cout << S << endl;
  cout << L << endl;
  cout << h << endl;
    
}
