#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<ll, ll> pr;
typedef pair<pr, ll> pr2;

const ll Max_V = 1000;
const ll INF = 2e9;

vector<ll> d(Max_V);



int main(void){
    ll V, E, r; cin >> V >> E >> r;
    vector<pr2> ed(E);
    for(ll i = 0; i < E; i++){
        ll s, t, d; cin >> s >> t >> d;
        ed[i] = {{s, t}, d};
    }
    for(ll i = 0; i < V; i++){
        d[i] = INF;
    }
    d[r] = 0;
    bool flag = false;
    for(ll i = 0; i < V; i++){
        for(ll j = 0; j < E; j++){
            pr e = ed[j].first;
            if(d[e.second] > d[e.first] + ed[j].second){
                d[e.second] = d[e.first] + ed[j].second;
                if(i == V - 1) flag = true;
            }
        }
    }
    if(flag){
        cout << "NEGATIVE CYCLE" << endl;
    } else {
        for(ll i = 0; i < V; i++){
            if(d[i] > 1e9){
                cout << "INF" << endl;
            } else {
                cout << d[i] << endl;
            }
        }
    }
}
