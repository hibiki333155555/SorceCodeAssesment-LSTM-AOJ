#include <bits/stdc++.h>
using namespace std;

int main(){
    string num;
    int m;
    cin>>num;
    while(1){
        if(num=="-") break;
        cin>>m;
        int vec[m];
        for(int i=0; i<m; i++){
            cin>> vec[i];
           num=num.substr(vec[i])+num.substr(0,vec[i]);
        }
        cout<< num<<endl;
        cin>>num;
    }           
}
