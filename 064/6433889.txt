#include<stdio.h>
int da[12];
void sort(){
    int t;
    for(int d=1;d<12;d++){
        for(int i=0;i<12-d;i++){
            if(da[i]>da[i+1]){
                t=da[i];
                da[i]=da[i+1];
                da[i+1]=t;
            };
        }
    }
}
int main(void){
    for(int i=0;i<12;i++) scanf("%d",&da[i]);
    sort();
    int j=1;
    for(int i=0;i<12;i+=4){
        if(da[i]!=da[i+3]) j=0;
    }
    printf(j?"yes\n":"no\n");
    return 0;
}
