#include<iostream>
#include<vector>
#include<string>
using namespace std;
int main(){
    int n,m;
    cin>>n;
    vector<int> data;
    for(int i=0;i<n;i++){
        int temp;
        cin>>temp;
        data.push_back(temp);
    }
    cin>>m;
    vector<int> test;
    for(int i=0;i<m;i++){
        int temp;
        cin>>temp;
        test.push_back(temp);
    }
    vector<int> dp(2001,0);
    for(int i=0;i<data.size();i++){
        dp[data[i]]=1;
    }
    dp[0]=1;
    for(int i=1;i<2001;i++){
        for(int j=1;j<i/2+1;j++){
            if(j==i-j)continue;
            dp[i]=dp[i]||(dp[j]&&dp[i-j]);
        }
    }
    for(int i=0;i<test.size();i++){
        if(dp[test[i]]==1){
            cout<<"yes"<<endl;
        }else{
            cout<<"no"<<endl;
        }
    }
    return 0;
}
