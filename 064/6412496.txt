#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int N, Q; cin >> N >> Q;
	vector<int> a(N);
	for (auto& an : a) cin >> an;
	for (int j = 0; j < Q; ++j) {
		int_fast64_t S = 0, A = 0, x;
		cin >> x;
		for (int i = 0, k = 0; i < N; ++i) {
			S += a[i];
			while (S > x) S -= a[k++];
			A += i - k + 1;
		}
		cout << A << endl;
	}
	return 0;
}
