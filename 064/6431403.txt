#include<bits/stdc++.h>
using namespace std;


void Bubble_Sort(vector<int> &arr)
{
    int count=0;
    int n=arr.size();
    for(int i=0;i<n-1;i++)
    {
        for(int j=0;j<n-1-i;j++)
        {
            if(arr[j]>arr[j+1])
            {
                count++;
            }
            
        }
        cout<<count<<"\n";
    }
}

int main(){
    int a;
    cin>>a;
    while(a--)
    {
    int n;
    cin>>n;
    vector<int>v(n);
    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }
     Bubble_Sort(v);
    }
    return 0;
}
