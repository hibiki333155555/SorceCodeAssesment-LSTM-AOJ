#include <iostream>
#include <cstring>
using namespace std;
#define NIL -1
#define MAX 1000
int c[MAX][MAX];

int lcs(char X[],int m,char Y[],int n){
    int i=m,j=n;
    if(i==0||j==0) return 0;
    else if(X[i]==Y[j]) return c[i][j]=lcs(X,i-1,Y,j-1)+1;
    else return max(lcs(X,i,Y,j-1),lcs(X,i-1,Y,j-1));
}
    
int main(void){
    // Your code here
    int n;
    cin>>n;
    char X[MAX],Y[MAX];
    for(int i=0;i<n;i++){
        cin>>X;
        cin>>Y;
        cout<<lcs(X,strlen(X),Y,strlen(Y))<<endl;
    }
}
