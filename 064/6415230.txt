ll MODpow(ll x, ll n, ll mod){
    ll res=1;
    while(n>0){
        if(n&1) res=res*x%mod;
        x=x*x%mod;
        n >>= 1;
    }
    return res;
}

ll POW(ll x, ll n){
    ll res=1;
    while(n>0){
        if(n&1) res=res*x;
        x=x*x;
        n >>= 1;
    }
    return res;
}

ll MODcomb(ll n, ll r, ll mod, vl& fact){
    ll res=1;
    if(n>=r){
        res=fact[n]%mod;
        res=res*MODpow(fact[r], mod-2, mod)%mod;
        res=res*MODpow(fact[n-r], mod-2, mod)%mod;
    }
    return res;
}
