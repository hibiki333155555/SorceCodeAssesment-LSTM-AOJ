import java.util.Scanner;

class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int W,H;
        for(;;){
            H= sc.nextInt();
            W= sc.nextInt();
            if(H==0)break;
            for(int i=0;i<W;i++){
                for(int ii=0;ii<H;ii++){
                System.out.println("#");
                }
                System.out.println();
            }
        }
    }
}
