#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;

    int A[n+1], L[n];
    for (int i = 0; i < n; ++i) cin >> A[i];

    L[0] = A[0];
    int length = 1;

    for (int i = 1; i < n; ++i) {
        // Lの末尾より大きい値ならばその後ろに追加する
        if (L[length-1] < A[i])
            L[length++] = A[i];
        // 値の昇順を保ちながらより小さい値でLが構成されるように置き換えていく
        else
            *lower_bound(L, L + length, A[i]) = A[i];
    }

    cout << length << endl;
}
