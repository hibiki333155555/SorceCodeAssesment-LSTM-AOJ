#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int r,c;
	cin>>r>>c;
	int a[r][c];
	for(int i=0;i<r;i++)
	{
		for(int j=0;j<c;j++)
		{
		cin>>a[i][j];
		}
	}
	int colsum=0;
	for(int i=0;i<r;i++)
	{
		int sum=0;
		for(int j=0;j<c;j++)
		{
			sum=sum+a[i][j];
			colsum=colsum+a[i][j];
		
			cout<<a[i][j]<<" ";
		
		}
		cout<<sum<<"\n";
	}
	for(int i=0;i<c;i++)
	{
		int csum=0;
		for(int j=0;j<r;j++)
		{
		   csum=csum+a[j][i];
		}
			cout<<csum<<" ";
	}
	cout<<colsum;
	cout<<"\n";
}
