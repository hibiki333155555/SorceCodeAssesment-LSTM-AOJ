#include <stdio.h>
#define N 100000

// cppでvectorを使って正答できたが、vectorを使わずにできると思ってやった
//　けどなぜかAOJではruntime errorになる

int id_list[N];
int G[N][80000];
int count[N];

void dfs(int from, int id) {
  id_list[from] = id;
  for (int i = 0; i < count[from]; i++) {
    int next_v = G[from][i];
    if (id_list[next_v] > 0) {
      continue;
    }
    dfs(next_v, id);
  }
}
void print(int x, int y) {
  if (id_list[x] == id_list[y]) {
    printf("yes\n");
  } else {
    printf("no\n");
  }
}

int main(void) {
  int n, m;
  scanf("%d %d", &n, &m);

  int a, b;
  for (int i = 0; i < m; i++) {
    scanf("%d %d", &a, &b);
    G[a][count[a]++] = b;
    G[b][count[b]++] = a;
  }
  int id = 0;
  for (int i = 0; i < n; i++) {
    if (id_list[i] > 0)
      continue;
    dfs(i, ++id);
  }
  int q;
  scanf("%d", &q);
  int x, y;
  for (int i = 0; i < q; i++) {
    scanf("%d %d", &x, &y);
    print(x, y);
  }
}
