#include <bits/stdc++.h>
using namespace std;

int main(){
  int n,q;
  cin >> n >> q;
  vector<stack<int>> v(n);
  
  for(int i=0; i<q; i++){
    int a,t;
    cin >> a >> t;
    if(a==0){
      int x;
      cin >> x;
      v.at(t).push(x);
    }else if(a==1){
      if(v.at(t).empty()) continue;
      cout << v.at(t).top() << endl;      
    }else{
      if(v.at(t).empty()) continue;
      v.at(t).pop();
    }
  }
  
}
