#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int partition(vector<int>& nums, int left, int right) {
	int x = nums[right];
	int i = left - 1;
	int j = left;
	for (j; j < right; j++) {
		if (nums[j] <= x) {
			i++;
			swap(nums[i], nums[j]);
		}
	}
	swap(nums[i+1] , nums[right]);
	return x;
}
int main() {
	int n;
	cin >> n;
	vector<int> nums(n);
	for (int i = 0; i < n; i++) {
		cin >> nums[i];
	}
	int x=partition(nums,0,n-1);
	cout << nums[0];
	for (int i = 1; i < n; i++) {
		if (nums[i] == x) {
			cout <<" " << "[" << x << "]";
		}
		else {
			cout <<" " << nums[i];
		}
	}
	return 0;
}
