#include <iostream>

using namespace std;

int main()
{
    int i = 1;
    int x;
    int n;
    cin >> n;

    for (i; i <= n; i++) {

        x = i;

        if (x % 3 == 0) {
            cout << " " << i;
            continue;
        }

        do {
            if (x % 10 == 3) {
                cout << " " << i;
                break;
            }
            x /= 10;
        } while (x);
    }
}
