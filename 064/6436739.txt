#include <bits/stdc++.h>
#define lol long long
using namespace std;
int main(){
    int h,w;
    bool flag = true;
    while(true){
        cin>>h>>w;
        if(h==0&&w==0){
            break;
        }
        for(int i=0;i<h;i++){
            for(int j=0;j<w;j++){
                if(flag){
                    cout<<"#";
                }else{
                    cout<<".";
                }
                flag = !flag;
            }
            cout<<"\n";
        }
        cout<<"\n";
    }
    return 0;
}
