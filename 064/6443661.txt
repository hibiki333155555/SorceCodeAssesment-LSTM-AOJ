#include <stdio.h>

int main(void) { 
	int W,H,x,y,r;
	scanf("%d %d %d %d %d\n",&W,&H,&x,&y,&r);
	if(x>=-100 && y<=100 && 0<W && W<=100 && H>0 && H<=100 && r<=100 && r>0){
	    if(W>x && H>y && x>=r && x<=W-r && r<=y && y<=H-r && r<=H/2 && r<=W/2){
	        printf("Yes\n");
	    }
	    else
	    printf("No\n");
	}
	return 0;
}
