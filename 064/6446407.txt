#include <stdio.h>

int main(void){
	int a[10000];
	int i,j;
	int n;
	int min,max;
	long sum;

	scanf("%d", &n);

	for(i=0;i<n;i++){
		scanf("%d", &a[i]);
	}

	min = a[0];
	max = a[0];
	sum = a[0];
	for(i=1;i<n;i++){
		if( min > a[i] ) min = a[i];
		if( max < a[i] ) max = a[i];
		sum += a[i];
	}
	printf("%d %d %ld\n", min, max, sum);

	return 0;
}
