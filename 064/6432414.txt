#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for(int i = 0; i < (n); i++)

int main(){
    //1.Input
    int n;
    cin >> n;
    //2.Process
    vector<ll> dp(n+1);
    dp[0] = 1;
    dp[1] = 1;
    for(int i = 2; i <= n; i++) dp[i] = dp[i-1]+dp[i-2];
    //3.Output
    cout << dp[n] << endl;
}
