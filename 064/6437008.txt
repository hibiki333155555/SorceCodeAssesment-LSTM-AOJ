#include <bits/stdc++.h>
using namespace std;

int main() {

  int n;
  cin >> n;
  vector<int> vc(n);
  for (int i = 0; i < n; i++) {
    cin >> vc.at(i);
  }
  for (int i = 0; i < n; i++) {
    cout << vc.at(i) << " ";
  }
  cout << endl;
  for (int i = 1; i < n; i++) {
    int a = vc.at(i), b = 0;
    for (int j = i - 1; j >= 0; j--) {
      if (a < vc.at(j)) {
        vc.at(j + 1) = vc.at(j);
      } else {
        b = j + 1;
        break;
      }
    }
    vc.at(b) = a;
    for (int j = 0; j < n; j++) {
      cout << vc.at(j) << " ";
    }
    if (i != i) {
      cout << endl;
    }
  }
}
