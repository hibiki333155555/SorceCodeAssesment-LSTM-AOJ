#include<iostream>
using namespace std;
int main(){
    int N;
    int i, j, k;
    int v;
    cin >> N;
    int arr[N];
    for( i = 0; i < N; i++){
        cin >> arr[i];
    }

    for( i = 0; i < N; i++){
        v = arr[i];
        for( j = i - 1; j >= 0; j--){
            if( v < arr[j] )arr[j+1] = arr[j];
            else break;
        }
        arr[j + 1] = v;
        for( k = 0; k < N; k++){
            cout << arr[k];
            if(k != N-1)cout << " ";
            else cout << "\n";
        }
    }
}
