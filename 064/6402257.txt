#include <iostream>
#include <cstring>
using namespace std;

int main(){
    while(1){
        char a[1001];
        int sum = 0;
        cin >> a;
        if(a[0] == '0') break;
        for(int i = 0; i < strlen(a); i++){
            sum += a[i] - '0';
        }
        cout << sum << endl;
        
    }
    
    return 0;
}
