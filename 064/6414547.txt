#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod=998244353;
int main(){
  int N;
  cin>>N;
  vector<int> A(N);
  vector<int> B(N,0);
  for(int i=0;i<N;i++){
    cin>>A[i];
    B[i+A[i]-1]++;
  }
  ll C=0;
  ll ANS=1;
  for(int j=N-1;j>=0;j--){
    C+=B[j];
    ANS=ANS*C%mod;
    C-=1;
  }
  cout<<ANS<<"\n";
}
