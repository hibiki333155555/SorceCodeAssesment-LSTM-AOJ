#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (true)
    {
        int x;
        cin >> x;
        if (x == 0)
        {
            break;
        }
        else
        {
            unsigned int sum = 0;
            while (x >= 10)
            {
                sum += x % 10;
                x /= 10;
            }
            sum += x;
            cout << sum << endl;
        }
    }

    return 0;
}
