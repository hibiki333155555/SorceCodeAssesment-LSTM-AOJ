#include <bits/stdc++.h>
#define lol long long
using namespace std;
int main(){
    int r,c;
    cin>>r>>c;
    int a[r+1][c+1] = {};
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            cin>>a[i][j];
            a[i][c] += a[i][j];
            a[r][j] += a[i][j];
        }
    }
    for(int i=0;i<=r;i++){
        for(int j=0;j<c;j++){
            cout<<a[i][j]<<" ";
        }
        cout<<a[i][c]<<endl;
    }
    return 0;
}
