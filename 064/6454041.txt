#include <iostream>

using namespace std;

const int N = 1010;

int n;
int w[N], f[N];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) 
		cin >> w[i];
	int maxx = 1;    
    for (int i = 0; i < n; i++) 
	{
        f[i] = 1;    
        for (int j = 0; j < i; j++) 
		{
            if (w[i] < w[j]) 
				f[i] = max(f[i], f[j] + 1);    		
        }
        maxx = max(maxx, f[i]);
    }
    cout << maxx << endl;
    return 0;
}
