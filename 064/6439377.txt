#include <iostream>
#include <vector>
#include <string>
using namespace std;
int calc(int n, int x) {
	int ans = 0;
	vector <int>numb(n);
	for (int i = 1; i <= n; ++i) {
		numb.at(i - 1) = i;
	}
	for (int i = 0; i <= n; ++i) {
		for (int j = i + 1; j <= n; ++j) {
			for (int k = j + 1; k <= n; ++k) {
				if (numb.at(i) + numb.at(k) + numb.at(j) == x) ans += 1;
			}
		}
	}
	return ans;
}

int main() {
	int n, x;
	for (int i = 0;; ++i) {
		cin >> n >> x;
		if ((n & x) == 0) break;
		cout << calc(n, x) << endl;
	}

}
