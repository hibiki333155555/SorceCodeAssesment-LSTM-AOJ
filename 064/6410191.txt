#include <bits/stdc++.h>
using namespace std;

int main(){
    string W,T;
    cin>>W;
    for(auto &w:W){
        if(w>='A' && w<='Z') w+=32;
    }
    int count=0;
    while(1){
        cin>>T;
        if(T=="END_OF_TEXT") break;
        else{
            for(auto &c: T){
                if(c>='A' && c<='Z') c+=32;
            }
            if(W==T) count++;
        }
    }
    cout<< count<<endl;
}
