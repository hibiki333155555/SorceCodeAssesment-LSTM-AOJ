#include <iostream>

using namespace std;

int buf[100];

int main()
{
	int n, cnt = 0, mini;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> buf[i];
	}
	for (int i = 0; i < n; ++i) {
		mini = i;
		for (int j = i; j < n; ++j ) {
			if (buf[j] < buf[mini]) {
				mini = j;
			}
		}
		if (i != mini) {

			swap(buf[i], buf[mini]);
			++cnt;
		}
	}
	for (int i = 0; i < n - 1; ++i) {
		cout << buf[i] << " ";
	}
	cout << buf[n-1] << endl;
	cout << cnt << endl;
	return 0;
}
