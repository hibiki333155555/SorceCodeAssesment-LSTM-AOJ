#include <iostream>
#include <algorithm>
#include <string>
#define rep(i, n) for(i = 0; i < n; i++)
using namespace std;

int h, w;
string s[100];

int main() {
	int i, j, k;
	
	cin >> h >> w;
	rep(i, h) cin >> s[i];
	
	int ans = h * w;
	for (i = 1; i < h; i++) {
		int cnt[4] = {0};
		rep(j, i) {
			rep(k, w) {
				if (s[j][k] == 'o') cnt[0]++;
				else cnt[1]++;
			}
		}
		for (j = i; j < h; j++) {
			rep(k, w) {
				if (s[j][k] == 'o') cnt[2]++;
				else cnt[3]++;
			}
		}
		ans = min(ans, cnt[0] + cnt[3]);
		ans = min(ans, cnt[1] + cnt[2]);
	}
	
	for (i = 1; i < w; i++) {
		int cnt[4] = {0};
		rep(j, i) {
			rep(k, h) {
				if (s[k][j] == 'o') cnt[0]++;
				else cnt[1]++;
			}
		}
		for (j = i; j < w; j++) {
			rep(k, h) {
				if (s[k][j] == 'o') cnt[2]++;
				else cnt[3]++;
			}
		}
		ans = min(ans, cnt[0] + cnt[3]);
		ans = min(ans, cnt[1] + cnt[2]);
	}
	
	cout << ans << endl;
	return 0;
}
