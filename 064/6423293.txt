#define rep(i, a, n) for (int i = (a); i < (n); ++i)
#define bit(n, k) ((n >> k) & 1)

using Vector = pair<int, int>;

class Program {
private:

    bool max(int &a, const int b) {
        if (a < b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    Vector p1, p2, p;
    int q;

public:

    Program() {
    }

    void input() {
        {
            auto& [x, y] = p1;
            cin >> x >> y;
        }

        {
            auto& [x, y] = p2;
            cin >> x >> y;
        }

        cin >> q;
    }

    int norm(const Vector &a, const Vector &b) {
        return a.first * b.first + a.second * b.second;
    }

    // a(p-p1)のb(p2-p1)への射影
    // 向きはb/|b|
    // 大きさは|a|cosθ
    // cosθはab/|a||b|
    // 大きさはab/|b|
    // |a|^2はaa
    void solve() {
        input();

        Vector b = make_pair(p2.first - p1.first, p2.second - p1.second);
        int bb = norm(b, b);

        rep(i, 0, q) {
            auto& [x, y] = p;
            cin >> x >> y;

            if (bb == 0) {
                printf("%.8f %.8f\n", 0.0, 0.0);
            } else if (b.first == 0) {
                printf("%.8f %.8f\n", 0.0, 1.0 * y);
            } else if (b.second == 0) {
                printf("%.8f %.8f\n", 1.0 * x, 0.0);
            } else {
                Vector a = make_pair(p.first - p1.first, p.second - p1.second);

                int ab = norm(a, b);

                printf("%.8f %.8f\n", 1.0 * ab / bb * b.first, 1.0 * ab / bb * b.second);
            }
        }

        // printf("%.8f\n", 0.00000001);
    }
};

int main() {

#ifdef DEV
    std::ifstream in("input");
    cin.rdbuf(in.rdbuf());

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();
#endif

    return 0;
}
