#include <bits/stdc++.h>
using namespace std;

constexpr int mod = 998244353;

long long modpow(long long a,long long b) {
    long long ans = 1;
    while(b) {
        if(b & 1) {
            (ans *= a) %= mod;
        }
        (a *= a) %= mod;
        b /= 2;
    }
    return ans;
}

int main() {
    int N,K;
    cin >> N >> K;
    if(K >= 3) {
        cout << modpow(2,N) << endl;
    }
    else if(K == 1) {
        cout << 1 << endl;
    }
    else {
        int a = 1,b = 1;
        for(int i = 0; i < N; i++) {
            int c = (a+b)%mod;
            a = b;
            b = c;
        }
        cout << b << endl;
    }
}
