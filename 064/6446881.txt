#include <iostream>
#include <bitset>
#include <vector>

using namespace std;

bool EnumerationOfSubsets1(unsigned int n) {

	for (int i = 0; i < (1 << n); i++) {
		cout << i << ':';
		for (int j = 0; j < n; j++) {
			if (i & (1 << j)) {
				cout << ' ' << j;
			}
		}
		cout << endl;
	}
	return true;

}

int main() {

	unsigned int n = 0;

	cin >> n;

	if (EnumerationOfSubsets1(n)) {
		return 0;
	}
	else {
		return -1;
	}

	return 0;
}
