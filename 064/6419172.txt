#include <bits/stdc++.h>
using namespace std;
using ll = long long;
int main(){
  while(true){  
    int n,sum=0;cin>>n;
    if(n==0){break;}
    vector<int> s(n);
    for(int i=0;i<n;i++){
        cin>>s.at(i);
        sum+=s.at(i);
    }
    double ave=(double)sum/n;
    //cout<<ave<<endl;
    double asum=0;
    for(int i=0;i<n;i++){
        double temp=s.at(i)-ave;
        //cout<<temp<<endl;
        temp=temp*temp;
        //cout<<temp<<endl;
        asum+=temp;
    }
    double ans=(double)asum/n;
    cout<<fixed<<setprecision(12)<<sqrt(ans)<<endl;
  }
  return 0;  
}
