#include <bits/stdc++.h>
using namespace std;

using ll = long long;


int main(int argc, char const *argv[])
{
	while(1){
		/* code */
		ll n = in_ll();
		ll x = in_ll();

		if(n == 0 && x == 0)
			break;

		ll cnt = 0;
		for (int i = 1; i <= n; i++)
		{
			for (int j = i + 1; j <= n; j++)
			{
				for (int k = j + 1; k <= n; k++)
				{
					if ((i + j + k) == x)
					{
						cnt++;
					}
				}
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
