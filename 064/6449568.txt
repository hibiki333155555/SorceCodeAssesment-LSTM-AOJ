void OutPutInteger(ll x) { cout << x << endl; }
void OutPutString(string x) { cout << x << endl; }


int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n; cin >> n;
    vector<string> s(n);
    rep(i, n) cin >> s[i];

    vector<bool> ng(n, false);

    rep(i, n)rep(j, n) {
        if (i == j) continue;
        int diff = 0;
        rep(k, SZ(s[0])) {
            if (s[i][k] != s[j][k]) diff++;
        }

        if (diff <= 1) {
            ng[i] = ng[j] = true;
        }
    }
    
    int res = 0;
    rep(i, n) {
        if (ng[i]) res++;
    }

    OutPutInteger(res);
}
