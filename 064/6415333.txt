#include <bits/stdc++.h>
using namespace std;
#ifdef tabr
#include "library/debug.cpp"
#else
#define debug(...)
#endif

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int tt = 1;
    //cin >> tt;
    while (tt--) {
        int n;
        cin >> n;
        vector<long long> h(n);
        for (int i = 0; i < n; i++) {
            cin >> h[i];
        }
        stack<int> st;
        int ans = 0;
        st.emplace(0);
        st.emplace(1);
        for (int i = 2; i < n; i++) {
            while ((int)st.size() >= 2) {
                int j = st.top();
                st.pop();
                int k = st.top();
                if ((h[i] - h[k]) * (j - k) >= (h[j] - h[k]) * (i - k)) {
                    continue;
                } else {
                    st.emplace(j);
                    break;
                }
            }
            st.emplace(i);
        }
        vector<int> t;
        while (!st.empty()) {
            t.emplace_back(st.top());
            st.pop();
        }
        for (int i = 1; i < (int)t.size(); i++) {
            ans = max(ans, t[i - 1] - t[i]);
        }
        cout << ans << '\n';
    }
    return 0;
}
