#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    int n, s, t, ans = 0, now;
    cin >> n;
    vector<pair<int, int>> ls;
    for (int i = 0; i < n; ++i)
    {
        cin >> s >> t;
        ls.emplace_back(t, s);
    }
    sort(ls.begin(), ls.end());
    now = ls[0].first;
    ans++;
    for (int i = 1; i < n; ++i)
    {
        if (ls[i].second >= now)
        {
            now = ls[i].first;
            ans++;
        }
        else
            continue;
    }
    cout << ans << endl;
    return 0;
}
