#include <bits/stdc++.h>
//#include<atcoder/all>
using namespace std;
//using namespace atcoder;
using ll = long long;
#define all(A) A.begin(),A.end()
using vll = vector<ll>;
#define rep(i, n) for (long long i = 0; i < (long long)(n); i++)

ll mod = 1e9 + 7;


int main() {
	ll N;
	cin >> N;
	ll M, P;
	P =997;
	M = 996;
	vector<vll> A(N+1, vll(M));
	rep(i, M)A[0][i] = i + 1;
	
	ll n=1;
	set<ll> S;
    rep(a,P){
        if(S.count(a))continue;
        rep(b,a){
            if(S.count(b))continue;
            rep(c,b){
                if(S.count(c))continue;
                if(a*b*c==0)continue;
                if(((a-b)*(a-b)+(b-c)*(b-c)+(c-a)*(c-a))%P==0){
                    
                    S.insert(a);
                    S.insert(b);
                    S.insert(c);

                    rep(m,M){
						if(m==a){
							A[n][m]=A[0][b];
						}
						else if(m==b){
							A[n][m]=A[0][c];
						}
						else if(m==c){
							A[n][m]=A[0][a];
						}
						else{
							A[n][m]=A[0][m];
						}
					}
					n++;
					if(n==N)break;
                }
				if(n==N)break;
            }
			if(n==N)break;
        }
		if(n==N)break;
    }

	cout<<M<<" "<<P<<endl;
	rep(i, N) {
		rep(m, M) {
			cout << A[i][m] << " ";
		}
		cout << endl;
	}

	


}
