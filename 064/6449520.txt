#include<bits/stdc++.h>
using namespace std;
using lint=long long;

int main(){
    lint a,b;
    cin>>a>>b;
    if(a<b)swap(a,b);
    vector<lint> s,t;
    while(a){
        s.emplace_back(a%2);
        a/=2;
    }
    while(b){
        t.emplace_back(b%2);
        b/=2;
    }
    while(s.size()!=t.size())t.emplace_back(0);
    lint ans=0;
    for(int i=int(s.size())-1;i>=0;--i){
        if(s[i]==t[i]){
            ans++;
        }else{
            break;
        }
    }
    cout<<ans<<endl;
}
