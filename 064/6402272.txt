#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
using namespace std;



int main()
{
    int r, c;
    cin >> r >> c;
    const int size = r*c;
    int *arr = new int[size];

    for(int i = 0; i < size; i++)
        cin >> arr[i];
    
    for(int i = 0, sum; i < r; i++)
    {
        sum = 0;
        for(int j = 0; j < c; sum+= arr[i*c+j], j++)
            cout << arr[i*c+j] << " ";
        cout << sum << endl;
    }

    int total = 0;
    for(int i = 0, sum; i < c; i++)
    {
        sum = 0;
        for(int j = 0; j < r; j++)
            sum += arr[j*c+i];
        cout << sum << " ";
        total += sum;
    }
    cout << total << endl;

    delete [] arr;
    return 0;
}
