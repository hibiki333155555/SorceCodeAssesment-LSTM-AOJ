template <class T>
T gcd(T a, T b) {
    if (b == 0)
        return a;
    else
        return gcd(b, a % b);
}

template <class T>
T lcm(T a, T b) {
    a /= gcd(a, b);
    return a * b;
}

template <class T>
T Pow(T x, i64 n) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = res * x;
        x = x * x;
        n >>= 1;
    }
    return res;
}

template <class T>
T mod_pow(T x, i64 n, i64 mod) {
    T res = 1;
    while (n > 0) {
        if (n & 1) res = (res * x) % mod;
        x = (x * x) % mod;
        n >>= 1;
    }
    return res;
}

void main_() {
    int n;
    while (std::cin >> n, n > 0) {
        std::vector<std::string> s(n);
        std::vector<int> a(26, -1);
        for (auto &str : s) {
            std::cin >> str;
            for (auto c : str) {
                a[c - 'A'] = 1;
            }
        }
        std::vector<i64> cf(26, 0), flag(26, 1);
        rep(i,0,n) {
            if((int)s[i].size() > 1) {
                flag[s[i][0]-'A'] = -1;
            }
            i64 ret = Pow(i64(10), s[i].size()-1);
            for(auto c: s[i]) {
                cf[c-'A'] += ret * (i == n-1 ? -1: 1);
                ret /= 10;
            }
        }
        int m = 0;
        std::vector<int> alpha;
        rep(i, 0, 26) {
            if (a[i] > 0) {
                a[i] = m++;
                alpha.emplace_back(i);
            }
        }
        std::vector<std::vector<int>> ps;
        std::vector<int> xs;
        auto dfs = [&ps, &xs, &m](auto &&self, int now) -> void {
            if ((int)xs.size() == m) {
                ps.emplace_back(xs);
                return;
            }
            rep(i, now, 10) {
                xs.emplace_back(i);
                self(self, i+1);
                xs.pop_back();
            }
            return;
        };
        dfs(dfs, 0);
        int ans = 0;
        for(auto p: ps) {
            do {
                i64 res = 0;
                for(auto idx: alpha) {
                    assert(a[idx] >= 0);
                    if(p[a[idx]] == 0 && flag[idx] < 0) {
                        res = -1;
                        break;
                    }
                    res += cf[idx] * p[a[idx]];
                }
                if(res == 0) ans++;
            } while(std::next_permutation(all(p)));
        }
        std::cout << ans << '\n';
    }
}

}  // namespace ebi

int main() {
    std::cout << std::fixed << std::setprecision(15);

    while (t--) {
        ebi::main_();
