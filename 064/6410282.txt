#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main(){
	cin.tie(nullptr);
	ios::sync_with_stdio(false);
	int n; cin >> n;
	ll K; cin >> K;
	vector<ll> a(n),b(n);
	for(int i=0;i<n;i++){
		cin >> a[i];
	}
	for(int i=0;i<n;i++){
		cin >> b[i];
	}
	ll l=0,r=1e17;
	while(r-l>1){
		ll mid=(l+r)/2;
		if(a.back()>mid){
			l=mid;
			continue;
		}
		ll cnt=0;
		ll ty=0;
		vector<ll> bb=b;
		for(int i=n-1;i>=0;i--){
			if(bb[i]<=ty){
				ty-=bb[i];
				continue;
			}
			else{
				bb[i]-=ty;
				ty=0;
			}
			if(mid-a[i]==0){
				cnt=1e18;
				break;
			}
			ll num=(bb[i]+(mid-a[i]-1))/(mid-a[i]);
			cnt+=num;
			ty=num*(mid-a[i])-bb[i];
		}
		if(cnt<=K){
			r=mid;
		}
		else{
			l=mid;
		}
	}
	cout << r << endl;
}
