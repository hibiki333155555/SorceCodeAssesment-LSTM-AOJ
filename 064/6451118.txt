#include <iostream>
#include <unordered_set>
#include <cstring>
using namespace std;

const int maxq = 10000 + 10;
const int maxl = 1000000 + 10;

char txt[maxl];
string sub[maxl];
char pat[maxq][1024];
bool len[1024];

unordered_set <string> S;

int main()
{
    //freopen("1.in","r",stdin);
    
    scanf("%s", txt);

    int q;
    scanf("%d", &q);

    int minLen = 1000;
    int maxLen = 0;
    for(int i = 0; i < q; ++i)
    {
        scanf("%s", pat[i]);
        int l = strlen(pat[i]);
        len[l] = true;
        minLen = min(minLen, l);
        maxLen = max(maxLen, l);
    }
    
    int txtLen = strlen(txt);

    for(int i = 0; i < txtLen ; ++i)
    {
        sub[i].assign(txt+i, min(txtLen-i, maxLen));
    }

    for(int iq = 0; iq < q; ++iq)
    {
        int l = strlen(pat[iq]);

        for(int j = 0; j < txtLen; ++j)
        {
            if(sub[j].length() >= l)
            {
                S.insert(sub[j].substr(0,l));
                //cout << "insertring: " << sub[j].substr(0,i) << endl;
            }
        }

        if(S.find(string(pat[iq])) != S.end())
            printf("1\n");
        else
            printf("0\n");
    }

}
