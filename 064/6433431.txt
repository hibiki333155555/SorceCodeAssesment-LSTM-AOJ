#include <iostream>
#define n_MAX 10000
#define a_MAX 1000000
#define a_MIN -1000000

int main() {
	int n, a;
	int max = a_MIN, min = a_MAX, sum = 0;

	std::cin >> n;

	for (int i = 0; i < n; i++) {
		std::cin >> a;
		sum += a;
		if (a > max) max = a;
		if (a < min) min = a;
	}

	std::cout << min << " " << max << " " << sum << std::endl;

	return 0;
}
