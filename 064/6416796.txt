#include<iostream>
#include<stdio.h>

using namespace std;
int main()
{
    int arr[3];
    int i,j;
    for(i=0;i<3;i++)
    {
        cin>>arr[i];
    }

    for(i=0;i<3;i++)
    {
      for(j=0;j<2;j++)
      {
          if(arr[j]>arr[j+1])
          {
              int temp;
              temp=arr[j];
              arr[j]=arr[j+1];
              arr[j+1]=temp;
          }
      }
    }

    for(i=0;i<3;i++)
    {
        cout<<arr[i]<<" ";
    }
   

    return 0;
}
