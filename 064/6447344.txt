#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[1000],b[1000],c[1000];
    for(int i=0; i<n; i++)
        cin>>a[i]>>b[i]>>c[i];

    for(int i=0; i<n; i++)
    {
        if(pow(a[i],2) == pow(b[i],2) + pow(c[i],2) || pow(b[i],2) == pow(a[i],2) + pow(c[i],2) || pow(c[i],2) == pow(a[i],2) + pow(b[i],2))
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
