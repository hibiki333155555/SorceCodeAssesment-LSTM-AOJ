#include <iostream>
#include <algorithm>
#include <vector>
#include <numeric>

using namespace std;


int main() {
    int n;
    vector<int> nums;
    int x;
    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> x;
        nums.push_back(x);
    }
    sort(nums.begin(), nums.end());
    cout << nums[0] << " " << nums[nums.size() - 1] << " " << accumulate(nums.begin(), nums.end(), 0) << endl;;
    return 0;
}
