#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string w, t;
    cin >> w;

    int wordCnt = 0;

    while (true)
    {
        cin >> t;
        if (t.compare("END_OF_TEXT") == 0)
        {
            break;
        }

        transform(t.begin(), t.end(), t.begin(), ::tolower);
        if(t.compare(w) == 0){
            wordCnt++;
        }


    }

    cout << wordCnt << endl;
    return 0;
}
