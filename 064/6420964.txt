int main() {
  constexpr int C = 26;
  string s; cin >> s;
  const int n = s.length();
  vector<int> right(n, -1), ls;
  REP(i, n) {
    if (s[i] == '(') {
      ls.emplace_back(i);
    } else if (s[i] == ')') {
      right[ls.back()] = i;
      ls.pop_back();
    }
  }
  string ans = "";
  auto f = [&](auto&& f, int l, int r, int n) -> void {
    if (r <= l) return;
    if (isdigit(s[l])) {
      f(f, l + 2, right[l + 1], (n + s[l] - '0') % C);
      f(f, right[l + 1] + 1, r, n);
    } else {
      while (l < r && islower(s[l])) ans += 'a' + (s[l++] - 'a' + n) % C;
      f(f, l, r, n);
    }
  };
  f(f, 0, n, 0);
  cout << ans << '\n';
  return 0;
}
