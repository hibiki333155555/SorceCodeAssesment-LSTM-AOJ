#include <iostream>
#include <stack>
using namespace std;
typedef pair<int, int> P;

int main()
{
    string s;
    cin >> s;
    int n = s.size();
    stack<int> st;
    int b[200005]{0};
    for(int i = 0; i < n; i++){
        if(s[i] == '('){
            st.push(s[i - 1] - '0');
            b[i] = s[i - 1] - '0';
        }
        if(s[i] == ')'){
            int x = st.top();
            st.pop();
            b[i] = -x;
        }
    }
    for(int i = 1; i < n; i++) b[i] += b[i - 1];
    for(int i = 0; i < n; i++){
        if(s[i] >= 'a' && s[i] <= 'z') cout << char('a' + (s[i] - 'a' + b[i]) % 26);
    }
    cout << endl;
}
