#include<iostream>
#include<algorithm>
using namespace std;
static const int maxn = 100010;
int n, a[maxn], l = 1, b[maxn];
int main() {
    cin>>n;
    for (int i = 1; i <= n; i++) {
        cin>>a[i];
    }
    b[l] = a[1];
    for (int i = 2; i <= n; i++) {
        if (a[i] > b[l]) {
            b[++l] = a[i];
        }
        else {
            *lower_bound(b+1, b + l+1, a[i]) = a[i];
        }
    }
    cout << l << endl;
}
