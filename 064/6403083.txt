#include <iostream>
#include <string>
using namespace std;

int main(){
    int n, x = 0, y = 0;
    cin >> n;
    for(int i = 0; i < n; i++){
        string a, b;
        cin >> a >> b;
        if(a == b){
            x++;
            y++;
        }
        else if(a > b) x += 3;
        else y += 3;
    }
    cout << x << " " << y << endl;
    return 0;
}
