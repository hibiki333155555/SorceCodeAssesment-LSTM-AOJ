#include <bits/stdc++.h>
#define lol long long
using namespace std;
int main(){
    int n,m;
    cin>>n>>m;
    int a[n][m];
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin>>a[i][j];
        }
    }
    int b[m];
    for(int i=0;i<m;i++){
        cin>>b[i];
    }
    int c[n] = {};
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            c[i] += a[i][j]*b[j];
        }
    }
    for(int i=0;i<n;i++){
        cout<<c[i]<<"\n";
    }
    return 0;
}
