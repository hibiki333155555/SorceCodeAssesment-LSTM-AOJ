#include<iostream>
#include<cstdio>
using namespace std;
int main() {
    int dice[6];
    for(int n = 0; n < 6; n++) {
        cin >> dice[n];
    }
    int vector[6][6] = {{0, 3, 5, 2, 4, 0}, {4, 0, 1, 6, 0, 3}, {2, 6, 0, 0, 1, 5},
                        {5, 1, 0, 0, 6, 2}, {3, 0, 6, 1, 0, 4}, {0, 4, 2, 5, 3, 0}};
    int count;
    cin >> count;
    for(int n = 0; n < count; n++) {
        int a, b;
        cin >> a >> b;
        cout << dice[vector[a - 1][b - 1] - 1] << endl;
    }
}
