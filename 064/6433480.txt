#include <iostream>
#include <string>
using namespace std;
int main() {
	int N;
	cin >> N;
	long long A[202020];
	for (int i = 1; i <= N; i++)cin >> A[i];
	long long answer = 0;
	long long left[202020];
	long long right[202020];
	for (int i = 1; i <= N - 1; i++) {
		left[i] = 292929292929;
		right[i] = 0;
	}
	left[1] = A[1];
	for (int i = 2; i <= N - 1; i++)left[i] = min(left[i - 1], A[i]);
	right[N - 1] = A[N - 1];
	for (int i = N - 2; i >= 1; i--)right[i] = max(right[i + 1], A[i]);
	for (int i = 1; i <= N - 1; i++)answer = max(answer, right[i] - left[i]);
	if (answer <= 0)cout << -1<<endl;
	else cout << answer<<endl;
    
  
}
