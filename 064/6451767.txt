#include <bits/stdc++.h>

using namespace std;

bool binarySearch(int S[], int len, int target) {
  int left = 0;
  int right = len - 1;
  int mid;
  while (left <= right) {
    mid = (left + right) / 2;
    if (target == S[mid]) return true;

    if (target > S[mid]) {
      left = mid + 1;
    } else {
      right = mid;
    }
  }
  return false;
}

int main() {
  int n;
  scanf("%d", &n);
  int S[n];
  for (int i = 0; i < n; i++) {
    scanf("%d", &S[i]);
  }
  int q;
  scanf("%d", &q);
  int T[q];
  for (int i = 0; i < q; i++) {
    scanf("%d", &T[i]);
  }

  sort(S, S + n);

  int count = 0;
  for (int i = 0; i < q; i++) {
    if (binarySearch(S, n, T[i])) count++;
  }

  cout << count << endl;
  return 0;
}
