import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        while (true) {
            String x = sc.next();
            if (x.equals("0")) break;
            int sum = 0;
            
            for (int i = 0; i < x.length(); ++i) {
//            	char値をinteger値に変換
                sum += x.charAt(i) - '0';
            }
            System.out.println(sum);
        }
    }
}
