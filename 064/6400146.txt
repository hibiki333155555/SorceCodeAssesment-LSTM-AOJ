int main() {
  int N;
  cin >> N;
  
  vector<int> dp(N+1, 0);
  dp.at(0) = 1;
  dp.at(1) = 1;
  
  rep(i,2,N+1) {
    dp.at(i) = dp.at(i-1) + dp.at(i-2);
