#include <iostream>
#include <string>
using namespace std;

int main() {
    string p, s;
    cin >> p >> s;
    string P = p + p;
    int cnt;
    for (int i = 0; i < p.size(); i++) {
        cnt = 0;
        for (int j = 0; j < s.size(); j++) {
            if (s[j] != P[i + j]) break;
            cnt++;
        }
        if (cnt == s.size()) break;
    }
    if (cnt == s.size()) cout << "Yes" << endl;
    else cout << "No" << endl;
    return 0;
}
