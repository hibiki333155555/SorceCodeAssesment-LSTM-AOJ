vector<int> palindrome(string &s){
  int n=s.size(),k;
  vector<int> r(n*2);
  for(int i=0,j=0;i<n*2;i+=k,j=max(j-k,0)) {
    while(i-j>=0&&i+j+1<n*2&&s[(i-j)/2]==s[(i+j+1)/2])j++;r[i]=j;
    for(k=1;i-k>=0&&r[i]-k>=0&&r[i-k]!=r[i]-k;k++)r[i+k]=min(r[i-k],r[i]-k);
  }
  return r;
}

void Main() {
  ll n;
  string s;
  cin >> n >> s;
  vector<int> v=palindrome(s);v.pop_back();
  ll ans=0;
  for(int i=0; i<v.size(); i++) {
    if(!v[i]) continue;
    int l=i/2,r=i/2,z=v[i]-1;
    if(i%2) r++,z--;
    else continue;
    l-=z/2,r+=z/2;
    if(!l||r==n-1) ans++;
  }
  pr(ans);
}
