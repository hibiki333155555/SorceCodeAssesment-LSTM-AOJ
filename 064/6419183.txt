#include <iostream>
#include <vector>
#include <queue>
using namespace std;
int n;
priority_queue <int,vector <int>,less <int>> q;
int main () {
	cin >> n;
	for (int i = 1;i <= n;i++) {
		int x;
		cin >> x;
		q.push (x);
	}
	cout << q.top ();
	q.pop ();
	for (int i = 2;i <= n;i++) {
		cout << ' ' << q.top ();
		q.pop ();
	}
    return 0;
}
