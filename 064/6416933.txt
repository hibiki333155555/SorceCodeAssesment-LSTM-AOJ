import java.util.*;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		PriorityQueue<Integer> pq = new PriorityQueue<>(Comparator.reverseOrder());
		
		while(true) {
			String s = sc.next();
			
			if(s.equals("end")) break;
			if(s.equals("insert")) {
				int n = sc.nextInt();
				pq.add(n);
			}
			if(s.equals("extract")) {
				System.out.println(pq.poll());
			}
		}
	}
}
