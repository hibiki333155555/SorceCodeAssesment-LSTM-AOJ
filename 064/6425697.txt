#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;
static const int INFINITY = INT_MAX;

int main() {
    int n, m;
    cin >> n >> m;

    int C[m], T[n + 1];
    for (int i = 0; i < m; ++i) {
        cin >> C[i];
    }
    for (int i = 0; i <= n; ++i) {
        T[i] = INFINITY;
    }

    T[0] = 0;

    for (int i = 0; i < m; ++i) {
        for (int j = 0; j + C[i] <= n; ++j) {
            T[j + C[i]] = min(T[j + C[i]], T[j] + 1);
        }
    }

    cout << T[n] << endl;
}
