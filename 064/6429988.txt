#include<bits/stdc++.h>
using namespace std;

int main(){
    int x,y;
    cin>>x>>y;
    int sum[1000001]={};
    sum[1]=1;
    sum[2]=2;
    int i=3;
    int ans;
    if(x==0&&y==0){
        ans=1;
    }else if(x==1&&y==0){
        ans=2;
    }else{
        int c=3;
        while(1){
            sum[i]=sum[i-2]+sum[i-1];
            if(sum[i]>=x&&sum[i]>=y){
                ans=c;
                break;
            }
            c++;
            if(c==4){
                c=1;
            }
            i++;
        }
        ans++;
    }
    cout<<ans<<"\n";
    return(0);
}
