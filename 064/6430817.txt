#include <bits/stdc++.h>
using namespace std;
int bubblesort(vector <int> &u)
{
	int n=u.size();
	int count=0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(u[j]>u[j+1]){
				swap(u[j],u[j+1]);
				count++;
			}
		}
		
	}
	return count;
}
int main() {
int n;
cin>>n;
vector<int>u(n);
for(int i=0;i<n;i++){
	cin>>u[i];
}
int res=bubblesort(u);
for(int i=0;i<n;i++){
	cout<<u[i];
	if(i!=n-1)
	cout << " ";
}
cout<<endl<<res<<endl;
return 0;
} //io
