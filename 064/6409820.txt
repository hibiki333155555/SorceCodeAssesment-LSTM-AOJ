#include <iostream>
#include <cstdio>

using namespace std;

void printArray(int *nums, int size)
{
  for (int i = 0; i < size; i++)
  {
    if (i != size - 1) printf("%d ", nums[i]);
    else printf("%d\n", nums[i]);
  }
}

int bubbleSort(int *nums, int size)
{
  bool flag = true;
  int count = 0;
  while (flag)
  {
    flag = false;
    for (int i = size-1; i >= 1; i--)
    {
      if (nums[i] < nums[i-1]) {
        int tmp = nums[i];
        nums[i] = nums[i-1];
        nums[i-1] = tmp;
        count++;
        flag = true;
      }
    }
  }

  return count;
}

int main(int argc, char const *argv[])
{
  int n;

  cin >> n;
  int *nums = new int[n];

  for (int i = 0; i < n; i++)
  {
    cin >> nums[i];
  }

  int count = bubbleSort(nums, n);

  printArray(nums, n);
  cout << count << endl;

  return 0;
}
