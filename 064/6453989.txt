using System;

namespace ConsoleApp3
{
    class Program
    {
        static void Main(string[] args)
        {
            int t = int.Parse(Console.ReadLine());
            int h = t / 3600;
            int m = (t % 3600) / 60;
            int s = t % 60;
            Console.WriteLine($"{h}:{m}:{s}");


        }
    }
}
