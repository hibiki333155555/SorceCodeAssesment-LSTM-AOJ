#define all(A) A.begin(),A.end()

vvll G;
vll A;
ll mod = 998244353;
vector<bool> seen;
vll mi;
vll nu;
vll T(2e6);
ll an = 0;
void dfs(ll n) {
    seen[n] = true;
    ll res = 1;
    ll m = 0;
    bool C = true;
    for (auto v : G[n]) {
        if (seen[v])continue;
        dfs(v);
        C = false;
        m += mi[v];
        res *= (nu[v]);
        res %= mod;
    }
    if (C) {
        nu[n] = 1;
        mi[n] = A[n];
        an += mi[n];
        an %= mod;
    }
    else {
        nu[n] += res;
        mi[n] = min(A[n], m);
        an += res * mi[n];
        an %= mod;
    }

}

ll inva(ll N) {
    ll a = N, b = mod, c = 1, d = 0;
    while (b > 0) {
        ll t = a / b;
        a -= t * b;
        swap(a, b);
        c -= t * d;
        swap(c, d);

    }
    c %= mod;
    if (c < 0)c += mod;
    return c;

}
