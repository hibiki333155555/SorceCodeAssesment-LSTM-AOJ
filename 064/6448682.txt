#include<bits/stdc++.h>
using namespace std;

const int N = 500;
int n;
vector<long long> e;
vector<long long> o;
long long b[N];

int main() {
	while(cin>>n && n) {
		e.clear();
		o.clear();
		for(int i=0;i<n*(n+1)/2;i++) {
			cin>>b[i];
			if(b[i]&1)
				o.push_back(b[i]);
			else 
				e.push_back(b[i]);
		}
		sort(e.begin(),e.end());
		sort(o.begin(),o.end());

		long long a0 = (long long)sqrt((double)e[0]*e[1]/o[0]+0.5);
		cout<<a0<<endl;
		for(int i=0;i<e.size();i++)
			cout<<e[i] / a0<<" ";
		cout<<endl;
	}
}
