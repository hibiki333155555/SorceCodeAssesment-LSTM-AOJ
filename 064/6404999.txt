#include <iostream>
#include <limits.h>

#include <algorithm>
using namespace std;
#define MAX_N  999999+1
#define MAX_M  10000
typedef long long ll;
ll M,H;
bool prime[MAX_N];
int prime_num(int n){
    int cnt = 0;
    for(int i = 2; i <= n; ++i){
        prime[i] = true;
    }
    for(int i = 2; i <= n; ++i){
        if(prime[i]){
            cnt++;
            for(int j =  i + i; j <= n; j+=i){
                prime[j] = false;
            }
        }
    }
    return cnt;
}

int main(){
    int n;
    while(cin >> n){
        int cnt = prime_num(n);
        cout << cnt;
    }
}
