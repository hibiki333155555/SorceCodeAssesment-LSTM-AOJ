#include <iostream>
using namespace std;

int main(){
    int H, W;
    int count = 0;
    
    while(true){
    cin >> H >> W;
    if(H == 0 && W == 0) break;
        for(int i = 0; i < H; i++){
            for(int j = 0; j < W; j++){
                if(count % 2 == 0) cout << '#';
                else cout << '.';
                count++;
            }
            cout << endl;
        }
    cout << endl
    }  

    return 0;
}
