#include <bits/stdc++.h>
using namespace std;

int main() {
    for (int i = 0; ; i++) {
        int a, b;
        char op;
        cin >> a >> op >> b;
        if (op == '?') break;
        else if (op == '+') cout << a + b << endl;
        else if (op == '*') cout << a * b << endl;
        else if ( op == '-') cout << a - b << endl;
        else if (op == '/') cout << a / b << endl;

    } 
}
