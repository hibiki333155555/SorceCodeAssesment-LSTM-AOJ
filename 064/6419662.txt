count=[[[0 for i in range(10)] for j in range(3)] for k in range(4)]
n=int(input())
for _ in range(n):
    b,f,r,v=map(int,input().split())
    count[b-1][f-1][r-1]=count[b-1][f-1][r-1]+v
for x in range(4):
    for y in range(3):
        print(" "+" ".join(map(str,count[x][y])))
    if x<3:
        print("#"*20)
