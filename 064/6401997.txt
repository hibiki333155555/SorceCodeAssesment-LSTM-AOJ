#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main(){
    vector<int> cnt(110, 0);
    for(int i = 0; i < 12; i++){
        int e; cin >> e;
        cnt[e]++;
    }
    string ans = "yes";
    for(auto &it: cnt){
        if(it%4 != 0) ans = "no";
    }
    cout << ans << endl;
    return 0;
}
