#include <bits/stdc++.h>
using namespace std;

int main(){
    while(true){
        int H,W;
        cin >> H >> W;
        if(H==0&&W==0){
            cout << endl;
            break;
        }
        for(int i = 0;i < H;i++){
            for(int j = 0;j < W;j++) cout << '#';
            cout << endl;
        }
        cout << endl;
    }
}
