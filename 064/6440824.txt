#include<iostream>
using namespace std;

int main() {
    int H, W;
    int i, j;
    while (true) {
        cin >> H >> W;
        if (H == 0 && W == 0) {
            break;
        }
        for (i = 0; i < H; i++) {
            if ((i % 2) == 0) {
                for (j = 0; j < W; j++) {
                    if ((j % 2) == 0) {
                        cout << "#";
                    }
                    else {
                        cout << ".";
                    }
                }
                cout << endl;
            }
            if ((i % 2) == 1) {
                for (j = 0; j < W; j++) {
                    if ((j % 2) == 0) {
                        cout << ".";
                    }
                    else {
                        cout << "#";
                    }
                }
                cout << endl;
            }
        }
        cout << endl;
    }
}
