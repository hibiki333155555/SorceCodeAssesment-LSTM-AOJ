def I(): return int(sys.stdin.readline())
def LI(): return list(map(int, sys.stdin.readline().split()))

def dfs(v, p, f=-1):
    global can
    can[v] = True
    for u in edges[v]:
        if u == p: continue
        if u == f or nuts[u]:
            for t in edges[u]:
                if t == v: continue
                if not nuts[t]:
                    dfs(u, v, t)
        else: dfs(u, v)
    return

n, m = LI()
edges = [[] for _ in range(n)]
for _ in range(n-1):
    x, y = LI()
    edges[x-1].append(y-1)
    edges[y-1].append(x-1)

nuts = [False] * n
can = [False] * n
depth = 0
for _ in range(m):
    a = I()
    nuts[a-1] = True
can[0] = True
dfs(0, -1)
