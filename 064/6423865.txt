#include <iostream>
#include <string>
using namespace std;
struct P {
    string name;
    int time;
};
P A[100000];
int tail, head, MAX;
void enqueue(int x){
    A[tail].time = x;
    if(tail + 1 == MAX) tail = 0;
    else tail ++;
}
int dequeue(){
    return A[head].time;
    if(head + 1 == MAX) head = 0;
    head++;
}
int main(){
    int n, q, sum = 0;
    P A[100000];
    cin >> n >> q;
    for(int i = 0; i < n; i++) cin >> A[i].name >> A[i].time;
    int head = 0, tail = n, MAX = n;
    while(head != tail){
        int a = dequeue();
        if(a <= q){
            sum += a;
            cout << A[head].name << sum << endl;
        }else{
            sum += q;
            enqueue(a - q);
        } 
    }
    
    return 0;
}
