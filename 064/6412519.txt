#include<stdio.h>

int main(void)
{
    int n,m,l;
    int x,y,z;
    
    scanf("%d %d %d",&n,&m,&l);
    int a[n][m];
    int b[m][l];
    long long c[n][l];
    for(x=0;x<n;x++)
    {
        for(y=0;y<m;y++)
        {
            scanf("%d",&a[x][y]);
        }
    }
    for(x=0;x<m;x++)
    {
        for(y=0;y<l;y++)
        {
            scanf("%d",&b[x][y]);
        }
    }
    for(x=0;x<n;x++)
    {
        for(y=0;y<l;y++)
        {
            c[x][y]=0;
            for(z=0;z<m;z++)
            {
                c[x][y]+=a[x][z]*b[z][y];
            }
            printf("%d",c[x][y]);
            if(y==l-1)
            {
                printf("\n");
            }
            else
            {
                printf(" ");
            }
        }
    }
    return 0;
}
