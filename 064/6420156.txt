#include <iostream>
#include <iomanip>
#include <vector>
#include <cmath>

using std::cin;
using std::cout; using std::endl;
using std::vector;

int main()
{
    cout << std::fixed << std::setprecision(6);

    int n;
    cin >> n;

    vector<int> x(n), y(n);
    for (int i = 0;i < n;++i)
        cin >> x[i];
    for (int i = 0;i < n;++i)
        cin >> y[i];

    double dist;

    dist = 0;
    for (int i = 0;i < n;++i)
        dist += std::abs(x[i] - y[i]);
    cout << dist << endl;

    dist = 0;
    for (int i = 0;i < n;++i)
        dist += ( x[i] - y[i] ) * ( x[i] - y[i] );
    cout << std::sqrt(dist) << endl;

    dist = 0;
    for (int i = 0;i < n;++i)
        dist += std::pow(std::abs( x[i] - y[i] ), 3);
    cout << std::pow(dist, (double)1/3) << endl;

    dist = 0;
    for (int i = 0;i < n;++i)
        dist = std::max(dist, std::fabs( x[i] - y[i] ) );
    cout << dist << endl;
}
