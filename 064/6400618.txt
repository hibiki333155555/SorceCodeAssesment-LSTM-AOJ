#include <stdio.h>
#include <iostream>
#include<algorithm>
using namespace std;
int n, m, a[100010], b[505], ans = 0;
int main()
{

	cin >> n;
	for (int i = 0; i < n; i++) cin >> a[i];
	cin >> m;
	for (int i = 0; i < m; i++) cin >> b[i];
	sort(a, a + n);
	sort(b, b + m);
	for (int i = 0; i < n; i++)
	{
		if (a[i] == a[i + 1]) a[i] = -1;
	}
	for (int i = 0; i < m; i++)
	{
		if (b[i] == b[i + 1]) a[i] = -1;
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (a[i] == b[j]&&a[i]!=0)
			{
				ans++;
				//cout << a[i] << b[j];
				break;
			}
		}
	}
	cout << ans << endl;
	return 0;
}
