#include <stdio.h>

void dfs(int from, int id, int idlistoo[], int count[], int G[][100000]);

void print(int x, int y, int idlistoo[]);

int main(void) {
  int idlistoo[100000];
  int G[100000][100000];
  int count[100000];
  int n, m;
  scanf("%d %d", &n, &m);

  int a, b;
  for (int i = 0; i < m; i++) {
    scanf("%d %d", &a, &b);
    G[a][count[a]++] = b;
    G[b][count[b]++] = a;
  }
  int id = 0;
  for (int i = 0; i < n; i++) {
    if (idlistoo[i] > 0) continue;
    dfs(i, ++id, idlistoo, count, G);
  }
  int q;
  scanf("%d", &q);
  int x, y;
  for (int i = 0; i < q; i++) {
    scanf("%d %d", &x, &y);
    print(x, y, idlistoo);
  }
}

void dfs(int from, int id, int idlistoo[], int count[], int G[][100000]) {
  idlistoo[from] = id;
  for (int i = 0; i < count[from]; i++) {
    int next_v = G[from][i];
    if (idlistoo[next_v] > 0) {
      continue;
    }
    dfs(next_v, id, idlistoo, count, G);
  }
}
void print(int x, int y, int idlistoo[]) {
  if (idlistoo[x] == idlistoo[y]) {
    printf("yes\n");
  } else {
    printf("no\n");
  }
}
