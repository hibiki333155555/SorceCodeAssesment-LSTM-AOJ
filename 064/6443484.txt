#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int main() {
   int n;
   cin >> n;
   n = 0;
   int b[n], f[n], r[n], v[n];

   int apart[4][3][10];
   for(int i=0; i<4; i++) {
      for(int j=0; j<3; j++) {
         for(int k=0; k<10; k++) {
            apart[i][j][k] = 0;
         }
      }
   }

   for(int i=0; i<n; i++) {
      cin >> b[i] >> f[i] >> r[i] >> v[i];
      apart[b[i]][f[i]][r[i]] += v[i];
   }

   for(int i=0; i<4; i++) {
      for(int j=0; j<3; j++) {
         for(int k=0; k<10; k++) {
            cout << " " << apart[i][j][k];
         }
         cout << "\n";
      }
      cout << "####################\n";
   }
}
