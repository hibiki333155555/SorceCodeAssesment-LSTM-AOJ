#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>

int main(){
    char ch[1000];
    int a = 0 ;
    int n;

    while(1)
    {
      
        scanf("%s",ch);        
        if(ch[n] == '0') break;

        for(int i = 0; ch[i] != '\0'; i++)
        {
            a += ch[i] - '0';
        }
    printf("%d\n",a);
    a = 0;
    }
    return 0;
}
