#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for (int i = 0; i < (n); ++i)

const int N = 100;
int main() {
    int M[N][N];
    int n, u, k, v;
    cin >> n;
    rep(i, n){
        rep(j, n){
            M[i][j] = 0;
        }
    }

    rep(i, n){
        cin >> u >> k;
        u--;
        rep(j, k){
            cin >> v;
            v--;
            M[u][v] = 1;
        }
    }

    rep(i, n){
        rep(j, n){
            if(j) cout << " ";
            cout << M[i][j];
        }
        cout << endl;
    }
    return 0;
}
