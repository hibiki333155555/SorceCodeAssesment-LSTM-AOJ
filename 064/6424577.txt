#include <iostream>

using namespace std;

void call(int n) {
    int i = 1;


    do {
        int x = i;
        if (x % 3 == 0) {
            cout << " " << i;
        } else {
            if (x % 10 == 3) {
                cout << " " << i;
            }
            while (x) {
                if (x % 10 == 3) {
                    cout << " " << i;
                }
                x /= 10;
            }
        }
    } while (++i <= n);
    cout << endl;
}

int main() {
    int n;
    cin >> n;
    call(n);
    return 0;
}
