void solve() {
  int n, m, v, a, b; cin >> n >> m >> v >> a >> b;
  if (n > m) {
    cout << "Yes\n";
    return;
  }
  v %= m;
  set<int> vs{v};
  FOR(i, 1, n) {
    v = (v * a + b) % m;
    if (!vs.emplace(v).second) {
      cout << "Yes\n";
      return;
    }
  }
  cout << "No\n";
}

int main() {
  int t; cin >> t;
  while (t--) solve();
  return 0;
}
