template <typename T>
vector<T> compress(vector<T> &v) {
  vector<T> val = v;
  sort(all(val)), val.erase(unique(all(val)), val.end());
  for (auto &&vi : v) vi = lower_bound(all(val), vi) - val.begin();
  return val;
}

void solve() {
  int n, m;
  cin >> n >> m;
  vector<string> s(n);
  rep(i, n) cin >> s[i];
  int csum[110][110] = {};
  rep(i, n) rep(j, m) {
    csum[i + 1][j + 1] =
        (s[i][j] == 'x') + csum[i + 1][j] + csum[i][j + 1] - csum[i][j];
  }
  auto sum = [&](int li, int lj, int ri, int rj, char c) {
    if (c == 'o')
      return csum[ri][rj] - csum[li][rj] - csum[ri][lj] + csum[li][lj];
    else
      return (ri - li) * (rj - lj) -
             (csum[ri][rj] - csum[li][rj] - csum[ri][lj] + csum[li][lj]);
  };

  int Min = INF;
  REP(i, 1, n) {
    chmin(Min, sum(0, 0, i, m, 'o') + sum(i, 0, n, m, 'x'));
    chmin(Min, sum(0, 0, i, m, 'x') + sum(i, 0, n, m, 'o'));
  }
  REP(i, 1, m) {
    chmin(Min, sum(0, 0, n, i, 'o') + sum(0, i, n, m, 'x'));
    chmin(Min, sum(0, 0, n, i, 'x') + sum(0, i, n, m, 'o'));
  }
  cout << Min << "\n";
}

int main() { solve(); }
