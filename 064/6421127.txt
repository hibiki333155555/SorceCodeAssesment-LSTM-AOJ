#include "cpp_library/string/str_009_manacher.cpp"
 
int main() {
    int N; cin >> N;
    string s; cin >> s;

    int ans = 0;
    Manacher<string, char> manacher(s, '#');
    for(int i=1; i<N; i++) {
        // i-1 と i の間に隙間がある
        int lhs = i, rhs = N - i;
        int len = min(lhs, rhs);

        int l = i - len, r = i + len;
        if(manacher.is_palindrome(l, r)) ans++;
    }
    cout << ans << endl;
    return 0;
}
