#include <iostream>
using namespace std;

int main(void){
    int x, y;
    int t;

    for (int i=1; i<=10000; i++) {
        cin >> x;
        cin >> y;

        if (x > y) {
            t = x;
            x = y;
            y = t;
        }

        cout << x << " " << y << endl;
    }
    
    return 0;
}
