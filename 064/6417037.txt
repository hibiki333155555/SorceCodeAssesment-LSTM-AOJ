#include<stdio.h>
#define MAX_N 100000
#define MAX_Q 50000

int S[MAX_N];
int T[MAX_Q];
int N, Q;

int binarySearch(int key)
{
  int ok = -1;
  int ng = N;
  int m;

  while(ok + 1 < ng){
    int m = (ok + ng) / 2;
    if(S[m] < key) ok = m;
    else ng = m;
  }

  return S[ng] == key;
}

int main()
{
  int cnt = 0;
  scanf("%d", &N);
  for(int i = 0; i < N; i++ ){
    scanf("%d", &S[i]);
  }
  scanf("%d", &Q);
  for(int i = 0; i < Q; i++ ){
    scanf("%d", &T[i]);
  }

  for(int i = 0; i < Q; i++ ){
    if(binarySearch(T[i])) cnt++;
  }

  printf("%d\n", cnt);
  return 0;
}
