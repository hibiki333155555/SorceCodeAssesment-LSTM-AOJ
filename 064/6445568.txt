using namespace std;
//分析见书《挑战程序设计竞赛2》上P

int main()
{	
	int n,i,j,t,sum=0;
	int a[200];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	
	for(i=0;i<n-1;i++)
	{
		for(j=0;j<n-1-i;j++)
		{
			if(a[j]>a[j+1])
			{
				sum++;
				t=a[j];
				a[j]=a[j+1];
				a[j+1]=t;
			}
		}
	}
	
	for(i=0;i<n;i++)
	{
		if(i==n-1)
			cout<<a[i]<<endl;
		else
		 	cout<<a[i]<<" ";
	
	}
	cout<<sum<<endl;
	
return 0;
}
