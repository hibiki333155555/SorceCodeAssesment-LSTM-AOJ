using namespace std;

void trace(vector<vector<int>> matrix) {
  REP(i, matrix.size()) {
    REP(j, matrix[i].size()) {
      cout << matrix[i][j];
      if(j != matrix[i].size() - 1) { cout << " "; }
    }
    cout << endl;
  }
}

signed main() {
  cin.tie(0);
  ios::sync_with_stdio(false);

  int n;
  cin >> n;

  vector<vector<int>> matrix(n, vector<int>(n, 0));

  REP(i, n) {
    int u, k;
    cin >> u >> k;
    REP(j, k) {
      int v;
      cin >> v;
      if(matrix[u - 1][v - 1] == 0) { matrix[u - 1][v - 1] = 1; }
    }
  }

  trace(matrix);

  return 0;
}
