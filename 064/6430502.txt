#include <bits/stdc++.h>
using namespace std;
void Print(vector<int>v){
	for(int i=0;i<v.size(); i++)
	{
		cout<<v[i]<<" ";
	}
	cout<<"\n";
}
void InsertionSort(vector<int>&a)
{
	int n=a.size(),j,temp;
	for(int i=0;i<n;i++)
	{
		j=i-1;
		temp=a[i];
		while(j>=0 && temp<a[j])
		{
			a[j+1]=a[j];
			j--;
		}
		a[j+1] = temp;
		Print(a);
	
	}
}
	
int main() {

	
	int n; cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
	}
       InsertionSort(v);
	
	return 0;
	
}
