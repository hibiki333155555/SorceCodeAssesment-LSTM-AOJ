#include <iostream>
using namespace std;
int main(){
    int i,n,a[100],j,tmp;
    cin >> n;
    for(i=0;i<n;i++){
        cin >> a[i]; 
    }
    for(i=1;i=n-1;i++){
        tmp = a[i];
        j=i-1;
    
        while(tmp<a[j] && j>=0){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = tmp;
        cout << a[j+1];
    }
    
}
