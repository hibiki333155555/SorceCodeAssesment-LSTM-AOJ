#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll pow(ll b, ll p, ll m){
    if (!p)
        return 1;
    if (p == 1)
        return b % m;
    ll ret = pow(b, p >> 1, m);
    ret = (ret * ret) % m;
    if (p & 1)
        ret = (ret * b) % m;
    return ret % m;
}
int main ()	{
	ios_base::sync_with_stdio(0);
	cin.tie(0);cout.tie(0);
	int a,b;
	cin>>a>>b;
	cout<<pow(a,b,1000000007)<<'\n';}
