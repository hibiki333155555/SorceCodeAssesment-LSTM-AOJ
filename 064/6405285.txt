#include<stdio.h>

int main(void)
{
    int h,w;
    int i,j;
    while(1)
    {
        scanf("%d %d",&h,&w);
        if(h==0,w==0)break;
        for(i=1;i<=h;i++)
        {
            for(j=1;j<=w;j++)
            {
                if(i==1||i==h||j==1||j==w)
                {
                    printf("#");
                }else{
                    printf(".");
                }
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
