#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,e;
    vector<int> src;
    cin>>n;
    while(n--){
        cin>>e;
        src.push_back(e);
    }
    cin>>n;
    int c = 0;
    while(n--){
        cin>>e;
        for(auto& s:src){
            if(s==e) c++;
        }
    }
    cout << c << endl;
}
