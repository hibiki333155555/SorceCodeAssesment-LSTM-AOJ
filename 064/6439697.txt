#include<bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)
//using ll=long long;
int ans=0,n,a[100000],maxi=-1;
int sum(int a[],int &maxi){
    int max=0,z=maxi+1;
    for(int i=z;i<n;i++){
        if(max<a[i]){
            max=a[i];
            maxi=i;
            //i++;
        }
    }ans++;
    cout<<maxi<<"\n";
    if(maxi<n-1){
        return sum(a,maxi);
    }else return ans;
}
int main(){
    cin>>n;
    rep(i,n)cin>>a[i];
    cout<<sum(a,maxi)<<"\n";
}
