#include <stdio.h>
int main()
{
    int n, m;
    scanf("%d %d", &n,&m);
    int a[n][m];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            scanf("%d", &a[i][j]);
        }
    }
    for(int i=0;i<m;i++)
    {
        int k;
        scanf("%d",&k);
        for(int j=0;j<n;j++)
        {
            a[j][i] *= k;
        }
    }
    int s=0;
    for(int i=0;i<n;i++)
    {
        s=0;
        for(int j=0;j<m;j++)
        {
            s+=a[i][j];
        }
        printf("%d\n",s);
    }
    return 0;
}
