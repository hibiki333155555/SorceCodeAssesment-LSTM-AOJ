#include<iostream>
#include<cmath>
#include<stdio.h>
using namespace std;
int main() {
    int n;
    while(1) {
        cin >> n;
        if(n == 0) {
            break;
        }
        int sum = 0;
        int sum2 = 0;
        for(int count = 0; count < n; count++) {
            int point;
            cin >> point;
            sum += point;
            int exp = point * point;
            sum2 += exp;
        }
        printf("%.8lf", sqrt(sum2 * 1.00000000 / n - (sum * 1.00000000 / n) * (sum * 1.00000000 / n)));
        cout << endl;
    }
}
