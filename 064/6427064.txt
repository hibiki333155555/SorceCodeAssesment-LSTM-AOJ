// D - Print a Frame

#include <iostream>
using namespace std;

int main() {
    int h,w;
    while (cin >> w >>h) {
        if ((h>=3 and h<=300) and (w>=3 and w<=300)) 
        {
            for (int i=0; i<w+1; i++) {
                cout << '#';
            }
            cout << endl;
            for (int i=0; i<w-2; i++) {
                cout << '#';
                for (int j=0; j<h-2; j++) {
                    cout << '*';
                }
                cout << '#'<< endl;
            }
            for (int i=0; i<w+1; i++) {
                cout << '#';
            }
            cout << endl;
        }
        else if (h==0 and w==0) break;
    }
}
