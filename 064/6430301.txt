#include <bits/stdc++.h>
using namespace std;
 
int main()
{	int n;
	int swaps = 0;
	cin >>n;
	vector<int> v(n);
	for(int i = 0; i<n; i++){
		cin >> v[i];
	}	
		int pos;
		for(int i = 0; i<n-1; i++){
			pos = i;
			for(int j = i+1; j<n;j++){
				if(v[pos] > v[j]){
					pos = j;
				}
			}
			swap(v[pos],v[i]);
			if(pos!=i){
				swaps = swaps + 1;	
			}
		
		}
	for(int i = 0; i<n; i++){
		cout << v[i] << " ";
	}
	cout << "\n";
	cout << swaps;
		
		
    return 0;
}
