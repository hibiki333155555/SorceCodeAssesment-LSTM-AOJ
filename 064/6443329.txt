#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int check(int x, int n) {
   if(++x > n) return 1;
}

int main() {
   int n, i=0;
   cin >> n;

   while(true) {
      i++;
      int x = i;

      if(x % 3 == 0) {
         cout << " " << i;
         if(check(i, n) == 1) break;
         else continue;
      }

      while(x!=0) {
         if(x % 10 == 3) {
            cout << " " << i;
            if(check(i, n) == 1) break;
            else continue;
         }
         x /= 10;
      }
      if(check(i, n) == 1) break;
      else continue;
   }
}
