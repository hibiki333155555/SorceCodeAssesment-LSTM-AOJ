#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include<string.h>

int main()
{
	char s[1200];
    scanf("%s",s);
    int n=strlen(s);
    for(int i=0; i<n; i++){
        if(s[i]>='a'&&s[i]<='z'){
        s[i]+='A'-'a';
        }
        else if(s[i]>='A'&&s[i]<='Z'){
        s[i]-='A'-'a';
        }
    }
    printf("%s",s);
	return 0;
}
