#include <bits/stdc++.h>

using namespace std;

int main(){
    int n, k;
    cin >> n >> k;
    vector<int> A(n);
    for (auto &a: A)
        cin >> a;
    
    vector<int> C(7);
    for (const auto &a: A)
        C[a]++;

    vector<int> f = {0, 5, 6, 6, 6, 6, 6};

    for (int i = 1; i <= 6; i++){
        int c = min(k, C[i]);
        k -= c;
        C[i] -= c;
        C[f[i]] += c;
    }

    int ans = 0;
    for (int i = 1; i <= 6; i++)
        ans += C[i] * i;
        
    cout << ans << endl;
}
