/* function */
template<class T>
T GCD(T a, T b) {
	if (b != 0) return GCD(b, a % b);
	return a;
}
template<class T>
T LCM(T a, T b) {
	return a / GCD(a, b) * b;
}

const int dy[4] = { 0, 1, -1, 0 };
const int dx[4] = { 1, 0, 0, -1 };

int main(){
	int n,k,tu=0;
	ll ans=0;
	cin >> n >> k;
	vector<int> A(n),dice(6,0);
	rep(i,0,n) {
		cin >> A[i];
		dice[A[i]-1]++;
		if(A[i]==1) tu+=2;
		else if(A[i]!=6) tu++;
		ans+=A[i];
	}
	if(tu<=k) {
		cout << 6*n << endl;
		return 0;
	}
	//rep(i,0,6) cout << dice[i] << endl;
	rep(i,0,5){ //1~5までの目を探索
		int cnt=dice[i];
		if(k<=cnt){ //つつける回数がcnt以下
			if(i==0&&dice[i]){
				ans+=4*k;
				break;
			}else if(dice[i]){
				ans+=(6-(i+1))*k;
				break;
			}
		}else{
			if(i==0){
				ans+=dice[i]*4;
				//dice[4]+=cnt;
				k-=dice[i];
				continue;
			}
			ans+=dice[i]*(6-(i+1));
			k-=dice[i];
		}
	}
	cout << ans << endl;
}
