#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   int n;
   string s;
   cin >> n >> s;
   int j, o, i;
   for (int t = 0; t < n; t++)
   {
      j = count(s.begin(), s.end(), 'J');
      o = count(s.begin(), s.end(), 'O');
      i = count(s.begin(), s.end(), 'I');
   }
   for (int t = 0; t < j; t++)
      cout << "J";

   for (int t = 0; t < o; t++)
      cout << "O";

   for (int t = 0; t < i; t++)
      cout << "I";

   cout << endl;
   return 0;
}
