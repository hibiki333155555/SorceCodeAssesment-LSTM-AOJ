#include <iostream>
#include <vector>
#include <queue>
using namespace std;
priority_queue <int,vector <int>,less <int>> q;
string s;
int main () {
	while (cin >> s && s != "end") {
		int x;
		if (s == "insert") {
			cin >> x;
			q.push (x);
		}
		else {
			cout << q.top () << endl;
			q.pop ();
		}
	}
    return 0;
}
