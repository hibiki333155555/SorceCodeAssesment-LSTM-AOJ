#include <bits/stdc++.h>
using namespace std;

void print(vector<int>v)
{
	for(int i=0;i<v.size();i++)
	{
		if(i!=v.size()-1)
		{
		cout<<v[i]<<" ";
		
		}
		else
		{
		cout<<v[i];
		}
		}
	cout<<"\n";
}

void Bubble(vector<int> &a)
{
	int n=a.size();
	int count=0;
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;j<n-1-i;j++)
		{
			if(a[j]>a[j+1])
			{
				swap(a[j],a[j+1]);
				count++;
			}
			
		}
	
	}
		print(a);
		cout<<count;
		cout<<"\n";
	
}


int main() {
	int s;
	cin>>s;
	vector<int>v(s);
	for(int i=0;i<s;i++)
	{
		cin>>v[i];
	}
	Bubble(v);
	return 0;
}
