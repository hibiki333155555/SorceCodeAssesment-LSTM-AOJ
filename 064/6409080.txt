#include <bits/stdc++.h>
using namespace std;

int V, E;
int const INF = numeric_limits<int>::max() / 2;
vector<vector<int>> G, dp;

int recur(int S, int v)
{
  if (dp.at(S).at(v) != -1) {
    return dp.at(S).at(v);
  }
  if (S == 0 && v == 0) {
    return dp.at(S).at(v) = 0;
  }
  if (!(S & (1 << v))) {
    return dp.at(S).at(v) = INF;
  }
  int res = INF;
  for (int u = 0; u < V; u++) {
    res = min(res, recur((S & ~(1 << v)), u) + G.at(u).at(v));
  }
  return dp.at(S).at(v) = res;
}

int main()
{
  cin >> V >> E;
  G.assign(V, vector<int>(V, INF));
  G.at(0).at(0) = 0;
  for (int i = 0; i < E; i++) {
    int s, t, d;
    cin >> s >> t >> d;
    G.at(s).at(t) = d;
  }
  dp.assign((1 << V), vector<int>(V, -1));
  int ans = recur((1 << V) - 1, 0);
  if (ans >= INF)
    ans = -1;
  cout << ans << '\n';
}
