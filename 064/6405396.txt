#include<stdio.h>
int main(void)
{
    int a,b,i,j,k;
    for(i = 0 ;i < 1000000000;i++)
    {
        scanf("%d %d",&a,&b);
        if(a == 0 && b == 0)
        {
            break;
        }
        for(j = 0;j <a;j++)
        {
            for(k = 0;k<b;k++)
            {
                printf("#");
            }
            printf("\n");
        }
    }
    return 0;
}
