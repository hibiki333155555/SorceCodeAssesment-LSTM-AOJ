#include <bits/stdc++.h>
using namespace std;

int main(){
    vector<vector<int>> vec1(3,vector<int>(10));
    vector<vector<int>> vec2(3,vector<int>(10));
    vector<vector<int>> vec3(3,vector<int>(10));
    vector<vector<int>> vec4(3,vector<int>(10));
    int n;
    cin>> n;
    for(int i=0; i<n; i++){
        int b,f,r,v;
        cin>> b >> f >> r >> v;
        if(b==1) vec1.at(f-1).at(r-1)+=v;
        else if(b==2) vec2.at(f-1).at(r-1)+=v;
        else if(b==3) vec3.at(f-1).at(r-1)+=v;
        else if(b==4) vec4.at(f-1).at(r-1)+=v;
    }
    for(int i=0; i<3; i++){
        for(int j=0; j<10; j++){
            if(i==2&&j==9) cout<< vec1.at(i).at(j);
            else cout<<vec1.at(i).at(j)<<" ";
        }
        cout<<endl;
    }
    for(int i=0; i<20; i++){
        cout<<"#";
    }
    cout<<endl;
    for(int i=0; i<3; i++){
        for(int j=0; j<10; j++){
            if(i==2&&j==9) cout<< vec2.at(i).at(j);
            else cout<<vec2.at(i).at(j)<<" ";           
        }
        cout<<endl;
    }
    for(int i=0; i<20; i++){
        cout<<"#";
    }
    cout<<endl;
    for(int i=0; i<3; i++){
        for(int j=0; j<10; j++){
            if(i==2&&j==9) cout<< vec3.at(i).at(j);
            else cout<<vec3.at(i).at(j)<<" ";
        }
        cout<<endl;
    }
    for(int i=0; i<20; i++){
        cout<<"#";  
    }
    cout<<endl;
    for(int i=0; i<3; i++){
        for(int j=0; j<10; j++){
            if(i==2&&j==9) cout<<vec4.at(i).at(j);
            else cout<<vec4.at(i).at(j)<<" ";
        }
        cout<<endl;
    }
}
