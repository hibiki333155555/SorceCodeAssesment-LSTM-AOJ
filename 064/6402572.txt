import java.util.Scanner;

// 成績クラス
class Grades {
    // 中間，期末，再試験のフィールド
    private int midtest, finaltest, retest;

    public Grades(int midtest, int finaltest, int retest) {
        this.midtest = midtest;
        this.finaltest = finaltest;
        this.retest = retest;
    }

    public void setMidtest(int midtest) {
        this.midtest = midtest;
    }

    public void setFinaltest(int finaltest) {
        this.finaltest = finaltest;
    }

    public void setRetest(int retest) {
        this.retest = retest;
    }

    public int getMidtest() {
        return midtest;
    }

    public int getFinaltest() {
        return finaltest;
    }

    public int getRetest() {
        return retest;
    }
}

// 生徒のクラス
class Student {
    // 学籍番号と成績情報
    private int number;
    private char result;
    private Grades grades;

    public Student(int midtest, int finaltest, int retest, int number) {
        grades = new Grades(midtest, finaltest, retest);
        this.number = number;
        setResult();
    }

    public char getResult() {
        return result;
    }

    // テストの結果を判定する
    public void setResult() {
        int midtest = grades.getMidtest();
        int finaltest = grades.getFinaltest();
        int sum = midtest + finaltest;
        int retest = grades.getRetest();

        if (midtest == -1 || finaltest == -1) {
            result = 'F';
        } else if (sum >= 80) {
            result = 'A';
        } else if (sum >= 65) {
            result = 'B';
        } else if (sum >= 50) {
            result = 'C';
        } else if (sum >= 30) {
            if (retest >= 50) {
                result = 'C';
            } else {
                result = 'D';
            }
        } else {
            result = 'F';
        }
    }

    public int getNumber() {
        return number;
    }

    public void setNumber(int number) {
        this.number = number;
    }

    public Grades getGrades() {
        return grades;
    }

    public void setGrades(Grades grades) {
        this.grades = grades;
    }

    // 成績結果を表示
    public void showResults() {
        System.out.println(result);
    }
}


public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 人数をカウント
        int numbers = 0;
        Student[] students = new Student[50];

        while (true) {
            // 値を取得
            int midtest, finaltest, retest;
            midtest = sc.nextInt();
            finaltest = sc.nextInt();
            retest = sc.nextInt();

            // 終了条件
            if (midtest == -1 && finaltest == -1 && retest == -1) {
                break;
            }

            // データを挿入する
            Student student = new Student(midtest, finaltest, retest, numbers);
            students[numbers] = student;
            numbers++;
        }

        // 結果を表示する
        for (int i = 0; i < numbers; i++) {
            students[i].showResults();
        }
    }
}
