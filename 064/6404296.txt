#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(int)(n);i++)

bool solve(int i,int rem,int n,vector<int> A){
    //rem:残り
    if(rem==0)  return true;
    // 再帰の出口
    if(i>=n)    return false;

    // その値を使わなかった場合と使った場合の分岐部(木構造のイメージでいい)
    bool result = solve(i+1,rem,n,A)||solve(i+1,rem-A[i],n,A);
    return result;
}

int main(){
    int n,q;
    cin>>n;
    vector<int> A(n);
    rep(i,n)    cin>>A[i];
    cin>>q;
    vector<int> m(q);
    rep(i,q)    cin>>m[i];
    
    rep(i,q){
        cout<<(solve(0,m[i],n,A) ? "yes":"no")<<endl;
    }
    
    return 0;
}
