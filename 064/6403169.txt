#include <stdio.h>

int main(void){
    int n;
    int a, min, max;
    long long int sum = 0;

    scanf("%d", &n);

    scanf("%d", &a);
    min = max = sum = a;

    while(n-- > 1){
        scanf("%d", &a);

        if(a < min) min = a;
        if(a > max) max = a;
        sum += a;
    }
    printf("%d %d %lld\n", min, max, sum);

    return 0;
}
