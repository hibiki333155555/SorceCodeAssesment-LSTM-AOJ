#include <bits/stdc++.h>
using namespace std;

int main() {
  int n; cin >> n;
  int tmp = n;
  vector<pair<int, int>> ans;
  for (int i = 2; i * i <= n; i++) {
    if (n % i != 0) continue;

    int ex = 0;
    while (n % i == 0) {
      ex++;
      n /= i;
    }

    ans.push_back({i, ex});
  }
  
  if (n != 1) ans.push_back({n, 1});

  cout << tmp << ":";
  for (auto x : ans) {
    for (int i = 0; i < x.second; i++) {
      cout << ' ' << x.first;
    }
  }
}
