#include<iostream>
using namespace std;

int main()
{
	int n,x,cou[1000000]={},t=0;

    while(true){
        cin>>n>>x;
        if(n==0 && x==0)break;
        for(int i=1;i<=n;++i){
            for(int j=1;j<=n;++j){
                for(int k=1;k<=n;++k){
                    if(x==(i+j+k) && i!=j && j!=k && k!=i)cou[t]+=1;
                }
            }
        }
        t+=1;
    }

    for(int i=0;i<t;++i){
        cout<<cou[i]/6<<endl;
    }

return 0;
}
