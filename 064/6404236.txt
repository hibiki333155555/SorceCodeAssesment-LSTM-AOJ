import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        while (true) {
            String str = sc.next();

            if (str == "0") {
                break;
            }

            int sum = 0;
            for (int i = 0; i < str.length(); i++) {
                int num = str.charAt(i) - '0';
                sum += num;
            }
            System.out.println(sum);
        }
    }
}
