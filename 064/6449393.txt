#include <bits/stdc++.h>
using namespace std;
int main(void){
    long long A,B;cin>>A>>B;
    long long X=1;
    while(X<A||X<B){
        X*=2;
    }
    X/=2;
    int ans=0;
    while((X&A)==(X&B)){
        ans++;
        X/=2;
        if(X==0) break;
    }
    cout<<ans<<endl;
    
    
    return 0;
}
