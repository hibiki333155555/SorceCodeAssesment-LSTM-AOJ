#include <bits/stdc++.h>
using namespace std;
const double PI =acos(-1);
const long double EPS = 1e-14;

int main(){
  stack<long> sta;
  string s;
  getline(cin, s);
  for(long i = 0; i < s.size(); i++){
    if(s[i] == ' '){
      continue;
    }
    if(isdigit(s[i])){
      sta.push(s[i] - '0');
    }else{
      long y = sta.top();
      sta.pop();
      long x = sta.top();
      sta.pop();
      switch(s[i]){
        case '+':
          x += y;
          sta.push(x);
          break;
        case '-':
          x -= y;
          sta.push(x);
          break;
        case '*':
          x *= y;
          sta.push(x);
          break;
      }
    }
  }
  cout << sta.top() << endl;
  return 0;
}
