{

  map<string, long long>m;
  map<string, long long>::iterator itr1, itr2;
  itr2 = m.end();
  while(q--)
  {
    long long f; cin >> f;
    string key;
    long long x;

    switch(f)
    {
      case 0:
        cin >> key >> x;
        if(m.find(key) == m.end())m.insert(pair<string, long long>(key, x));
        break;
      case 1:
        cin >> key;
        if(m.find(key) != m.end())cout << m[key] << endl;
        else cout << 0 << endl;
        break;
      case 2:
        cin >> key;
        if(m.find(key) != m.end())m.erase(key);
        break;
      case 3:
        string l, r; cin >> l >> r;
        itr1 = m.lower_bound(l);
        itr2 = m.upper_bound(r);
        //cout << itr1->first << " " << itr2->first << endl;
        for (; itr1 != itr2 && itr1->first.compare(l) >= 0 && itr1->first.compare(r) <= 0; itr1++)
        {
          
          cout << itr1->first << " " << itr1->second << endl;
        }
        
        break;

    }

  }
  
  return 0;
}
