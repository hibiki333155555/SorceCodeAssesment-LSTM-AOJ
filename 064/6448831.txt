#include<iostream>
#include<vector>
using namespace std;
int main()
{
    int n,q;
    cin>>n>>q;
    vector<int>v[n];
    for(int i=0;i<q;i++)
    {
        int op,num,x;
        cin>>op>>num;
        if(op==0)
        {
            cin>>x;
            v[num].push_back(x);
        }
        else if(op==1)
        {
            int s=v[num].size();
            if(s==0)
            {
                cout<<endl;
            }
            else {

            for(int i=0;i<v[num].size();i++)
            {
                cout<<v[num][i]<<" ";
            }
            cout<<endl;
            }
        }
         else if(op==2)
   {
       v[num].pop_back();
   }

    }

    return 0;
}
