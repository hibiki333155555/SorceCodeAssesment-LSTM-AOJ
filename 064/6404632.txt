#include <iostream>
#include <math.h>

using namespace std;

int main() {

    int a,b,c;
    int count;
    int flag;
    double n;

    cin >> a >> b >> c;

    n = sqrt(c);

    for(int i = a; i <= n;i++){
        if(c%i == 0){
            if(i != flag){
                count++;
                flag = c/i;
            }
            if(c/i <= b && c/i != flag){
                count++;
                flag = c/i;
            }
        }
    }

    cout << count << "\n";

}
