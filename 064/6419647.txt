import java.util.*;
public class Main {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int num = sc.nextInt();
    int key = 0;
    String word = "";
    int no = 0;
    HashMap<String, Integer> meow = new HashMap<>();
    for(int a = 0; a<num; a++){
      no = sc.nextInt();
      word = sc.next();
      if(no==1){
        System.out.println(meow.get(word));
      }
      else{
        key= sc.nextInt();
        meow.put(word, key);
      }
    }
  }
}
