int calc(string S){
  if(S.size() == 1){
    return S[0] - 'A';
  }
  int x = S[0] - 'A';
  int y = S[1] - 'A';
  return 26 + x*26 + y;
}

int main(){
  string A, B;
  int a, b;
  cin >> A >> a >> B >> b;
  ll X = calc(B) - calc(A) + 1;
  ll Y = b - a + 1;
  cout << X*Y << endl;
}
