#include<bits/stdc++.h>
using namespace std;

int main() {
  int N;
  cin >> N;
  vector<int> A(N);
  for(int i = 0; i < N; i++) cin >> A[i];
  int Q;
  cin >> Q;
  vector<bool> result(40001, false);
  for(int n = 0; n < (1 << N); n++) {
      int sum = 0;
      for(int i = 0; i < N; i++) {
          if(n & (1 << i)) sum += A[i];
      }
      result[sum] = true;
  }
  for(int q = 0; q < Q; q++) {
    int m;
    cin >> m;
    cout << (result[m] ? "yes" : "no") << '\n';
  }
}
