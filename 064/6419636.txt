
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        List<Integer> x = new ArrayList<>();

        while (scan.hasNextInt()) {
            int i = scan.nextInt();
            if (i == 0) {
                break;
            }
            x.add(i);


        }
        for (int j = 0; j < x.size(); j++) {
            System.out.println("Case " + (j + 1) + ": "+ x.get(j));
        }

    }
}
