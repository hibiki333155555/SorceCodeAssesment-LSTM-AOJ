#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef long double LD;

#define rep(i,n) for(LL i=0;i<(n);i++)

int main(){
    LD n, m, k, x;
    LD max = 0;

    cin >> n >> m >> k >> x;

    rep(i,n){
        LD a; cin >> a;

        if(a/m > max){
            max = a/m;
        }

    }

    LD pos = max * k;

    if(pos >= x){
        cout << "Yes" << endl;
    }
    else{
        cout << "No" << endl;
    }

    return 0;
}
