import scala.io.StdIn

object Main {
  def main(args: Array[String]): Unit = {
    val n = StdIn.readInt()
    val table = collection.mutable.Map.empty[Int, String]
    for {
      i <- 1 to n;
      (j, v) <- (1 to n).zip(StdIn.readLine().split(" ").map(_.toInt))
    } {
      if (!table.contains(v)) table += v -> s"$i $j"
    }
    val q = StdIn.readInt()
    for (_ <- 1 to q) {
      println(table.getOrElse(StdIn.readInt(), "-1"))
    }
  }
}
