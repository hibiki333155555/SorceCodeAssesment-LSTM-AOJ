#include<stdio.h>
#include<string.h>

int main(void)
{
    int n,taro=0,hanako=0,i;
    char a[101],b[101];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%s %s",a,b);
        if(strcmp(a,b)==0)
        {
            taro++;hanako++;
        }
        else if(strcmp(a,b)>0)
        {
            taro+=3;
        }
        else if(strcmp(a,b)<0)
        {
            hanako+=3;
        }
    }
    printf("%d %d\n",taro,hanako);
    
    return 0;
}
