#include <iostream>
#include <algorithm>
#include <vector>
#include <string>

using namespace std;
using ull = unsigned long long;

void solve(){
    int N; cin >> N;
    string s; cin >> s;

    const ull B = 998244353;
    vector<ull> powB(N + 1), sumH(N + 1), invH(N + 2);
    powB[0] = 1, sumH[0] = invH[N + 1] = 0;
    for(int i = 0; i < N; i++) powB[i + 1] = powB[i] * B;
    for(int i = 0; i < N; i++) sumH[i + 1] = sumH[i] * B + (s[i] - 'a' + 1);
    for(int i = N + 1; i >= 2; i--) invH[i - 1] = invH[i] * B + (s[i - 2] - 'a' + 1);

    int ans = 0;
    for(int i = 1; i < N; i++){
        int len = min(i, N - i);
        ull HL = 0, HR = 0;

        HL = invH[i - len + 1] - invH[i + 1] * powB[len];
        HR = sumH[i + len] - sumH[i] * powB[len];

        //cout << i << " : "HL << " " << HR << "\n";

        if(HL == HR){
            ans++;
        }
    }

    cout << ans << "\n";
}

int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    solve();
    return 0;
}
