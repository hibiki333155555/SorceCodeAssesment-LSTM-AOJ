#include <bits/stdc++.h>
using namespace std;

void selection(vector<int>v){
	int n=v.size();
	int min,count=0;
	for(int i=0;i<n;i++){
		min=i;
		for(int j=i+1;j<n;j++){
			if(v[j]<v[min])
				min=j;
		}
		if(i!=min){
		swap(v[min],v[i]);
		count++;
		}
	}
	
	for(int i=0;i<n;i++){
		cout<<v[i];
		if(i!=n)
		cout<<" ";
	}
		
	cout<<"\n";
	cout<<count;
}

int main()
{
	int n;
	cin>>n;
	
	vector<int>v(n);
	for(int i=0;i<n;i++)
		cin>>v[i];
		
	selection(v);
	return 0;
}
