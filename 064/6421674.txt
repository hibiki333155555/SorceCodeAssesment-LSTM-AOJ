#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define rep(i,a,b) for(int i=a;i<b;i++)
#define fore(i,a) for(auto &i:a)

vector<int> dp;
vector<int> dp2;
int INF = 100000000;

int main() {
  vector<int> tetra(181);
  rep(i,1,181) {
    tetra.at(i) = i*(i+1)*(i+2)/6;
  }
  
  while (true) {
    int num;
    cin >> num;
    if (num==0) break;
    
    vector<int> dp(num+1, INF);
    vector<int> dp2(num+1, INF);
    rep(i,0,num+1) {
      dp.at(i) = i;
      dp2.at(i) = i;
    }
    
    rep(i,0,num+1) {
      rep(j,2,181) {
        if (i-tetra.at(j) >= 0) {
          dp.at(i) = min(dp.at(i), dp.at(i-tetra.at(j))+1);
          if (tetra.at(j) % 2 == 1) {
            dp2.at(i) = min(dp2.at(i), dp2.at(i-tetra.at(j))+1);
          }
        }
      }
    }
    
    cout << dp.at(num) << " " << dp2.at(num) << endl;
  }
  return 0;
}
