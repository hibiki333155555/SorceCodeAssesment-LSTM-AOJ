/*-- 入力処理&出力 --*/
function main(raw) {
    // 入力処理
    const args = raw.split("\n");
    const input = args[0].split(' ').map(Number);
    // 計算
    const output = input*input*input;
    // 表示
    console.log(`${output}`);
}
