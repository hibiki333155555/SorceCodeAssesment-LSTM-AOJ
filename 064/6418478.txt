#include <bits/stdc++.h>
using namespace std;

int main(){
        int n=0; cin>>n;
        int hoge=0, j=0;
        int A[n+1];
        for(int i=0;i<n;i++) cin>>A[i];
        for(int s=0;s<n;s++){
                cout<<A[s]<<endl;
        }
        for(int i=1;i<n;i++){
                hoge = A[i];
                j=i-1;
                while(j>=0&&A[j]>hoge){
                        A[j+1]=A[j];
                        j--;
                        A[j+1]=hoge;
                }
                for(int s=0;s<n;s++){
                        cout<<A[s]<<endl;
                }
        }
        return 0;;
}
