#include <bits/stdc++.h>
#define lol long long
using namespace std;
int main(){
    int n,m,l;
    cin>>n>>m>>l;
    int a[n][m];
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin>>a[i][j];
        }
    }
    int b[m][l];
    for(int i=0;i<m;i++){
        for(int j=0;j<l;j++){
            cin>>b[i][j];
        }
    }
    int c[n][l] = {};
    for(int i=0;i<n;i++){
        for(int j=0;j<l;j++){
            for(int k=0;k<m;k++){
                c[i][j] += a[i][k]*b[k][j];
            }
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<l-1;j++){
            cout<<c[i][j]<<" ";
        }
        cout<<c[i][l-1]<<"\n";
    }
    return 0;
}
