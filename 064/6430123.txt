#include <string>
#include <vector>
#include <iostream>
using namespace std;

int main() {
	int N;
	cin >> N;
	vector<string> S(N);
	for (int i = 0; i < N; i++) {
		cin >> S[i];
	}
	int L = S[0].size();
	vector<bool> ok(N, true);
	for (int i = 0; i < N; i++) {
		for (int j = i + 1; j < N; j++) {
			int cnt = 0;
			for (int k = 0; k < L; k++) {
				if (S[i][k] != S[j][k]) {
					cnt += 1;
				}
			}
			if (cnt <= 1) {
				ok[i] = false;
				ok[j] = false;
			}
		}
	}
	int answer = 0;
	for (int i = 0; i < N; i++) {
		if (!ok[i]) {
			answer += 1;
		}
	}
	cout << answer << endl;
	return 0;
}
