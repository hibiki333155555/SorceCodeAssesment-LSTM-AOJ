#include <bits/stdc++.h>
using namespace std;

bool isPrime(int N) {
    bool primeFlag = true;
    for(int i = 2;i * i <= N;i++) {
        primeFlag = true;
        if(N % i == 0)primeFlag = false;
    }
    if(primeFlag) return true;
    else return false;
}

int main() {
    int n; cin >> n;
    int number;
    int count = 0;
    for(int i = 0;i<n;i++) {
        cin >> number;
        isPrime(number);
        if(isPrime) count++;
    }
    cout << count << endl;
}
