#include<iostream>

using namespace std;

int main()
{
    int n = 1;
    cin>>n;

    int max1 = -1000000;
    int min1 = 1000000;
    int sum1 = 0;
    for(int i = 0;i < n; i++)
    {
        int x = 0;
        cin>>x;
        if(max1 < x)
        {
            max1 = x;
        }

        if(min1 > x)
        {
            min1 = x;
        }
        sum1 += x;
    }
    cout<<min1<<" "<<max1<<" "<<sum1<<endl;
    return 0;
}
