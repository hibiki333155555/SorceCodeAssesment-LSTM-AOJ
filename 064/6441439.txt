#include<bits/stdc++.h>

#define ll long long

#define fi first

int mod=1e9+7;

int main(){
    cin.tie(nullptr);
    ios_base::sync_with_stdio(false);

    int n; cin>>n;

    bool v[4][13]={};
    char c[]={'S','H','C','D'};

    for(int a=0;a<n;a++){
        char m; cin>>m;
        int x; cin>>x;
        for(int i=0;i<4;i++){
            if(m==c[i]){
                v[i][x-1]=1;
            }
        }
    }

    for(int i=0;i<4;i++){
        for(int j=0;j<13;j++){
            if(!v[i][j]){
                cout<<c[i]<<" "<<j+1<<"\n";
            }
        }
    }

    return 0;
}
