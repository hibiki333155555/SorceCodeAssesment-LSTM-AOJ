#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

int main(){
    ll n = cinll();
    ll maxv = -1;
    ll v = -1;
    ll minv = INF32;

    while(n--){
        v = cinll();
        maxv = max(maxv, v - minv);
        minv = min(minv, v);
    }
    if(maxv!=0)
    coutll(maxv);
    else
    coutll(-1);
}
