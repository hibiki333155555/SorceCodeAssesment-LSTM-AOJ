#include <stdio.h>
#include <string.h>

int main(){
    char w[11],t[1000];
    char *start,*re;
    int cnt=0,i=0;
    char end[]="END_OF_TEXT";
    
    scanf("%s",w);

    while(1){
        scanf("%s",t);
        if(strcmp(t,end)==0){
            break;
        }
        start=t;
        do{
            re=strstr(start,w);
            if(re!=NULL){
                re++;
                cnt++;
                start=re;
            }
        }while(re!=NULL);
    }
    printf("%d\n",cnt);
}
