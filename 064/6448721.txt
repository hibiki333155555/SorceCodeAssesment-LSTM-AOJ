#include <bits/stdc++.h>
#define endl '\n'
 
using namespace std;
 
int r, c, q;
 
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    while (cin >> r >> c >> q, r|| c || q) {
        int n = ceil(log2(r + 1)), m = ceil(log2(c + 1));
        int a[r][c][n + 1][m + 1];
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j ++) {
                cin >> a[i][j][0][0];
            }
        }
        for (int k = 1; k < n; k ++) {
            for (int i = 0; i < r; i ++) {
                for (int j = 0; j < c; j ++) {
                    int u = min(r - 1, i + (1 << (k - 1)));
                    a[i][j][k][0] = min(a[i][j][k - 1][0], a[u][j][k - 1][0]);
                }
            }
        }
        for (int s = 1; s < m; s ++) {
            for (int i = 0; i < r; i ++) {
                for (int j = 0; j < c; j ++) {
                    int k = min(c - 1, j + (1 << (s - 1)));
                    a[i][j][0][s] = min(a[i][j][0][s - 1], a[i][k][0][s - 1]);
                }
            }
            for (int k = 1; k < n; k ++) {
                for (int i = 0; i < r; i ++) {
                    for (int j = 0; j < c; j ++) {
                        int u = min(r - 1, i + (1 << (k - 1)));
                        a[i][j][k][s] = min(a[i][j][k - 1][s], a[u][j][k - 1][s]);
                    }
                }
            }
        }
        while (q --) {
            int r1, c1, r2, c2; cin >> r1 >> c1 >> r2 >> c2;
            int k = log2(r2 - r1 + 1), s = log2(c2 - c1 + 1);
            int u = r2 - (1 << k) + 1, v = c2 - (1 << s) + 1;
            cout << min({a[r1][c1][k][s], a[u][c1][k][s], a[r1][v][k][s], a[u][v][k][s]}) << endl;
        }
    }
    return 0;
}
