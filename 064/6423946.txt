n = int(input())

d = {}
for i in ['S', 'H', 'C', 'D']:
  d[i] = []

i = 1
while i <= int(n):
  line = input()
  if not line:
    break
  x = list(map(str, line.split()))
  d[x[0]].append(int(x[1]))
  
  i += 1

for i in ['S', 'H', 'C', 'D']:
  for j in range(1, 14):
    if j not in d[i]:
      print(f"{i} {j}")
