using System;
using System.Collections.Generic;
using System.Linq;

namespace AOJ_ICPC_Prelim_1608
{
    internal class Program
    {
        public static void Main()
        {
            while (true)
            {
                if (Console.ReadLine() == "0")
                    break;
                var data = Console.ReadLine().Split().Select(d => int.Parse(d)).ToArray();
                var sortedData = new SortedDictionary<int, int>();
                foreach (var elem in data)
                {
                    if (sortedData.ContainsKey(elem))
                        sortedData[elem]++;
                    else
                        sortedData[elem] = 1;
                }

                (int temp, int sub) work = (-1, int.MaxValue);
                foreach (var elem in sortedData)
                {
                    if(elem.Value >= 2)
                    {
                        work.sub = 0;
                        break;
                    }
                    if(work.temp == -1)
                    {
                        work.temp = elem.Key;
                        continue;
                    }
                    if(elem.Key - work.temp < work.sub)
                        work.sub = elem.Key - work.temp;
                    work.temp = elem.Key;
                }
                Console.WriteLine(work.sub);
            }
        }
    }
}
