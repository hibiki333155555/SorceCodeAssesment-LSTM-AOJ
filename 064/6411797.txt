#include <stdio.h>
int main(){
    int h, w;
    int i, j;
    while(1){
        scanf("%d%d", &h, &w);
        if (h==0 && w==0) break;
        for(j=0; j<h; j++){
            if(j%2==0){
                for(i=0; i<w; i++){
                    if(i%2==0){
                    putchar('#');
                    }else{
                        putchar('.');
                    }
                    }
                }else{
                for(i=0; i<w; i++){
                    if(i%2==1){
                    putchar('#');
                    }else{
                        putchar('.');
                    }
            }
            }
            putchar('\n');
        }
        putchar('\n');
    }
    return 0;
}
