#include<iostream>
#include<algorithm>
using namespace std;
int N,M;
char S[100][100];
char T[100][100];
int solve(int H,int W,char S[100][100])
{
	int ret=1e9;
	int bc=0;
	for(int i=0;i<H;i++)for(int j=0;j<W;j++)
	{
		if(S[i][j]=='o')bc++;
	}
	int ac=0;
	for(int k=1;k<H;k++)
	{
		for(int j=0;j<W;j++)
		{
			if(S[k-1][j]=='o')
			{
				bc--;
				ac++;
			}
		}
		ret=min(ret,min(ac+(H-k)*W-bc,k*W-ac+bc));
	}
	return ret;
}
int main()
{
	cin>>N>>M;
	for(int i=0;i<N;i++)for(int j=0;j<M;j++)cin>>S[i][j],T[j][i]=S[i][j];
	int X=solve(N,M,S);
	int Y=solve(M,N,T);
	cout<<min(X,Y)<<endl;
}
