#include <stdio.h>
#include <math.h>

int main() 
{
 long n, i, j;
 scanf("%d", &n);
 int a[n];
 for( i=0 ; i<n ; i++ )
 {
  scanf("%d", &a[i]);
 }
 long temp = -99999999;
 for( i=0 ; i<n ; i++ )
 {
  for( j=i+1 ; j<n ; j++ )
  {
   if(a[j]-a[i] > temp)
   {
    temp = a[j] - a[i];
   }
   
  }
 }
 printf("%ld", temp);
 
 return 0;
}
