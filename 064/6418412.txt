#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<vector<bool>> a(4,vector<bool>(13));
  
  for(int i=0; i<n; i++){
    char c;
    int x,s;
    cin >> c >> x;
    if(c=='S') s=0;
    else if(c=='H') s=1;
    else if(c=='C') s=2;
    else s=3;
    a.at(s).at(x-1) = true;
  }
  
  for(int i=0; i<4; i++){
    for(int j=0; j<13; j++){
      if(!a.at(i).at(j)){
        if(i==0) cout << "S";
        else if(i==1) cout << "H";
        else if(i==2) cout << "C";
        else cout << "D";
        cout << " " << j+1 << endl;
      }
    }
  }
  return 0;
        
}
