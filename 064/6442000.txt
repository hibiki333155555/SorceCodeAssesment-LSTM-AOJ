#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int n,i,sum=0;
	int min=+1000001,max=-1000001;
	int a;
	
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a;
		
		if(a>max)
			max=a;
		if(a<min)
			min=a;
		sum+=a;
	}
	
	cout<<min<<" "<<max<<" "<<sum<<endl;

return 0;
}
