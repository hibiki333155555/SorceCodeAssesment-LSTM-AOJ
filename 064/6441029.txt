#define PROBLEM "https://judge.u-aizu.ac.jp/onlinejudge/description.jsp?id=GRL_4_B"

#include <bits/stdc++.h>
using namespace std;

vector<int> topo_sort(const vector<vector<int>> &G) {
	int n = G.size();
	vector<bool> vis(n, false);
	vector<int> ord;
	ord.reserve(n);

	const auto dfs = [&](const auto &self, int u) -> void {
		vis[u] = true;
		for (int v : G[u])
			if (!vis[v])
				self(self, v);
		ord.push_back(u);
	};

	for (int i = 0; i < n; i++)
		if (!vis[i])
			dfs(dfs, i);
	reverse(ord.begin(), ord.end());
	return ord;
}

int main() {
	int N, M;
	cin >> N >> M;
	vector<vector<int>> G(N);
	for (int i = 0; i < M; i++) {
		int u, v;
		cin >> u >> v;
		G[u].push_back(v);
	}
	auto ord = topo_sort(G);
	for (int u : ord)
		cout << u << '\n';
}
