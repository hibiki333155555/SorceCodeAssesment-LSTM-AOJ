#include<stdio.h>

int main(void)
{
    int n,x;
    int a,b,c;
    
    while(1)
    {
        int count=0;
        scanf("%d %d",&n,&x);
        if(n==0&&x==0)break;
        for(a=1;a<=n-2;a++)
        {
            for(b=a+1;b<=n-1;b++)
            {
                for(c=b+1;c<=n;c++)
                {
                    if(a+b+c==x)
                    {
                        count+=1;
                    }
                }
            }
        }
        printf("%d\n",count);
    }
    return 0;
}
