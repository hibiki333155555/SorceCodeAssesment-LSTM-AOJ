#include <bits/stdc++.h>
using namespace std;
long long n;

int main(int argc, char** argv) {
	cin>>n;
	long long div = 2;
	long long m = n;
	cout<<m<<":";
		if(m <2)
		{
			cout<<1;
			return 0;
		} 
	while(m > 1)
	{
		if(m%div == 0)
		{
			m /= div;
			cout<<" "<<div;
		}
		if(m%div != 0)
		{
			div++;
		}
	}
	cout<<endl;
	return 0;
}
