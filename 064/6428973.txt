int main(void) {
	int a, b, c;
	
	/*入力*/
	scanf("%d %d %d", &a, &b, &c);
	
	/*処理*/
	if ( a < b ) {
		if ( a > c ) {
			printf("%d %d %d\n", c, a, b);
		} else if ( b < c ) {
			printf("%d %d %d\n", a, b, c);
		} else {
			printf("%d %d %d\n", a, c, b);
		}
	} else {
		if ( b > c ) {
			printf("%d %d %d\n", c, b, a);
		} else if ( a < c ) {
			printf("%d %d %d\n", b, a, c);
		} else {
			printf("%d %d %d\n", b, c, a);
		}
	}
	
	return 0;
}
