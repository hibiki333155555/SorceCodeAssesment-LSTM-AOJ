#include<stdio.h>
#define MAX 252

int main()
{
  int n;
  int H[MAX];
  scanf("%d", &n);
  for(int i = 1; i <= n; i++ ) scanf("%d", &H[i]);

  for(int i = 1; i <= n; i++ ){
    printf("node %d: key = %d, ", i, H[i]);
    if(i / 2 >= 1) printf("parent key = %d ", H[i / 2]);
    if(i * 2 <= n) printf("left ley = %d, ", H[i * 2]);
    if(i * 2 + 1 <= n) printf("right key = %d, ", H[i * 2 + 1]);
    printf("\n");
  }

  return 0;
}
