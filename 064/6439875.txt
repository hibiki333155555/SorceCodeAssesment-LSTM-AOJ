#include <iostream>
using namespace std;

static const int N=100;
int m[N+1][N+1];
int p[N+1];

int matrixCainMultiplication(int n){
    for(int i=1;i<=n;i++) m[i][i]=0;
    for(int l=2;l<=n;l++){
        for(int i=1;i<=n-l+1;i++){
            int j=i+l-1;
            m[i][j]=(1<<21);
            for(int k=i;k<j;k++){
                m[i][j]=min(m[i][j],m[i][k]+m[k+1][j]+p[i-1]*p[k]*p[j]);
                
            }
        }
    }
    return m[1][n];
}
int main(void){
    // Your code here!
    int n;cin>>n;
    for(int i=1;i<=n;i++){
        cin>>p[i-1]>>p[i];
    }
    
    cout<<matrixCainMultiplication(n)<<endl;
}
