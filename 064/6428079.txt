#include <iostream>
#include <string>
using namespace std;

int N;
string S[1009];
bool fusei[1009];

bool check(string T1, string T2) {
	int cnt = 0;
	for (int i = 0; i < (int)T1.size(); i++) {
		if (T1[i] != T2[i]) cnt++;
	}
	if (cnt <= 1) return true;
	return false;
}

int main() {
	// Step #1. 入力
	cin >> N;
	for (int i = 1; i <= N; i++) cin >> S[i];

	// Step #2. チェック
	for (int i = 1; i <= N; i++) fusei[i] = false;
	for (int i = 1; i <= N; i++) {
		for (int j = i + 1; j <= N; j++) {
			if (check(S[i], S[j]) == true) { fusei[i] = true; fusei[j] = true; }
		}
	}
	
	// Step #3. 答えを出力
	int Answer = 0;
	for (int i = 1; i <= N; i++) {
		if (fusei[i] == true) Answer += 1;
	}
	cout << Answer << endl;
	return 0;
}
