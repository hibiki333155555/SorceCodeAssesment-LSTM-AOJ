N = int(input())
A = list(map(int, input().split()))

def bubble_sort(A):
    flag = True
    cnt = 0
    while flag:
        flag = False
        for j in range(1, N)[::-1]:
            if A[j] < A[j - 1]:
                A[j], A[j-1] = A[j-1], A[j]
                cnt += 1
                flag = True
    print(*A)
    print(cnt)

bubble_sort(A)
