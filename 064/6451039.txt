using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace CsharpforAizeOnlineJudge
{
    class Program
    {
        static void Main(string[] args)
        {
            int i = 1;
            while(true)
            {
                int x = Int32.Parse(Console.ReadLine());
                if (x == 0) break;
                Console.WriteLine("Case {0}: {1}", i++, x);
            }
        }
    }
}
