    ll modpow(ll x, ll b,ll mod=MOD){
        ll res = 1;
        while(b){
            if(b&1)res = res * x % mod;
            x = x * x % mod;
            b>>=1;
        }
        return res;
    }

    ll modinv(ll x){
        return modpow(x, MOD-2);
    }

const int N = 1010;

class bitmatrix{
public:
    vector<bitset<N>> data;
    bitmatrix():data(N){}
    static bitmatrix Identity(){
        bitmatrix res;
        rep(i,N){
            res.data[i][i] = 1;
        }
        return res;
    }
    bitmatrix rotate(){
        bitmatrix res;
        rep(i,N){
            rep(j,N){
                res.data[j][i] = data[i][j];
            }
        }
        return res;
    }
    bitmatrix operator*(bitmatrix that){
        that = that.rotate();
        bitmatrix res;
        rep(i,N){
            rep(j,N){
                res.data[i][j] = (data[i] & that.data[j]).any();
            }
        }
        return res;
    }
    bitmatrix pow(int y){
        bitmatrix res = Identity();
        bitmatrix sum = *this;
        while(y){
            if(y&1)res = res * sum;
            sum = sum * sum;
            y >>= 1;
        }
        return res;
    }
    bitset<N>& operator[](int i){
        return data[i];
    }
};

vector<int> func(){
    int n = in();
    int d = in();
    vvector<pii> edges(n);
    vector<int> adds(n+1,0);
    rep(i,n){
        int k = in();
        chmax(adds[i+1],1);
        rep(_,k){
            int v = in()-1;
            int d = in();
            edges[i].emplace_back(v,d);
            chmax(adds[i+1],d);
        }
    }
    rep(i,n){
        adds[i+1] += adds[i];
    }
    int N = adds.back();
    bitmatrix mat;
    rep(i,n){
        rep(j,adds[i],adds[i+1]-1){
            mat[j][j+1] = 1;
        }
        foreach(e,edges[i]){
            mat[adds[i]+e.second-1][adds[e.first]] = 1;
        }
    }
    mat = mat.pow(d);
    vector<int> res;
    rep(i,n){
        int cnt = 0;
        rep(j,n){
            if(mat[adds[j]][adds[i]])++cnt;
        }
        if(cnt==n)res.emplace_back(i+1);
    }
    return res;
}
