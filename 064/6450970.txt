#include<bits/stdc++.h>
//#include<atcoder/all>

using namespace std;
//using namespace atcoder;
using ll = long long;

int main() {
    int a, b; cin >> a >> b;

    if (a == 0 && b == 0) {
        cout << 1 << endl;
        return 0;
    }

    auto bit_len = [&](int x) {
        int ret = 0;
        while (x) {
            ++ret;
            x >>= 1;
        }
        return ret;
    };

    int l = max(bit_len(a), bit_len(b));
    vector<int> ipa(l), ipb(l);
    for (int i = 0; i < l; ++i) {
        ipa[l - 1 - i] = a & 1;
        ipb[l - 1 - i] = b & 1;
        a >>= 1;
        b >>= 1;
    } 

    int ans = 0;
    for (int i = 0; i < l; ++i) {
        if (ipa[i] == ipb[i]) ++ans;
        else break;
    }

    cout << ans << endl;
    return 0;
}
