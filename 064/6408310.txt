#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main(){
    int n, e;
    cin>>n;
    vector<int> v;
    while(n--){
        cin>>e;
        v.push_back(e);
    }
    cin>>n;
    int a,b,c;
    while(n--){
        cin>>a>>b>>c;
        cout << v.count(v.begin()+a, v.begin()+b, c) << endl;
    }
}
