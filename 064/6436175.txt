#include <stdio.h>

int main(void) { 
    int a,b,c;
    // scanf("%d%d%d",&a,&b,&c);
    scanf("%d",&a);
    scanf("%d",&b);
    scanf("%d",&c);
    int i;
    int count;
    for(i=a;i<=b;i=i+1){
        if(c%i==0){
            count=count+1;
        }
    }
    printf("%d\n",count);
    
	return 0;
}
