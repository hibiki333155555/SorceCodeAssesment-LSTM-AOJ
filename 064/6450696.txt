#include <bits/stdc++.h>
using namespace std;

void bubbleSort(vector<int> A,int N){
  bool flag = true;
  int num =0;
  while(flag){
    flag = false;
    for(int j=N-1;j>= 1;j--){
      if(A[j] < A[j-1]){
        num++;
        swap(A[j], A[j-1]);
        flag = true;
      }
    }
    
  }
  
  for(int a=0;a<N;a++){
    if(a != 0)
      cout << " ";
    cout << A[a];
  }
  cout << "\n" <<num <<  endl;
  return;
}
int main(){
  int N;
  cin >> N;
  vector<int> A(N);
  for(int i =0;i<N;i++)
    cin >> A[i];
  
  bubbleSort(A,N);

  return 0;
}
