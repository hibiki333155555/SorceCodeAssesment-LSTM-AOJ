#include<iostream>
using namespace std;

int main()
{
	int m,f,r,i;
	int score[50][1][1][1];
    char grade[50];

	for(i=0;i<50;++i){
        cin>>score[i][m][f][r];
        if(m==-1 && f==-1 &&r==-1)break;
    }


	for(int j = 0; j < i; j++){
        if(m==-1 || f==-1)grade[j]=='F';
        else if((m+f)>=80)grade[j]=='A';
        else if((m+f)>=65 && (m+f)<80)grade[j]=='B';
        else if((m+f)>=50 && (m+f)<65)grade[j]=='C';
        else if((m+f)>=30 && (m+f)<50 && r>=50)grade[j]=='C';
        else if((m+f)>=30 && (m+f)<50)grade[j]=='D';
        else if((m+f)<30)grade[j]=='F';
        
    }

    for(int k = 0; k< i; k++){
        cout<<grade[k]<<endl;
    }

return 0;
}
