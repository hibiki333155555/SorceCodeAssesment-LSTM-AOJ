// メイン------------------
int main() {
  // デバッグ用
  ifstream in("input.txt");
  if (in.is_open()) {
    cin.rdbuf(in.rdbuf());
  }

  int n, q;
  cin >> n >> q;
  ll a[n];
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }

  // クエリ
  for (int i = 0; i < q; i++) {
    ll x;
    cin >> x;

    ll len = 0;

    int r = 0;
    ll total = 0;
    for (int l = 0; l < n; l++) {
      while (r < n && total + a[l] <= x) {
        total += a[r];
        r++;
      }

      len += (r - l);
      if (r == l) {
        r++;
      } else {
        total -= a[l];
      }
    }
    cout << len << endl;
  }
}
