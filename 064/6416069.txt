#include <iostream>
#include <cmath>
#include <cstdlib>
#include <stdio.h>
#include <vector>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)

int N;
vector<int> A(N);

int main() {
    int N;
    cin >> N;
    vector<int> A(N);
    rep(i,N) cin >> A[i];
    rep(i,N) cout << A[i] << " ";
        cout << endl;

    for(int i=1;i<N-1;i++) {
        int v = A[i];
        int j = i - 1;
        while(j>=0 && A[j] > v) {
            A[j+1] = A[j];
            j--;
        }
        A[j+1] = v;
        rep(i,N) cout << A[i] << " ";
        cout << endl;
    }
    rep(i,N) cout << A[i] << " ";
        cout << endl;
}
