#include <iostream>
#include <vector> // vector
#include <algorithm> // min, max, swap, sort, reverse, lower_bound, upper_bound
using namespace std;
static const int MAX = 200000;

int main() {
    int R[MAX], n;

    cin >> n;
    for(int i = 0;i < n;i++)cin >> R[i];

    int maxv = -2000000000;
    int minv = R[0];

    for(int i = 1;i < n;i++){
        maxv = (maxv, R[i] - minv);
        minv = (R[i], minv);
    }
    cout << maxv << endl;
}
