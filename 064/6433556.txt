using namespace std;
using namespace __gnu_pbds;
void solve(){
    i64 q; cin >> q;
    map<string, i64> hesh;
    while(q--){
        int t; cin >> t;
        if(!t){
            string s; cin >> s;
            i64 a; cin >> a;
            hesh.insert({s, a});
        }
        else if(t == 1){
            string s; cin >> s;
            if(hesh.find(s) != hesh.end()){
                cout << hesh[s] << '\n';
            }
            else cout << 0 << '\n';
        }
        else if(t == 2){
            string s; cin >> s;
            if(hesh.find(s) != hesh.end()){
                hesh.erase(s);
            }
        }
        else{
            string x, y;
            cin >> x >> y;
            auto it_a = hesh.lower_bound(x);
            auto it_b = hesh.upper_bound(y);
            for(auto it = it_a; it != it_b; it++){
                cout << it -> ff << " " << it -> ss << '\n';
            }
        }
    }
}
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int tc; tc = 1;
    while(tc--){
        solve();
    }
    return 0;
}
