template<class T> inline bool chmin(T& a, T b) {
    if (a > b) {
        a = b;
        return true;
    }
    return false;
}
template<class T> inline bool chmax(T& a, T b) {
    if (a < b) {
        a = b;
        return true;
    }
    return false;
}
ll det(pl a,pl b,pl c){//a->bとb->cの周り方、正なら時計回り
    ll ax=b.first-a.first,ay=b.second-a.second;
    ll bx=c.first-b.first,by=c.second-b.second;
    return ax*by-ay*bx;
}
int main(){
    ll n;cin >> n;
    vl v(n);rep(i,n)cin >> v[i];
    vpl st;
    rep(i,n){
        pl now={i,v[i]};
        if(st.size()<2){
            st.push_back(now);
            continue;
        }
        ll len=st.size();
        while(len>=2&&det(st[len-2],st[len-1],now)>=0){
            st.pop_back();len--;
        }
        st.push_back(now);
    }
    ll ans=0;
    rep(i,st.size()-1){
        chmax(ans,st[i+1].first-st[i].first);
    }
    cout << ans << endl;
}
