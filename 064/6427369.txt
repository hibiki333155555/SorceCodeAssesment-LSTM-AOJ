#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    while(n!=0){
        int k,one=0,zero=0;
        for(int i=0;i<n;i++){
            cin>>k;
            if(k<=1) one++;
            else if(k==0) zero++;
        }
        if(one==n) cout<<"NA"<<endl;
        else cout<<n-zero+1<<endl;
        cin>>n;
    }
    return(0);
}
