int main(){
    int n; cin>>n;
    int r[n]; rep(i,0,n) cin>>r[i];
    int big=r[0],col=1;
    rep(i,1,n){
        if(r[i]<r[i-1]){
            col++;
        }else if(r[i]>r[i-1]){
            if(big<=r[i]){
                col=1;
                big=r[i];
            }else{
                col=1;
                int j=i-1;
                while(j>=0){
                    if(r[j]>r[i]) col++;
                    if(r[j]==big) j=-99999;
                    j--;
                }
            }
        }
    }
    cout<<col<<"\n";
