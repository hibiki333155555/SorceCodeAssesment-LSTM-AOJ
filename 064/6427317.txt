#include<iostream>
using namespace std;

void insertion_sort(int a[],int n){
    int,i,j,tmp;
    for(i=0;i<n;i++){
        tmp = a[i];
        j = i - 1;
        while(j>=0 && a[j]>tmp){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = tmp;
    }
}
int main(){
    int a[3];
    for(int i=0;i<3;i++){
        cin >> a[i];
    }
    insertion_sort(a,3);
    for(int i=0;i<3;i++){
        if(i!=0) cout << " ";
    cout << a[i]; 
    }
    cout << endl;
}
