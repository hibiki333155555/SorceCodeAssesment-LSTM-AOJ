#include <bits/stdc++.h>
#define rep(i,n) for(int i = 0; i < (n); i++)
using namespace std;
typedef long long ll;

int main(){
    cin.tie(0);
    ios::sync_with_stdio(0);
    
    string S; cin >> S;
    string ans = "";
    stack<int> st;
    int sum = 0;
    for(char c : S) {
        if(c == '(') continue;
        if(isdigit(c)) {
            st.push(c - '0');
            sum += c - '0';
        } else if(c == ')') {
            sum -= st.top();
            st.pop();
        } else {
            ans += (char)((c - 'a' + sum) % 26 + 'a');
        }
    }
    cout << ans << endl;
}
