#include <stdio.h>

void chmax(int* a, int b)
{
	if (*a < b) *a = b;
}

int main()
{
	int i, N, M, K, X, A, max;
	scanf("%d %d %d %d", &N, &M, &K, &X);
	for (i = 1; i <= N; i++) {
		scanf("%d", &A);
		chmax(&max, A);
	}
	if ((long long)max * K >= (long long)M * X) printf("Yes\n");
	else printf("No\n");
	fflush(stdout);
	return 0;
}
