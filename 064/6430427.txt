void Print(vector<int>v){
	for(int i=0;i<v.size(); i++){
		if(i<v.size()-1){
			cout<<v[i]<<" ";
		}
		else{
			cout<<v[i];
			
		}
		
		
		
	}
	cout<<"\n";
}
void BubbleSort(vector<int>&arr){
	int n=arr.size(),temp=0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(arr[j]>arr[j+1]){
				swap(arr[j],arr[j+1]);
				temp++;
				
			}
		}
	
	}
		Print(arr);
		cout<<temp<<"\n";
}
