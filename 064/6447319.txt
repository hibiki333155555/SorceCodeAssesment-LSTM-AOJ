#include <cstdio>
#include <cmath>

int main() {
    while (true) {
        int n, m;
        scanf("%d%d", &n, &m);
        if (n == 0 && m == 0) {
            break;
        }
        int c[m];
        for (int i = 0; i < m; i++) {
            scanf("%d", &c[i]);
        }
        int x[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &x[i]);
        }
        int dp[n + 1][256];
        for (int i = 0; i < n + 1; i++) {
            for (int j = 0; j < 256; j++) {
                dp[i][j] = 0x3f3f3f3f;
            }
        }
        dp[0][128] = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < 256; k++) {
                    int s = std::min(255, std::max(0, k + c[j]));
                    dp[i+1][s] = std::min(dp[i+1][s], dp[i][k] + (s - x[i]) * (s - x[i]));
                }
            }
        }
        int ans = 0x3f3f3f3f;
        for (int i = 0; i < 256; i++) {
            ans = std::min(ans, dp[n][i]);
        }
        printf("%d\n", ans);
    }
    return 0;
}
