#include <stdio.h>
#include <string.h>
int main()
{
    int i, j, k, len;
    char str[1200];
    fgets(str, 1200, stdin);
    len = strlen(str);
    char alph[27] = "abcdefghijklmnopqrstuvwxyz";
    int count[27] = {0};
    while (fgets(str, 1200, stdin) != NULL)
    {
        for (i = 0; i < len; i++)
        {
            if (str[i] >= 'A' && str[i] <= 'Z')
            {
                str[i] = str[i] + 32;
            }
        }
        for (i = 0; i < len; i++)
        {
            for (j = 0; j < 27; j++)
            {
                if (str[i] == alph[j])
                {
                    count[j]++;
                }
            }
        }
        for (j = 0; j < 26; j++)
        {
            printf("%c : %d\n", alph[j], count[j]);
        }
    }

    // printf("%s", str);
    return 0;
}
