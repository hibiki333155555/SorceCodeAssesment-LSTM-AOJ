using System;

namespace ConsoleApp3
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arr = Array.ConvertAll(Console.ReadLine().Split(),k =>int.Parse(k));
            Array.Sort(arr);
            int i = 0;
            while(i < 2){
                Console.Write(arr[i] + " ");
                i++;
            }
            Console.WriteLine(arr[2]);



        }
    }
}
