using namespace std;
void print(vector<int>v){
	for(int i=0;i<v.size();i++){
		if(i!=v.size()-1){
		cout<<v[i]<<" ";
		
		}
		else{
		cout<<v[i];
		}
		}
	cout<<"\n";
}
void Selection(vector<int> &arr){
	int n=arr.size();
	int pos,count=0;
	for(int i=0;i<n-1;i++){
		pos=i;
		for(int j=i+1;j<n;j++){
			if(arr[pos]>arr[j]){
				pos=j;
				count++;
			}
		
		}
			swap(arr[pos],arr[i]);
			
		}
		print(arr);
			if(count%2==0){
				cout<<count/2<<"\n";
			}
			else{
				cout<<count-2<<"\n";
			}
	}
