#include <bits/stdc++.h>
using namespace std;

#define rep(i,n) for (int i = 0; i < (n); ++i)
#define N 100
#define WHITE 0
#define GRAY 1
#define BLACK 2

int n, M[N][N];
int color[N], d[N], f[N], tt;

void dfs_visit(int u){
    int v;
    color[u] = GRAY;
    d[u] = ++tt;
    rep(v, n) {
        if(M[u][v] == 0) continue;
        if(color[v] == WHITE){
            dfs_visit(v);
        }
    }
    color[u] = BLACK;
    f[u] = ++tt;
}

void dfs() {

    rep(i, n){
        color[i] = WHITE;
    }
    tt = 0;
    rep(i, n) {
        if(color[i] == WHITE) dfs_visit(i);
    }
    rep(i, n) {
        printf("%d %d %d\n", i+1, d[i], f[i]);
    }
}

int main() {
    int u, v, k, i, j;
    cin >> n;
    rep(i, n) rep(j, n) M[i][j] = 0;
    rep(i, n){
        cin >> u >> k;
        u--;
        rep(j,k){
            cin >> v;
            v--;
            M[u][v] = 1;
        }
    }

    dfs();
    return 0;
}
