#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    scanf("%d", &n);
    
    vector<int> a;
    int temp;
    
    for(int i = 0; i < n; i++){
        scanf("%d", &temp);
        a.push_back(temp);
    }
    
    int q;
    scanf("%d", &q);
    int k;
    
    for(int i = 0; i < q; i++){
        scanf("%d", &k);
        
        printf("%lld ", lower_bound(a.begin(), a.end(), k) - a.begin());
        printf("%lld\n", upper_bound(a.begin(), a.end(), k) - a.begin());
    }
    
    return 0;
}
