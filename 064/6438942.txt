#include<bits/stdc++.h>
using namespace std;
int main(){
  int k;
  cin>>k;
  int r[k],c[k];
  for(int i=0;i<k;i++)cin>>r[i]>>c[i];
  int p[8]={0,1,2,3,4,5,6,7};
  string record[8];
  for(int i=0;i<8;i++){
    for(int j=0;j<8;j++)record[i].push_back('.');
  }
  do{
    bool x=true;
    for(int i=0;i<k;i++){
      if(p[r[i]]!=c[i])x=false;
    }
    if(x){
      bool y=true;
      for(int i=1;i<8;i++){
        for(int j=0;j<i;j++){
          if(p[i]-i==p[j]-j||p[i]+i==p[j]+j)y=false;
        }
      }
      if(y){
        for(int i=0;i<8;i++)record[i][p[i]]='Q';
      }
    }
  }while(next_permutation(p,p+8));
  for(int i=0;i<8;i++)cout<<record[i]<<endl;
}
