#include<bits/stdc++.h>
using namespace std;
int main(){
  int n,k;
  cin>>n>>k;
  int ans=0;
  vector<int>T(n);
  for(int i=0;i<n;i++){
    int a;
    cin>>a;
    ans+=a;
    if(a==1)T[i]=5-a;
    else T[i]=6-a;
  }
  sort(T.rbegin(),T.rend());
  for(int i=0;i<min(k,n);i++)ans+=T[i];
  cout<<ans<<endl;
}
