process.stdin.resume();
process.stdin.setEncoding('utf8');

var input_lines = [];
var reader = require('readline').createInterface({
  input: process.stdin,
  output: process.stdout
});

reader.on('line', (line) => {
    input_lines = line.split(" ")
  });
  
reader.on('close', () => {
  console.log(input_lines);
});
