#include <stdio.h>
int main(){
    int h, w;
    int i, j;
    while(1){
        scanf("%d%d", &h, &w);
        if (h==0 && w==0)break;
        for(j=0; j < h;j++){
             if (j ==0 || j == h-1){
                for(i=0; i < w; i++){
                    putchar('#');
                    }
            }
            else{
                putchar('#');
                for(i=0; i < w-2; i++){
                    putchar('.');
                    }
                putchar('#');
            }
            putchar('\n');
        }
        putchar('\n');
    }
    return 0;
}
