#include <bits/stdc++.h>
using namespace std;


int select(int A[], int N){
        int time=0, minv;
        for(int i=0;i<N-1;i++){
                minv=i;
                for(int j=i+1;j<N;j++){
                        if(A[j]<A[minv]){
                                minv=j;
                        }
                }
                swap(A[i],A[minv]);
                if(i!=minv) time++;
        }
        return time;
}

int main(){
        int A[100], N, time=0; cin>>N;
        for(int i=0;i<N;i++) cin>>A[i];

        time=select(A, N);

        for(int i=0;i<N;i++){
                if(i) cout<<" ";
                cout<<A[i];
        }
        cout<<endl;
        cout<<time<<endl;

        return 0;
}
