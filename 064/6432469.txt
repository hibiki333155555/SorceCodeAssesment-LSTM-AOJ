#include<iostream>
using namespace std;
int main(){
  int n, m;
  cin >> n >> m;
  
  //n×m行列の定義
  int matrix_nm[n][m];
  
  //配列の定義
  int matrix_m[m];
  
  //n×m行列のデータ取得
  for (int i=0; i<n; i++){
    for (int j=0; j<m; j++){
      cin >> matrix_nm[i][j];
    }
  }
  
  //配列データの取得
  for (int i=0; i<m; i++){
    cin >> matrix_m[i];
  }
  
  //出力
  int sum = 0;
  
  for (int i=0; i<n; i++){
    for (int j=0; j<m; j++){
      sum += matrix_nm[i][j] * matrix_m[j];
    }
    
    cout << sum << endl;
    sum=0;
  }
  
  
  
}
