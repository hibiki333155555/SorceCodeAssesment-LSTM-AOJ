#pragma GCC target("avx2")
#pragma GCC optimize("O3")
#pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9;
const ll inf = 1LL<<62;

void solve() {
  string s; cin >> s;
  ll num = 0;
  int n = s.size();
  vector<ll> ans;
  stack<int> st;
  for (auto c : s) {
    if (isdigit(c)) {
      num += c - '0';
      st.push(c - '0');
    } else if (c == '(') {
      continue;
    } else if (c == ')') {
      num -= st.top();
      st.pop();
    } else {
      ans.push_back((c - 'a' + num)%26);
    }
  }
  for (int p : ans) cout << (char)(p + 'a');
  cout << '\n';
}

int main() {
  ios::sync_with_stdio(false);
  std::cin.tie(nullptr);
  // int t; cin >> t;
  /*while (t--)*/ solve();
}
