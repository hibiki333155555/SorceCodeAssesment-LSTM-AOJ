#include<bits/stdc++.h>
using namespace std;

int main(){
  for(;;){
    int N,M;
    cin >> N >> M;
    if(N == 0 && M == 0) break;
    vector<int> A(N);
    for(int i=0;i<N;i++) cin >> A[i];
    sort(A.begin(),A.end());
    int ans = -1;
    for(int i=0;i<N;i++){
      int t = lower_bound(A.begin(),A.end(),M+1-A[i])-A.begin();
      if(t == 0) continue;
      t--;
      if(A[t] == A[i]) t--;
      if(t == -1) break;
      ans = max(ans,A[i]+A[t]);
    }
    if(ans == -1) cout << "NONE" << endl;
    else cout << ans << endl;
  }
}
