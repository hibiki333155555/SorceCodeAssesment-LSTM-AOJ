import java.util.Scanner;

class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int H= sc.nextInt();
        int W= sc.nextInt();
        if(H==0&&W==0){
            break;
        }
        else{
            for(int i=0;i<H;i++;){
                for(int i=0;i<W;i++){
                System.out.println("#");
                }
                System.out.println("")
            }
            
        }
    }
}
