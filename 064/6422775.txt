#include <iostream>
using namespace std;

void bubblesort(int a[],int n){
    int i,flag,j;
    int count = 0;
    flag = 1;
    for(i=0;flag;i++){
        flag = 0;
        for(j=n-1;j>=i+1;j--){
           if(a[j] < a[j-1]){
               flag = 1;
               swap(a[j],a[j-1]);
               count++;
           }
        }
    }
    for(i=0;i<n;i++){
        cout << a[i] << " ";
    }
    cout << endl;
}

int main(){
    int i,n;
    cin >> n;
    int a[n];
    for(i=0;i<n;i++){
        cin >> a[i];
    }
    bubblesort(a,n);
    return 0;
}
