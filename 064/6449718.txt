

#include<iostream>
using namespace std;

int main(){
    int a, b;
    int c, d;
    double e;
    cin >> a >> b >> c >> d >> e;
    
    c = a / b;
    d = a % b;
    e = 1.0 * a / b;
    
    cout << c << " " << d << " " << e << endl;
    printf("%.8lf\n %.8lf\n %.8lf\n", c, d, e);
    

    return 0;
}
