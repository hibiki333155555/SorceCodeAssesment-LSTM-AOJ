#include <iostream>
#include <map>
using namespace std;

int main()
{
    map<string, int> mp;
    for(int i = 0; i < 26; i++){
        string s;
        s += char('A' + i);
        mp[s] = i;
    }
    for(int i = 0; i < 26; i++){
        for(int j = 0; j < 26; j++){
            string s;
            s += char('A' + i);
            s += char('A' + j);
            mp[s] = (i + 1) * 26 + j;
        }
    }
    string c1, c2;
	int r1, r2;
    cin >> c1 >> r1 >> c2 >> r2;
    cout << (abs(r1 - r2) + 1) * (abs(mp[c1] - mp[c2]) + 1) << endl;
}
