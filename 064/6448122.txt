#include <iostream>
using namespace std;

int main(){
    int n;
    int x;
    int count = 0;
    while (true)
    {
        cin >> n >> x;
        if(n == 0 && x == 0) break;
        for(int i = 1;i <= x;i++){
            for(int j = i + 1;j <= x;j++){
                for(int k = j + 1;k <= x;k++){
                    if(i + j + k == x){
                        count++;
                    }
                }
            }
        }
    }
    cout << count - 1 << endl;
    return 0;
}
