#include <iostream>
#include <queue>

using namespace std;

queue<int> Q;
int d[100];
int G[100][100];
int n;
int visited[100];

void bfs(int start) {
  Q.push(start);
  d[start] = 0;
  visited[0] = 1;
  while (!Q.empty()) {
    int front = Q.front();
    Q.pop();
    cout << front << endl;
    for (int i = 0; i < n; i++) {
      if (G[front][i] != 0 && !visited[i]) {
        visited[i] = 1;
        Q.push(i);
        d[i] = d[front] + 1;
      }
    }
  }
}

int main(void) {
  for (int i = 0; i < 100; i++) {
    d[i] = -1;
  }
  cin >> n;
  for (int i = 0; i < n; i++) {
    int to, m;
    cin >> to >> m;
    for (int k = 0; k < m; k++) {
      int v;
      cin >> v;
      G[to - 1][v - 1] = 1;
    }
  }
  bfs(0);
  for (int i = 0; i < n; i++) {
    cout << i + 1 << ' ' << d[i] << endl;
  }
}
