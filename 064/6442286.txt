#include<iostream>
using namespace std;

int main() {
    int n, r, a[4][12] = {{}, {}, {}, {}};
    char s;
    cin >> n;
    for(int i = 0; i < n; i++) {
        cin >> s >> r;
        switch(s) {
            case 'S':
                a[0][r - 1] = r;
                break;
            case 'H':
                a[1][r - 1] = r;
                break;
            case 'C':
                a[2][r - 1] = r;
                break;
            case 'D':
                a[3][r - 1] = r;
                break;
        }
    }
    for(int i = 0; i < 4; i++) {
        for(int j = 0; j < 12; j++) {
            if(i == 0 && a[i][j] == 0) cout << "S " << j + 1 << endl;
            else if(i == 1 && a[i][j] == 0) cout << "H " << j + 1 << endl;
            else if(i == 2 && a[i][j] == 0) cout << "C " << j + 1 << endl;
            else if(i == 3 && a[i][j] == 0) cout << "D " << j + 1 << endl;
        }
    }
    return 0;
}
