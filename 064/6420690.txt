#include <bits/stdc++.h>
using namespace std;
int main(){
  string S;
  cin >> S;
  int N = S.size();
  int p = 0;
  stack<int> st;
  for (int i = 0; i < N; i++){
    if (isdigit(S[i])){
      p += S[i] - '0';
      st.push(S[i] - '0');
    } else if (isalpha(S[i])){
      int x = S[i] - 'a';
      x += p;
      x %= 26;
      cout << (char) ('a' + x);
    } else if (S[i] == ')'){
      p -= st.top();
      st.pop();
    }
  }
  cout << endl;
}
