#include <bits/stdc++.h>
using namespace std;

int main() {
	int A,B;cin>>A;
	vector<int> a(A);
	for(int i=0;i<A;i++)
	{
		cin>>a[i];
	}	
	cin>>B;
	vector<int> b(B);
	for(int i=0;i<B;i++)
	{
		cin>>b[i];
	}
	int j=0,c=0;
	for(int i=0;i<A;i++)
	{
		if(j==b.size())
		{
			break;
		}
		if(a[i]==b[j])
		{
			c++;
			j++;
		}
	}
	if(c==b.size())
	{
		cout<<"1";
	}
	else
	{
		cout<<"0";
	}
	return 0;
}
