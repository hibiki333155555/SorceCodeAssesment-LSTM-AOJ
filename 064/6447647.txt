    cin.getline(s,maxn);
    int len = strlen(s);
    for(int i = 0; i < len; i++) {
        cnt[int(s[i])]++;
    }
    int ans = 0;
    priority_queue<int, vector<int>, greater<int>> q;
    for(int i = 0; i < maxm; i++) {
        if(cnt[i]) q.push(cnt[i]);
    }
    if(q.size() <= 1) {
        cout<<len<<endl;
        return 1;
    }
    while(q.size() > 1) {
        int a = q.top(); q.pop();
        int b = q.top(); q.pop();
        ans += a+b;
        q.push(a+b);
    }
    cout<<ans<<endl;
    
    
    
    //cerr<<"Time : "<<1000*((double)clock())/(double)CLOCKS_PER_SEC<<"ms";

}
