def check(type_, l):
    for i, j in enumrate(l):
        if j == 0:
            print('{} {}'.format(type_, i+1))


n = input()
type_list = ['S', 'H', 'C', 'D']
list_ = [[0 for i in range(13)] for j in range(4)]

for i in range(n):
    type, num = input().split()
    for j in range(len(type_list)):
        if type == type_list[j]:
            list_[j][int(num)-1] = int(num)

for i, l in enumrate(list_):
    check(type_list[i], l)
