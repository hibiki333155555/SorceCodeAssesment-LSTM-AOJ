#include <bits/stdc++.h>
#include<iostream>
#include<algorithm>
#include<cstdio>
#include<string>
#include<climits>
#include<vector>
#define check cout<<"check"<<endl
#define rep(a,b,c)   for(int a = b ; a < c ;a++)
#define roop(i, n) for (int i = 0; i < (n); ++i)
#define INF 1000100100
using namespace std;
using REF = vector< vector<int> > ;
using namespace std;


int V, E;
int G[20][20];  // グラフ
int dp[50000][20];
int main() {
    cin >> V >> E;
    // グラフの初期化
    // dp の初期化
   roop(i,50000)roop(j,20)dp[i][j] = INF;
   roop(i, 20) roop(j, 20) G[i][j] = INF; 
   roop(i,E)
    {
        int a,b,c;
        cin>>a>>b>>c;
        G[a][b] = c;
    }
    
    dp[0][0] = 0;  // スタートを頂点 0 とする
    rep(s,0,1<<V)
    {
        rep(t,0,V)
        {
            if(s & (1<<t))continue;
            rep(u,0,V)
            {
                if(s != 0 && !(s & 1 << u))continue;
                int a = dp[s][u] + G[u][t];
                if(a < dp[s | 1 << t][t])dp[s | 1 << t][t] = a;
            }

        }
    }
    if (dp[(1 << V) - 1][0] != INF) {
        cout << dp[(1 << V) - 1][0] << endl;
    } else {
        cout << -1 << endl;
    }
    return 0;
}
