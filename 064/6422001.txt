int main() {
    int N, Q;
    cin >> N;
    vector<int> a(N);
    REP(i, N) cin >> a[i];

    vector<int> ord(N);
    vector<pair<int, int>> st;
    iota(ALL(ord), 0);
    cin >> Q;
    while(Q--) {
        int t;
        cin >> t;
        if(t == 1) {
            int x, y;
            cin >> x >> y;
            x--, y--;
            st.emplace_back(x, y);
        } else if(t == 2) {
            st.pop_back();
        } else {
            int K;
            cin >> K;
            
            vector<int> c(N), now(N);
            iota(ALL(now), 0); iota(ALL(c), 0);
            REP(i, SZ(st)) swap(c[st[i].first], c[st[i].second]);
            vector<int> nxt(N);
            while(K) {
                if(K & 1) {
                    REP(i, N) nxt[i] = now[c[i]];
                    now = nxt;
                }
                REP(i, N) nxt[i] = c[c[i]];
                c = nxt;
                K >>= 1;
            }
            REP(i, N) nxt[i] = ord[now[i]];
            ord = nxt; 
            debug(ord);
        }
    }
    REP(i, N) cout << a[ord[i]] << " \n"[i == N-1];
}
