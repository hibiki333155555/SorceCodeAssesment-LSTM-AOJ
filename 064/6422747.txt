#include <bits/stdc++.h>
#define rep(i,n) for(int i = 0; i < (n); i++)
using namespace std;
typedef long long ll;

int main(){
    cin.tie(0);
    ios::sync_with_stdio(0);
    
    // 300,000 yen : oooooo
    //   4,000 yen : xxoooo
    //     500 yen : xxxxoo

    ll n; cin >> n;
    map<string,ll> mp4, mp2;
    rep(i,n) {
        string num; cin >> num;
        mp4[num.substr(2, 4)]++;
        mp2[num.substr(4, 2)]++;
    }

    vector<pair<ll,string>> four, two;
    for(auto [num, cnt] : mp4) four.push_back({cnt, num});
    for(auto [num, cnt] : mp2) two.push_back({cnt, num});
    four.push_back({0, "xxxx"});
    sort(four.rbegin(), four.rend());
    sort(two.rbegin(), two.rend());

    ll ans = 300'000;
    for(auto [cnt4, num4] : four) {
        vector<ll> a;
        for(int i = 0; i < (int)two.size() && (int)a.size() < 4; i++) {
            auto [cnt2, num2] = two[i];
            if(num4.substr(2, 2) != num2) a.push_back(cnt2);
        }

        int len = (int)a.size();
        vector<ll> sum(len + 1, 0);
        for(int i = 0; i < len; i++) sum[i + 1] += sum[i] + a[i];

        for(ll cnt2 : sum) {
            ll cur = 0;
            if(mp2[num4.substr(2, 2)] + cnt2 < n) cur += 300'000;
            cur += 4'000 * cnt4;
            cur +=   500 * cnt2;
            ans = max(ans, cur);
        }
    }
    cout << ans << endl;
}
