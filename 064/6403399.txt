#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for (int i = 0; i < (n); ++i)

struct Node {
    int key;
    Node *left, *right, *parent;
};

Node *root, *NIL;

void insert(int key){
    Node *x = root;
    Node *y = NIL;
    Node *z;

    z = (Node *)malloc(sizeof(Node));
    z->key = key;
    z->left = NIL;
    z->right = NIL;
//節点を探す,葉になるまで  
    while(x!=NIL){
        y = x;
        if(x->key > z->key){
            x = x->left;
        } else {
            x = x->right;
        }
    }

    z->parent = y;
    if(y == NIL){
        root = z;
    } else {
        if(z->key < y->key){
            y->left = z;
        } else {
            y->right = z;
        }
    }
}

void inorder(Node *u){
    if(u == NIL) return;
    inorder(u->left);
    printf(" %d", u->key);
    inorder(u->right);
}

void preorder(Node *u){
    if(u == NIL) return;
    printf(" %d", u->key);
    preorder(u->left);
    preorder(u->right);
}

int main() {
    int n;
    cin >> n;
    rep(i, n){
        string com;
        cin >> com;
        if(com == "insert"){
            int a;
            cin >> a;
            insert(a);
        } else {
            inorder(root);
            cout << endl;
            preorder(root);
            cout << endl;
        }
    }
    return 0;
}
