import java.util.*;
public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int n = sc.nextInt();
		
		HashMap<String, Integer> map = new HashMap<>();
		
		for(int i = 0; i < n; i++) {
			if(sc.nextInt() == 0) {
				map.put(sc.next(), sc.nextInt());
			}
			else {
				System.out.println(map.get(sc.next()));
			}
		}
	}
}
