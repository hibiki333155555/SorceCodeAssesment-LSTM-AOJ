import java.util.Scanner;

// 部屋の情報
class Room {
    // 部屋番号
    private int roomnumber;

    // 人数
    private int number;

    public Room() {
        roomnumber = 0;
        number = 0;
    }

    public void setRoomNumber(int roomnumber) {
        this.roomnumber = roomnumber;
    }

    // 人数を設定する
    public void setNumber(int number) {
        this.number = number;
    }

    public void addNumber(int number) {
        this.number = this.number + number;
    }

    public void subNumber(int number) {
        this.number = this.number + number;
    }

    public void updateNumber(int number) {
        if (number >= 0) {
            addNumber(number);
        } else {
            subNumber(number);
        }
    }

    public int getRoomnumber() {
        return roomnumber;
    }

    public int getNumber() {
        return number;
    }
}

// 階の情報
class Floor extends Room {
    // 1フロア10部屋ある
    Room[] rooms = new Room[10];

    public Floor() {
        for (int i = 0; i < rooms.length; i++) {
            rooms[i] = new Room();
        }
    }

    public void setRooms(Room[] rooms) {
        this.rooms = rooms;
    }

    public Room[] getRooms() {
        return rooms;
    }
}

// 棟の情報
class Building extends Floor {
    // 棟一個に3階ある
    Floor[] floors = new Floor[3];

    public Building() {
        for (int i = 0; i < floors.length; i++) {
            floors[i] = new Floor();
        }
    }

    // 棟の情報を表示
    public void showBuilding() {
        for (int i = 0; i < floors.length; i++) {
            Room[] rooms = floors[i].getRooms();
            for (int j = 0; j < rooms.length; j++) {
                // 今住んでいる人数を表示
                System.out.print(" " + rooms[j].getNumber());
            }
            System.out.println("");
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // クラスを宣言
        // 4棟ある
        Building[] buildings = new Building[4];

        //　初期化
        for (int i = 0; i < buildings.length; i++) {
            buildings[i] = new Building();
        }

        // 情報を読み込む
        int n = sc.nextInt();
        for (int i = 0; i < n; i++) {
            int buildingsnum = sc.nextInt() - 1;
            int floorsnum = sc.nextInt() - 1;
            int roomsnum = sc.nextInt() - 1;
            int numbers = sc.nextInt();

            // 入居していなければ入居させる
            if (buildings[buildingsnum].floors[floorsnum].rooms[roomsnum].getNumber() == 0) {
                buildings[buildingsnum].floors[floorsnum].rooms[roomsnum].setRoomNumber(roomsnum);
                buildings[buildingsnum].floors[floorsnum].rooms[roomsnum].setNumber(numbers);
            } else {
                // すでに入居しているなら人数を更新する
                buildings[buildingsnum].floors[floorsnum].rooms[roomsnum].updateNumber(numbers);
            }
        }

        // 棟の情報を表示
        for (int i = 0; i < buildings.length; i++) {
            if (i != 0) {
                System.out.println("####################");
            }
            buildings[i].showBuilding();
        }
    }
}
