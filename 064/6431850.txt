template<class T>
class RSQ_tree{
    T E;
    vector<T> dat;
    int N;
    T _RSQ(int qL, int qR, int k, int l, int r){
        if(r <= qL || qR <= l) return E;
        if(qL <= l && r <= qR) return dat[k];
        int vl = _RSQ(qL, qR, (k << 1) + 1, l, (l + r) >> 1);
        int vr = _RSQ(qL, qR, (k << 1) + 2, (l + r) >> 1, r);
        return vl + vr;
    }

    public:
    RSQ_tree(int _N) : E(0) {
        for(N = 1; N < _N; N <<= 1);
        dat.resize((N << 1) - 1, E);
    }
    void add(int idx, T x){
        idx += N - 1;
        dat[idx] += x;
        while(idx > 0){
            idx = (idx - 1) >> 1;
            dat[idx] = dat[(idx << 1) + 1] + dat[(idx << 1) + 2];
        }
    }
    T RSQ(int l, int r){ return _RSQ(l, r, 0, 0, N); }
};

void Main(){
    srand(time(nullptr));
    int N, Q; cin >> N >> Q;
    RSQ_tree<int> sg(N);
    while(Q--){
        int a, b, c; cin >> a >> b >> c;
        if(a == 0) sg.add(b-1, c);
        else cout << sg.RSQ(b-1, c) << endl;
    }
    if(rand() % 8 == 0) cout << "p" << endl;
}
