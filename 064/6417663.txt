/* function */
template<class T>
T GCD(T a, T b) {
	if (b != 0) return GCD(b, a % b);
	return a;
}
template<class T>
T LCM(T a, T b) {
	return a / GCD(a, b) * b;
}

const int dy[4] = { 0, 1, -1, 0 };
const int dx[4] = { 1, 0, 0, -1 };

int main(){
	int n,k,tu=0;
	cin >> n >> k;
	vector<int> A(n);
	rep(i,0,n) {
		cin >> A[i];
		if(A[i]==1) tu+=2;
		else if(A[i]!=6) tu++;
	}
	ll ans=0;
	if(tu<k) {
		cout << n*6 << endl;
		return 0;
	}
	int index=0;
	rep(i,0,n) {
		if(A[i]==6) {
			ans+=A[i];
			continue;
		}else if(A[i]==1){
			ans+=5;
			k--;
		}else{
			ans+=6;
			k--;
		}
		if(k<=0){
			index=i+1;
			break;
		}
	}
	rep(i,index,n) {
		ans+=A[i];
	}
	cout << ans << endl;
}
