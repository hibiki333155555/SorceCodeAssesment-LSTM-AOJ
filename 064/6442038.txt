#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int H,W,i,j;
	
	cin>>H>>W;
	while(!(H==0&&W==0))
	{
	
		for(i=0;i<H;i++)
		{
			if(i==0||i==H-1)
				for(j=0;j<W;j++)
					cout<<"#";	
			else
			{
				cout<<"#";
				for(j=0;j<W-2;j++)
					cout<<".";	
				cout<<"#";
			}
			
			cout<<endl;
		
		}
		
		
		cout<<endl;
		cin>>H>>W;
		
	}
	
return 0;
}
