#include <bits/stdc++.h>	
using namespace std;	
typedef long long ll;

int main(){
  int n;
  cin>>n;
  vector<int>a(n);
  for(int i=0;i<n;i++){
    cin>>a[i];
  }
  bool flag=1;
  int ans=0;
  while(flag){
    flag=0;
    for(int i=n-1;i>0;i--){
      if(a[i]<a[i-1]){
        swap(a[i],a[i-1]);
        flag=1;
        ans++;
      }
    }
  }
  for(int i=0;i<n;i++){
    cout<<a[i];
    if(i<n-1){
      cout<<" ";
    }
    else{
      cout<<endl;
    }
  }
  cout<<ans<<endl;
 return 0;
  
}
