import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a, b, c;
        a = sc.nextInt();
        b = sc.nextInt();
        c = sc.nextInt();

        int ans = 0;
        for (int i = a; i <= b; i++) {
            // aからbまでの整数の中にcの約数がいくつあるか数える
            if (c % i == 0) {
                ans++;
            }
        }
        System.out.println(ans);
    }
}
