template<class T> inline T gcd(T a, T b) {
  while (b > 0) {
    T r = a % b;
    a = b;
    b = r;
  }

  return a;
}

// gcd(a, b) * lcm(a, b) == a * b
template<class T> inline T lcm(T a, T b) {
    return a * b / gcd(a, b);
}
