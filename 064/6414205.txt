import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        while(true){
            int x = sc.nextInt();
            int y = sc.nextInt();
            int tmp;
            
            if(x==0 && y==0)break;
            
            if(x < y){
                tmp = x;
                x = y;
                y = tmp;
            }
            System.out.printf("%d %d\n",x,y);
        }
    }
}
