#include <iostream>
#include <iomanip>
#include <complex>

using namespace std;

complex<double> in()
{
	double x, y;
	cin >> x >> y;
	return {x, y};
}

int main()
{
	int q;
	for (cin >> q; q > 0; --q) {
		complex<double> p0 = in(), p1 = in(), p2 = in(), p3 = in();
		p1 -= p0; p3 -= p2;
		auto op = p1.real() * p3.imag() - p3.real() * p1.imag();
		auto ip = p1.real() * p3.real() + p1.imag() * p3.imag();
		cout << (op == 0 ? 2 : ip == 0 ? 1 : 0) << endl;
	}
	return 0;
}
