#include <iostream>
using namespace std;

int isPrime(int x) {
    // 2より小さい整数は素数ではない
    if (x < 2) return 0;
    // 2は素数である
    if (x == 2) return 1;
    // 2以外の偶数は素数ではない
    if (x % 2 == 0) return 0;
    // 3以降の整数では、xの平方根以下の間までに割り切れる整数があるか判定していく
    for (int i = 3; i*i <= x; i+=2) {
        if (x % i == 0) return 0;
    }
    // 割り切れる整数がなければxは素数である
    return 1;
}

int main() {
    int n, x, cnt = 0;
    cin >> n;

    for (int i = 0; i < n; ++i) {
        cin >> x;
        if (isPrime(x)) cnt++;
    }

    cout << cnt << endl;
}
