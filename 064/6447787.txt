import java.util.Arrays;
import java.util.Scanner;

class Main {
	static public void main(String[] args) {
		// input 3 integers in scanner
		Scanner sc = new Scanner(System.in);
		int a, b, c;
		a = sc.nextInt();
		b = sc.nextInt();
		c = sc.nextInt();
		sc.close();

		// sort them
		int[] arr = { a, b, c };
		Arrays.sort(arr);

		// output the result
		System.out.println(arr[0] + " " + arr[1] + " " + arr[2]);

	}
}
