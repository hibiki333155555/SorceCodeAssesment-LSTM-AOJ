#include <bits/stdc++.h>
using namespace std;
const double pi = 3.1415926535897932384626533832795;

int main()
{
	vector<int> a(0);
	int q, type, x;
	cin >> q;
	for (int i = 0; i < q; i++){
		cin >> type;
		if (type == 0){
			cin >> x;
			a.push_back(x);
		}else if(type == 1){
			cin >> x;
			cout << a.at(x) << endl;
		}else{
			a.pop_back();
		}
	}
}
