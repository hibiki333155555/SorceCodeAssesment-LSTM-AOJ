#include "bits/stdc++.h"
using namespace std;
int main()
{
  int r1, r2;
  string c1, c2;
  cin >> c1 >> r1 >> c2 >> r2;

  map<string, int> row_to_num;
  for (char i = 'A'; i <= 'Z'; ++i)
  {
    string s;
    s.push_back(i);
    row_to_num[s] = i - 'A';
  }
  for (char i = 'A'; i <= 'Z'; ++i)
  {
    for (char j = 'A'; j <= 'Z'; ++j)
    {
      string s;
      s.push_back(i);
      s.push_back(j);
      row_to_num[s] = 26 + (i - 'A') * 26 + j - 'A';
    }
  }
  cout << (r2 - r1 + 1) * (row_to_num[c2] - row_to_num[c1] + 1) << endl;
}
