#include<stdio.h>
#include<stdlib.h>
int N, A[100], i, j, k;
void d() {
    int x;
    for ( x=0; x<N; x++ )
    printf("%d%s", A[x], x==N-1?"/n":" ");
}
int main()  {
    scanf("%d", &N);
    for (i=0; i<N; i++) 
        scanf("%d", &A[N]);
    for (i=0; A[j+1]=k; i++) 
      for (k=A[i], j=i-1,d(); A[j]>k; j--)
     A[j+1]=A[j];
  d();
  return 0;
}
