#include<iostream>
using namespace std;
int main(){
  int r, c;
  cin >> r >> c;
  
  //r+1×c+1行列の定義
  int table[r+1][c+1];
  
  //初期化
  for (int i=0; i<=r; i++){
   for (int j=0; j<=c; j++){
     table[i][j] = 0;
   }
  }
  
  //r+1×c+1行列の取得
  for (int i=0; i<r; i++){
    for (int j=0; j<c; j++){
      cin >> table[i][j]; 
      table[r][j] += table[i][j];
      table[i][c] += table[i][j];
      table[r][c] += table[i][j];
      
    }
  }
  
  //出力
  for (int i=0; i<=r; i++){
    for (int j=0; j<c; j++){
      cout << table[i][j] << " ";
    }
    cout << table[i][c] << endl;
  }
  
  
}
