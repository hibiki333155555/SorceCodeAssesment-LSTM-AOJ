#include<iostream>
#include<string>
using namespace std;
int main(){
    int n;
    string s;
    cin >> n >> s;
    int a[5];
    for(int i=0; i<5; i++){
        a[i]=0;
    }
    for(int i=0; i<n; i++){
        if(s[i]=='A')
            a[0]=1;
        if(s[i]=='B')
            a[1]=1;
        if(s[i]=='C')
            a[2]=1;
        if(s[i]=='D')
            a[3]=1;
        if(s[i]=='E')
            a[4]=1;
    }
    int sum=0;
    for(int i=0; i<5; i++){
        sum+=a[i];
    }
    if(sum>=3){
        cout << "Yes" << endl;
    }
    else{
        cout << "No" << endl;
    }
}
