import java.math.BigDecimal;
import java.math.RoundingMode;
import java.text.DecimalFormat;
import java.util.Scanner;

public class Main {
	
	@SuppressWarnings("resource")
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);
		int a = sc.nextInt();
		int b = sc.nextInt();
		
		BigDecimal intDecimal1 = new BigDecimal(a);
		BigDecimal intDecimal2 = new BigDecimal(b);

		String format = "0.00000";
		DecimalFormat decimalFormat = new DecimalFormat(format);
		
		int d = a / b;
		int r = a % b;
		BigDecimal bd = intDecimal1.divide(intDecimal2, 8 ,RoundingMode.HALF_DOWN);
		System.out.println(d + " " + r + " " + decimalFormat.format(bd));
    }
}
