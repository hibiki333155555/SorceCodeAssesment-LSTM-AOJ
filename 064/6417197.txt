#include <iostream>
#include <vector>

using namespace std;
using uint = unsigned int;
using ll = long long;
using ull = unsigned long long;
constexpr ll TEN(int n) { return (n == 0) ? 1 : 10 * TEN(n - 1); }
const ll MOD = 998244353;
template <class T> using V = vector<T>;
template <class T> using VV = V<V<T>>;

pair<ll,ll> dp[3010][3010][2]; //dp[i桁目][立ってるbitの個数][同じかどうか] = {通り数,総和}

int main() {

    string N;int k;
    cin>>N;
    cin>>k;
    dp[0][0][1] = {1,0};
    for(int i = 0; (int)N.size() > i; i++){
      for(int j = 0; k >= j; j++){
        if(N[i] == '1'){
          dp[i+1][j+1][1].first = (dp[i+1][j+1][1].first+dp[i][j][1].first)%MOD;
          dp[i+1][j+1][1].second = (dp[i+1][j+1][1].second+dp[i][j][1].second*2+dp[i][j][1].first)%MOD;
          dp[i+1][j][0].first = (dp[i+1][j][0].first+dp[i][j][1].first+dp[i][j][0].first)%MOD;
          dp[i+1][j][0].second = (dp[i+1][j][0].second+(dp[i][j][1].second+dp[i][j][0].second)*2)%MOD;
          dp[i+1][j+1][0].first = (dp[i+1][j+1][0].first+dp[i][j][0].first)%MOD;
          dp[i+1][j+1][0].second = (dp[i+1][j+1][0].second+(dp[i][j][0].second*2+dp[i][j][0].first))%MOD;
        }else{
          dp[i+1][j][1].first = (dp[i+1][j][1].first+dp[i][j][1].first)%MOD;
          dp[i+1][j][1].second = (dp[i+1][j][1].second+dp[i][j][1].second*2)%MOD;
          dp[i+1][j][0].first = (dp[i+1][j][0].first+dp[i][j][0].first)%MOD;
          dp[i+1][j][0].second = (dp[i+1][j][0].second+dp[i][j][0].second*2)%MOD;
          dp[i+1][j+1][0].first = (dp[i+1][j+1][0].first+dp[i][j][0].first)%MOD;
          dp[i+1][j+1][0].second = (dp[i+1][j+1][0].second+dp[i][j][0].second*2+dp[i][j][0].first)%MOD;
        }
      }
    }
    // for(int i = 0; N.size() >= i; i++){
    //   for(int j = 0; k >= j; j++){
    //     cout << "{(" << dp[i][j][1].first << "," << dp[i][j][1].second << ") (" << dp[i][j][0].first << "," << dp[i][j][0].second << ")} ";
    //   }
    //   cout << endl;
    // }
    cout << (dp[N.size()][k][1].second+dp[N.size()][k][0].second)%MOD << endl;
    return 0;
}
