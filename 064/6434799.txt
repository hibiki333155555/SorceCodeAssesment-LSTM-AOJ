#include <stdio.h>
#define N 100000

int G[N][N];
int visited[N];

int ok;

void dfs(int from, int tar) {
  if (ok)
    return;
  visited[from] = 1;
  for (int i = 0; G[from][i] != -1; i++) {
    int next_v = G[from][i];
    if (visited[next_v]) {
      continue;
    }
    if (next_v == tar) {
      ok = 1;
    } else {
      dfs(next_v, tar);
    }
  }
}

int main(void) {
  int n, m;
  scanf("%d %d", &n, &m);
  int a, b, k = 0, tmp = -1;
  for (int i = 0; i < m; i++) {
    for (int k = 0; k < m; k++) {
      G[i][k] = -1;
    }
  }

  for (int i = 0; i < m; i++) {
    scanf("%d %d", &a, &b);
    if (tmp != a) {
      k = 0;
    }
    tmp = a;
    G[a][k++] = b;
  }

  int q;
  scanf("%d", &q);
}
