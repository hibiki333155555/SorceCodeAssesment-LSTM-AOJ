#include <iostream>
#include <iomanip>
#include <vector>
#include <complex>

using namespace std;

complex<double> in() {double x, y; cin >> x >> y; return {x, y};}

int main()
{
	int n; cin >> n;
	vector<complex<double>> P(n);
	for (auto& p : P) p = in();
	double d = abs(P[0] - P[1]);
	for (int i = 0; i < n; ++i) for (int j = i + 1; j < n; ++j) d = min(abs(P[i] - P[j]), d);
	cout << fixed << setprecision(6) << d << endl;
	return 0;
}
