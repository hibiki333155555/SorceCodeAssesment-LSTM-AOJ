int mat[101][101];

int main() {


	/*
	(void)scanf("%d", &);
	(void)scanf("%d %d", & ,& );
	*/
	
	memset(mat, 0, sizeof(mat));

	int n; cin >> n;

	rep(i, n) {
		int u, k; (void)scanf("%d%d", &u, &k);
		rep(j, k) {
			int x; (void)scanf("%d", &x);
			mat[u - 1][x - 1] = 1;
		}
	}

	rep(i, n) {
		rep(j, n-1) {
			printf("%d ", mat[i][j]);
		}
		printf("%d\n", mat[i][n - 1]);
	}

	return 0;
