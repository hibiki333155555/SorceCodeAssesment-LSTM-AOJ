using namespace std;

int main(){
    int a,b;
    string s;
    int ans;
    while(true){
        cin >> a >> s >> b;
        if (s == "?") break;
        if (s=="+") ans = a+b;
        else if (s=="-") ans = a-b;
        else if (s=="*") ans = a*b;
        else ans = a/b;
        cout << ans << endl;
    }
    //system("pause");
    return 0;
}
