#include <iostream>
using namespace std;
int selectionSort(int A[], int N){
    for(int i = 0; i < N - 1; i++){
        int x = 0;
        for(int j = i; j < N; j++){
            int minj = 101, n;
            if(A[j] < minj) {
                min = A[j], n = j;
            }
        }
        if(n != i) swap(A[n], A[i]);
        x++;
    }
    return x;
}
int main(){
    int N, A[100];
    cin >> N;
    for(int i = 0; i < N; i++) cin >> A[i];
    x = selectionSort(A, N)
    for(int i = 0; i < N; i++){
        if(i) cout << " ";
        cout << A[i];
    }
    cout << endl;
    cout << x << endl;
    return 0;
}
