import java.util.Scanner;

public class Main {
    void solve() {
        Scanner sc = new Scanner(System.in);
        int dress = sc.nextInt();
        int party = sc.nextInt();
        System.out.println((int) Math.ceil(1.0 * party / dress));
    }

    public static void main(String[] args) {
        new Main().solve();
    }
}
