#include<iostream>
#include<string>
using namespace std;
int main() {
    string x;
    while(1) {
        cin >> x;
        if(x == "0") {
            break;
        }
        int sum = 0;
        for(int count = 0; count < x.size(); count++) {
            sum += x[count] - '0';
        }
        cout << sum << endl;
    }
}
