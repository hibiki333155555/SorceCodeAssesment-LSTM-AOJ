#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int a, x;
  cin >> a;
  
  int m = 10000000;
  int M = -10000000;
  long long int sum=0;
  
  for (int i = 0; i < a; i++){
    cin >> x;
    m = min(m, x);
    M = max(M, x);
    sum = sum + x;
  }
  
  cout << m << " " << M << " " << sum << endl;
}
