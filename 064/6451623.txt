import java.util.Scanner;

public class Main{
	public static void main(String[] args) {
		Scanner scan=new Scanner(System.in);

		int count=0;//除数カウンター変数
		int a=scan.nextInt();
		int b=scan.nextInt();
		int c=scan.nextInt();

		while(a<=b) {//aからbまで一つずつ
			if(c%a==0) {//除数か検討
				count++;//カウント
			}
			a++;//繰り返し変数
		} 
		System.out.println(count);//出力
	}
}
