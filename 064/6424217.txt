#include <bits/stdc++.h>
using namespace std;
int main() {
    int r, c;
    cin >> r >> c;
    vector<vector<int>> g(r + 1, vector<int>(c + 1));
    int total = 0;
    for (int i=0; i<r; i++) {
        int row_sum = 0;
        for (int j=0; j<c; j++) {
            cin >> g[i][j];
            row_sum += g[i][j];
        }
        g[i][c] = row_sum;
        total += row_sum;
    }
    for (int i=0; i<c; i++) {
        int line_sum = 0;
        for (int j=0; j<r; j++) {
            line_sum += g[j][i];
        }
        g[r][i] += line_sum;
    }
    g[r][c] = total;
    for (int i=0; i<r+1; i++) {
        for (int j=0; j<c+1; j++) {
            if (j == c) cout << g[i][j] << endl;
            else cout << g[i][j] << " ";
        }
    }
    return 0;
}
