#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#define MAX 100000000
void insert(int);
void delete(int);
void deleteF();
void deleteL();
void show();
typedef struct node{
    struct node *prev;
    int date;
    struct node *next; 
}nd;
nd head;
int main(){
    head.date=MAX;
    head.prev=head.next=&head;
    int n,num;
    char order[12];
    scanf("%d",&n);
    while(n--){
        scanf("%s",order);
        if(0==strcmp(order,"insert")){
            scanf("%d",&num);insert(num);
        }else if(0==strcmp(order,"delete")){
            scanf("%d",&num);delete(num);
        }else if(0==strcmp(order,"deleteFirst")){
            deleteF();
        }else if(0==strcmp(order,"deleteLast")){
            deleteL();
        }else exit(8);
        //show();
    }
    show();
    return 0;
}
void insert(int x){
    nd *tmp=(nd *)malloc(sizeof(nd));
    tmp->prev=&head;
    tmp->next=head.next;
    head.next->prev=tmp;
    head.next=tmp;
    tmp->date=x;
}
void delete(int x){
    nd *tmp=head.next;
    while(1){
        if(tmp->date==MAX)return;
        else if(tmp->date==x){
            tmp->prev->next=tmp->next;
            tmp->next->prev=tmp->prev;
            free(tmp);
            return;
        }
        tmp=tmp->next;
    }
}
void deleteF(){
    delete(head.next->date);
}
void deleteL(){
    nd *tmp=head.prev;
    tmp->prev->next=tmp->next;
    tmp->next->prev=tmp->prev;
    free(tmp);
}
void show(){
        nd *tmp=head.next;
    while(1){
        if(tmp->next->date==MAX){printf("%d\n",tmp->next->date);break;}
        printf("%d ",tmp->date);
        tmp=tmp->next;
    }
}
