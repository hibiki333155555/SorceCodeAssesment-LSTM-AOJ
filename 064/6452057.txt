using namespace std;
typedef long long int ll;
typedef unsigned long long ull;

mt19937_64 rng(chrono::steady_clock::now().time_since_epoch().count());
ll myRand(ll B) {
    return (ull)rng() % B;
}
inline ll time() {
    return static_cast<long double>(chrono::duration_cast<chrono::nanoseconds>(chrono::steady_clock::now().time_since_epoch()).count()) * 1e-9;
}

int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    int h,w; cin >> h >> w;
    vector<ll> a(h),b(w);
    for(int i=0;i<h;i++){
        cin >> a[i];
    }
    for(int i=0;i<w;i++){
        cin >> b[i];
    }
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    a.push_back(2e9);
    b.push_back(2e9);
    ll res = 0;
    for(int i=0,j=0;i<h and j<w;){
        if(a[i] <= b[j]){
            res += a[i]*(w-j);
            i++;
        }
        else{
            res += b[j]*(h-i);
            j++;
        }
    }
    cout << res << endl;
}
