int T(int i){
    if(i>=11 && i<=14) return 0;
    else if(i>=18 && i<=20) return 1;
    else if(i>=21 || i<=1) return 2;
    else return 4;
}
 
int cnt[5];
int ok[5];
 
void pt(int i){
    if(cnt[i] == 0){
        printf("no guest\n");
    }else{
        printf("%.0lf\n",(double)100*ok[i]/cnt[i]);
    }
}
