using System;
using System.Linq;

namespace AOJ_ITP1_9_A
{
    internal class Program
    {
        public static void Main()
        {
            var W = Console.ReadLine();
            var line = string.Empty;
            var Ti = Enumerable.Empty<string>();
            while ((line = Console.ReadLine()) != "END_OF_TEXT")
            {
                Ti = Ti.Concat(line.Split());
            }
            Console.WriteLine(Ti.Count(t => t.Equals(W, StringComparison.OrdinalIgnoreCase)));
        }
    }
}
