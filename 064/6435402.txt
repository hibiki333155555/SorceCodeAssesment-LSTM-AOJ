#include<bits/stdc++.h>
using namespace std;

int main()
{
    map<string,int>mp;

    int q,m,x;
    //char l,r;
    string key,l,r;
    cin>>q;

    for(int i=0;i<q;i++)
    {
        cin>>m;
        if(m==0)
        {
            cin>>key>>x;
            mp[key]=x;
        }
        else if(m==1)
        {
            cin>>key;
            if(!mp.empty())
            {
                cout<<mp[key]<<endl;
            }
            else
            {
                cout<<"0"<<endl;
            }
        }
        else if(m==2)
        {
            cin>>key;
            mp.erase(key);
        }
        else
        {
            cin>>l>>r;
            map<string,int>::iterator it;
            for(it=mp.lower_bound(l);it!=mp.upper_bound(r);it++)
            {
                cout<<it->first<<" "<<it->second<<endl;
            }

        }
    }
}
