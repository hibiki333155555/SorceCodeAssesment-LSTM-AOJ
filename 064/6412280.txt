#include<stdio.h>
int main(){
    int r,c,i,j,sum=0,q,z;
    scanf("%d %d",&r,&c);
    int a[r][c];
    q=c+1;
    z=r+1;
    for(i=0;i<r;i++){
        for(j=0;j<c;j++)
        scanf("%d",&a[i][j]);
    }
    int b[r+1][c+1];
     for(i=0;i<r;i++){
        for(j=0;j<c;j++)
        b[i][j]=a[i][j];
    }
    for(i=0;i<r;i++){
        for(j=0;j<c;j++)
        sum+=a[i][j];
        b[i][q-1]=sum;
        sum=0;}
        for(j=0;j<q;j++){
        for(i=0;i<r;i++)
        sum+=b[i][j];
        b[z-1][j]=sum;
        sum=0;}
    for(i=0;i<z;i++){
        for(j=0;j<q;j++)
        if(j==q-1)
        printf("%d",b[i][j]);
        else
        printf("%d ",b[i][j]);
        printf("\n");
    }
}
