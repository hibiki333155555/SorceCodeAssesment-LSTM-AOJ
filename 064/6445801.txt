#include <iostream> // cout, endl, cin
#include <vector> // vector
using namespace std;

int main() {
	int n;
	cin >> n;
	vector<vector<int>> v(n, vector<int>(n,0));
	for (int i = 0; i < n; i++) {
		int u, k;
		cin >> u >> k;
		vector<int> s(k);
		for (int j = 0; j < k; j++) {
			cin >> s.at(j);
			v.at(i).at(s.at(j) - 1) = 1;
		}
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			if (j == n - 1) {
				cout << v.at(i).at(j) << endl;
			}
			else {
				cout << v.at(i).at(j) << " ";
			}
		}
	}
}
