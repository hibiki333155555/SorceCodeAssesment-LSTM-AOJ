using namespace std;
bool  solve(int h,int w,vector<vector<bool>>&vec){
    queue<pair<int,int>> q;
  bool flag=false;
  for (int i = 0; i < h; i++){
    for(int j=0;j<w;j++){
      if(vec[i][j]==true){
        //親決定
        q.push(make_pair(i,j));
        flag=true;
        break;
      }
    }
    if(flag==true)break;
  }
  if(flag==false)return false;//もう島がない
  while (q.empty()==false)
  {
    pair<int,int> v =q.front();q.pop();
    vec[v.first][v.second]=false;
    for (int dx = -1; dx <=1; dx++)
    {
        for (int dy = -1; dy <=1; dy++)
      {
          if(dx==0&&dy==0)continue;
          int x=v.first+dx;
          int y=v.second+dy;
          if(x<0||x>=h)continue;
          if(y<0||y>=w)continue;
          if(vec[x][y]==true){
            q.push(make_pair(x,y));
          }
      }
    }
    
  }
  return true;//まだ島があった
}
int main() {
  while (true)
  {
    
    int w,h;
    cin >>w>>h;
    if(w==0&&h==0)break;
    vector<vector<bool>> vec(h, vector<bool>(w));
    for (int i = 0; i < h; i++){
      for(int j=0;j<w;j++){
        int temp;

        cin >> temp;
        if(temp==1){vec[i][j]=true;
        }else {vec[i][j]=false;}
      }
    }
    bool ret=true;
    int count=0;
    while(ret==true){
      ret =solve(h,w,vec);
      if(ret==true)count++;
    }
    cout << count << endl;
    vec.clear();
  }

return 0;
}
