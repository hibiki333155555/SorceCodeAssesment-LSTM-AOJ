#include <bits/stdc++.h>

using namespace std;

int binary_search_iterator(vector<int> v, int left, int right, int item)
{
  int mid = (left + right) / 2;

  if (v[mid] == item)
    return 1;
  if (v[mid] < item)
    left = mid + 1;
  if (v[mid] > item)
    right = mid;
  if (left == right)
    return 0;

  return binary_search_iterator(v, left, right, item);
}

int binary_search(vector<int> v, int item)
{
  int left = 0;
  int right = v.size();

  return binary_search_iterator(v, left, right, item);
}

int main()
{
  int n, q;
  cin >> n;
  vector<int> S(n);
  int s;

  for (int i = 0; i < n; i++)
  {
    cin >> s;
    S.push_back(s);
  }

  cin >> q;
  long long int ans = 0;

  for (int i = 0; i < q; i++)
  {
    int t;
    cin >> t;
    if (binary_search(S, t) == 1)
    {
      ans++;
    }
  }

  cout << ans << endl;
  return 0;
}
