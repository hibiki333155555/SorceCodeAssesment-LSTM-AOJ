#include<bits/stdc++.h>
using namespace std;

int main(){
  int N,A=0,Un=0;
  cin >> N;
  bool flag = true;
  for(int i=0;i<N;i++){
    string S;
    cin >> S;
    if(S == "A") A++;
    else Un++;
    if(A < Un) flag = false;
  }
  if(A != Un) flag = false;
  if(flag) cout << "YES" << endl;
  else cout << "NO" << endl;
}
