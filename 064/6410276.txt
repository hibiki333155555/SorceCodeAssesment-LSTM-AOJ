#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main(){
	cin.tie(nullptr);
	ios::sync_with_stdio(false);
	int n; cin >> n;
	string s; cin >> s;
	int res=0;
	for(int i=0;i<n;i++){
		if(i%2==0 and s[i]!='I')res++;
		if(i%2==1 and s[i]!='O')res++;
	}	
	cout << res << endl;
}
