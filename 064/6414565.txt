#include <bits/stdc++.h>
using namespace std;

constexpr int mod = 998244353;

int main() {
    int N;
    cin >> N;
    vector<int>A(N+1),B(N+1);
    for(int i = 1; i <= N; i++) {
        cin >> A[i];
        B[i+A[i]-1]++;
    }
    long long ans = 1;
    for(int i = N; i >= 1; i--) {
        ans *= B[i];
        ans %= mod;
        B[i-1] += B[i]-1;
    }
    cout << ans << endl;
}
