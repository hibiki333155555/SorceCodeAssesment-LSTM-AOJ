#include <bits/stdc++.h>
using namespace std;

int main() {
    
    int N;
    cin >> N;

    vector<int> A(N);
    for (int i = 0; i < N; i++) {
        cin >> A.at(i);
    }

    for (int i = 0; i < N; i++) {
        int k = A[i], j = i - 1;
        while (j >= 0 && A[j] > k) {
            A[j + 1] = A[j];
            j--;
        }
        A[j + 1] = k;

        for (int j = 0; j < N - 1; j++) cout << A[j] << ' ';
        cout << A[N - 1] << endl;;
    }

}
