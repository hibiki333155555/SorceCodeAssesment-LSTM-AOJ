n, m, l = gets.split(' ').map(&:to_i)
a = (0..n-1).map { gets.split.map(&:to_i) }
b = (0..m-1).map { gets.split.map(&:to_i) }
c = (0..n-1).map { (0..l-1).map { 0 } }

n.times do |i|
    l.times do |j|
        m.times do |k|
            c[i][j] += a[i][k] * b[k][j]
        end
    end
end
c.each { |row| puts row.join(' ') }
