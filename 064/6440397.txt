#include<bits/stdc++.h>
using namespace std;
int a[10], b[10], sa, si;
int main() {
//1 2 3 4 5 6
//6 2 4 3 5 1
	for (int i = 1; i <= 6; i++) cin>>a[i];
	for (int i = 1; i <= 6; i++) cin>>b[i];
	for (int i = 1; i <= 6; i++) {
		if (a[i] == b[i]) {
			sa++;
			si = i;//3 /4  2 /5  1 /6
		}
	}
	if (sa >= 5) {
		cout<<"Yes"<<endl;
		return 0;
	}
	if (sa != 2) {
		cout<<"No"<<endl;
		return 0;
	}
	if (si == 4) {
		if (a[3] != b[3]) {
			cout<<"No"<<endl;
			return 0;
		}
		if ((a[1]==b[6] && b[1]==a[6]) && (a[2]==b[5] && b[2]==a[5])) {
			cout<<"Yes"<<endl;
			return 0;
		}
	}
	if (si == 5) {
		if (a[2] != b[2]) {
			cout<<"No"<<endl;
			return 0;
		}
		if ((a[1]==b[6] && b[1]==a[6]) && (a[3]==b[4] && b[3]==a[4])) {
			cout<<"Yes"<<endl;
			return 0;
		}
	}
	if (si == 6) {
		if (a[1] != b[1]) {
			cout<<"No"<<endl;
			return 0;
		}
		if ((a[2]==b[5] && b[2]==a[5]) && (a[3]==b[4] && b[3]==a[4])) {
			cout<<"Yes"<<endl;
			return 0;
		}
	}
	cout<<"No"<<endl;
	
//固定面 25 16 34   有一个面不变 
	
	
	
	
	return 0;
}
