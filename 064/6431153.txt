#include <bits/stdc++.h>
using namespace std;
int main(){
    
    int n,count=0; cin>>n;
    vector<int>v(n);
    for(int i=0;i<n;i++)
        cin>>v[i];
        
    for(int i=0;i<n;i++){
        for(int j=n-1;j>=0;j--){
            if(v[j]<v[j-1]){
                swap(v[j],v[j-1]);
                count++;
            }
        }
    }
    
    
    for(int i=0;i<n;i++)
        cout<<v[i]<<" ";
        
    cout<<"\n";
    cout<<count<<"\n";
    return 0;
}
