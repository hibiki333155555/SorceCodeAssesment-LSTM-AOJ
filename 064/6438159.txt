#include <iostream>
using namespace std;
#define NIL -1
#define N 44
int F[N];

int fib(int n){
    if (n==0||n==1) return F[n]=1;
    if (F[n]!=NIL) return F[n];
    return F[n] =fib(n-1)+fib(n-2);
}
int main(void){
    // Your code here!
    int n;
    for(int i=0;i<N;i++){F[i]=NIL;}
    cin>>n;
    cout<<fib(n)<<endl;
}
