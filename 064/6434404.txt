#include<stdio.h>
int main(void){
    int bage[300001];
    for(int i=0;i<300001;i++) bage[i]=0;
    int n,Q,c,j;
    scanf("%d %d",&n,&Q);
    for(int i=0;i<n;i++){
        scanf("%d",&c);
        bage[c]=1;
    }
    for(int i=0;i<Q;i++){
        int q;
        j=0;
        scanf("%d",&q);
        for(int qes=q-1;qes>0;qes--){
            for(int t=qes;t<300000;t+=q){
                if(bage[t]){j=qes;break;};
            }
            if(j) break;
        }
        printf("%d\n",j);
    }
    return 0;
}
