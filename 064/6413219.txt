#include <cstdio>
#include <iostream>

using namespace std;

int a[100];

int main(void){
	int n;	
	scanf("%d",&n);
	for (int i = 0; i < n; ++i) 
	    scanf("%d",a+i);
	//for (int i = n-1; i >= 0; --i)	printf("%d%c",a[i]," \n"[i==0]);
	for (int i = 0; i < n/2; ++i)	
	    swap(a[i],a[n-1-i]);

	for (int i = 0; i < n; ++i)
		printf("%d%c",a[i]," \n"[i==n-1]);
	//" \n" = {' ','\n','\0'}
	//i != n-1 -> 0
	//i == n-1 -> 1
}
