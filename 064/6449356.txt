template<typename T> void drop(const T &x){cout<<x<<endl;exit(0);}
template<typename T=Int>
vector<T> read(size_t n){
  vector<T> ts(n);
  for(size_t i=0;i<n;i++) cin>>ts[i];
  return ts;
}

signed main(){
  cin.tie(0);
  ios::sync_with_stdio(0);


  Int a,b;
  cin>>a>>b;

  auto to_v=[&](Int x){
    vector<Int> vs;
    while(x) vs.emplace_back(x&1),x>>=1;
    reverse(vs.begin(),vs.end());
    if(vs.empty()) vs.emplace_back(0);
    return vs;
  };

  auto va=to_v(a);
  auto vb=to_v(b);
  while(va.size()<vb.size()) va.emplace(va.begin(),0);
  while(vb.size()<va.size()) vb.emplace(vb.begin(),0);
  for(Int i=0;i<(Int)min(va.size(),vb.size());i++){
    if(va[i]!=vb[i]) drop(i);
  }
  drop(min(va.size(),vb.size()));
  return 0;
}
