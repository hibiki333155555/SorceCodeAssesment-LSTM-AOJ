N = int(input())
vs = [int(v) for v in input().split()]
print(' '.join([str(v) for v in vs]))

for i in range(1, N):
    v = vs[i]
    j = i - 1
    while j >= 0 and vs[j] > v:
        vs[j+1] = vs[j]
        j -= 1
    vs[j+1] = v
    print(' '.join([str(v) for v in vs]))
