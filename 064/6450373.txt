#include <bits/stdc++.h>
#define rep(i,n) for(int i = 0; i < (n); i++)
using namespace std;
typedef long long ll;

int main(){
    cin.tie(0);
    ios::sync_with_stdio(0);
    
    int A,B; cin >> A >> B;
    auto f = [&](int x) {
        if(x == 0) return (string)"0";
        string s = "";
        while(x > 0) {
            s += (char)(x % 2 + '0');
            x /= 2;
        }
        reverse(s.begin(), s.end());
        return s;
    };

    string a = f(A), b = f(B);
    while(a.size() > b.size()) b = '0' + b;
    while(a.size() < b.size()) a = '0' + a;
    a += '0'; b += '1';
    int i;
    while(a[i] == b[i]) i++;
    cout << i << endl;
}
