#include<iostream>
using namespace std;

int main()
{
	int r,c,S=0;
    cin>>r>>c;
    int sh[r+1][c+1];
    int sum_a[r]={},sum_b[c]={};

    for (int i = 0; i < r; i++){
        for(int j=0;j<c;++j){
            cin>>sh[i][j];
        }
    }

    for (int i = 0; i < r; i++){
        for(int j=0;j<c;++j){
            sum_a[i]+=sh[i][j];
            sum_b[j]+=sh[i][j];
        }
    }

    for (int i = 0; i < r; i++){
        for(int j=0;j<c;++j){
            cout<<sh[i][j];
            if(j!=(c))cout<<" ";
        }
    cout<<sum_a[i];
    cout<<endl;
    }
    for(int i=0;i<c;++i){
        cout<<sum_b[i];
        S+=sum_b[i];
        if(i!=(c))cout<<" ";
    }
    cout<<S<<endl;
return 0;
}
