#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, x;
    while (true)
    {
        cin >> n >> x;
        if (n == 0 && x == 0)
        {
            break;
        }
        int cnt = 0;
        // 全探索で行ってみる

        // 3つの数の取りえる値の最大値はnまたはx-3
        for (int i = 1; i <= n; i++)
        {
            for (int j = i + 1; j <= n; j++)
            {
                for (int k = j + 1; k <= n; k++)
                {
                    if (i + j + k == x)
                    {
                        cnt++;
                        break;
                    }
                }
            }
        }

        cout << cnt << endl;
    }

    return 0;
}
