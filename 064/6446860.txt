#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main()
{
    char s[1201]; unsigned long long int count[26] = {0};
    while (scanf("%s", s) != EOF){
    
    for (int i = 0; i < strlen(s) ; i++)
    {
        if ((tolower(s[i]) - 97) < 0) {continue;}
        count[tolower(s[i]) - 97]++;
    }
    for (int j = 0; j < 26; j++)
    {
        printf("%c : %llu\n", j + 97, count[j]);
    }
    }
    return 0;
}
