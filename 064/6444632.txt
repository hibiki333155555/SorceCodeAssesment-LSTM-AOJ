#include <iostream>
using namespace std;

int main() {
    int N,A[109],sw=0;
    cin >> N;
    for (int i = 0; i < N; i++)cin >> A[i];

    for (int i = 0; i < N; i++) {
        int minj = A[i];
        for (int j = i; j < N; j++) {
            if (A[j] < minj) {
                minj = A[j];
                swap(A[j], A[i]);
                sw++;
            }
    }
    for (int i = 0; i < N; i++) {
        if (i)cout << " ";
        cout << A[i];
    }
    cout << endl;
    cout << sw << endl;

    return 0;
}
