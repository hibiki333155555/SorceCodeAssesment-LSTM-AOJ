#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (int)n; i++)
using ll = long long;

int main(){
  int a, b;
  cin >> a >> b;
  if(a == 0 and b == 0) {
    cout << 1 << endl;
    return 0;
  }
  vector<int> ba, bb;
  while(a) {
    ba.push_back(a % 2);
    a /= 2;
  }
  while(b) {
    bb.push_back(b % 2);
    b /= 2;
  }
  int l = max(ba.size(),bb.size());
  ba.resize(l,0);
  bb.resize(l,0);
  reverse(ba.begin(),ba.end());
  reverse(bb.begin(),bb.end());
  int ans = 0;
  rep(i,l) {
    if(ba[i] == bb[i]) ans++;
    else break;
  }
  cout << ans << endl;
  return 0;
}
