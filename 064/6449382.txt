#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    int ans = 0;
    vector<string>S(N);
    for(int i = 0; i < N; i++) {
        cin >> S[i];
    }
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            if(i == j) {
                continue;
            }
            int res = 0;
            for(int k = 0; k < S[i].size(); k++) {
                if(S[i][k] != S[j][k]) {
                    res++;
                }
            }
            if(res <= 1) {
                ans++;
                break;
            }
        }
    }
    cout << ans << endl;
}
