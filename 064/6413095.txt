#include<stdio.h>
#include<string.h>

int main{
    
    char W[10],T[1000];
    int i,n=0;
    
    scanf("%s", &W);
    
    while(1){
        scanf("%s", &T);
        if(strcmp(T,"END_OF_TEXT")==0)break;
        for(i=0;T[i]=="\0"){
            if(T[i] >= 'A' && T[i] <= 'Z')T[i]=T[i]+32;
        }
        if(strcmp(W,T)==0)n++;
    }
    
    printf("%d", n);
    
    return 0;
}
