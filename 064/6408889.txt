#include <iostream>
using namespace std;
int a[100000],n;
int binarysearch(int key){
    int left = 0;
    int right = n-1;
    while(left < right){
        int mid = (left + right) / 2;
        if(a[mid] == key){
            return 1;
        }else if(key < a[mid]){
            right = mid;
        }else{
            left = mid + 1;
        }
    }
    return 0;
}
int main(void){
    int q,key,sum = 0;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    
    cin >> q;
    for(int i = 0; i < q; i++){
        cin >> key;
        if(binarysearch(key)){
            sum++;
        }
    }
    
    cout << sum << endl;
}
