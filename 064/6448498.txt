#include<bits/stdc++.h>
using namespace std;
int main(){
    int a,b,c;
    cin>>a>>b>>c;
    double rad = c*3.141592 / 180.0;
    double me = (double)(0.5 * a  * b * sin(rad));
    cout<<fixed<<setprecision(8)<<me<<"\n";
    double hen = sqrt(a*a+b*b);
    double shu = a + b + hen;
    cout<<fixed<<setprecision(8)<<shu<<"\n";
    double taka = (2.0 * me) / (double)a;
    cout<<fixed<<setprecision(8)<<taka<<"\n";
    return(0);
}
