#include <iostream>
using namespace std;

int main()
{
    int i, j;
    int h, w;
    
    while(1) {
		cin >> h >> w;
		if ( ( h | w ) != 0 ) {
			for ( i = 0; i < h; i++ ) {
				for ( j = 0; j < w; j++ ) {
					cout << "#";
				}
				cout << endl;
			}
			cout << endl;
		} else {
			break;
		}
	}
    
    return 0;
}
