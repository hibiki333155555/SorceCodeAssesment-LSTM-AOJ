#include<bits/stdc++.h>
#include <vector> 
using namespace std;

void insertionSort(vector<int>& A, int n) { 
    for (int i=1; i < n; i++) {
        int v = A.at(i);
        int j = i- 1;
        while (j >= 0 && A.at(j) > v) {
            A.at(j + 1) = A.at(j);
            j -= 1;
        A.at(j + 1) = v;    
        }
        for (int k = 0; k < n ; k++){
            if (k == n-1) {
                cout << A.at(k) << endl; 
            }
            else {
                cout << A.at(k) << " "; 
            }
        }
    } 
}

int main () {
    int n;
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
        cin >> A.at(i);
    }
    insertionSort(A, n);
    
    return 0;
}
