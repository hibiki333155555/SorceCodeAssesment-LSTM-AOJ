#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < n; ++i)
typedef long long ll;
using namespace std;

void solve(ll P) {
    ll ans = P + 2;
    for (ll i = 1; i * i * i <= P; ++i) {
        if (P % i != 0) continue;

        ll d = P / i;
        for (ll j = i; j * j <= d; ++j) {
            if (d % j != 0) continue;

            ll k = d / j;
            ans = min(ans, i + j + k);
        }
    }
    cout << ans << "\n";
}

int main() {
    cin.tie(0);
    ios_base::sync_with_stdio(false);

    ll P;
    while (cin >> P, P)
        solve(P);

    return 0;
}
