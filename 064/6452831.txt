#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
#define rep(i,a,b) for(int i=a;i<b;i++)
#define fore(i,a) for(auto &i:a)

ll modpow(ll a, ll n, ll mod) {
  ll res = 1;
  while (n > 0) {
    if (n & 1) res = res * a % mod;
    a = a*a%mod;
    n = n >> 1;
  }
  return res;
}

int main() {
  ll m,n;
  cin >> m >> n;
  
  cout << modpow(m,n,1000000007) << endl;
}
