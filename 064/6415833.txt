#include "bits/stdc++.h" 
using namespace std; 
typedef long long ll;


int main() {
    int m,n;cin >> m >> n;
    vector<string> s(m);
    vector<vector<int>> cnt(m+1,vector<int>(n+1,0));
    for (int i = 0; i < m; i++) {
        cin >> s[i];
        for (int j = 0; j < n; j++) {
            cnt[i+1][j+1] = cnt[i][j+1] + cnt[i+1][j] - cnt[i][j] + (s[i][j]=='o'?1:0);
        }
    }
    int ans = n*m;
    for (int i = 0; i < m-1; i++) {
        ans = min(ans,cnt[i+1][n]+(m-i-1)*n-cnt[m][n]+cnt[i+1][n]);
        ans = min(ans,(i+1)*n-cnt[i+1][n]+cnt[m][n]-cnt[i+1][n]);
    }
    for (int i = 0; i < n-1; i++) {
        ans = min(ans,cnt[m][i+1]+(n-i-1)*m-cnt[m][n]+cnt[m][i+1]);
        ans = min(ans,(i+1)*m-cnt[m][i+1]+cnt[m][n]-cnt[m][i+1]);
    }
    cout << ans << endl;
    return 0;
}
