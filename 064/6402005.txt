#include <iostream>
using namespace std;
int LIS(int arr[], int n) {
    int len = n;
    if (len == 0) return 0;
    int L[len];
    int f[len];
    for (int i = 0; i < len; i++) {
        L[i] = 1;
        f[i] = 0;
    }
    f[0] = arr[0];
    for (int i = 1; i < len; i++) {
        for (int j = 0; j < i; j++) {
            if (f[j] < f[i] && L[j] > L[i])
                L[i] = L[j];
        }
        ++L[i];
        if (f[L[i]-1] > arr[i])
            f[L[i]-1] = arr[i];
    }
    int max = L[0];
    for (int i = 0; i < len; i++) {
        if (L[i] >= max)
            max = L[i];
    }
    return max;
}

int main(int argc, char *argv[]) {
    int n;
    int arr[n];
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> arr[i];
    cout << LIS(arr, n) << endl;
    return 0;
}
