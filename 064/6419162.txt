#include<stdio.h>
int main()
{
    int a,b,c,x,y,z,N;


    scanf("%d", &N);
    while (N--)
    {
        scanf("%d %d %d",&x,&y,&z);
        a=x*x;
        b=y*y;
        c=z*z;
        if (a==b+c || b==a+c || c==a+b)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }

    }


    return 0;
}
