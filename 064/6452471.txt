#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main()
{
	double x1, y1, x2, y2, x3, y3;
	cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3;
	x2 -= x1; y2 -= y1; x3 -= x1; y3 -= y1;
	auto t = (x3 * x3 + y3 * y3 - (x2 * x3 + y2 * y3)) / (x3 * y2 - x2 * y3);
	auto cx = 0.5 * (x2 + t * y2), cy = 0.5 * (y2 - t * x2);
	auto r = sqrt(cx * cx + cy * cy);
	cout << fixed << setprecision(20) << x1 + cx << ' ' << y1 + cy << ' ' << r << endl;
	return 0;
}
