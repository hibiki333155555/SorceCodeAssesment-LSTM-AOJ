using System;
using System.Linq;

internal class Program
{
    public static void Main()
    {
        Console.ReadLine();
        var S = Console.ReadLine().Split().Select(s => s.ParseInt()).ToArray();
        Console.ReadLine();
        var T = Console.ReadLine().Split().Select(s => s.ParseInt());
        var counter = 0;
        foreach (var t in T)
        {
            if (ContainsBinary(S, t))
                counter++;
        }
        Console.WriteLine(counter);
    }

    private static bool ContainsBinary(int[] S, int t)
    {
        var head = 0;
        var tail = S.Length;
        while (head != tail)
        {
            var half = (head + tail) / 2;
            if (t < S[half])
            {
                tail = half;
                continue;
            }
            else if (S[half] == t)
            {
                return true;
            }
            else
            {
                head = half + 1;
                continue;
            }
        }
        return false;
    }
}



internal static class Extensions
{
    public static int ParseInt(this string s) => int.Parse(s);
}
