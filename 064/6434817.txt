#include <bits/stdc++.h>
using namespace std;
const int INF = 1001001001;
const int MOD = 1000000007;

int cnt = 0;

void insert(vector<int>& a, int n, int g) {
    for(int i=g;i<n;i++) {
        int v = a.at(i);
        int j = i-g;
        while(j>=0 && a.at(j) > v) {
            a.at(j+g) = a.at(j);
            j = j-g;
            cnt++;
        }
        a.at(j+g) = v;
    }
}

void shell(vector<int>& a, int n) {
    cnt = 0;
    vector<int> g(0);
    for(int i=0;;i++) {
        if(i==0) g.push_back(1);
        else g.push_back(g.at(i-1)*3+1);
        if(i!=0 && g.at(i)*3+1>n) break;
    }
    int m = g.size();
    cout << m << endl;
    reverse(g.begin(),g.end());
    for(int i=0;i<m;i++) {
        cout << g.at(i);
        if(i==m-1) cout << endl;
        else cout << " ";
    }
    for(int i=0;i<m;i++) {
        insert(a, n, g.at(i));
    }
} 

int main(){
    int n;cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;i++) cin>>a.at(i);
    shell(a,n);
    cout << cnt << endl;
    for(int i=0;i<n;i++) cout << a.at(i) << endl;
}
