#include <iostream>
#include <vector>

using namespace std;

int main(){
    int i,j,k,tmp,n;
    cin >> n;
    vector<int> a(n);
    for(i=0;i<n;i++){
        cin >> k;
        a[i] = k;
    }
    for(i=1;i<n;i++){
        tmp = a[i];
        j = i - 1;
        while(j >= 0 && a[j] > tmp){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = tmp;
        for(i=0;i<n;i++){
            cout << a[i] << "Â¥n";
        }
    }
    return 0;
}
