X=[]
while True:
    a,b,c=map(int,raw_input().split())
    if b== int(?):
        break
    X.append(a)
    Y.append(b)
    Z.append(c)
for i in range(len(X)):
    if Y[i] == int(+):
        print X[i]+Z[i]
    elif Y[i] == int(-):
        print X[i]-Z[i]
    elif Y[i] == int(*):
        print X[i]*Z[i]
    elif Y[i] == int(/):
        print X[i]/Z[i]
