void solve() {
    ll a, b;
    cin >> a >> b;
    if (a > b) swap(a, b);
    ll num = 1;
    while (b >= (1ll << num)) num++;
    num--;
    ll res = 0;
    while (1) {
        ll pos = (1ll << num);
        if ((pos & a) == (pos & b))
            res++;
        else
            break;
        num--;
        if (num < 0) break;
    }
    cout << res << endl;
}

    while (t--) solve();
