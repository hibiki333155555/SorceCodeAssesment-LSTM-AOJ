void solve()
{
    int n, K;
    scanf("%d%d", &n, &K);
    vector<int> v;
    ll s = 0;
    for (int i = 0; i < n; i++) {
        int t;
        scanf("%d", &t);
        s += t;
        if (t == 1) v.push_back(4);
        else if (t < 6) v.push_back(6 - t);
    }
    sort(v.rbegin(), v.rend());
    ll ans = 0;
    for (int i = 0; i < min(K, (int)v.size()); i++) {
        ans += v[i];
    }
    printf("%lld\n", ans+s);

    return;
}

int main()
{

    solve();

    for (t = 0; t < T; t++) {
        //printf("Case #%d: ", t + 1);
        solve();
    }
#endif
    return 0;
}
