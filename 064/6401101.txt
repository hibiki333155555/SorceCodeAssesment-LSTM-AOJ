#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int n;

int main(){
  cin >> n;
  for (int i = 1; i <= n; i++){
    if (i % 3 == 0){
      cout << ' ' << i;
      continue;
    }
    int j = i;
    while(j != 0){
      if (j % 10 == 3){
        cout << ' ' << j;
        break;
      }
      j /= 10;
    }
  }
  cout << endl;
}
