#include<bits/stdc++.h>
using namespace std;

int main(){
    vector<int> e(12);bool tf=true;
    for(int i=0;i<12;i++)cin>>e[i];
    sort(e.begin(),e.end());
    for(int i=0;i<3;i++){
        for(int j=1;j<4;j++){
            if(e[4*i+j]!=e[4*i]){
                tf=false;
            }
        }
    }
    if(tf)cout<<"yes"<<endl;
    else cout<<"no"<<endl;
    return 0;
}
