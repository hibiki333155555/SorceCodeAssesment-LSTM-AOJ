int main(){
    int n,m; cin >> n >> m;
    vector<tapu> v(n);
    vl imo(m+1,0), pl(m+1,0);
    rep(i,n){
        int a,b,c; cin >> a >> b >> c; b--;
        v[i] = tie(b,c,a);
        imo[b] += a;
        imo[c] -= a;
    }
    rep(i,m) imo[i+1] += imo[i];
    int q; cin >> q;
    while(q--){
        int d,e; cin >> d >> e; e--;
        imo[e] -= d;
    }
    sort(all(v));
    int now = 0;
    ll ans = 0;
    int mx = -1;
    rep(i,m){
        if(i > 0) pl[i] += pl[i-1];
        imo[i] += pl[i];
        if(imo[i] < 0){
            while(now < n){
                if(get<0>(v[now]) <= i){
                    chmax(mx, get<1>(v[now]));
                    now++;
                }else{
                    break;
                }
            }
            if(mx <= i){
                cout << -1 << endl;
                return 0;
            }
            ll val = -imo[i];
            ans += val;
            pl[i] += val;
            pl[mx] -= val;
        }
    }
    cout << ans << endl;
}
