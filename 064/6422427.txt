
#include <bits/stdc++.h>
using namespace std;

void printMinSubArray(int arr[], int n, int k)
{
	int j, min;

	for (int i = 0; i <= n - k; i++)
	{
		min = arr[i];

		for (j = 1; j < k; j++)
		{
			if (arr[i + j] < min)
				min = arr[i + j];
		}
		cout << min << " ";
	}
}

int main()
{   int n,k;
cin>>n>>k;
	int arr[n];
	for(int i=0;i<n;i++)
	{  
	    cin>>arr[i];
	}
	
	
	printMinSubArray(arr, n, k);
	return 0;
}
