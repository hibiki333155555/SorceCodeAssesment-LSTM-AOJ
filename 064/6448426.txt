// #define DEBUG
#define FOR(i, a, b) for (int i = (int)(a); i < (int)(b); ++i)
#define rep(i, n) for (int i = 0; i < (int)(n); ++i)

struct schedule
{
    string s;
    int time;
};

int main()
{
#ifndef DEBUG

    int n, q, count = 0;
    cin >> n >> q;
    queue<schedule> task, res;
    schedule tmp;

    rep(i, n)
    {
        cin >> tmp.s >> tmp.time;
        task.push(tmp);
    }

    while (!task.empty())
    {
        tmp = task.front();
        task.pop();

        tmp.time -= q;
        if (tmp.time <= 0)
        {
            count = count + tmp.time + q;
            tmp.time = count;
            res.push(tmp);
            continue;
        }
        count += q;
        task.push(tmp);
    }

    while (!res.empty())
    {
        cout << res.front().s << " " << res.front().time << endl;
        res.pop();
    }
#else

#endif
}
