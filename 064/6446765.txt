n = int(input())

for k in range(n):
    S = input()
    T = input()
    
    dp = [[0] * (len(T) + 1) for _ in range(len(S) + 1)]
    for i in range(1, len(S) + 1):
        for j in range(1, len(T) + 1):
            indicator = 1 if S[i - 1] == T[j - 1] else 0
            dp[i][j] = max(dp[i - 1][j - 1] + indicator, dp[i - 1][j], dp[i][j - 1])
    print(dp[len(S)][len(T)])
