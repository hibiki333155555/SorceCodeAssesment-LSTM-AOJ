#include <iostream>
#include <bitset>
#include <vector>

using namespace std;

bool EnumerationOfSubsets2(unsigned int n) {

	unsigned int k = 0;
	cin >> k;
	unsigned int T = 0;
	for (int i = 0; i < k; i++) {
		unsigned int tmp = 0;
		cin >> tmp;
		T |= (1 << tmp);
	}
	vector<unsigned int> t;
	for (int i = 0; i < n; i++) {
		if (!((T >> i) & 1)) {
			t.push_back(i);
		}
	}

	for (int i = 0; i < (1 << (n - k)); i++) {
		int b = T;
		for (int j = 0; j < n - k; j++) {
			if (i & (1 << j)) {
				b += (1 << t[j]);
			}
		}
		cout << b << ':';
		for (int j = 0; j < n; j++) {
			if (b & (1 << j)) {
				cout << ' ' << j;
			}
		}
		cout << endl;
	}
	return true;

}

int main() {

	unsigned int n = 0;

	cin >> n;

	if (EnumerationOfSubsets2(n)) {
		return 0;
	}
	else {
		return -1;
	}

	return 0;
}
