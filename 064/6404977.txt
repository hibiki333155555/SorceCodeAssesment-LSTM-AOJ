#include <bits/stdc++.h>
using namespace std;

int main(){
    while(1){
        int H,W;
        cin>> H>>W;
        vector<vector<char>> vec(H,vector<char>(W,'#'));
        if(H !=0 && W !=0){
        for(int i=0; i<H; i++){
            for(int j=0; j<W; j++){
                if(i%2==0&& j%2 !=0){
                    vec.at(i).at(j)='.';
                }
                else if(i%2 !=0 && j%2==0){
                    vec.at(i).at(j)='.';
                }
                else{
                    continue;
                }
            }
        }
            for(int i=0; i<H; i++){
                for(int j=0; j<W; j++){
                 cout<< vec.at(i).at(j);
                }             
                cout<<endl;
            }
        }
        else{
            break;
        }
        cout<<endl;
    }    
}
