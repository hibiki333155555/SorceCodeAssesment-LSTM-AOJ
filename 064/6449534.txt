#include <bits/stdc++.h>
using namespace std;
int main(void){
    int N;cin>>N;
    vector<vector<char>> A(N,vector<char>(100,'?'));
    
    for(int i=0;i<N;i++){
        string s;cin>>s;
        for(int j=0;j<s.size();j++){
            A[i][j]=s[j];
        }
    }
    
    int ans=0;
    
    for(int i=0;i<N;i++){
        bool flag=false;
        for(int j=0;j<N;j++){
            if(i==j) continue;
            int sa=0;
            for(int k=0;k<100;k++){
                if(A[i][k]!=A[j][k]) sa++;
            }
            if(sa<=1){
                flag=true;
                break;
            }
        }
        if(flag){
            ans++;
        }
    }
    
    cout<<ans<<endl;
    
    
    return 0;
}
