#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int n,x,i,j,k,sum=0;
	
	cin>>n>>x;
	
	while(!(n==0&&x==0))
	{
		for(i=1;i<=n;i++)
		{
			for(j=i+1;j<=n;j++)
			{
				for(k=j+1;k<=n;k++)
				{
					if(i+j+k==x)
						sum++;
				}
			}
		}
		
		
		cin>>n>>x;
		
	}
		
	cout<<sum<<endl;
return 0;
}
