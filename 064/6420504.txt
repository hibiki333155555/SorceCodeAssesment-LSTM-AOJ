#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int l, n;
    cin >> l >> n;
    for (int i = 0; i < n; ++i) {
        int a;
        cin >> a;
        l -= a;
        if (l <= 0) {
            cout << (i + 1) << '\n';
            return 0;
        }
    }
    cout << "No\n";
    return 0;
}
