#include <bits/stdc++.h>
using namespace std;

int check(int a) {
    if (a == 1) return 0;
    int r = sqrt(a);
    for (int i = 2; i <= r; i++) {
        if (a % i == 0) return 0;
    }
    return 1;
}

int main() {
    int n, ans = 0;
    cin >> n;

    for (int i = 0; i < n; i++) {
        int a;
        cin >> a;
        ans += check(a);
    }
    cout << ans << endl;
    
}
