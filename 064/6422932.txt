#include<iostream>
#include<string>

using namespace std;

int main() {
    string taroCard, hanakoCard;
    int taroPoint = 0, hanakoPoint = 0, n;
    int cp;
    cin >> n;
    while(n > 0) {
        cin >> taroCard >> hanakoCard;
        cp = taroCard.compare(hanakoCard);
        if(cp == -1) {
            hanakoPoint += 3;
        }
        if(cp == 1) {
            taroPoint += 3;
        }
        if(cp == 0) {
            hanakoPoint++;
            taroPoint++;
        }
        n--;
    }
    cout << taroPoint << " " << hanakoPoint ;
    return 0;
}
