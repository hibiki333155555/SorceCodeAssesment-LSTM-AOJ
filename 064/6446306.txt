#include <stdio.h>

int main()
{
    int i,j,k,n,b,f,r,v;
    int a[4][3][10]={0};
    scanf("%d",&n);
    for (i=0;i<n;i++)
    {
        scanf("%d %d %d %d",&b,&f,&r,&v);
        a[b-1][f-1][r-1] = v;
    }
    
    for (i=0;i<4;i++)
    {
        for(j=0;j<3;j++)
        {
            for(k=0;k<10;k++)
            {
                printf("%d",a[i][j][k]);
                if(k < 9)
                {
                    printf(" ");
                }
            }
            printf("\n");
        }
        printf("####################\n");
    }

    return 0;
}
