#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < n; ++i)
typedef long long ll;
using namespace std;

void solve(int N, int M, int P) {
    vector<bool> flag(N);
    flag[P - 1] = true;
    rep(i, M) {
        int a, b;
        cin >> a >> b;
        a--, b--;

        flag[a] = flag[b] = (flag[a] || flag[b]);
    }

    int ans = count(flag.begin(), flag.end(), true);
    cout << ans << "\n";
}

int main() {
    cin.tie(0);
    ios_base::sync_with_stdio(false);

    int N, M, P;
    while (cin >> N >> M >> P, N)
        solve(N, M, P);

    return 0;
}
