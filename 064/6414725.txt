#include <iostream>
using namespace std;
int selectionSort(int A[], int N){
    int x = 0, minj;
    for(int i = 0; i < N - 1; i++){
        minj = i;
        for(int j = i; j < N; j++){
            if(A[j] < A[minj]) minj = j;
        }
        if(minj != i) {
            swap(A[minj], A[i]);
            x++;
        }
    }
    return x;
}
int main(){
    int N, A[100];
    cin >> N;
    for(int i = 0; i < N; i++) cin >> A[i];
    int x = selectionSort(A, N);
    for(int i = 0; i < N; i++){
        if(i) cout << " ";
        cout << A[i];
    }
    cout << endl;
    cout << x << endl;
    return 0;
}
