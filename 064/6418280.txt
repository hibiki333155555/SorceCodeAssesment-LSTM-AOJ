N, K = map(int, input().split())
A = list(map(int, input().split()))

maxi = {1:5, 2:6, 3:6, 4:6, 5:6, 6:5}
used = [False]*N

sum = 0

for i in range(1, 6):
    for j in range(N):
        if used[j]==False:
            if K>0 and A[j]==i and A[j]!=6:
                used[j] = True
                K -= 1
                A[j] = maxi[A[j]]

for i in range(N):
    if K>0:
        if A[i]==5:
            A[i] = 6
    else:
        break
for v in A:
    sum += v

print(sum)
