#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i = 0; i < (int)n; i++)
using ll = long long;

int main(){
  int n;
  cin >> n;
  vector<string> s(n);
  rep(i,n) cin >> s[i];
  int ans = 0;
  rep(i,n) {
    bool cheat = false;
    rep(j,n) {
      if(i == j) continue;
      int cnt = 0;
      rep(k,s[i].size()) if(s[i][k] != s[j][k]) cnt++;
      if(cnt <= 1) cheat = true;
    }
    if(cheat) ans++;
  }
  cout << ans << endl;
  return 0;
}
