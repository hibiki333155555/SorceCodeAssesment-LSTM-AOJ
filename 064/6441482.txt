#include <bits/stdc++.h>
using namespace std;

int main(void) {
    string c1, c2;
    int r1, r2;
    cin >> c1 >> r1 >> c2 >> r2;
    function<int(string)> f = [](string str) {
        if(str.length() == 1) return str[0] - 'A';
        return (str[0] - 'A' + 1) * 26 + (str[1] - 'A');
    };
    cout << (f(c2) - f(c1) + 1) * (r2 - r1 + 1) << '\n';
}
