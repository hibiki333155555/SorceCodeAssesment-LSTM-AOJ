int main(){
    int n, m;
    cin >> n >> m;

    vector<int> coins(m);
    rep(i, m){
        int c;
        cin >> c;
        coins[i] = c;
    }
    vector<int> dp(n + 1, 100010);
    dp[0] = 0;

    rep(i, n){
        rep(j, m){
            if(i + coins[j] < n + 1){
                if(dp[i + coins[j]] > dp[i] + 1)dp[i + coins[j]] = dp[i] + 1;
            }
        }
    }

    cout << dp[n] << "\n";

}
