#include <iostream>
using namespace std;
void insertion_sort(int a[],int n){
    int i,j,tmp;
    for(i=1;i=n-1;i++){
        tmp = a[i];
        j=i-1;
    
        while(tmp<a[j] && j>=0){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = tmp;
        cout << a[j+1];
    }
}
int main(){
    int i,n,a[100];
    cin >> n;
    for(i=0;i<n;i++){
        cin >> a[i]; 
    }
    insretion_sort(a[],n);
    
}
