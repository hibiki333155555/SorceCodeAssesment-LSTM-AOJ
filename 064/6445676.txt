#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int main() {
    int count = 0;
    int vector[6][6] = {{0, 2, 4, 1, 3, 0}, {3, 0, 0, 5, 0, 2}, {1, 5, 0, 0, 0, 4},
                        {4, 0, 0, 0, 5, 1}, {2, 0, 5, 0, 0, 3}, {0, 3, 1, 4, 2, 0}};
    string message = "No";
    int face;
    int a, b;
    int dice1[6];
    for(int n = 0; n < 6; n++) {
        cin >> dice1[n];
    }
    face = dice1[vector[0][1]];
    a = dice1[0];
    b = dice1[1];
    int dice2[6];
    for(int n = 0; n < 6; n++) {
        cin >> dice2[n];
    }
    for(int num = 0; num < 6; num++) {
        if(dice2[num] == a) {
            a = num;
            break;
        }
    }
    for(int num2 = 0; num2 < 6; num2++) {
        if(dice2[num2] == b) {
            b = num2;
            break;
        }
    }
    if(face == dice2[vector[a][b]]) {
        message = "Yes";
    }
    cout << message << endl;
}
