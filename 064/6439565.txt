#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <map>
#include <set>

using namespace std;

const long long INF = 1LL << 60;
const int INT_INF = 2e9;

template<class T> inline bool chmin(T& a, T b) {
    if (a > b) {
        a = b;
        return true;
    }
    return false;
}
template<class T> inline bool chmax(T& a, T b) {
    if (a < b) {
        a = b;
        return true;
    }
    return false;
}

using namespace std;
vector<vector<int>> dp;

int main(){
    string s1, s2;

    cin >> s1 >> s2;

    dp.assign((int)s1.length()+10, vector<int>((int)s2.length()+10, INT_INF));
    dp[0][0] = 0;

    for(int ii1=-1; ii1<(int)s1.length(); ii1++){
        for(int ii2=-1; ii2<(int)s2.length(); ii2++){
            if(ii1==-1 && ii2==-1) continue;

            /* 削除 */
            if(ii1 >= 0) chmin(dp[ii1+1][ii2+1], dp[ii1][ii2+1]+1);

            /* 追加 */
            if(ii2 >= 0) chmin(dp[ii1+1][ii2+1], dp[ii1+1][ii2]+1);

            if(ii1 >= 0 && ii2 >= 0){
                /* 変更処理が不要な場合 */
                if(s1[ii1] == s2[ii2])  chmin(dp[ii1+1][ii2+1], dp[ii1][ii2]);
                /* 変更処理が必要な場合 */
                else chmin(dp[ii1+1][ii2+1], dp[ii1][ii2]+1);
            }
        }
    }

    cout << dp[(int)s1.length()][(int)s2.length()] << endl;

    return 0;
}
