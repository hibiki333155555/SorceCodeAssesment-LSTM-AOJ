#include <stdio.h>

int main(void)
{
    unsigned int A, B, T;
    int i;
    int n;
    
    scanf("%u%u", &A, &B);
    if (A < B){
        T = A;
        A = B;
        B = T;
    }
    for (i = 31; i >= 1; i--){
        if ((A & (1 << i)) != 0){
            break;
        }
    }
    
//    printf("<%d>", i);
    
    n = 0;
    while (i >= 0){
        if ((A & (1 << i)) != (B & (1 << i))){
            break;
        }
        n++;
        i--;
    }
    
    printf("%d\n", n);
    
    return (0);
}
