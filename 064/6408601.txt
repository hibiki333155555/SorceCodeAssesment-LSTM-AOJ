#include <iostream>
using namespace std;

const int N = 200000;
int main(void){
    int n;
    cin >> n;
    int op;
    int x;
    int p;
    int A[N];
    int nEnd;
    nEnd = 0;
    for( int i = 0; i < n; i++)
    {
        cin >> op;
        switch( op)
        {
            case 0:
                cin >> x;
                A[nEnd] = x;
                nEnd++;
                break;
            case 1:
                cin >> p;
                cout << A[p] << endl;
                break;
            case 2:
                
                A[nEnd] = -1;
                nEnd--;
                break;
        
        }
    }
}
