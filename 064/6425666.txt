#include <iostream>
#include<stdio.h>

using namespace std;

int main() {
	while (1) {
		char op;
		int a, b;
		cin >> a >> op >> b;
		if (op == '+') {
			cout << a + b <<endl;
		}
		else if (op == '-') {
			cout << a - b << endl;
		}
		else if (op == '*') {
			cout << a * b << endl;
		}
		else if (op == '/') {
			cout << a / b << endl;
		}
		else {
			break;
		}
	}
	

}
