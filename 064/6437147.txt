import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner input=new Scanner(System.in);
        int a = input.nextInt();
        int b = input.nextInt();
        String msg;
        
        if(a==b){
            msg="a == b";
        }else if(a<b){
            msg="a < b";
        }else{
            msg="a > b";
        }
        
        System.out.println(msg);
    }
}
