#include<stdio.h>
#include<math.h>
int main()   {
    int n;
    double average1, average2, sum1, sum2;
    scanf("%d", &n);
    int num[n];
   while ( n>0 )  {
       if (n == 0) break;
   
    for ( int i=0; i<n; i++ ) {
        scanf("%d", &num[i]);
        sum1 += num[i];
    }
  average1 = sum1/n;
   for ( int i=0; i<n; i++ ) {
       sum2 += (average1 - num[i])*(average1 - num[i]);
   }   
   average2 = sum2/n;
   }

}
