#include <bits/stdc++.h>
using namespace std;
string binary(int A){
  string S;
  while (A > 0){
    S += '0' + A % 2;
    A /= 2;
  }
  reverse(S.begin(), S.end());
  if (S.empty()){
    S = "0";
  }
  return S;
}
int main(){
  int A, B;
  cin >> A >> B;
  string S1 = binary(A);
  string S2 = binary(B);
  if (S1.size() != S2.size()){
    cout << 0 << endl;
  } else {
    int L = S1.size();
    for (int i = L; i >= 0; i--){
      if (S1.substr(0, i) == S2.substr(0, i)){
        cout << i << endl;
        break;
      }
    }
  }
}
