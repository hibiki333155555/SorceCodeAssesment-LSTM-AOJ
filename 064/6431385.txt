#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int bubulesort(int a[], int m) {
	bool flag =1;
	int sw=0;
	for (int i = 0; flag; i++) {
		flag = 0;
		for (int j = m - 1; j >= i + 1; j--) {
			
			if (a[j]<a[j-1])
			{
			
				swap(a[j], a[j - 1]);
				
				flag = 1;
				sw++;
			}
		
		
		}
	
	}
	return sw;
}

int main() {
	int a[100];
	int m;
	int sw;
	cin >> m;
	for (int i = 0; i < m; i++)cin >> a[i];
	 sw = bubulesort(a, m);
	for (int i = 0; i < m; i++) {
	if(i)
		cout << " ";
		cout << a[i] ;
		
	}
	cout << endl;
	cout << sw << endl;
	return 0;
}
