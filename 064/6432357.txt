#include <bits/stdc++.h>
using namespace std;
int main(){
    stack<int> st;
    char ch;
    
    while (cin>>ch){
        switch (ch){
            case '+':{
                int sum = st.top();
                st.pop();
                sum += st.top();
                st.pop();
                st.push(sum);
                break;
            }
            case '-':{
                int sub = st.top();
                st.pop();
                sub = st.top() - sub;
                st.pop();
                st.push(sub);
                break;
            }
            case '*':{
                int mul = st.top();
                st.pop();
                mul *= st.top();
                st.pop();
                st.push(mul);
                break;
            }
            default:{
                // cout<<ch<<endl;
                int i = ch-'0';
                st.push(i);
                break;
            }
        }
        // cout << ch<<endl;
    }
    cout<<st.top()<<endl;
    return 0;
}
