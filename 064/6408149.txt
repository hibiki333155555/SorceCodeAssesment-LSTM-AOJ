#include <iostream>

using namespace std;

int  n, a[50];

int solve(int i, int m){
    if(!m) return 1;
    if(i >= m) return 0;
    int res = solve(i + 1, m) || solve(i + 1, m - a[i]);
    return res;
}

int main()
{
    int m;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    cin >> m;
    for(int i = 0; i < m; i++){
        int x;
        cin >> x;
        if(solve(0, x)) cout << "yes\n";
        else cout << "no\n"; 
    }
    return 0;
}
