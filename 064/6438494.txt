#include <iostream>

using namespace std;

int main () {
    int n, cards[4][13] = {};
    char mark;
    int rank;
    
    cin >> n;
    
    for(int i = 0; i < n; i++) {
        cin >> mark >> rank;
        
        if(mark == 'S') {
            cards[0][rank - 1] = 1;
        } else if(mark == 'H') {
            cards[1][rank - 1] = 1;
        } else if(mark == 'C') {
            cards[2][rank - 1] = 1;
        } else if(mark == 'D') {
            cards[3][rank - 1] = 1;
        }
        
    }
    
    for(int a = 0; a < 4; a++) {
        for(int b = 0; b < 13; b++) {
          char numToMark;
          if(a == 0) {
            numToMark = 'S';
          } else if(a == 1) {
            numToMark = 'H';
          } else if(a == 2) {
            numToMark = 'C';
          } else if(a == 3) {
              numToMark = 'D';
          }
          if(cards[a][b] == 0) {
              cout << numToMark << " " << b + 1 << endl;
          }
        }
    }
    
}
