	int row,col;

int calc(int *list, int size){
	int left, right, tmp;
	int min = 1000000000;
	//要素数１のとき
	if(size == 1) return 0;
	//要素数２のとき
	if(size == 2) return  *(list - 1) * (*list) * (*(list + 1));
	//それ以上
	for(int i = 1; i < size; i++){
		left = calc(list, i);
		right = calc(list+i, size-i);
		tmp = *(list - 1) * (*(list + (i-1))) * (*(list + (size-1))) + left + right;
		min = (tmp < min)? tmp : min;
	}
	return min;
}

int main(){
	int n, r,c, ans;
	int list[101];
	
	scanf("%d", &n);
	for(int i = 1; i < n+1; i++){
		scanf("%d %d", &r, &c);
		if(i == 1) list[0] = r;
		list[i] = c;
	}
	ans = calc(list + 1, n);
	printf("%d\n", ans);
