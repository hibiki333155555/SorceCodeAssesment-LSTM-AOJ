#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    int a[n],i,j,k=0;
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(a[i]<a[j]){
                int t=a[i];
                a[i]=a[j];
                a[j]=t;
                k++;
            }
        }
    }
    for(i=0;i<n;i++){
        if(i==n-1)
         printf("%d",a[i]);
         else
         printf("%d ",a[i]);
    }
   
    printf("\n%d",k);
}
