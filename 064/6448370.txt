inline ll read()
{
    ll x=0, f=1; char ch=getchar();
    while(ch<'0' || ch>'9') {if(ch=='-') f=-1; ch=getchar();}
    while(ch>='0' && ch<='9') {x=x*10+ch-'0'; ch=getchar();}
    return x*f;
}


ll gcd(ll a,ll b)
{
    if (b==0) return a;
    return gcd(b,a%b);
}

ll n_gcd(vector<ll> a)
{
    ll res=a[0];
    int siz=a.size();
    for (int i=1;i<siz;++i) res=gcd(res,a[i]);
    return res;
}
