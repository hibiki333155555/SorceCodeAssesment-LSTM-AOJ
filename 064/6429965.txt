#include<bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)
//using ll=long long;
int main(){
    while(true){
        int n,ans=0,cun=0;cin>>n;vector<int> k(n);
        rep(i,n){
            cin>>k[i];
            if(k[i]<=1){
                cun++;
                ans+=k[i];
            }
        }
        sort(k.begin(),k.end());
        reverse(k.begin(),k.end());
        if(n==0)break;
        else if(n==1)cout<<2<<"\n";
        else if(n==cun)cout<<"NA"<<"\n";
        else {
            cout<<n+1<<"\n";
        }


    }
    
}
