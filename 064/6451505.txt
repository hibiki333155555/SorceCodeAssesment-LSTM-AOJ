#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int a, b;
	cin >> a >> b;
	if (a < b) swap(a, b);
	
	int i;
	for (i = 30; i > 0; i--) if ((a >> i) % 2) break;
	
	int cnt = 0;
	for (; i >= 0; i--) {
		if ((a >> i) % 2 != (b >> i) % 2) break;
		cnt++;
	}
	
	cout << cnt << endl;
	return 0;
}
