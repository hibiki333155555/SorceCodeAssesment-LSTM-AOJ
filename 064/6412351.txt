#include <stdio.h>
#include <cassert>
#include <iostream>
#include <fstream>

#include <string>

#include <map>

using std::cin;
using std::cout;
using std::endl;
using std::terminate;
using std::vector;
using std::string;
using std::set;
using std::map;

#define rep(i, a, n) for (int i = (a); i < (n); ++i)
#define bit(n, k) ((n >> k) & 1)

class Program {
private:

public:

    Program() {
    }

    void solve() {
        int n, m; cin >> n >> m;

        vector<int> a;
        a.resize(n);
        rep(i, 0, n) cin >> a[i];

        map<int, int> b;
        rep(i, 0, m) {
            int c; cin >> c;
            b[c] = 1;
        }

        int ans = 0;
        for (auto c : a) {
            if (b.count(c) == 1) {
                ans++;
            }
        }

        cout << ans << endl;
    }
};

int main() {

#ifdef DEV
    std::ifstream in("input");
    cin.rdbuf(in.rdbuf());

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();
#endif

    return 0;
}
