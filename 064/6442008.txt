#include <stdio.h>
#include <string.h>

int main(){
    int t,i,hana=0,taro=0,j;
    char card1[101],card2[101];
    
    scanf("%d",&t);
    
    for(i=1;i<=t;i++){
        scanf("%s %s",card1,card2);
        j=strcmp(card1,card2);
        //printf("%d\n",j);
        if(j==0){
            hana+=1;taro+=1;
        }
        else if(j>0){
            taro+=3;
            //printf("%d\n",j);
        }
        else{
            hana+=3;
            //printf("%d\n",j);
        }
    }
    
    printf("%d %d\n",taro,hana);
}
