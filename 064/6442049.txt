#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    long long N;
    long long num[11000];
    
    cin >> N;
    for(int i = 0; i < N; i++){
        cin >> num[i]; 
    }
    
    sort(num, num + N);
    long long Min = num[0];
    long long Max = num[N - 1];
    long long sum;
    for(int i = 0; i < N; i++){
        sum += num[i];
    }
    
    cout << Min << " " << Max << " " << sum << endl;
    
    return 0;
}
