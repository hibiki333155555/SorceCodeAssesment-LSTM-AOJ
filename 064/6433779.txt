#include <bits/stdc++.h>
using namespace std;

int main() {
	int m;
	int count=0;
	cin>>m;
	vector<int> v(m);
	for(int i=0;i<m;i++){
		cin>>v[i];
	}
	int k;
	cin>>k;
	vector<int> b(k);
	for(int i=0;i<k;i++){
		cin>>b[i];
	}
	int l=0;
		for(int i=0;i<m;i++){
			if(b[l]==v[i]){
				count++;
				l++;
			}
			else if(l==k){
				break;
			}
	}
	if(count==k){
		cout<<"1"<<"\n";
	}
	else{
		cout<<"0"<<"\n";
	}
	return 0;
}
