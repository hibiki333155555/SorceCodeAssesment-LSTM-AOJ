#include <iostream>
#include <string>
using namespace std;

int main() {
    int n, score_t = 0, score_h = 0;
    cin >> n;
    string t, h;
    for (int i = 0; i < n; i++) {
        cin >> t >> h;
        if (t < h) score_h += 3;
        else if (t > h) score_t += 3;
        else {score_t += 1; score_h += 1;}
    }
    cout << score_t << score_h << endl;
    return 0;
}
