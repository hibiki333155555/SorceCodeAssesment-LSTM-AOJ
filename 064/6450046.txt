#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n, m;
ll x[10000010], odd;
void _Init() {
    m = n * (n + 1) / 2;
    ll cnt = 0;
    odd = 0x7fffffffffffffffLL;
    for(int i = 0; i < m; i++) {
        ll t;
        cin >> t;
        if((t % 2) == 0)
            x[cnt++] = t;
        else
            odd = min(odd, t);
    }
}
 
void solve() {
    sort(x, x + n);
    ll base = (ll)sqrt((double)(x[0] * x[1]) / odd);
    cout << base << endl;
    for(int i = 0; i < n; i++) {
        if(i)
            cout << " ";
        cout << x[i] / base;
    }
}
int main() {
#ifdef LOCAL
    freopen("data.in", "r", stdin);
    freopen("data.out", "w", stdout);
#endif
    int ttt = 0;
    while(cin >> n) {
        if(n == 0)
            break;
        if(ttt++)
            cout << endl;
        _Init();
        solve();
    }
 
    return 0;
}
