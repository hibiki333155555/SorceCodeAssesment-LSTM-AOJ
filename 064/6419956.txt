#define MAX 101

int selectionSort( int A[] ,int  N );

int main(){

	int i,count,A[MAX],N;

	scanf("%d",&N);

	for( i = 0 ; i < N ; i++ ){
		scanf( "%d" , &A[i] );
	}
	
	count = selectionSort( A , N );

	for( i = 0 ; i < N ; i++ ){
		if( i == N - 1){
			printf( "%d\n",A[i] );
		}
		else printf( "%d ", A[i] );
	}

	printf("%d\n",count);

	return 0;
}

int selectionSort( int A[] , int N ){

	int i,j,minj,k,temp;
	int count = 0;

	for( i = 0 ; i < N ; i++ ){
		minj = i;
		
		for( j = i ; j <= N - 1 ; j++ ){

			if( A[j] < A[minj]){
				minj = j;
				}
		}

		temp = A[i];
		A[i] = A[minj];
		A[minj] = temp;
	  	count++;


//		for( k = 0 ; k < N ; k++ ){

//		}

	}
	return count;
