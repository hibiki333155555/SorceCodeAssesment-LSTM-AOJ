#include <iostream>
#include <iomanip>
#include <vector>
#include <cmath>

using namespace std;

int main()
{
	int n; cin >> n;
	vector<pair<double, double>> P(n);
	for (auto& [x, y] : P) cin >> x >> y;
	int d = 0;
	for (int i = 0; i < n; ++i) for (int j = i + 1; j < n; ++j) {
		int dx = P[j].first - P[i].first, dy = P[j].second - P[i].first;
		int t = dx * dx + dy * dy;
		if (d < t) d = t;
	}
	cout << fixed << setprecision(12) << sqrt(d) << endl;
	return 0;
}
