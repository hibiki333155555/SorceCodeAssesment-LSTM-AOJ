#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define all(x) x.begin(),x.end()
#define fol(i,s,g) for(int i=(int)(s); i<(int)(g); i++)
#define fix(ans,k) fixed << setprecision(k) << ans

int main(){

    vector<int> fib(2,1);
    int now_i=1;
    while(fib[now_i] < 1e6){
        int k = fib[now_i-1]+fib[now_i-2];
        fib.push_back(k);
        now_i++;
    }

    int x,y;
    cin>>x>>y;
    int rl,ud;
    x>0 ? rl=1 : rl=3;
    y>0 ? ud=2 : ud=0;
    int xk=0,yk=0;
    while(xk<abs(x)||yk<abs(y)){
        xk += fib[rl];
        yk += fib[ud];
        rl += 4;
        ud += 4;
    }
    int ans;
    xk>yk ? ans=xk : ans=yk;
    ans = (ans+1)%3;
    if(ans==0) ans=3;
    cout<<ans<<'\n';
    return 0;
}
