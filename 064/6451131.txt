#include<stdio.h>
int main()   {
    int A[101], n, t, sw;
    int j;
    scanf("%d", &n);
    for (int i=1;i<=n;i++) {
        scanf("%d", &A[i]);
        for (int i=1;i<=n;i++)
        mini=i;
        for (j=i:j<=n;j++) {
            if (A[j]<A[mini]) mini=j;
        }
    t=A[i]; A[i]=A[mini];A[mini]=t;
    if (i!=mini) sw++;
    }
  for (int i=1;i<=n;i++)  {
      if (i>1) printf(" ");
      printf("%d", A[i]);
  }
  printf("/n");
  printf("%d/n", sw);
  return 0;
    
}
