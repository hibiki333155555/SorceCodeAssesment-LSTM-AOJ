import java.util.Scanner;
class Main
{
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int x;
        for(int i = 3; i <= n; i++){
            x = i / 10;
            if(i % 3 == 0){
                System.out.print(" " + n);
            }else if(x == 3){
                System.out.print(" " + n);
            }else if(i % 10 == 3 ){
                System.out.print(" " + n);
            }
        }
    }
}
