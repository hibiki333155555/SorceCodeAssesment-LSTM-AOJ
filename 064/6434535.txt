que = {}
sum = 0
order = {}
n, q = map(int, input().split())
for i in range(n):
    n, q = map(str, input().split())
    que[n] = q

while que != {}:
    rest = 0
    order = dict.fromkeys(que, 0)
    rest = int(que.popitem()[1]) - n
    if rest != 0:
        que[q[i]] = rest
        sum += 100
    elif rest == 0:
        sum += rest
        order[q[i]] = sum
    else:
        break
    
print(order)
