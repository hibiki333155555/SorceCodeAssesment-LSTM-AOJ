template <typename T1, typename T2>
inline bool chmax(T1 &a, T2 b)
{
    return a < b && (a = b, true);
}
template <typename T1, typename T2>
inline bool chmin(T1 &a, T2 b)
{
    return a > b and (a = b, true);
}

signed main()
{
    while (1)
    {
        int n, x;
        cin >> n >> x;
        if (n == 0 and x == 0)
            break;
        int ans = 0;
        rep(i, 1, n + 1) rep(j, i + 1, n + 1)
        {
            int cp = x;
            cp -= i + j;
            if (j + 1 <= cp and cp <= n)
                ans++;
        }

        cout << ans << newl;
    }
}
