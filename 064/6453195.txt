#include <stdio.h>

int main(void)
{
    int a,b,c,d;
    do
    {
        scanf("%d %d %d",&a,&b,&c);
    }while(!((1 <= a && a <= 10000)&&(1 <= b && b <= 10000)&&(1 <= c && c <= 10000)));
    while(!(a<b<c))
    {
        if(a<b)
        {
            d = a;
            a = b;
            b = d;
        }
        if(b<c)
        {
            d = b;
            b = c;
            c = d;
        }
    }
}
