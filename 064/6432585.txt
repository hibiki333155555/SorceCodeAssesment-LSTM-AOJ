#include <iostream>
#include <string>
using namespace std;
long long gcd(long long x, long long y) {
    if (x < y) swap(x, y);
    int b;
    while (x % y != 0) {
        b = x;
        x = y;
        y = b % y;
    }
    return y;
}
long long A, B;
int main() {
	cin >> A >> B;
    cout << gcd(A, B);
    
  
}
