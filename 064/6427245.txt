#include <bits/stdc++.h>
using namespace std;

int main(){
    int N;
    cin>>N;
    int K[N],P[N];
    for(int i=0;i<N;i++){
        cin>>K[i]>>P[i];
    }
    for(int i=0;i<N;i++){
        if(K[i]%P[i]!=0){
            cout<<K[i]%P[i];
        }else{
            cout<<P[i];
        }
        cout<<"\n";
    }
    return(0);
}
