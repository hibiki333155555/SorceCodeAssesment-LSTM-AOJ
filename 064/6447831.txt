#include <stdio.h>

int main(){
    int n,i,a[100];
    
    scanf("%d",&n);
    
    scanf("%d",&a[0]);
    printf("%d",a[0]);
    
    for(i=1;i<n;i++){
        scanf("%d",&a[i]);
        printf(" %d",a[i]);
    }
    printf("\n");
    
    int j,t,key,k;
    
    for(i=1;i<n;i++){
        key=a[i];
        j=i-1;
        while(a[j]>key && j>=0){
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=key;

        for(k=0;k<n;k++){
            if(k!=0){
               printf(" ");
            }
            printf("%d",a[k]);
        }
        printf("\n");
    }
    
    return 0;
}
