#else

using namespace std;

int main() {
  int N,W;
  cin >>N>>W;
  vector<vector<int>> vec(N+1, vector<int>(2));
  for (int i = 1; i <= N; i++){
    for(int j=0;j<2;j++){
      cin >> vec[i][j];
    }
  }

vector<vector<int>> memo (N+1,vector<int>(W+1,0));
for (int i = 1; i <= N; i++)
{
  for (int j = 0; j <= W; j++){
    int b=(int)j/vec[i][1];
      if(j-b*vec[i][1]>=0){
        memo[i][j]=max(memo[i-1][j],memo[i-1][j-b*vec[i][1]]+b*vec[i][0]);
      }
     
  }
  

  
}
dprint(memo);
int ret=0;
for (int i = 0; i <= W; i++)
{
  ret =max(ret,memo[N][i]);
}

cout << ret << endl;
return 0;
}
