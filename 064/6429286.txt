#include<stdio.h>
int main(){
    int H, W, h, w;
    int i = 1;
    
    while(i){
        scanf("%d %d", H, W);
        if(H == 0 && W == 0)i--;
        
        for(h = 1; h <= H; h++){
            
            for(w = 1; w <= W; w++){
                if(h * w /2 == 0)printf(".");
                else printf("#");
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
