#include<iostream>
#include<algorithm>
#include<vector>
#include<cmath>
#include<set>

using namespace std;

using graph = vector<vector<int>>;

using ll=long long;

ll h[500001]={};

void max_heap(int i){
    if(i==1) return;
    if(h[i]>h[i/2]){
        int tmp = h[i];
        h[i]=h[i/2];
        h[i/2]=tmp;
    }
    max_heap(i/2);
}

int main(void){
    int N;

    cin>>N;

    for(int i=1;i<=N;i++){
        cin>>h[i];
        max_heap(i);
    }

    for(int i=1;i<=N;i++){
        cout<<" "<<h[i];
    }
    cout<<endl;

    return 0;
}
