#include <iostream>

using namespace std;

int main () {
    int n, cards[4][13];
    char mark;
    int rank;
    
    cin >> n;
    
    for(int i = 0; i < n; i++) {
        cin >> mark >> rank;
        
        if(mark == 'S') {
            cards[0][rank - 1] = 1;
        } else if(mark == 'H') {
            cards[1][rank - 1] = 1;
        } else if(mark == 'C') {
            cards[2][rank - 1] = 1;
        } else {
            cards[3][rank - 1] = 1;
        }
        
    }
    
    for(int a = 0; a < 4; a++) {
        for(int b = 0; b < 13; b++) {
            if(cards[0][b] == 0) cout << 'S' << " " << b + 1 << endl;
            if(cards[1][b] == 0) cout << 'H' << " " << b + 1 << endl;
            if(cards[2][b] == 0) cout << 'C' << " " << b + 1 << endl;
            if(cards[3][b] == 0) cout << 'D' << " " << b + 1 << endl;
        }
    }
    
}
