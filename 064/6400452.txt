#include<iostream>
using namespace std;
const int MAX = 10000;
void countingSort(int* A, int* B, int n, int Max);

int main()
{
	int n{ 0 };
	cin >> n;
	int* A = new int[n+1] {0};
	int* B = new int[n+1] {0};
	for (int i = 1; i < n+1; ++i)
		cin >> A[i];
	countingSort(A, B, n+1, MAX);
	for (int i = 1; i < n+1; ++i)
	{
		cout << B[i];
		if (i < n)
			cout << " ";
		else
			cout << endl;
	}
	delete[]A;
	delete[]B;
	return 0;
}
void countingSort(int* A, int* B, int n, int Max)
{
	int* C = new int[Max] {0};
	for (int i = 1; i < n; ++i)
		C[A[i]] += 1;
	for (int i = 1; i < Max; ++i)
		C[i] += C[i - 1];
	for (int i = n - 1; i >= 0; --i)
	{
		B[C[A[i]]] = A[i];
		--C[A[i]];
	}

	delete[] C;
}
