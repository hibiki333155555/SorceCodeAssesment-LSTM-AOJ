#include<stdio.h>
#include<stdlib.h>
void makebit(int,int);
void init(int);
int *B;
int main(){
    int na,nm,sum,flag,lim=1;
    int *a,*m;
    scanf("%d",&na);
    a=(int *)malloc(sizeof(int)*na);
    B=(int *)malloc(sizeof(int)*na);
    for(int i=0;i<na;i++){scanf("%d",a+i);lim*=2;}
    //printf("%d\n",lim);
    scanf("%d",&nm);
    m=(int *)malloc(sizeof(int)*nm);
    for(int i=0;i<nm;i++){
        flag=0;
        scanf("%d",m+i);
        for(int j=0;j<lim;j++){
            sum=0;
            makebit(j,na);
            for(int k=0;k<na;k++)sum+=B[k]*a[k];
            //for(int l=0;l<na;l++)printf("%d|",B[l]);printf(" (%d) -> sum : %d\n",j,sum);
            if(sum==m[i]){flag=1;break;}
        }
        if(flag==0)printf("no\n");
        else printf("yes\n");
    }
    return 0;
}
void makebit(int num,int na){
    init(na);
    int now=na-1;
    if(num==0)B[now--]=0;
    else if(num==1)B[now--]=1;
    else{
        while(1){
            B[now--]=num%2;
            num/=2;
            if(num==1){B[now--]=1;break;}
        }
    }
    while(now>0)B[now--]=0;
}
void init(int na){
    for(int i=0;i<na;i++)B[i]=0;
}
