    void solve(){
        int n;
        string s;
        cin>>n>>s;
        vector<ll>a(n);
        rep(i,n)cin>>a[i];

        ll ans=0;
        rep(i,n){
            if(s[i]=='('){
                ans+=a[i];
                a[i]*=-1;
            }
        }
        priority_queue<int,vector<int>,greater<int>>q;
        rep(i,n){
            q.push(a[i]);
            if(i%2==0){
                ans+=q.top();
                q.pop();
            }
        }
        cout<<ans<<endl;
    }  

    int main(){
        ios_base::sync_with_stdio(false);
        cin.tie(NULL);
        
        solve();
