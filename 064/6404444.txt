while True:
    H,W = map(int, input().split())
    
    if(H==0 and W ==0):
        break
    
    for i in range(H):
        a = ""
        for j in range(W):
            a += "#"
    
    for i in range(1,H-1):
        for j in range(1, W-1):
            a += "."
        print(a)
    print("")
