#include <stdio.h>

int main()
{
    int i,j,k=0,n,m,buff=0;
    scanf("%d %d",&n,&m);
    int a[n][m],b[m],p[n];
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<m;i++)
    {
        scanf("%d",&b[i]);
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            buff += a[i][j]*b[j];
        }
        p[k] = buff;
        k++;
        buff = 0;
    }
    for(i=0;i<n;i++)
    {
        printf("%d%s",p[i],i==(n-1)?"":"\n");
    }
    return 0;
}
