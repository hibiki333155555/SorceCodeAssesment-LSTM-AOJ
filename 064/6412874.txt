#include<stdio.h>
#include<iostream>
#define MAX 20000
using namespace std;
int C[MAX];
int A[MAX], B[MAX];
int n;
void CountingSort(int A[], int B[], int k)
{
	for (int j = 1; j <= n; j++) C[A[j]]++;
	for (int i = 1; i <= k; i++) C[i] = C[i] + C[i - 1];
	for (int j = n; j >= 1; j--)
	{
		B[C[A[j]]] = A[j];
		C[A[j]]--;
	}
}

int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d", &A[i]);
	CountingSort(A, B, MAX);
	for (int i = 1; i <= n; i++)
	{
		if (i>1) printf(" ");
		printf("%d", B[i]);
	}
	printf("\n");
	return 0;
}
