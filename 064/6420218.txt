import java.util.Scanner;

public class Main {
	
	@SuppressWarnings("resource")
	public static void main(String[] args){
		Scanner sc = new Scanner(System.in);
				
		while (true) {
			int x = sc.nextInt();
			int y = sc.nextInt();
			if (x == 0 && y == 0) {
				break;
			}
			//昇順ソート			
            if (x>y){
                int escape=x;
                x=y;
                y=escape;
            }
            System.out.println(x+" "+y);
        }
    }
}
