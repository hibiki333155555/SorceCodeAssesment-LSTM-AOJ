#include<iostream>
#include<vector>
#include<string>
using namespace std;

int main(){
    int H, W; cin >> H >> W;
    vector<string> field(H);
    for(int i=0; i<H; ++i) cin >> field[i];

    int R, C; cin >> R >> C;
    vector<string> p(R);
    for(int i=0; i<R; ++i) cin >> p[i];

    for(int i=0; i<=H-R; ++i){
        for(int j=0; j<=W-C; ++j){
            for(int k=0; k<R; ++k){
                if(field[i+k].substr(j,C) != p[k]) break;
                if(k==R-1) cout << i << " " << j << endl;
            }
        }
    }
    return 0;
}
