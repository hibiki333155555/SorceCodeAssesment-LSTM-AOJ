int assign[101][7];
int tigau(int hanni1,int hanni2){
    int i,j,b[7],dice[7];
    for (j = 0; j < 24; j++) {
		for (i = 1; i <= 6; i++) dice[i] = rot[j][i];
		for (i = 1; i <= 6; i++) b[i] = assign[hanni1][dice[i]];
		for (i = 1; ; i++) {
			if (i > 6) { return 1; }
			if (assign[hanni2][i] != b[i]) break;
		}
	}
	return 0;
}
int main()
{
