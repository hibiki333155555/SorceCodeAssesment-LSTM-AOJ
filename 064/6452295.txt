#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
  int n, k;
  cin >> n >> k;

  int *weights = new int[n];
  int *trucks = new int[k];
  int sum = 0;;

  for (int i = 0; i < n; i++)
  {
    cin >> weights[i];
    sum += weights[i];
  }
  
  int P = (sum + k - 1) / k;
  int truck_i = 0;
  int truck_w = 0;
  for (int i = 0; i < n; i++)
  {
    if (truck_w + weights[i] > P )
    {
      trucks[truck_i] = truck_w;
      truck_w = weights[i];
      truck_i++;
    }
    else truck_w += weights[i];
  }
  
  cout << P << endl;

  return 0;
}
