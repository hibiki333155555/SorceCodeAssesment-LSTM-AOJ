#include<stdio.h>

int main(void)
{
    int n;
    int x;
    int y=1000000;
    int z=-1000000;
    long long sum=0;
    
    scanf("%d",&n);
    
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        
        if(x<y)
        {
            y=x;
        }
        if(x>z)
        {
            z=x;
        }
        sum=sum+x;
    }
    
    printf("%d %d %10d\n",y,z,sum);
    
    return 0;
    
}
