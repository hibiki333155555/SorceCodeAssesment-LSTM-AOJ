#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> e(4);
    for(int i = 0; i < 4; i++){
        cin >> e[i];
    }
    sort(e.begin(), e.end());
    if(e[0] == e[1] && e[2] == e[3]){
        cout << "yes" << endl;
    } else {
        cout << "no" << endl;
    }
    return 0;
}
