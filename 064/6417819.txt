#include "bits/stdc++.h"
using namespace std;
using ll=long long;

int main(){
    ll n,k; cin>>n>>k;
    vector<int>a(n);
    for(int i=0;i<n;i++)cin>>a[i];
    sort(a.begin(),a.end());
    ll c=0,o=0,sum=0;
    for(int i=0;i<n;i++){
        if(i<k){
            if(i>n-1&&c>0){
                o++;c--;continue;
            }
            if(a[i]==1){
                c++;continue;
            }
            if(1<a[i]&&6>a[i]){
                o++;continue;
            }
            if(c>0){
                o++;c--;
            }
            else o++;
        }
        else sum+=a[i];
    }
    sum+=o*6+c*5;
    cout<<sum<<endl;
}
