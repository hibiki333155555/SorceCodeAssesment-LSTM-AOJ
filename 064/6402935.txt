#include<stdio.h>
int main(void)
{
    int i,a,min,max,sum=0;
    scanf("%d",&a);
    int b[a];
    for(i = 0;i < a;i++)
    {
        scanf("%d",&b[i]);
    }
    i = 1;
    min = b[0];
    max = b[0];
    sum = b[0];
    while(i<a)
    {
        sum += b[i];
        if(min > b[i])
        {
            min = b[i];
        }
        if(max < b[i])
        {
            max = b[i];
        }
        i++;
    }
    printf("%d %d %d\n",min,max,sum);
    return 0;
}
