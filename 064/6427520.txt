#include <iostream>
using namespace std;
#include <stdio.h>

int binary_search(int A[],int n,int key){
    int left=0;
    int right=n;
    int mid;
    while (left<right){
        mid=int((left + right)/2);
        if (A[mid]==key){
            return mid;
        }else if(key<A[mid]){
            right=mid;
        }else {
            left=mid+1;
        }
    }
    return -1;
}
int main(void){
    // Your code here!
    int n;
    scanf("%d",&n);
    int A[n];
    for (int i=0;i<n;i++) scanf("%d",&A[i]);
    int key,sum=0,q;
    scanf("%d",&q);
    for(int i=0;i<q;i++){
        scanf("%d",&key);
        if (binary_search(A,n,key)!=-1)sum++;
    }
    printf("%d\n",sum);
    return 0;
    
}
