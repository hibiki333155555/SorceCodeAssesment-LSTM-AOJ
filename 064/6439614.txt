#include <stdio.h>

int primeNum[20000];

void primeNumStr(void){
    int i,j,num=1;
    
    primeNum[0]=2;
    for(i=3;i<=60000;i+=2){
        for(j=3;j<i;j+=2){
            if(i%j==0) break;
        }
        if(i==j){
            primeNum[num]=i;
            num++;
        }
    }
}

int main(void){
    int i,j,n,toimen,flg,counter;
    
    primeNumStr();
    
    while(1){
        scanf("%d",&n);
        if(n==0) break;
        
        counter=0;
        for(i=0;primeNum[i]<=n/2;i++){
            flg=-1;
            toimen = n-primeNum[i];
            //printf("%5d: %5d + %5d = %d ",i,primeNum[i],toimen,n);
            for(j=0;primeNum[j]<n;j++){
                if(toimen==primeNum[j]) counter++;
            }
        }
        printf("%d\n",counter);
    }
}
