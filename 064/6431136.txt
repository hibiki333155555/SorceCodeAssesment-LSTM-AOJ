#include <stdio.h>

int main(void){
    int t,h,s,r;
    
    while(1){
        scanf("%d %d %d",&t,&h,&s);
        if((t+1)*(h+1)*(s+1)==0) break;
        r=(120*60-t*3600-h*60-s);
        printf("%02d:%02d:%02d\n",(int)floor((double)r/3600),(int)(floor((double)r/60))%60,r%60);
        r*=3;
        printf("%02d:%02d:%02d\n",(int)floor((double)r/3600),(int)(floor((double)r/60))%60,r%60);
    }
}
