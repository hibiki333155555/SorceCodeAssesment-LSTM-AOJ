#include<bits/stdc++.h>
using namespace std;
void print(vector<int>v){
 for(int i=0;i<v.size();i++){
  cout<<v[i]<<" ";
 }
 cout<<"\n";
}
void InsertionSort(vector<int>&A){
 int n=A.size();
 for(int i=1;i<n;i++){
  int j=i-1;
  int temp=A[i];
  while(j>=0&&temp<A[j]){
   A[j+1]=A[j];
   j--;
  }
  A[j+1]=temp;
  print(A);
 }
}
int main(){
 int n,i;
 cin>>n;
 vector<int>v(n);
 for(i=0;i<n;i++){
  cin>>v[i];
 }
  cout<<"\n";
  for(i=0;i<n;i++){
   if(i==n-1){
    cout<<v[i];
   }
   else{
    cout<<v[i]<<" ";
   }
  
 }
 cout<<"\n";
 InsertionSort(v);
 cout<<"\n";
 return 0;
 
}
