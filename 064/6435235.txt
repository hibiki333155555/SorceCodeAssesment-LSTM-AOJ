#include<iostream>
using namespace std;

int main() {

    int h, r, h_head, h_bottom;
    cin >> h >> r;
    h_head = h + r;
    h_bottom = h - r;

    if (h_head > 0) {
        cout << 1 << endl;
    } else if (h_head < 0) {
        cout << -1 << endl;
    } else {
        cout << 0 << endl;
    }

    return 0;
}
