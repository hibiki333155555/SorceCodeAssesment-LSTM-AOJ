#include <bits/stdc++.h>
#include <string>
using namespace std;

int main() {

    int n,min=-2000000,max=2000000,sum=0,a;
    cin>>n;
    

    for(int i=0;i<n;++i){
        cin>>a;
        if(a<min)min=a;
        if(a>max)max=a;
        sum+=a;
    }
    
    cout<<min<<" "<<max<<" "<<sum<<endl;

    return 0;
}
