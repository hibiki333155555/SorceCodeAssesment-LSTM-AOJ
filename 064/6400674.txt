#include <cstdio>

int main(void){
    while(1){
        int H, W;
        scanf("%d %d", &H, &W);
    
        if(H<3 || H>300 || W<3 || W>300){
            return 0;
        }
        for(int i=0; i<H; ++i){
            for(int j=0; j<W; ++j){
                if(i == 0 || i == H-1 || j == 0 || j == W-1)
                    putchar('#');
                else
                    putchar('.');
            }
        }
    }
    return 0;
}
