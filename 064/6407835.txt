#include<stdio.h>
int main(){
    int a,i;
    int hai[100];
    scanf("%d",&a);
    
    
    for(i=0;i<a;i++){
        scanf("%d ",&hai[i]);
    }
    for(i=a;i>0;i--){
        printf("%d",hai[i-1]);
         if(i>1){
            printf(" ");
        }
    }
    printf("\n");
    
    return 0;
}
