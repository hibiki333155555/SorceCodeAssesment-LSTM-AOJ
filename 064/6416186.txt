#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    int a[100];
    for (int i=0; i<n; i++) {
        cin >> a[i];
    }
    for (int i=n-1; i>=0; i--) {
        if (i == 0) {
            cout << a[i] << endl;
        } else {
            cout << a[i] << " ";
        }
    }
    return 0;
}
