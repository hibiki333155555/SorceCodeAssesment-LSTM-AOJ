#include <stdio.h>

int main(void){
    int n;
    long long int a1;

    scanf("&d&d", &n, &a1);

    int min, max, sum;
    min = max = sum = a1;

    int i = 0;
    while(i < n - 1){
        long long int ai;
        scanf("%d", &ai);
        if(ai < min) min = ai;
        if(ai > max) max = ai;
        sum += ai;
        i++;
    }

    printf("%d %d %d\n", min, max, sum);

    return 0;
}
