#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<double> x(n),y(n),a(n);
  for(int i=0; i<n; i++) cin >> x.at(i);
  for(int i=0; i<n; i++) cin >> y.at(i);
  double m=0;
  for(int i=0; i<n; i++){
    a.at(i) = abs(x.at(i)-y.at(i));
    m = max(m,a.at(i));
  }
  
  cout << fixed << setprecision(15);
  
  for(int i=1; i<=3; i++){
    double d=0;
    for(int j=0; j<n; j++){
      d += pow(a.at(j),i);
    }
    cout << pow(d, (double)1/i) << endl;
  }
  
  cout << m << endl;
  
}
