#include<stdio.h>
int main()
{
    int n,x,b,c;
    scanf("%d%d%d%d",&n,&x,&b,&c);
    int count=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            for(int k=1;k<=n;k++)
            {
                if(i!=j && j!=k && i!=k)
                {
                    if(i+j+k==x) count++;
                }
            }
        }

    }
    printf("%d\n",count/6);
    return 0;
}
