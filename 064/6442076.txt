#include<bits/stdc++.h>
using namespace std;
int a[1010];
int main() {
	int n;
	while (cin>>n && n) {
		double sum = 0, ans = 0;
		for (int i = 0; i < n; i++) {
			cin>>a[i];
			sum += a[i];
		}
		sum /= n;
		for (int i = 0; i < n; i++) {
			ans += pow(a[i]-sum, 2); 
		}
		ans = sqrt(ans/n);
		cout<<fixed<<setprecision(8)<<ans<<endl;
	}

	
	
	return 0;
}
