#include <bits/stdc++.h>
using namespace std;

int main() {
int n;
cin>>n;
vector<int> v(n);
for(int i=0;i<n;i++){
	cin>>v[i];
}
int n1,c=0,c1=0;
cin>>n1;
vector<int> v1(n1);
for(int i=0;i<n1;i++){
	cin>>v1[i];
}
for(int i=0;i<n;i++){
	for(int j=0;j<n1;j++)
	if(v1[j]==v[i])
	c++;
	else
	c1++;
}
if(c==n1){
	cout<<"1"<<endl;
}
else{
	cout<<"0"<<endl;
}
}
