#include <iostream>

using namespace std;

int main()
{
    int H = 3;
    int W = 3;
    cin>>H>>W;
    while(W != 0 && H != 0)
    {
    for(int x = 0;x < W;x++)
    {
        cout<<"#";
    }
    cout<<endl;
    
    for(int i = 0;0 < i && i < H-1;i++)
    {
        cout<<"#";
        for(int j = 0;0 < j && j < W-1;j++)
        {
            cout<<".";
        }
        cout<<"#"<<endl;
    }
    
    for(int y = 0; y < W;y++)
    {
        cout<<"#"<<endl;
    }
    cout<<endl;
    }
    return 0;
}
