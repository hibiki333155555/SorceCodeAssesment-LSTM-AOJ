#include <iostream>

using namespace std;

int data[10000];
bool isPrime(int n)
{

  bool is_prime = true;
  // 0 and 1 are not prime numbers
  if (n == 0 || n == 1) {
    is_prime = false;
  }

  // loop to check if n is prime
  for (i = 2; i <= n/2; ++i) {
    if (n % i == 0) {
      is_prime = false;
      break;
    }
  }
  return is_prime;
}
int main()
{
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> data[i];
    }
    int cnt = 0;
    for (int i = 0; i < n; ++i) {
        if (isPrime(data[i])) {
            ++cnt;
        }
    }
    cout << cnt << endl;
    return 0;
}
