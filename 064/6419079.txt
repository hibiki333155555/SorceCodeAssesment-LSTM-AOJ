#include <bits/stdc++.h>
using namespace std;

int V, E;
int const INF = numeric_limits<int>::max() / 2;
vector<vector<int>> G, dp;

int main()
{
  cin >> V >> E;
  G.assign(V, vector<int>(V, INF));
  G.at(0).at(0) = 0;
  for (int i = 0; i < E; i++) {
    int s, t, d;
    cin >> s >> t >> d;
    G.at(s).at(t) = d;
  }
  dp.assign((1 << V), vector<int>(V, INF));
  dp.at(0).at(0) = 0;
  int ans = INF;
  for (int S = 0; S < (1 << V); S++) {
    for (int u = 0; u < V; u++) {
      for (int v = 0; v < V; v++) {
        if (S & (1 << v)) continue;
        int &temp = dp.at(S | (1 << v)).at(v);
        temp = min(temp, dp.at(S).at(u) + G.at(u).at(v));
      }
    }
  }
  ans = dp.at((1 << V) - 1).at(0);
  if (ans >= INF)
    ans = -1;
  cout << ans << '\n';
}
