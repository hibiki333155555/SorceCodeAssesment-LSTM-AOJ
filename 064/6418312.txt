from heapq import *

N, K = map(int, input().split())
A = list(map(int, input().split()))

maxi = {1:5, 2:6, 3:6, 4:6, 5:6, 6:5}
heapify(A)
ans = []

while A:
    if A[0]==6:
        ans.append(heappop(A))
    else:
        if K>0:
            K -= 1
            heappush(A, maxi[heappop(A)])
        else:
            ans.append(heappop(A))
print(sum(ans))
