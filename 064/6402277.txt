#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main(){
    string str;
    vector<int> cnt(26, 0);
    while(true){
        cin>>str;
        for(int i=0; i < str.size();i++){
            if('A' <= str.at(i) && str.at(i) <= 'Z'){
                cnt[(char)str.at(i)-'A']++;
            }
            else if('a' <= str.at(i) && str.at(i) <= 'z'){
                cnt[(char)str.at(i)-'a']++;
            }
        }
    }
    for(char c = 'a'; c <= 'z'; c++){
        cout << c << " : " << cnt[c-'a'] << endl;
    }
    return 0;
}
