#include <bits/stdc++.h>
using namespace std;
 
int main() {
  string str, str2;
  int m, h;
  
  while(true){
    //文字列取得
    cin >> str;
    
    //ループ抜ける条件
    if (str == "-"){
      break;
    }
    
    //シャッフル回数取得
    cin >> m;
    
    //m回シャッフル
    for (int i=0; i<m; i++){
      cin >> h;
      
      for (int i=h; i<str.size(); i++){
        str2 += str[i];
      }
      
      for (int i=0; i<h; i++){
        str2 += str[i];
      }
      
      str = str2;
      str2.clear();
      
    }
    
    cout << str << endl;
  }
  
}
