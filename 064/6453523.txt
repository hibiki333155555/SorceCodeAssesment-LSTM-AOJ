#include <bits/stdc++.h>
using namespace std;

void BubbleSort(vector<string> a, int n){
  for(int i = 0; i < n; i++){
    for(int j = n - 1; j > i; j--){
      if(a[j].at(1) < a[j-1].at(1)) swap(a[j], a[j-1]);
    }
  }
  for(int i = 0; i < n; i++){
    if(i == n-1) cout << a[i] << endl;
    else cout << a[i] << " ";
  }
}

void SelectionSort(vector<string>a, int n){
  bool check = false;
  for(int i = 0; i < n; i++){
    int count = 0;
    int minj = i;
    for(int j = i; j < n; j++){
      if(a[j].at(1) < a[minj].at(1)){
        minj = j;
        count++;
      }
    }
    swap(a[i], a[minj]);
    if(count > 1) check = true;
  }

  for(int i = 0; i < n; i++){
    if(i == n-1) cout << a[i] << endl;
    else cout << a[i] << " ";
  }
  if(check) cout << "Not stable" << endl;
  else cout << "Stable" << endl;
}

int main(){
  int n;
  cin >> n;
  vector<string> a(n), b(n);
  for(int i = 0; i < n; i++) cin >> a.at(i);
  for(int i = 0; i < n; i++) b.at(i) = a.at(i);
  BubbleSort(a, n);
  cout << "Stable" << endl;
  SelectionSort(b, n);

  return 0;
}
