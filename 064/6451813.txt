long long int search(long long int target,long long int i,long long int a[],long long int n){
   if(target<0||i>=n){
       return 0;
   }
   if(target==0){
       return 1;
   }
   if( search(target,i+1,a,n)==0&&search(target-a[i],i+1,a,n)==0){
       return 0;
   }else{
       return 1;
   }
}

     cin>>a[i];

      }
      putchar('\n');
  }
  return 0;
}
