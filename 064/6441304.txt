#include<iostream>
#include<vector>
#include<stack>
#include<string>
#include<algorithm>
using namespace std;
int main(){
    stack<int> st1;
    char ch;
    int sum=0;
    int a=0;
    stack<pair<int,int>> st2;
    for(int i=0;cin>>ch;i++){
        if(ch=='\\'){
            st1.push(i);
        }else if(ch=='/'&&st1.size()>0){
            int j=st1.top();
            st1.pop();
            sum+=i-j;
            a=i-j;
            while(st2.size()>0&&st2.top().first>j){
                a+=st2.top().second;
                st2.pop();
            }
            st2.push({j,a});
        }
    }
    cout<<sum;
    vector<int> ans;
    while(!st2.empty()){
        ans.push_back(st2.top().second);
        st2.pop();
    }
    reverse(ans.begin(),ans.end());
    cout<<ans.size();
    for(int i=0;i<ans.size();i++){
cout<<" ";
        cout<<ans[i];
    }
    
    return 0;
}
