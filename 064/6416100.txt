#include<bits/stdc++.h>
#include<cmath>
#include<numeric>
using namespace std;

int main(){

    int n;
    cin>>n;
    int score[1000],sum,t;
    double h,ans,b;
    
    while(1){
        for(int i=0;i<n;++i)cin>>score[i];
        if(n==0)break;
        ++t;

        h=accumulate(score,score+n,0)/n;

        for(int i=0;i<n;++i){
            b+=pow(score[i]-h,2);
        }

        ans=sqrt(b/n);

        cout<<fixed<<setprecision(10)<<ans<<endl;
    }

    return 0;
}
