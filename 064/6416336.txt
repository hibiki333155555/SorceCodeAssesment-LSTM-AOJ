import java.util.Scanner;
import java.util.Arrays;

public class Main {
    public static void main(String[] args){
        Scanner scan=new Scanner(System.in);
        int a[]=new int[3];
        for(int i=0;i<3;i++){
            a[i]=scan.nextInt();
        }
        Arrays.sort(a);
        for(int i=0;i<3;i++){
            System.out.print(a[i]);
            if(i+1<3)System.out.print(" ");
        }
        System.out.println();
        scan.close();
    }
}
