#include<bits/stdc++.h>
using namespace std;
int main(){
    string s; getline(cin, s);
    istringstream ss(s);
    string temp;
    vector<string> word;
    while(ss >> temp){
        word.push_back(temp);
    }
    for(int j = 0; j < word.size(); j++){
        if(word[j].size() < 5)
            continue;
        for(int i = 0; i + 5 <= word[j].size(); ){
            if(word[j].substr(i, 5) == "apple"){
                word[j] = word[j].replace(i, 5, "peach");
                //cout << word[j].substr(i, 5) << endl;
                i += 5;
            }
            else if(word[j].substr(i, 5) == "peach"){
                word[j] = word[j].replace(i, 5,"apple");
                //cout << word[j].substr(i, 5) << endl;
                i += 5;
            }
            else
                i++;
        }
        //cout << j << " " << word[j] << endl;
    }
    for(int i = 0; i < word.size(); i++){
        cout << word[i];
        if(i != word.size() - 1)
            cout << " ";
    }
}
