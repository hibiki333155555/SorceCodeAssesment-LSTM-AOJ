#include<bits/stdc++.h>
using namespace std;
int main(){
  int N,M;
  cin>>N>>M;
  vector<string> S(N);
  int X=0;
  int Y=0;
  for(int i=0;i<N;i++){
    cin>>S[i];
    for(int j=0;j<M;j++){
      if(S[i][j]=='o'){
        X++;
      }else{
        Y++;
      }
    }
  }
  int ANS=N*M;
  int A=0,B=0;
  for(int i=0;i<N-1;i++){
    for(int j=0;j<M;j++){
      if(S[i][j]=='o'){
        A++;
      }else{
        B++;
      }
    }
    ANS=min(ANS,min(A+Y-B,X-A+B));
  }
  A=0;
  B=0;
  for(int i=0;i<M-1;i++){
    for(int j=0;j<N;j++){
      if(S[j][i]=='o'){
        A++;
      }else{
        B++;
      }
    }
    ANS=min(ANS,min(A+Y-B,X-A+B));
  }
  cout<<ANS<<"\n";
}
