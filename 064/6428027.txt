#include<bits/stdc++.h>
using namespace std;

int main(){
  cin.tie(0);
  ios::sync_with_stdio(false);

  int n;
  cin >> n;
  vector<vector<int> > g(n);
  vector<int> dist(n, -1);
  queue<int> que;

  int u, k, v;
  for(int i=0;i<n;i++){
    cin >> u >> k;
    u--;
    for(int j=0;j<k;j++){
      cin >> v;
      v--;
      g[u].push_back(v);
      g[v].push_back(u);
    }
  }

  dist[0]=0;
  que.push(0);
  while(!que.empty()){
    int x = que.front();
    que.pop();
    for(auto y : g[x]){
      if(dist[y] != -1) continue;
      dist[y] = dist[x] + 1;
      que.push(y);
    }
  }

  for(int i=0;i<n;i++){
    cout << i+1 << " " << dist[i] << endl;
  }
  
  return 0;
}
