#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;
#define drep(i, cc, n) for (ll i = (cc); i <= (n); ++i)
#define rep(i, n) drep(i, 0, n - 1)
#define all(a) (a).begin(), (a).end()
#define pb push_back
#define fi first
#define se second

const ll MOD = 1000000007;
const ll MOD2 = 998244353;
const ll INF = 1LL << 60;
const ll N_MAX = 2e5;

int main(){
    ll n;
    cin >> n;

    ll x_min = INF;
    vector<ll> x(n);
    rep(i, n){
        cin >> x[i];
        x_min = min(x_min, x[i]);
    }

    for(ll i=1; i<=x_min; i++){
        bool flag = true;
        rep(j, n){
            if(x[j] % i != 0){
                flag = false;
                break;
            }
        }

        if(flag) cout << i << endl;
    }
}
