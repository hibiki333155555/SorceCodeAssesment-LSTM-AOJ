#include <iostream>
#include <cstdio>
#include <utility>
#include <vector>
#include <algorithm>

int dp[110][1010];

using pi = std::pair<int, int>;

int main() {
    int N, W; std::cin >> N >> W;
    std::vector<pi> items(N);
    for (auto &[v, w] : items) std::cin >> v >> w;

    for (int i = 0; i < N; i++) {
        int value = items[i].first;
        int weight = items[i].second;
        for (int w = 0; w <= W; w++) {
            if (w < weight) continue;
            dp[i+1][w] = std::max(dp[i][w-weight] + value, dp[i][w]);
        }
    }

    std::cout << dp[N][W] << std::endl;
}
