#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    getline(cin, s);
    for(int i=0;i<s.length();i++){
        if('A' <= s[i] && s[i] <= 'z'){
            if(s[i] <= 'Z'){
                s[i] = s[i] + ('a' - 'A');
            }else if('a' <= s[i]){
                s[i] = s[i] - ('a' - 'A');
            }
        }
    }
    cout << s << "\n";
    return(0);
}
