import java.util.Scanner;

public class Main {
    void solve() {
        Scanner sc = new Scanner(System.in);
        int my = sc.nextInt();
        int friend = sc.nextInt();
        int book = sc.nextInt();
        String minimumAmount = null;

        if (book - my - friend > 0) {
            minimumAmount = "NA";
        } else if (book - my >= 0) {
            minimumAmount = String.valueOf(book - my);
        } else {
            minimumAmount = "0";
        }

        System.out.println(minimumAmount);
    }

    public static void main(String[] args) {
        new Main().solve();
    }
}
