#include <iostream>
#include <vector>
#include <algorithm>

int dp[20+1][50010];

int main() {
    int n, m; std::cin >> n >> m;
    std::vector<int> coins(m); for (auto &e : coins) std::cin >> e;

    for (int i = 0; i < m; i++) {
        for (int j = 0; j <= n; j++) {
            if (j != 0) dp[i][j] = (1<<30LL)-1;
        }
    }

    for (int i = 0; i < m; i++) {
        for (int j = 0; j <= n; j++) {
            if (j >= coins[i]) {
                dp[i+1][j] = std::min(dp[i][j - coins[i]] + 1, dp[i][j]);
            } else {
                dp[i+1][j] = dp[i][j];
            }
        }
    }

    std::cout << dp[m][n] << std::endl;
}
