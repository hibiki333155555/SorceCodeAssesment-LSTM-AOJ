#include <iostream>
#include <stdio.h>
#include <cmath>
#include <iomanip>
#include <stack>
#include <vector>
using namespace std;

int main(void){
    int n, m;
    cin >> n >> m;

    vector< vector<int > > a;
    a.resize(n);
    for(int i=0; i<n; i++) {
        a[i].resize(m);
    }
    for (int i=0; i<n; i++) {
        for (int j=0; i<m; j++) {
            cin >> a[i][j];
        }
    }

    int b[m];
    for (int j=0; j<m; j++) {
        cin >> b[j];
    }

    int c[n];
    for (int i=0; i<n; i++) {
        for (int j=0; j<m; j++) {
            c[n] += a[i][j]*b[j];
        }
    }

    for (int i=0; i<n; i++) {
        cout << c[i] << endl;
    }

    return 0;
}
