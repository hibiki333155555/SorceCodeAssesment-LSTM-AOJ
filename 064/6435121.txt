#include<iostream>
using namespace std;
//int a[7]={5,2,4,6,1,3};
int a[110];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
//	for(int i=0;i<n;i++)cout<<a[i]<<" ";
//	cout<<"\n";
	int len = n;
	
	for(int i=0;i<=len-1;i++){
		int key=a[i];
		int j=i-1;
		while(j>=0&&a[j]>key){
			a[j+1]=a[j];
//			for(int i=0;i<len;i++){
//				cout<<a[i]<<" ";
//			}
//			cout<<"\n";
			j--;
		}
		a[j+1]=key;
		for(int i=0;i<len;i++){
			cout<<a[i];
			cout<<' ';
		}
		cout<<"\n";
	}
	return 0;
}
