#include <iostream>
#include <map>
#include <vector>
#include <queue>
using namespace std;

typedef pair<int, int> PII;
int n, m, q, id = 0;
bool vis[100010];
vector<int> ls[100010];
int block[100010];

void dfs(int x)
{
	if(vis[x]) return ;
	vis[x] = true;
	block[x] = id;
	for(int i = 0; i < ls[x].size(); i++) dfs(ls[x][i]);
	return ;
}

int main()
{
	cin >> n >> m;
	for(int i = 0; i < m; i++)
	{
		int u, v;
		cin >> u >> v;
		ls[u].push_back(v);
		ls[v].push_back(u);
	}
	for(int i = 0; i < n; i++)
	{
		if(!vis[i])
		{
			dfs(i);
			id++;
		}
	}
	cin >> q;
	for(int i = 0; i < q; i++)
	{
		int u, v;
		cin >> u >> v;
		if(block[u] == block[v]) cout << "yes" << endl;
		else cout << "no"  << endl; 
	}
	return 0;
}
