#include <iostream>
using namespace std;
int main(void) {
	
	int n,a,m,t; 
	string s(1024,'0'), r(2500, '0');
	cin >> n;
	for (int iiyokoiyo = 0; iiyokoiyo < n; iiyokoiyo++) {
		m = 1000;
		t = 1000;
		cin >> s;
		a = (s.size() - 1) / 3;
		r[1000] = s[0];
		for (int i = 0; i < a; i++) {
			if (s[(i + 1) * 3 - 1] == '>') 
				r[++m] = s[(i + 1) * 3];
			else 
				r[--m] = s[(i + 1) * 3];
			if (t > m)
				t = m;
		}
		for (int i = 0; i < a+1; i++) {
			cout << r[t + i];
		}
		cout << endl;
	}
	return 0;
}
