#include <bits/stdc++.h>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<int> a(n);
  for(int i = 0; i < n; i++) cin >> a.at(i);

  bool flag = true;
  int ans = 0;
  while(flag){
    flag = false;
    for(int j = n - 1; j > 0; j--){
      if(a.at(j) < a.at(j - 1)){
        swap(a.at(j), a.at(j - 1));
        flag = true;
        ans++;
      }
    }
  }
  
  for(int i = 0; i < n; i++){
    if(i == n - 1) cout << a.at(i) << endl;
    else cout << a.at(i) << " "; 
  }
  cout << ans << endl;

  return 0;
}
