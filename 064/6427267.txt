#include <stdio.h>

int main()
{
	int A, B;
	scanf("%d %d", &A, &B);
	if (A == 0 && B == 0) {
		printf("1\n");
		fflush(stdout);
		return 0;
	}
	
	int i, bit[31], ans = 0;
	for (i = 1, bit[0] = 1; i <= 30; i++) bit[i] = bit[i-1] << 1;
	for (i = 30; i >= 0; i--) if ((A & bit[i]) != 0 || (B & bit[i]) != 0) break;
	for (; i >= 0; i--, ans++) if ((A & bit[i]) != (B & bit[i])) break;
	printf("%d\n", ans);
	fflush(stdout);
	return 0;
}
