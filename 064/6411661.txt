#include <iostream>
using namespace std;

int main() {
    int n, count[5][4][11] = {};
    cin >> n;
    for (int i = 0; i < n; i++) {
        int b, f, r, v;
        cin >> b >> f >> r >> v;
        count[b][f][r] += v;
    }
    for (int i = 1; i <= 4; i++) {
        for (int j = 1; j <= 3; j++) {
            for (int k = 1; k <= 10; k++) {
                cout << " " << count[i][j][k];
            }
            cout << endl;
        }
        if (i != 4) {
                cout << "####################" << endl;
            }
    }
}
