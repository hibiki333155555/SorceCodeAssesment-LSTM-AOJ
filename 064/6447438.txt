#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,a,b,c;
    cin>>n;


    for(int i=0; i<n; i++)
    {
        cin>>a>>b>>c;
  
        if( pow(a,2)+pow(b,2)==pow(c,2) || pow(b,2)+pow(c,2)==pow(a,2) || pow(a,2)+pow(c,2)== pow(b,2) ){
            cout<<"YES"<<endl;
        }
        else{
            cout<<"NO"<<endl;
        }
        a=0,b=0,c=0;
 
    }
}
