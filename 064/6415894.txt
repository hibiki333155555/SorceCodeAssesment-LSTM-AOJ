#include<iostream>
#include<string>
using namespace std;
int main() {
    int Taro = 0;
    int Hanako = 0;
    int turn;
    cin >> turn;
    for (int n = 0; n < turn; n++) {
        string taro, hanako;
        cin >> taro >> hanako;
        if(taro == hanako) {
            Taro++;
            Hanako++;
        } else {
            if(taro > hanako) {
                Taro += 3;
            } else if(taro < hanako) {
                Hanako += 3;
            }
        }
    }
    cout << Taro << " " << Hanako << endl;
}
