#include<stdio.h>
#define N 200000
#define INF 1e9

int n;

int MaxProf(int A[]){
  int i,max = -INF,min = INF;

  for(i=0 ;i<n ;i++){
    // if(min > A[i]) min = A[i];
    if(A[i]-min > max) max =  A[i] - min;
    if(min > A[i]) min = A[i];
     }

  return  max;
}


int main(){
  int A[N],i;

  scanf("%d",&n);

  for(i=0 ;i<n ;i++){
    scanf("%d",&A[i]);
  }

  printf("%d\n",MaxProf(A));

  return 0;
}
