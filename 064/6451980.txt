#include <bits/stdc++.h>
using namespace std;

int main() {
    int a,b;
    cin >> a >> b;
    int N;
    cin >> N;
    vector<int>s(N),f(N);
    for(int i = 0; i < N; i++) {
        cin >> s[i] >> f[i];
        if(max(s[i],a) < min(b,f[i])) {
            cout << 1 << endl;
            return 0;
        }
    }
    cout << 0 << endl;
}
