import java.util.*;

class Solve{
    public void Do(){
        final Scanner sc = new Scanner(System.in);
        int s=sc.nextInt();
        System.out.println( s/3600 +":"+ (s%3600)/60 +":"+ s%60 );
        sc.close();
    }
}
class Main{
    public static void main(final String[] _args){
        new Solve().Do();
    }
}
