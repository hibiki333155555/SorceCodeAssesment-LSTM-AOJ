#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

const int inf = 100000;

int main(){
    int n, m;
    cin >> n >> m;

    vector<int> coin(m+1);
    for(int i=1; i<=m; ++i) cin >> coin[i];

    // dp[i][j] := i番目までのコインでj円を支払う最小枚数
    vector<vector<int>> dp(m+1, vector<int>(n+1));

    const int inf = 100000;
    for(int j=1; j<=n; ++j) dp[0][j] = inf;
    for(int i=0; i<=m; ++i) dp[i][0] = 0;

    for(int i=1; i<=m; ++i){
        for(int j=1; j<=n; ++j){
            if(j >= coin[i]) dp[i][j] = min(dp[i-1][j], dp[i][j-coin[i]] + 1);
            else dp[i][j] = dp[i-1][j];
        }
    }

    cout << dp[m][n] << endl;
    return 0;
}
