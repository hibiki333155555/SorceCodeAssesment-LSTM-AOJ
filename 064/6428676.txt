#include <iostream>
#include <vector>
#include <algorithm>
#define tt temtuple
#define ff first.first
#define fs first.second
#define s second
using namespace std;

vector<pair<pair<int, int>,int>> temtuple; // first was used to rank, second where to start
vector<int> sA;
vector<int> r;
string t;
int len;
void sACon()
{
    for (int i = 0; i<len; i++) {
        tt[i].ff=t[i];
        tt[i].s = i;
    }
    for (int i = 1; i<=len; i *= 2) {
        sort(tt.begin(), tt.begin()+len);
        r[tt[0].s] = 0;
        int j = 0;
        for (int k = 0; k<len; k++) {
            if(tt[k]==tt[k-1])
                r[tt[k].s] = j;
            else
                r[tt[k].s] = ++j;
        }
        for (j = 0; j<len; j++) {
            tt[j].ff = r[tt[j].s];
            tt[j].fs = r[tt[j].s+i];
        }
    }
    // last time
    sort(tt.begin(), tt.begin()+len);
    r[tt[0].s] = 0;
    int j = 0;
    for (int k = 0; k<len; k++) {
        if(tt[k]==tt[k-1])
            r[tt[k].s] = j;
        else
            r[tt[k].s] = ++j;
    }
    for (int i = 0; i<len; i++) {
        sA[r[i]] = i;
    }



}

int contain(string s)
{
    int l = 0, r = len - 1;
    while (l<=r)
    {
        int mid = (l+r)/2;
        if(t.compare(sA[mid], s.size(), s)==0)
            return 1;
        else if(t.substr(sA[mid])<s)
        {
            l = mid +1;
        }
        else
        {
            r = mid - 1;
        }

    }
    return 0;
}

int main(void)
{

    cin >> t;
    len = int(t.size());
    temtuple.resize(len*2, {{0, 0}, 0});
    sA.resize(len);
    r.resize(len);
    sACon();
    int q;
    string s;
    cin >> q;
    while(q--)
    {
        cin >> s;
        cout << contain(s) << endl;
    }
    return 0;
}
