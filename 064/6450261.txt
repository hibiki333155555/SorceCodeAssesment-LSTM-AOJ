bool CheckReport(string rp1, string rp2){
    int dif=0;
    for(int i=0;i<(int)rp1.size();i++){
        if(rp1[i]!=rp2[i]) dif++;
    }
    if(dif>=2) return false;
    return true;
}

int main(){
    int N;
    cin >> N;
    vs report(N);
    vector<bool> fusei(N, false);
    for(int i=0;i<N;i++){
        cin >> report[i];
    }
    for(int i=0;i<N-1;i++){
        for(int j=i+1;j<N;j++){
            bool CheckFlag=CheckReport(report[i], report[j]); 
            if(CheckFlag){
                fusei[i]=true;
                fusei[j]=true;
            }
        }
    }
    int ans=0;
    for(int i=0;i<N;i++){
        if(fusei[i]) ans++;
    }
    cout << ans << endl;
}
