#include <stdio.h>
int main() {
    int n, b, f, r, v;
    scanf("%d", &n);
    int floors[4][3][10];
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 3; j++)
            for (int k = 0; k < 10; k++) floors[i][j][k] = 0;

    for (int i = 0; i < n; i++) {
        scanf("%d %d %d %d", &b, &f, &r, &v);
        floors[b - 1][f - 1][r - 1] += v;
    }
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 10; k++) {
                printf(" %d", floors[i][j][k]);
            }
            printf("\n");
        }
        if (i != 3) puts("####################");
    }
}
