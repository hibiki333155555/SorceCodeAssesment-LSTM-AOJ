#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int N, S; cin >> N >> S;
	vector<int> A(N);
	int k = N;
	for (int i = 0, j = 0, s = 0; i < N; ++i) {
		cin >> A[i];
		s += A[i];
		if (s < S) continue;
		while (s >= S) s -= A[j++];
		k = min(k, i - j + 2);
	}
	cout << k << endl;
	return 0;
}
