#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    int n,h;
    while(true){
        cin>>s;
        if(s == "-"){
            break;
        }
        cin>>n;
        char t[300],p[300];
        for(int i=0;i<n;i++){
            cin>>h;
            int atolen = s.size() - h;
            for(int i=0;i<h;i++){
                t[i] = s[i];
            }
            for(int i=0;i<atolen;i++){
                p[i] = s[i + h];
            }
            for(int i=0;i<atolen;i++){
                s[i] = p[i];
            }
            for(int i=atolen;i<s.size();i++){
                s[i] = t[i - atolen];
            }
        }
        cout<<s<<"\n";
    }
    return(0);
}
