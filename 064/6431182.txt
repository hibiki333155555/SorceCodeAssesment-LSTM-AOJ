#include <bits/stdc++.h>
using namespace std;

void print (vector<int> &v)
{
	for(int i = 0 ; i<v.size() ; i++)
	{
		cout<<v[i]<<" ";
	}
	cout<<"\n";
}
void BubbleSortAscendingLeftToRight(vector<int>&arr)
{
	int n = arr.size();
	for(int i = 0 ; i<n-1 ; i++)
	{
		for(int j = 0 ; j<n-1 ; j++)
		{
			if(arr[j]<arr[j+1])
			{
				swap(arr[j],arr[j+1]);
			}
			print (arr);
		}
		cout<<"\n";
	}
}
int main() {
	int n;
	cin>>n;
	vector<int>v(n);
	for(int i = 0 ; i<n; i++)
	{
		cin>>v[i];
	}
	BubbleSortAscendingLeftToRight(v);
	return 0;
}
