#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,c; cin>>n>>c;
    int p[c],ans=0;
    n+=1;
    for(int i=0;i<c;i++){
        cin>>p[i];
        ans+=p[i];
    }
    if(ans>n){
        if(ans%n==0){
            cout<<ans/n<<endl;
        }else{
            cout<<ans/n+1<<endl;
        }
    }else{
        cout<<1<<endl;
    }
    return 0;
}
