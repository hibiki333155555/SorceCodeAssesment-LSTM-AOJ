#include <iostream>

using namespace std;

int main()
{
    int n , m;
    cin>>n>>m;
    int A[100][100]={};
    int b[100]{};
    for(int i = 0; i < n;i++)
    {
        for(int j = 0;j < m;j++)
        {
            cin>>A[i][j];
        }
    }
    for(int j = 0;j < m;j++)
    {
        cin>>b[j];
    }
    for(int  i = 0;i < n;i++)
    {
        int x = 0;
        for(int j = 0; j < m;j++)
        {
            x += A[i][j]*b[j];
        }
        cout<<x<<endl;
    }


    return 0;
}
