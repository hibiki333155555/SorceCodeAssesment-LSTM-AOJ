#include<stdio.h>

int main(void)
{
    int came[4][3][10];
    int b,f,r,v;
    int x,y,z;
    int a,i;
    for(x=0;x<4;x++)
    {
        for(y=0;y<3;y++)
        {
            for(z=0;z<10;z++)
            {
                came[x][y][z]=0;
            }
        }
    }
    scanf("%d",&a);
    for(i=0;i<a;i++)
    {
        scanf("%d %d %d %d",&b,&f,&r,&v);
        came[b-1][f-1][r-1]+=v;
    }
    
    for(x=0;x<4;x++)
    {
        for(y=0;y<3;y++)
        {
            for(z=0;z<10;z++)
            {
                printf(" %d",came[x][y][z]);
            }
            printf("\n");
        }
        if(x!=3)
        {
            printf("####################\n");
        }
    }
    return 0;
}
