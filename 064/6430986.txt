#include <stdio.h>

int main(void){
    int field[16][16];
    int i,j;
    
    while(1){
        for(i=0;i<8;i++){
            for(j=0;j<8;j++){
                if(scanf("%1d",&field[i][j])==EOF) goto finish;
                //printf("%d ",field[i][j]);
            }
            //puts("");
        }
        for(i=0;i<8;i++){
            for(j=0;j<8;j++){
                if(field[i][j]==1&&field[i+1][j]==1&&field[i][j+1]==1&&field[i+1][j+1]==1){
                    puts("A");
                }
                else if(field[i][j]==1&&field[i+1][j]==1&&field[i+2][j]==1&&field[i+3][j+1]==1){
                    puts("B");
                }
                else if(field[i][j]==1&&field[i][j+1]==1&&field[i][j+2]==1&&field[i][j+3]==1){
                    puts("C");
                }
                else if(field[i][j+1]==1&&field[i+1][j]==1&&field[i+1][j+1]==1&&field[i+2][j]==1){
                    puts("D");
                }
                else if(field[i][j]==1&&field[i][j+1]==1&&field[i+1][j+1]==1&&field[i+1][j+2]==1){
                    puts("E");
                }
                else if(field[i][j]==1&&field[i+1][j]==1&&field[i+1][j+1]==1&&field[i+1][j+2]==1){
                    puts("F");
                }
                else if(field[i][j+1]==1&&field[i][j+2]==1&&field[i+1][j]==1&&field[i+1][j+1]==1){
                    puts("G");
                }
            }
        }
    }
    finish:
    return 0;
}
