#include<bits/stdc++.h>
using namespace std;
#define endl "\n"
#define ll long long
#define fix(ans,x) fixed << setprecision(x)<<ans
int main(){
    int n;
    cin>>n;
    while(n!=0){
        int k[n],ans=0;
        for(int i=0;i<n;i++){
            cin>>k[i];
        }
        for(int i=0;i<n;i++){
            if(k[i]<2){
                ans++;
            }
        }
        if(ans==n){
            cout<<"NA"<<endl;
        }else{
            cout<<n+1<<endl;
        }
        cin>>n;
    }
    return 0;
}
