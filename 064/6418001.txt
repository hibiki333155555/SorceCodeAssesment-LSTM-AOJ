struct bsmat{
  array<bitset<1100>,1100> bs,nbs;
  inline void add_edge(int x,int y){
    bs[x].flip(y);
    nbs[y].flip(x);
  }
  bsmat operator *=(bsmat b){
    bsmat c;
    for(int i=0;i<1100;++i)for(int j=0;j<1100;++j){
      if((bs[i]&b.nbs[j]).any())c.add_edge(i,j);
    }
    (*this)=move(c);
    return (*this);
  }
  bsmat operator *(bsmat c){
    return bsmat(*this)*=c;
  }
  bsmat operator ^=(int K){
    bsmat c;
    for(int i=0;i<1100;++i)c.add_edge(i,i);
    if(K&1)c*=(*this);
    while(K){
      K>>=1;
      (*this)*=(*this);
      if(K&1)c*=(*this);
    }
    (*this)=move(c);
    return (*this);
  }
  bsmat operator ^(ll K){
    return bsmat(*this)^=K;
  }
};
int N,D,cnt;
int main(){
  startupcpp();
  cin>>N>>D;
  cnt=N;
  bsmat g;
  for(int i=0;i<N;++i){
    int K;
    cin>>K;
    vector<pair<int,int>> ch(K);
    for(auto&[v,d]:ch){
      cin>>v>>d;
      --v;
    }
    sort(all(ch),[](pair<int,int> l,pair<int,int> r){return l.second<r.second;});
    int nowd=1,bf=i;
    for(auto&[v,d]:ch){
      while(nowd<d){
        g.add_edge(bf,cnt);
        bf=cnt++;
        ++nowd;
      }
      g.add_edge(bf,v);
    }
  }
  g^=D;
  vector<int> ans;
  for(int i=0;i<N;++i){
    bool f=true;
    for(int j=0;j<N;++j)f&=g.nbs[i][j];
    if(f)ans.emplace_back(i+1);
  }
  print(ans.size());
  output(ans);
}
