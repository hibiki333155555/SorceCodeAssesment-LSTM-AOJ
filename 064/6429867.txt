#include <bits/stdc++.h>
using namespace std;
int main(){
    while(true) {
        int a, ans=0;
        cin>>a;
        if(a == 0) return(0);
        for(int i=0;i<a;i++) {
            int b;
            cin>>b;
            ans = max(b,ans);
        }
        if(ans >= 2){
            cout<<a+1<<endl;
        }else{
            cout<<"NA"<<endl;
        }
    }
    return(0);
}
