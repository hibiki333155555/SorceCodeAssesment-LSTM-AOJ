#include <bits/stdc++.h>
#define rep(i, f, n) for (int i = f; i < (int)(n); i++)
using namespace std;

int main()
{
    int n, a[22], q, m[300];
    cin >> n;
    rep(i, 0, n) cin >> a[i];
    cin >> q;
    rep(i, 0, q) cin >> m[i];
    bool v[505000] = {false};
    rep(j, 0, (1 << n))
    {
        int val = 0;
        rep(k, 0, n)
        {
            if (j & 1 << k)
                val += a[k];
        }
        v[val] = true;
    }
    rep(i, 0, q)
    {
        if (v[m[i]])
            cout << "yes" << endl;
        else
            cout << "no" << endl;
    }
}
