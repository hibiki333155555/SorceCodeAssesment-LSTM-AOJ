#include <iostream>
#include <string>
#include <queue>
using namespace std;

struct process {
	string name;
	long t;
};

int main()
{
	long n = 0, q = 0, time = 0, u = 0;
	string tu;
	queue<process> a;

	cin >> n >> q;
	
	for (int i = 0; i < n; i++) {
		cin >> tu >> u;
		a.push(process{tu, u});
	}

	while (!a.empty()) {
		if (a.front().t > q) {
			time += q;
			a.push(process{ a.front().name, a.front().t - q });
			a.pop();			
		}
		else {
			time += a.front().t;
			cout << a.front().name << " " << time << endl;
			a.pop();
		}
	}

	return 0;
}
