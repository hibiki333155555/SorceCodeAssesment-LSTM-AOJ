#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    long long int merg = -10000000000;
    int m;
    cin >> m;


    for (int i = 1; i < n; i++) {
        int a;
        cin >> a;
        m = min(m, a);
        if ((a - m) > merg) {
            merg = a - m;
        }
    }
    cout << merg << endl;
}
