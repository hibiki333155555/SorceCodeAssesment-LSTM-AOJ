#include <iostream>
#include <string>
using namespace std;

int EditDistance(string a, string b) {
    int distance = 0;
    int arr[a.size()][b.size()];
    for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            arr[i][j] = 0;
        }
    }
    for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            if (i != 0 && j != 0 && a.at(i) != b.at(j)) {
                ++arr[i-1][j-1];
            }
        }
    }
    distance = arr[0][0];
    for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            if (arr[i][j] > distance) {
                distance = arr[i][j];
            }
        }
    }
    return distance;
}

int main() {
    string a;
    string b;
    cin >> a >> b;
    cout << EditDistance(a, b);
    return 0;
}
