int main() {
    string w, t;
    cin >> w >> t;
    int ans = 0;
    while(t != "END_OF_TEXT") {
        if(w.size() == t.size()) {
            for(char &ch : t) {
                if(isupper(ch)) ch = tolower(ch);
            }
            if(w == t) ans++;
        }
        cin >> t;
    }
    cout << ans << endl;
    return 0;
}
