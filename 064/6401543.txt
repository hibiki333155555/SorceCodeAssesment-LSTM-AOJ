#include <stdio.h>

int main(void){
    int n,m,i,j;
    scanf("%d %d",&n,&m);
    int A[n][m],b[m],c[n];
    
    c[0]=0;
    
    for(i=0;i<n;i++){
        for(j=0;j<m;m++){
            scanf("%d",&A[i][j]);
        }
    }
    for(i=0;i<m;i++){
        scanf("%d",&b[i]);
    }
    for(i=0;i<n;i++){
        for(j=0;j<m;j++){
            c[i]=c[i]+A[i][j]*b[j];
        }
        printf(" %d",c[i]);
    }
    printf("\n");
}
