#include<stdio.h>
int main(void)
{
    int a,b,i,j,k;
    for(i = 0 ;i < 1000000000;i++)
    {
        if(i != 0)
        {
            printf("\n");
        }
        scanf("%d %d",&a,&b);
        if(a == 0 && b == 0)
        {
            break;
        }
        for(j = 0;j <a;j++)
        {
            for(k = 0;k<b;k++)
            {
                if(j%2 == 1 && k%2 ==1)
                {
                    printf("#");
                }
                else if(j%2 ==1 && k%2==0)
                {
                    printf(".");
                }
                else if(j%2 == 0 && k%2 == 1)
                {
                    printf(".");
                }
                else if(j%2 ==0 && k%2 ==0)
                {
                    printf("#");
                }

            }
            printf("\n");
        }
    }
    printf("\n");
    return 0;
}
