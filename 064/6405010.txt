import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
	Scanner sc=new Scanner(System.in);
	int k=0, n=sc.nextInt();
	int[] a=new int[1001];
	for(int i=1;i<n;i++) {
		a[i]=sc.nextInt();
	}
	for(int i=1;i<n;i++) {
		if(a[i-1]<a[i]) {
			k=k+a[i]-a[i-1];
		}else if(a[i-1]>a[i-1]) {
			k=k+a[i-1]-a[i];
		}
	}
	System.out.println(k);
	sc.close();
	}
}
