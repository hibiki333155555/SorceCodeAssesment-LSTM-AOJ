#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    while(n!=0){
        int k,ans=0;
        for(int i=0;i<n;i++){
            cin>>k;
            ans=max(k,ans);
        }
        if(ans<2) cout<<"NA"<<endl;
        else if(n==1) cout<<2<<endl;
        else if(ans==2||ans==n) cout<<n+1<<endl;
        else cout<<ans<<endl;
        cin>>n;
    }
    return(0);
}
