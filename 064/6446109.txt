#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    int n;
    int num[1100];
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> num[i];
    }
    
    reverse(num, num + n);
    for(int i = 0; i < n; i++){
        cout << num[i];
        if(i != n-1) cout << " ";
    }
    cout << endl;
    
    return 0;
}
