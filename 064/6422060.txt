#include <iostream>

using namespace std;

int main()
{
    int n = 0;
    cin>>n;
    for(int i = 1; i <= n; i++)
    {
        if(i % 3 == 0)
        {
            cout<<" "<<i;
        }
        else if(i % 10 == 3)
        {
            cout<<" "<<i;
        }
        cin>>n;
    }
    cout<<endl;

    return 0;
}
