#include <algorithm>
#include <iostream>
#include <vector>
#include <map>
#include <tuple>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector A(n,vector<int>(n));
  map<int,vector<tuple<int,int>>> m;
  for(int i = 0; i < n; ++i){
    for(int j = 0; j < n; ++j){
      cin >> A[i][j];
      m[A[i][j]].emplace_back(i,j);
    }
  }
  for(auto& [k,v] : m)
    sort(v.begin(),v.end());
  int q;
  cin >> q;
  while(q--){
    int t;
    cin >> t;
    if(m[t].empty()){
      cout << -1 << '\n';
    }else{
      auto [i,j] = m[t][0];
      cout << i+1 << ' ' << j+1 << '\n';
    }
  }
}
