//main("2");
main(require("fs").readFileSync("/dev/stdin", "utf8"));

/*-- 計算&出力 --*/
function main(raw) {
    // 入力処理
    const args = raw.split("\n");
    const input = args[0].split(' ').map(Number);
    const r = input[0];
    // 計算
    const sq = Math.pow(r,2)*Math.PI;
    const len = 2*r*Math.PI;
    // 表示
    console.log(`${sq.toFixed(5)} ${len.toFixed(5)}`);
}
