template <class T>
void vdeb(const std::vector<T> &da) {
    auto n = da.size();
    for (size_t i = 0; i < n; i++) {
        if (i == n - 1)
            std::cout << da[i];
        else
            std::cout << da[i] << " ";
    }
    std::cout << std::endl;
}
template<class T>
void vdeb(const std::vector<std::vector<T>> &da) {
    auto n = da.size();
    for (size_t i = 0; i < n; i++) {
        std::cout << i << " : ";
        vdeb(da[i]);
    }
    std::cout << std::endl;
}

}

using namespace std;

int solve(int n) {
    vector<int> da(n);
    rep(i,n,0) cin >> da[i];
    sort(all(da));
    int ans = da.back();
    rep(i,n,1) {
        chmin(ans, da[i] - da[i-1]);
    }
    return ans;
}

int main() {
    vector<int> ans;
    while (1)
    {
        int n; cin >> n;
        if(n == 0) break;
        ans.emplace_back(solve(n));
    }
    for(auto &i : ans) cout << i << endl;
}
