import java.util.*;


class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        List<Integer> list = new ArrayList<>();
        
        for(int i=0;i<n;i++){
            list.add(Integer.valueOf(sc.nextInt()));
        }
        
        for(i=n-1;i>=0;i--){
            if(i==0) System.out.printf("%d\n",list.get(i));
            else System.out.printf("%d ",list.get(i));
        }
    }
}
