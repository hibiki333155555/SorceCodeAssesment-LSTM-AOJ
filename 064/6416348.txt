template <typename T1, typename T2>

{
    return a < b && (a = b, true);

constexpr char newl = '\n';

using ll = long long;
using vi = vector<int>;
using vll = vector<ll>;
using vvi = vector<vi>;
using vvll = vector<vll>;
using vp = vector<pair<int, int>>;
using vs = vector<string>;

int mod = 1000000007;

signed main() {
    int n, m; cin >> n >> m;
    ll res = 1;
    while (m > 0) {
        if (m & 1) res = res * n % mod;
        n = n * n % mod;
        m >>= 1;
    }
    cout << res << newl;
}
