using namespace std;
using ll=long long;

int main(){
  int N;
  cin>>N;
  string S;
  cin>>S;
  ll ans=0;
  if(string(N,S[0])==S){
    cout<<N-1<<endl;
  }else{
    for(int i=0;i<N-1;i++){
      bool flag=true;
      for(int j=0;j<min(i+1,N-i-1);j++){
        if(S[i+1+j]!=S[i-j]){
          flag=false;
          break;
        }
      }
      if(flag){
        ans++;
      }
    }
    cout<<ans<<endl;
  }
}
