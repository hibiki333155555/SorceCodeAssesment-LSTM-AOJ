n=int(input())
a=[]
dic={}
b=[]
for i in range(n):
  A=list(map(str,input().split()))
  A[1]=int(A[1])
  if A[1] not in dic:
    dic[A[1]]=[A[0]]
  else:
    dic[A[1]].append(A[0])
  a.append(A)
suzi=[]
for i in dic:
  suzi.append(i)
suzi.sort()
for i in suzi:
  for j in range(len(dic[i])):
    b.append([dic[i][j],i])

if a==b:
  print('Stable')
else:
  print('Not stable')
for i in range(n):
  print(*b[i])
