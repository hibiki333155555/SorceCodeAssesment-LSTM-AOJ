#include <bits/stdc++.h>
#include <complex>
#include <random>
#include <functional>
using namespace std;
using ll = long long;
#define all(A) A.begin(),A.end()
using vll = vector<ll>;
using vvll = vector<vll>;
using vvvll = vector<vvll>;
using vvvvll = vector<vvvll>;
#define rep(i, n) for (long long i = 0; i < (long long)(n); i++)

int main() {
    ll N;
    cin>>N;
    string T;
    cin>>T;
    string S="";
    rep(i,T.size()-1){
        S.push_back(T[i]);
        S.push_back('+');
    }
    S.push_back(T[N-1]);
    int i = 0, j = 0;
    vll R(2*N-1);
while (i < S.size()) {
  while (i-j >= 0 && i+j < S.size() && S[i-j] == S[i+j]) ++j;
  R[i] = j;
  int k = 1;
  while (i-k >= 0 && k+R[i-k] < j) R[i+k] = R[i-k], ++k;
  i += k; j -= k;
}
ll an=0;
//rep(i,N-1)cout<<R[2*i+1]<<endl;
rep(i,N-1){
    if(min(2*(i+1),(N-i-1)*2)==R[2*i+1]){
        an++;
        //cout<<i<<endl;
    }
}
cout<<an<<endl;

}
