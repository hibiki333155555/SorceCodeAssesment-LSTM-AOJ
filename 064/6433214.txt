#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int a[n],w[n],rw=10000,lw=10000;
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    for(int i=0;i<n;i++){
        cin>>w[i];
        if(a[i]) rw=min(rw,w[i]);
        else lw=min(lw,w[i]);
    }
    if(rw==10000||lw==10000){
        cout<<0<<endl;
    }else cout<<rw+lw<<endl;
    return(0);
}
