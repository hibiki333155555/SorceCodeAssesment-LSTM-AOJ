#include <iostream>
#include <vector>

using namespace std;

int f(const vector<vector<pair<int, int>>>& P, int n, int p)
{
	int r = 0;
	for (auto [k, w] : P[n]) if (k != p) r = max(r, f(P, k, n) + w);
	return r;
}

int main()
{
	int n; cin >> n;
	vector<vector<pair<int, int>>> P(n);
	vector<int> A(n, 0);
	for (int i = 1, s, t, w; i < n; ++i) {
		cin >> s >> t >> w;
		P[t].push_back({s, w});
		P[s].push_back({t, w});
	}
	for (int i = 0; i < n; ++i) cout << f(P, i, i) << endl;
	return 0;
}
