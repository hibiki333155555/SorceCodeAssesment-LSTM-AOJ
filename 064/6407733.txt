#include<bits/stdc++.h>
using namespace std;

inline int in() { int x; scanf("%d", &x); return x; }
inline long long inll() { long long x; scanf("%lld", &x); return x; }
inline double indb() { double x; scanf("%lf", &x); return x; }
inline void printline(int x) { printf("%d\n", x); }
inline void printline(long long x) { printf("%lld\n", x); }
inline void printline(double x) { printf("%lf\n", x); }
inline void printline(string str) { cout << str << "\n"; }

struct Edge {
    int to;
    long long cost;
    Edge() {}
    Edge(int to, long long cost) : to(to), cost(cost) {}
};
using Graph = vector<vector<Edge>>;

void dfs(Graph &G, int v, vector<int> &used, vector<int> &res) {
    used[v] = true;
    for ( auto &e : G[v] ) {
        if ( !used[e.to] ) { dfs(G, e.to, used, res); }
    }
    res.emplace_back(v);
}
vector<int> topological_sort(Graph &G) {
    int n = (int)G.size();
    vector<int> res;
    vector<int> used(n, false);
    for ( int v = 0; v < n; ++v ) {
        if ( !used[v] ) { dfs(G, v, used, res); }
    }
    reverse(res.begin(), res.end());
    return res;    
}

int main() {
    int N = in(), M = in();
    Graph G(N);
    for ( int i = 0; i < M; ++i ) {
        int s = in(), t = in();
        G[s].emplace_back(t, 1);
    }

    auto ans = topological_sort(G);
    for ( int i = 0; i < N; ++i ) {
        printline(ans[i]);
    }
    return 0;
}
