#include "iostream"
using namespace std;
#define N 200100
int A[N],B[N/2],C[N/2];
long long sum=0;
void merge(int left,int mid,int right){
    int n1 = mid-left;
    int n2 = right-mid;
    for (int i = 0; i <n1 ; ++i) {
        B[i] = A[left+i];
    }
    for (int i = 0; i < n2; ++i) {
        C[i] = A[mid+i];
    }
    int i=0,j=0,k=left;
    while (i<n1&&j<n2&&left<right){
        if(B[i]<=C[j]){
            A[k++] = B[i++];
        }else{
            A[k++] = C[j++];
            sum += n1-i;
        }
    }
    while(i<n1) A[k++] = B[i++];
    while(j<n2) A[k++] = C[j++];
}
void mergeSort(int left,int right){
    if(right-left>1){
        int mid = (left+right)/2;
        mergeSort(left,mid);
        mergeSort(mid,right);
        merge(left,mid,right);
    }
}
int main(){
    int n;
    cin>>n;
    for (int i = 0; i < n; ++i) {
        cin>>A[i];
    }
    mergeSort(0,n);
    cout<<sum<<endl;
    return 0;
}
