#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define rep(i, n) for(int i = 0; i < (n); i++)
#define coutf(f) cout << fixed << setprecision(f)
#define all(v) (v).begin(), (v).end()
#define pb push_back
using Graph = vector<vector<int>>;

int main(){
    //1.Input
    int n, w;
    cin >> n >> w;
    vector<pair<double, pair<int, int>>> v(n);
    rep(i,n){
        int x, y;
        cin >> x >> y;
        v[i].second.first = x;
        v[i].second.second = y;
        v[i].first = (float)x / (float)y;
    }
    sort(all(v));
    reverse(all(v));
    //2.Process
    int ans = 0;
    rep(i,n){
        if(w >= v[i].second.second){
            ans += v[i].second.first * (w / v[i].second.second);
            w -= (w / v[i].second.second) * v[i].second.second;
        }
    }
    //3.Output
    cout << ans << endl;
}
