#include <iostream>
#include<queue>
#include <iomanip>
#include<math.h>

using namespace std;

/*
 コッホ曲線
 */
void printPos(double x,double y){
    cout << fixed << setprecision(8) << x << " " << y << endl;
}

void kohho(int n,double x0,double y0,double x1,double y1){
    if (n == 0) return;
    double sx = (2.0/3.0) * x0 + (1.0/3.0) * x1;
    double sy = (2.0/3.0) * y0 + (1.0/3.0) * y1;
    kohho(n-1,x0,y0,sx,sy);
    printPos(sx,sy);
    
    double len = sqrt((x0 - sx)*(x0 - sx) + (y0 - sy)*(y0 - sy));
    double ux = cos(60.0 / 360.0 * M_PI * 2) * len + sx;
    double uy = sin(60.0 / 360.0 * M_PI * 2) * len + sy;
    kohho(n-1,sx,sy,ux,uy);
    printPos(ux,uy);
    
    double tx = (1.0/3.0) * x0 + (2.0/3.0) * x1;
    double ty = (1.0/3.0) * y0 + (2.0/3.0) * y1;
    kohho(n-1,ux,uy,tx,ty);
    printPos(tx,ty);
    
    kohho(n-1,tx,ty,x1,y1);
}

int main(){
    int n;
    cin >> n;
    
    printPos(0,0);
    kohho(n,0,0,100,0);
    printPos(100,0);
}
