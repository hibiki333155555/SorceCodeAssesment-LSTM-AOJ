#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main()
{
    char s[1200]; int count[26] = {0};
    gets(s);
    for (int i = 0; i < strlen(s) ; i++)
    {
        count[tolower(s[i]) - 97]++;
    }
    for (int i = 0; i < 26; i++)
    {
        printf("%c : %d\n", i + 97, count[i]);
    }

    return 0;
}
