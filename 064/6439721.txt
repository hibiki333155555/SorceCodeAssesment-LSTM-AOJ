#include<iostream>
using namespace std;

int main(){
    int H,W;
    int i,j;
    while(true){
        cin >> H >> W;
        if(H==0 && W==0){
            break;
        }
        for(i=0;i<H;i++){
            for(j=0;j<W;j++){
                if(j != 1 && j !=W){
                    cout << ".";
                }else{
                                    cout << "#";
                }
            }
            cout << endl;
        }
        cout << endl;
    }
}
