int main() {
    int N, M;
    cin >> N >> M;
    vector<vector<pair<int, ll>>> G(N);
    REP(i, M) {
        int a, b, c;
        cin >> a >> b >> c;
        a--, b--;
        G[a].emplace_back(b, c);
        // G[b].emplace_back(a, c);
    }

    set<pair<ll, int>> seen;
    ll mx = -LLINF, mn = LLINF;
    auto dfs = [&](auto&& self, ll l, int u) -> void {
        if(seen.count({l, u})) return;
        seen.emplace(l, u);
        if(u == N-1) {
            chmax(mx, l);
            chmin(mn, l);
        }
        for(auto [v, c] : G[u]) {
            self(self, lcm(l, c), v);
        }
    };
    dfs(dfs, 1, 0);
    
    if(mx == -LLINF) {
        cout << "IMPOSSIBLE\n";
    } else {
        cout << mn << ' ' << mx << endl;
    }
}
