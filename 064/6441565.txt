#include<bits/stdc++.h>
using namespace std;

#define WHITE 1
#define GRAY 2
#define BLACK 3

/*int stack[100];

int G[100][100];
int findTime[100], finishTime[100];
int color[100];
int ttime = 0, n;

void depthFirstSearch(int i);

void initColor(){
    for(int i = 0; i < n; i++){
        color[i] = WHITE;
    }

    for(int i = 0; i < n; i++){
        if(color[i] == WHITE){
            depthFirstSearch(i);
        }
    }
}

void depthFirstSearch(int i){
    color[i] = GRAY;
    ttime++;
    if(findTime[i] == -1){
        findTime[i] = ttime;
    }
    for(int j = 0; j < n; j++){
        if(G[i][j] == 1 && color[j] == WHITE){
            depthFirstSearch(j);
        }
    }
    color[i] = BLACK;
    ttime++;
    finishTime[i] = ttime;
}

int main(){
    int u, k, v;
    cin >> n;
    for(int i = 0; i < n; i++){
        findTime[i] = finishTime[i] = -1;
        for(int j = 0; j < n; j++){
            G[i][j] = 0;
        }
    }
    for(int i = 0; i < n; i++){
        cin >> u >> k;
        u--;
        for(int j = 0; j < k; j++){
            cin >> v;
            v--;
            G[u][v] = 1;
        }
    }

    initColor();
    for(int i = 0; i < n; i++){
        cout << i + 1 << " " << findTime[i] << " " << finishTime[i] << endl;
    }
}
