#include <stdio.h>

int main(void){
    int a,t;
    
    scanf("%d",&a);
    int b[a-1],c[a-1];
    
    for(int i=1;i<=a-1;i++){
        scanf("%d",b[a-1]);
    }
    for(int i=0;i<a-1/2;i++){
        t=c[i];
        c[i]=b[a-1-i];
        b[a-1-i]=t;
    }
    for(int i=0;i<=a-1;i++){
        printf(" %d",b[i]);
    }
    printf("\n");
    
    
    
    
}
