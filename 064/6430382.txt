#include <bits/stdc++.h>
using namespace std;
void Print(vector<int> v,int n){
	for(int i=0;i<n;i++){
		if(i==n-1){
		cout<<v[i];
		}
		else{
			cout<<v[i]<<" ";
		}
	}
}
void bubblesort(vector<int> &v,int n){
	int cnt=0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(v[j]>v[j+1]){
			swap(v[j],v[j+1]);
			cnt+=1;
			}
		}
	}
	Print(v,n);
	cout<<"\n"<<cnt<<"\n";
	
}


int main(){
	int n;cin>>n;
	vector<int>v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
	}
	bubblesort(v,n);
	return 0;
}
