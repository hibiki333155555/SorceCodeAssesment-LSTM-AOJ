#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int main() {
   int n, x;

   while(true) {
      cin >> n >> x;
      if(n >= x-3) n = x-3;

      int num[n];
      for(int i=0; i<n; i++) {
         num[i] = i + 1;
      }

      int count = 0;
      for(int i=0; i<n; i++) 
      {
         for(int j=1; j<n; j++) 
         {
            if(i!=j && num[i]+num[j] <= x-1) 
            {
               for(int k=0; k<n; k++) {
                  if(i!=k && j!=k)
                  {
                     int sum = num[i] + num[j] + num[k];
                     if(sum == x) count++;
                  }
               }
            }


         }
      }

      cout << count << "\n";
   }
}
