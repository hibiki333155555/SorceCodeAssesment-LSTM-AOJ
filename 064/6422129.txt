#include <bits/stdc++.h>
using namespace std;
int main() {
    while (true) {
        int x;
        cin >> x;
        if (x == 0) {
            break;
        } else {
            int sum = 0;
            while (x > 0) {
                sum += x % 10;
                x /= 10;
            }
            cout << sum << endl;
        }
    }
    return 0;
}
