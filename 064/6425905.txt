#include<stdio.h>
#include<math.h>
#include<iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	for (int i = 1; i <= n; ++i) {
		if (i % 3 == 0 || i % 10 == 3 || floor(i/ 10) == 3 || floor(i / 100) == 3 || floor(i/ 1000) == 3 || floor(i / 10000) == 3) {
			printf(" %d", i);
		}
		
	}
	printf("\n");

	return 0;
}
