#include<iostream>
using namespace std;
#define MAXN 100000
typedef long long ll;
#define fr(n) for(int i = 0; i < n; i++)

int n, k;
ll T[MAXN];

int check(ll p){
	int i = 0;
	for(int j = 0; j < k; j++){
		ll s = 0;
		while(s + T[i] <= p){
			s += T[i];
			i++;
			if(i == n)	return n;
		}
	}
	return i;
}

int solve(){
	ll left = 0;
	ll right = 100000 * 10000;
	ll mid = 0;
	while(right > left + 1){
		mid = (right + left) / 2;
		int v = check(mid);
		if(v >= n)	right = mid;
		else{
			left = mid;
		}
	}
	return right;
}
