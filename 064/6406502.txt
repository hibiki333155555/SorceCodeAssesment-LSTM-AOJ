#include<stdio.h>

int main(){
    int a,b,i=1;
    char op;
    
    while(i){
     scanf("%d%c%d",&a,op,&b);
        switch(op){
            case '+':
             printf("%d\n",a+b);
             break;
            case '-':
             printf("%d\n",a-b);
             break;
            case '/':
             printf("%d\n",a/b);
             break;
            case '*':
             printf("%d\n",a*b);
             break;
            default:
             i=0;
        }
    }
    return 0;
}
