#include <iostream>
#include<cstring>
using namespace std;
const int N=1000;
int main(void){
    // Your code here!
    char T[N],P[N];
    cin>>T>>P;
    int N=strlen(T);
    int M=strlen(P);
 
    for(int i=0;i<=N-M;i++){
        int j;
        for(j=0;j<M;j++){
            if(T[i+j]!=P[j]) break;
        }
        if(j==M) cout<<i<<endl;
    }
}
