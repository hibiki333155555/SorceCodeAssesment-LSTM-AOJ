#include <iostream>

using std::cout; using std::cin;
using std::endl;

const int max = 100;

int main()
{
    int mat1[max][max], mat2[max][max], mat3[max][max] = { { 0 } };

    int n, m, l;
    cin >> n >> m >> l;
    for (int i = 0;i < n;++i)
        for (int j = 0;j < m;++j)
            cin >> mat1[i][j];
    for (int i = 0;i < m;++i)
        for (int j = 0;j < l;++j)
            cin >> mat2[i][j];

    for (int i = 0;i < n;++i)
        for (int j = 0;j < l;++j)
            for (int k = 0;k < m;++k)
                mat3[i][j] += mat1[i][k] * mat2[k][j];

    for (int i = 0;i < n;++i) {
        cout << mat3[i][0];
        for (int j = 1;j < l;++j)
            cout << ' ' << mat3[i][j];
        cout << endl;
    }

    return 0;
}
