#include<iostream>
using namespace std;

int main()
{ 
	int n,b,f,r,v ;
	int a[4][3][10];

	//初期値の代入
	for (int i = 0; i < 4; i++) {
		for (int j = 0; j < 3; j++) {
			for (int k = 0; k < 10; k++) {
				a[i][j][k] = 0;
			}
		}
	}

	cin >> n;

	//入力のループ
	for (int i=0; i < n; i++) {
		cin >> b >> f >> r >> v;
		a[b-1][f-1][r-1] += v;//部屋に入居者情報を追加
	}

	//出力のループ

	for (int i = 0; i < 4; i++) {
		for (int j = 0; j < 3; j++) {
			for (int k = 0; k < 10; k++) {
				cout << " " << a[i][j][k];
			}
			cout << "\n";

			//#を打つための行
			if (j == 2 && i!=3 ) {
				for (int l=0; l < 10; l++) {
					cout << "##";
				}
				cout << "\n";
			}
		}
	}



}
