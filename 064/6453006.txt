#include <bits/stdc++.h>
using namespace std;
const long long MOD = 998244353;
int main(){
  int N, M;
  cin >> N >> M;
  vector<int> A(M);
  for (int i = 0; i < M; i++){
    cin >> A[i];
  }
  vector<vector<long long>> dp(M + 1, vector<long long>(N + 1, 0));
  dp[0][0] = 1;
  for (int i = 0; i < M; i++){
    vector<long long> S(N + 2, 0);
    for (int j = 0; j <= N; j++){
      S[j + 1] = (S[j] + dp[i][j]) % MOD;
    }
    for (int j = 0; j <= N; j++){
      if (j < A[i]){
        dp[i + 1][j] = S[j + 1];
      } else {
        dp[i + 1][j] = (S[j + 1] - S[j - A[i]] + MOD) % MOD;
      }
    }
  }
  long long ans = dp[M][N];
  for (int i = 1; i <= N; i++){
    ans *= i;
    ans %= MOD;
  }
  cout << ans << endl;
}
