#include <stdio.h>

void extend(char **s, int n)
{
    int n1;
    
//printf("[s=%s n=%d]", *s, n);
    
    while (1){
        if (**s == NULL){
            return;
        }
        if (**s == ')'){
            (*s)++;
            return;
        }
        if ('0' <= **s && **s <= '9'){
            n1 = (**s - '0');
            (*s)++;  // skip 0-9
            (*s)++;  // skip '('
            extend(s, n + n1);
//            (*s)++;  // skip ')'
        }
        else {
            while ('a' <= **s && **s <= 'z'){
                printf("%c", (((**s - 'a') + n) % 26) + 'a');
                (*s)++;
            }
        }
    }
}

int main(void)
{
    char S[200001];
    char *s;
    
    scanf("%s", S);
    
    s = S;
    extend(&s, 0);
    
    printf("\n");
    
    return (0);
}
