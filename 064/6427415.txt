#include<bits/stdc++.h>
using namespace std;
int main(){
    int n=1,ans=0,sum=0,z=0;
    vector<int> a;
    while(n){
        ans=0;
        sum=0;
        cin>>n;
        int k[n];
        for(int i=0;i<n;i++){
            cin>>k[i];
        }
        for(int i=0;i<n;i++){
            if(k[i]!=0) ans+=1;
            if(k[i]<2) sum+=1;
        }
        if(sum==n){
            a.push_back(-1);
        }else{
            a.push_back(ans+1);
        }
        z++;
    }
    for(int i=0;i<z;i++){
        if(a[i]==-1){
            cout<<"NA"<<endl;
        }else{
            cout<<a[i]<<endl;
        }
    }
    return 0;
}
