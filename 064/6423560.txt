#include <iostream>

using namespace std;

int main()
{
    int  n = 1;
    int  arr[n];
    int a = 0;
    int b = n-1;
    cin>>n;
    cin>>arr[n];
    for(int c = arr[a];a < b;)
    {
        arr[a] = arr[b];
        arr[b] = c;
        a++;
        b--;
    }
    for(int i = 0; i < n;i++)
    {
        if( i != n - 1 )
        {
            cout<<arr[i]<<" ";
        }
        else if(i = n - 1)
        {
        cout<<arr[n - 1];
        }
    }
    return 0;
}
