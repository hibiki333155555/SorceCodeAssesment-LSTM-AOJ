#include <stdio.h>

int main(void)
{
    int a,b,c;
    do
    {
        scanf("%d %d %d",&a,&b,&c);
    }while(!((0 <= a && a <= 100) && (0 <= b && b <= 100) && (0 <= c && c <= 100)));
    if(a < b && b < c)
    {
        printf("Yes\n");
    }
    else
    {
        printf("No\n");
    }
    return 0;
}
