NM = input().split()
times = int(NM[1])
ball_num = int(NM[0])
#print(times)


d_def = {}
i = 0
a = 1
while a < ball_num + 1: #初始状态所有球在对应的箱子里
    d_def[a] = a
    a = a + 1
#print(d_def)

d = d_def
while i < times:
    XY = input().split()
    X = int(XY[0])  #球号码
    Y = int(XY[1])  #箱子号码
    d[X] = Y
    i = i + 1
    #print(type(X))
    #print(X[0])
#print(d)

x = 1
while x < ball_num + 1:
    print(d.get(x))
    x = x + 1
