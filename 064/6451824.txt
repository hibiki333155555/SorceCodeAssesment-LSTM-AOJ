#include<bits/stdc++.h>
//#include<atcoder/all>

using namespace std;
//using namespace atcoder;
using ll = long long;

int main() {
    int n; cin >> n;
    vector<string> report(n);
    for (int i = 0; i < n; ++i) cin >> report[i];

    int ans = 0, l = report[0].size();
    for (int i = 0; i < n - 1; ++i) {
        for (int j = i + 1; j < n; ++j) {
            int dif = 0;
            for (int k = 0; k < l; ++k) {
                if (report[i][k] != report[j][k]) ++dif;
            }

            if (dif <= 1) ++ans;
        }
    }

    cout << ans << endl;
    return 0;
}
