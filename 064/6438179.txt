N = int(input())
heap = list(map(int, input().split()))

for i, a in enumerate(heap, start=1):  
    print(f'node {i}: key = {a}, ', end='')
    #親が存在するときに出力
    if i//2 != 0:
        print(f'parent key = {heap[i//2 -1 ]}, ', end='')
    #左の子が存在するときに出力
    if 2 * i <= N:
        print(f'left key = {heap[2*i - 1]}, ', end='')
    #右の子
    if 2 * i + 1 <= N:
        print(f'right key = {heap[2*i]}, ', end='')
    print()
