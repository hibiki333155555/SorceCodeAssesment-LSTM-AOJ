#include <bits/stdc++.h>
using namespace std;
struct node{
    int id;
    node* left;
    node* right;
};
void preorder(node* node)
{
    if (node != NULL)
    {
         cout << node->id << " ";
        preorder(node->left);
        preorder(node->right);
    }
}
void inorder(node* node)
{
    if (node != NULL)
    {
        inorder(node->left);
        cout << node->id << " ";
        inorder(node->right);
    }
}
void postorder(node* node)
{
    if (node != NULL)
    {
        postorder(node->left);
        postorder(node->right);
        cout << node->id << " ";
    }
}
int main()
{
    node* pnew;
    node* run;
    int n;
    cin>>n;
    node* a[1005];
    for(int i=0;i<n;i++)
    {
        pnew=new node();
        a[i]=pnew;
    }
    for(int i=0;i<n;i++)
    {
        int x;
        int y;
        int z;
        cin>>x>>y>>z;
        a[i]->id=x;
        if(y==-1)
            a[i]->left=NULL;
        else
            a[i]->left=a[y];
        if(z==-1)
            a[i]->right=NULL;
        else
            a[i]->right=a[z];
    }
    cout<<"Preorder"<<endl<<" ";
    preorder(a[0]);
    cout<<endl;
    cout<<"Inorder"<<endl<<" ";
    inorder(a[0]);
    cout<<endl;
    cout<<"Postorder"<<endl<<" ";
    postorder(a[0]);
    return 0;
}
