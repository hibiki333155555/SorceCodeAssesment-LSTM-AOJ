#include <iostream>

using namespace std;

int main()
{
    int H = 1;
    int W = 1;
    cin>>H>>W;
    while(H != 0 && W != 0)
    {
        for(int i = 0;i < H;i++)
        {
            if(i % 2 == 0)
            {
                for(int x = 0; x < W;x++)
                {
                    if(x % 2 == 0)
                    {
                    cout<<"#";
                    }
                    else
                    {
                        cout<<".";
                    }
                    cout<<endl;
                }
            }
            else if(i % 2 == 1)
            {
                for(int y = 0; y < W;y++)
                {
                    if(y % 2 == 0)
                    {
                        cout<<".";
                    }
                    else
                    {
                        cout<<"#";
                    }
                    cout<<endl;
                }
            }
        }
        cout<<endl;
        cout<<endl;
        cin>>H>>W;
    }

    return 0;
}
