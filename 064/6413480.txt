#include <bits/stdc++.h>

using namespace std;
using ll = long long;

int dfs(const vector<int>& w_list) {
   int max_count = 0;

   for (int i = 1; i < w_list.size(); i++) {
      int w0 = w_list[i - 1];
      int w1 = w_list[i];

      if (abs(w0 - w1) > 1) {
         continue;
      }

      vector<int> child;

      for (int j = 0; j < w_list.size(); j++) {
         if (j == i || j == i - 1) {
            continue;
         }

         child.emplace_back(w_list[j]);
      }

      int count = dfs(child);
      count += 2;

      max_count = max(max_count, count);
   }

   return max_count;
}

int main() {
   cout << fixed << setprecision(15);

   while (true) {
      int n;
      cin >> n;

      if (n == 0) {
         break;
      }

      vector<int> w_list;

      for (int i = 0; i < n; i++) {
         int w;
         cin >> w;

         w_list.emplace_back(w);
      }

      cout << dfs(w_list) << endl;
   }

   return 0;
}
