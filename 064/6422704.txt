#include<bits/stdc++.h>
using namespace std;
typedef long long lint;

int main(){
  int n;
  cin >> n;
  vector<int> a(n);
  for(int i=0;i<n;i++)cin >> a[i];
  
  set<int> db;
  for(int i=0;i<(1<<n);i++){
    bitset<20> bs(i);
    int tmp=0;
    for(int j=0;j<n;j++){
      if(bs & 1<<j){
        tmp += a[j];
      }
    }
    db.insert(tmp);
  }
  
  int q, m;
  cin >> q;
  for(int i=0;i<q;i++){
    cin >> m;
    if(db.find(m) == db.end()){
      cout << "no" << endl;
    }else{
      cout << "yes" << endl;
    }
  }
  return 0;
}
