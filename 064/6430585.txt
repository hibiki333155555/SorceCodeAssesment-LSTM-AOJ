#include <string>
#include <iostream>
using namespace std;

int getcode(string s) {
	if (s.size() == 1) {
		return int(s[0] - 'A');
	}
	return 26 + int(s[0] - 'A') * 26 + int(s[1] - 'A');
}

int main() {
	string SA, SB; int NA, NB;
	cin >> SA >> NA >> SB >> NB;
	cout << (getcode(SB) - getcode(SA) + 1) * (NB - NA + 1) << endl;
	return 0;
}
