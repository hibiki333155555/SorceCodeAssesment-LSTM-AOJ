#include<stdio.h>

int main(void){
    int num,i,counter[26];
    char ch,chcou='a';
    
    for(i = 0;i < 26;i++){
        counter[i] = 0;
    }
    
    while(scanf("%c",&ch) != EOF){
        num = ch - 'a';
        counter[num]++;
    }
    
    for(i = 0;i < 26;i++){
        if(i == 25){
            printf("%c : %d",chcou,counter[i]);
            break;
        }
        printf("%c : %d\n",chcou,counter[i]);
        chcou++;
    }
    
    return 0;
}
