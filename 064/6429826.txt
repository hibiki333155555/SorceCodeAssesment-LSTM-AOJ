#include <bits/stdc++.h>
using namespace std;

/* 選択ソート (0オリジン)*/

int selectionSort(int A[], int N) {
    int i, j, minj, sw = 0;
    for (i = 0; i < N - 1; i++) {
        minj = i;
        for (j = i + 1; j < N; j++) {
            if (A[j] < A[minj])minj = j;
        }
        swap(A[minj], A[j]);
        if (i != minj)sw++;
    }
    return sw;
}

int main() {
    int N, sw;
    cin >> N;
    int A[N];
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    sw = selectionSort(A, N);

    for (int i = 0; i < N; i++) {
        if (i) cout << ' ';
        cout << A[i];
    }
    cout << endl;
    cout << sw << endl;

    return 0;
}
