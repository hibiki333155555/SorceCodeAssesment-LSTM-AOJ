#include <bits/stdc++.h>
using namespace std;
int bubblesort(vector <int> &b)
{
	int n=b.size();
	int cnt=0;
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;j<n-1-i;j++)
		{
			if(b[j]>b[j+1])
			{
				swap(b[j],b[j+1]);
				cnt++;
			}
		}
		
	}
	return cnt;
}
int main() {
int n;
cin>>n;
vector<int>b(n);
for(int i=0;i<n;i++)
{
	cin>>b[i];
}
int c=bubblesort(b);
for(int i=0;i<n;i++)
{
	cout<<b[i];
	if(i!=n-1)
	cout << " ";
}
cout<<endl<<c<<endl;
return 0;
}
