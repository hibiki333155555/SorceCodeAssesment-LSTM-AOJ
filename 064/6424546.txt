#include <iostream>
using namespace std;

int main()
{
   int n;
   string s;
   cin >> n >> s;
   bool I1 = false, O = false, I2 = false;
   for (int i = 0; i < n; i++)
   {
      if (s[i] == 'I' && O == false && I2 == false)
         I1 = true;

     else if (s[i] == 'O' && I1 == true && I2 == false)
         O = true;

      else if (s[i] == 'I' && I1 == true && O == true)
         I2 = true;
   }
   cout << (I2 == true ? "Yes" : "No") << endl;
   return 0;
}
