#include <iostream>
#include <stdio.h>

using namespace std;

int ch_to_int(char ch) {
    if (ch == 'S') {
        return 0;
    } else if (ch == 'H') {
        return 1;
    } else if (ch == 'C') {
        return 2;
    } else if (ch == 'D') {
        return 3;
    }
}

char int_ch_to_char(int int_ch) {
    if (int_ch == 0) {
        return 'S';
    } else if (int_ch == 1) {
        return 'H';
    } else if (int_ch == 2) {
        return 'C';
    } else if (int_ch == 3) {
        return 'D';
    }
}

int main(){

    int trump[4][13];

    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 13; j++) {
            trump[i][j] = false;
        }
    }

    int n;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {

        char ch;
        int int_ch, rank;

        // scanf("%c %d", &ch, &rank);
        cin >> ch >> rank;

        int_ch = ch_to_int(ch);

        trump[int_ch][rank - 1] = true;
    }

    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 13; j++) {
            if (trump[i][j] == false) {
                char c_ch = int_ch_to_char(i);
                cout << c_ch << ' ' << j + 1 << endl;
                // printf("%s %d\n", c_ch, j+1);
            }
        }
    }
}
