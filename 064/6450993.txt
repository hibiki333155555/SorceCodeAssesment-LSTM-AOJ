#include<stdio.h>
int main()   {
    int N, cnt;
    int a[101];
    int temp;
    scanf("%d", &N);
    for (int i=0;i<N;i++) {
        for (int j=N-1;j>i;j--) {
            if (a[j] < a[j-1]) {
                temp=a[j];
                a[j]=a[j-1];
                a[j-1]=temp;
            cnt++;
            }
        }
    }
    for (int i=0;i<N;i++) {
        if(i) printf(" ");
        printf("%d", a[i]);
    }
    printf("/n%d/n", cnt);
     return 0;
}
