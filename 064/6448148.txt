#include <bits/stdc++.h>
using namespace std;

void insertionSort(vector<int> A,int N){
  for(int i =0;i<N;i++){
    int v = A[i];
    int j = i -1;
    while(j >= 0 && A[j] > v){
      A[j+1] = A[j];
      j--;
    }
    A[j+1] = v;
    for(int a=0;a<N;a++){
      if(a != 0)
        cout << " ";
      cout << A[a];
    }
  cout << endl;
  }
  
  return;
}
int main(){
  int N;
  cin >> N;
  vector<int> A(N);
  for(int i =0;i<N;i++)
    cin >> A[i];
  
  insertionSort(A,N);

  return 0;
}
