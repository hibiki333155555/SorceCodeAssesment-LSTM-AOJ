def solution(l,n,q):
    st,end,c,s = 0,0,0,l[0]
    while st<n and end<n:
	    if (s<=q):
	        end+=1
	        if end>=st:
	            c+=end-st
	        if end<n:
	            s+=l[end]
	    else:
	        s-=l[st]
	        st+=1
    return c


n,q=map(int,input().split())
l=list(map(int,input().split()))
ql=list(map(int,input().split()))
for q in ql:
    print(solution(l,n,q))
