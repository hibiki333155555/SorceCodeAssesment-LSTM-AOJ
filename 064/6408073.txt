#include<stdio.h>
#include<stdlib.h>
int n,count = 0;
int *A;

void swap(int *x,int *y){
  int temp;
  temp = *x;
  *x = *y;
  *y = temp;
}

void section(){
  int i,j,minj;

  for(i=0 ;i<n ;i++){
    minj = i;
    for(j=n-1 ;j>i ;j--){
      if(*(A+minj) > *(A+j)) minj = j;
    }
    if(minj != i) count++;
    swap(A+minj,A+j);
  }

  for(i=0 ;i<n ;i++){
    if(i) printf(" ");
    printf("%d",A[i]);
  }

  printf("\n");
  printf("%d\n",count);
     
}

int main(){
  int i;
  scanf("%d",&n);

  A = malloc(sizeof(int)*n);

  for(i=0 ;i<n ;i++) scanf("%d",A+i);

  section();

  return 0;
}
