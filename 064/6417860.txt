require "set"
T = gets.to_i
T.times do
    n, m, v, a, b = gets.split.map(&:to_i)
    set = Set.new
    x = v % m
    occur = false
    [n, m * 2].min.times {
        if set.include?(x)
            occur = true
            break
        end
        set << x
        x = (x * a + b) % m
    }
    puts occur ? "Yes" : "No"
end
