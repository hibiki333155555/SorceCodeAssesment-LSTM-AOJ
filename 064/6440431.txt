#include<iostream>
using namespace std;
#include<algorithm>
int main(){
 int price[INT_MAX],n;
 cin>>n;
 for(int i=0;i<n;i++){cin>>price[i];}
 int maxvalue=INT_MIN;
 int minvalue=price[0];
 for(int i=0;i<n;i++){
  maxvalue=max(maxvalue,price[i]-minvalue);
 minvalue=min(minvalue,price[i]);
}
cout<<maxvalue;

}
