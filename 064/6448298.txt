const int N = 2e6 + 10;
ll a[N];
ll b[N];
int main()
{
    ll n;
    while(cin >> n)
    {
        if(!n) break;
        double t = 1, s = 1;
        ll nn = n*(n+1)/2 - 1;
        for(ll i = 0; i <= nn; i++){
            cin >> b[i];
            s *= pow((double)b[i], 1.0/n);
            if(b[i] % 2 == 0) t *= b[i];
        }
        ll aa = t / s;
        ll cnt = 0;
        aa = pow((double)aa, (double)1/(n-1));
        for(ll i = 0; i <= nn; i++){
            if(b[i] % 2 == 0){
                a[++cnt] = b[i] / aa;
            }
        }
        sort(a+1, a+cnt+1);
        cout << aa << endl;
        for(ll i = 1; i <= cnt; i++){
            if(i > 1) cout << ' ';
            cout << a[i];
        }
        cout << endl;
    }
    return 0;
}
