#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < n; ++i)
typedef long long ll;
using namespace std;

int main() {
    cin.tie(0);
    ios_base::sync_with_stdio(false);

    int P, Q;
    cin >> P >> Q;

    int g = __gcd(P, Q);
    Q /= g;

    int ans = 1;
    for (int i = 2; i * i <= Q; ++i) {
        if (Q % i == 0) {
            ans *= i;

            while (Q % i == 0) {
                Q /= i;
            }
        }
    }
    if (Q != 1) ans *= Q;
    cout << ans << "\n";

    return 0;
}
