#include<iostream>
#include<string>
using namespace std;

int main(){

    string str,str1,str2,ans;
    int num,k;

    while(1){
        cin>>str;
        if(str == "_")break;
        cin>>num;
        for(int i=0;i<num;++i){
            cin>>k;
            str2=str.substr(0,k);
            str1=str.substr(k);
            ans=str2+str1;
        }
        cout<<ans<<endl;
    }

    

    return 0;
}
