#include <bits/stdc++.h>
using namespace std;

int main() {
	int n = 3;
	int a[ 100 ];

	for (int i = 0; i < n; i++) {
		cin >> a[ i ];
	}

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			if (a[ i ] < a[ j ]) {
				int a2 = a[ i ];
				a[ i ] = a[ j ];
				a[ j ] = a2;
			}
		}
	}

	for (int i = 0; i < n; i++) {
		printf("%d ", a[ i ]);
	}
	putchar('\n');

	return (0);
}
