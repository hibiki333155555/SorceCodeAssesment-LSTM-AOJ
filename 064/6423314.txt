#include <bits/stdc++.h>
using namespace std;

int main(){
  int V,E,r;
  cin >> V >> E >> r;
  vector<vector<vector<int>>> node(V);
  for(int i=0; i<E; i++){
    int s,t,d;
    cin >> s >> t >> d;
    node.at(s).push_back({t,d});
  }
  
  vector<long> ans(V,-1);
  priority_queue<long long, vector<long long>, greater<long long>> que;
  que.push(r);
  
  while(!que.empty()){
    long long d = que.top()/V, v = que.top()%V;
    que.pop();
    if(ans.at(v) == -1){
      ans.at(v) = d;
    }else{
      continue;
    }
    for(int i=0; i<node.at(v).size(); i++){
      long long nv = node.at(v).at(i).at(0);
      if(ans.at(nv) != -1) continue;
      long long nd = d + node.at(v).at(i).at(1);
      que.push(nd*V + nv);
    }
  }
  
  for(int i=0; i<V; i++){
    if(ans.at(i) == -1) cout << "INF" << endl;
    else cout << ans.at(i) << endl;
  }
  
  return 0; 
}
