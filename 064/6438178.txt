#define FOR(i, a, b) for (int i = (int)(a); i < (int)(b); ++i)
#define rep(i, n) for (int i = 0; i < (int)(n); ++i)

void is_factor(int a, int b, int N)
{
#ifdef DEBUG
    int c = 0;
    int tmp;
    if (N == 1 || N == 2)
    {
        if (a == 1)
        {
            cout << 1 << endl;
            return;
        }
        else
        {
            cout << 0 << endl;
            return;
        }
    }

    for (int i = 2; i * i <= N; ++i)
    {
        if (N % i == 0)
        {
            if (a <= i && i <= b)
            {
                c++;
            }
            tmp = N / i;
            if (a <= tmp && tmp <= b)
            {
                c++;
            }
        };
    }
    cout << c << endl;
    return;
#else
    int c = 0, tmp = 0;
    vector<int> fact;
    fact.push_back(1);

    for (int i = 2; i * i <= N; ++i)
    {
        if (N % i == 0)
        {
            tmp = N / i;
            if (tmp == i)
            {
                fact.push_back(i);
            }
            else
            {
                fact.push_back(i);
                fact.push_back(tmp);
            }
        }
    }
    if (N != 1)
        fact.push_back(N);

    sort(fact.begin(), fact.end());
    for (vector<int>::iterator itr = fact.begin(); itr != fact.end(); ++itr)
    {
        // cout << *itr << endl;
        if (a <= *itr && *itr <= b)
            c++;
    }
    cout << c << endl;
#endif
}

int main()
{

#ifndef DEBUG
    int a, b, c;
    cin >> a >> b >> c;
    is_factor(a, b, c);

#else

#endif
}
