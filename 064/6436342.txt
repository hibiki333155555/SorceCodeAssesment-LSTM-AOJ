#include <bits/stdc++.h>
using namespace std;

int main() {
  int n; cin >> n;
  vector<int> b(n),f(n),r(n),v(n);
  for(int i = 0;i < n;i++) {
      cin >> b[i] >> f[i] >> r[i] >> v[i];
  }
  int build1[3][10];
  int build2[3][10];
  int build3[3][10];
  int build4[3][10];
  for(int i = 0;i<3;i++) {
      for(int j = 0;j<10;j++) {
          build1[i][j] = 0;
          build2[i][j] = 0;
          build3[i][j] = 0;
          build4[i][j] = 0;
      }
  }
//   vector<vector<int>> build2(3,vector<int>(10));
//   vector<vector<int>> build3(3,vector<int>(10));
//   vector<vector<int>> build4(3,vector<int>(10));

  for(int i = 0;i < n;i++) {
      if(b[i] == 1) {
          if(f[i] == 1) {
              build1[0][r[i]-1] += v[i];
          }else if(f[i] == 2) {
              build1[1][r[i]-1] += v[i];
          }else {
              build1[2][r[i]-1] += v[i];
          }
      }else if(b[i] == 2) {
          if(f[i] == 1) {
              build2[0][r[i]-1] += v[i];
          }else if(f[i] == 2) {
              build2[1][r[i]-1] += v[i];
          }else {
              build2[2][r[i]-1] += v[i];
          }
      }else if(b[i] == 3) {
          if(f[i] == 1) {
              build3[0][r[i]-1] += v[i];
          }else if(f[i] == 2) {
              build3[1][r[i]-1] += v[i];
          }else {
              build3[2][r[i]-1] += v[i];
          }
      }else {
          if(f[i] == 1) {
              build4[0][r[i]-1] += v[i];
          }else if(f[i] == 2) {
              build4[1][r[i]-1] += v[i];
          }else {
              build4[2][r[i]-1] += v[i];
          }
      }
  }
  for(int i = 0;i<3;i++) {
      for(int j = 0;j<10;j++) {
          cout << build1[i][j];
          if(j < 9) cout << " ";
          else cout << endl;
      }
  }
  cout << "####################" << endl;
  for(int i = 0;i<3;i++) {
      for(int j = 0;j<10;j++) {
          cout << build2[i][j];
          if(j < 9) cout << " ";
          else cout << endl;
      }
  }
  cout << "####################" << endl;
  for(int i = 0;i<3;i++) {
      for(int j = 0;j<10;j++) {
          cout << build3[i][j];
          if(j < 9) cout << " ";
          else cout << endl;
      }
  }
   cout << "####################" << endl;
  for(int i = 0;i<3;i++) {
      for(int j = 0;j<10;j++) {
          cout << build4[i][j];
          if(j < 9) cout << " ";
          else cout << endl;
      }
  }
}
