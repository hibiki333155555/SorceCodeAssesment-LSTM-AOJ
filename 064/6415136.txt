template <typename T> inline bool chmax(T &a, T b) {
  return ((a < b) ? (a = b, true) : (false));
}
template <typename T> inline bool chmin(T &a, T b) {
  return ((a > b) ? (a = b, true) : (false));
}
int main(){
  ll n;cin >> n;
  string s;cin >> s;
  vl a(n);rep(i,n)cin >> a[i];
  ll ans=0;
  priority_queue<ll,vl,greater<ll>> lt,rt;
  ll cum=0;
  rep(i,n){
    if(s[i]=='('){
      cum++;
      if(cum==1){
        continue;//とるしかない
      }
      if(cum>n-i-1){
        lt.push(a[i]);
        ans+=lt.top();cum-=2;
        lt.pop();
      }
      else{
        lt.push(a[i]);
      }
    }
    else{
      cum--;
      if(cum==n-i-1){
        continue;//とるしかない
      }
      if(cum<0){
        rt.push(a[i]);
        ans+=rt.top();cum+=2;
        rt.pop();
      }
      else {
        rt.push(a[i]);
      }
    }
  }
  cout << ans << endl;
}
