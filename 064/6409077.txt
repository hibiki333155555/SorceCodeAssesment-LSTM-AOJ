import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        String tarou, hanako;
        int taroupoints, hanakopoints;
        taroupoints = hanakopoints = 0;

        for (int i = 0; i < n; i++) {
            tarou = sc.next();
            hanako = sc.next();

            // 判定
            int judge = tarou.compareTo(hanako);

            // 文字列が同じ
            if (judge == 0) {
                taroupoints++;
                hanakopoints++;
            }
            // tarouの文字列の方が辞書順で後ろになる
            else if (judge > 0) {
                taroupoints += 3;
            }
            // hanakoの文字列の方が辞書順で後ろになる
            else {
                hanakopoints += 3;
            }
        }
        System.out.println(taroupoints + " " + hanakopoints);
    }
}
