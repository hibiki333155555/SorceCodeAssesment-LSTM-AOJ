#include <iostream>
#include <vector>
using namespace std;

int main() {
    string s; getline(cin,s);
    vector<int> c(26,0);
    for(auto t : s) {
        if('a'<=t&&t<='z') c[t-'a']++;
        else if('A'<=t&&t<='Z') c[t-'A']++;
    }
    for(int i=0;i<26;i++) {
        cout<<char('a'+i)<<" : "<<c[i]<<endl;
    }
    return 0;
}
