#include<stdio.h>
int min(int a,int b){
    return a<b?a:b;
}
int main(void){
    int n,a[11],w[11];
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n;i++) scanf("%d",&w[i]);
    int rmin=1005,lmin=1005;
    for(int i=0;i<n;i++){
        if(a[i]){rmin=min(rmin,w[i]);}
        else{lmin=min(lmin,w[i]);};
    }
    if(rmin==1005 || lmin==1005) printf("0\n");//皆利き手同じ
    else{printf("%d\n",rmin+lmin);};
    return 0;
}
