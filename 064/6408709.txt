#include<iostream>
#include<cstdio>
using namespace std;

int main(){
    int *A,*B;
    int C[2000001];
    int n,i;

    printf("%d",&n);
    A=new int[n];
    B=new int[n];

    for(i=0;i<2000001;i++){
        C[i]=0;
    }

    for(i=0;i<n;i++){
        printf("%d",&A[i]);
        C[A[i]]++;
    }

    for(i=1;i<2000001;i++){
        C[i]=C[i]+C[i-1];
    }

    for(i=n-1;i<=n;i++){
        if(i>1)printf(" ");
        printf("%d",B[i]);
    }
    printf("\n");
    return 0;
}
