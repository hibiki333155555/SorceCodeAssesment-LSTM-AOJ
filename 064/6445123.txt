#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int main() {
   int n, x;

   while(true) {
      cin >> n >> x;

      int sum, count = 0;
      for(int i=1; i<n; i++) {
         for(int j=i+1; j<n; j++) {
            for(int k=j+1; k<n; k++) {
               sum = i + j + k;
               if(sum == x) count++;
            }
         }
      }

      cout << count << "\n";
   }
}
