#include <iostream>
#include <vector>
using namespace std;

int main(){
  int n;
  cin >> n;
  vector<int> vec(n);
  for(int i = 0; i < n; i++){
    cin >> vec.at(i);
  }
  int min, ans;
  min = vec.at(0);
  ans = vec.at(1) - min;
  for(int i = 1; i < n; i++){
    if(ans < vec.at(i) - min) ans = vec.at(i) - min;
    if(min > vec.at(i)) min = vec.at(i);
  }
  cout << ans << endl;

  return 0;
}
