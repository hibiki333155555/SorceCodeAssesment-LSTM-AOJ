#include<iostream>
#include<algorithm>
#include<string>
#include<math.h>
#include <map>
#include <vector>
using namespace std;
int main() {
	int N;
	cin >> N;
	long long A[2020];
	for (int i = 1; i <= N; i++)cin >> A[i];
	map<long long, int>Map;
	for (int i = 1; i <= N; i++)Map[A[i]]++;
	int Q;
	cin >> Q;
	long long  B[2020];
	for (int i = 1; i <= Q; i++)cin >> B[i];
	int count = 0;
	for (int i = 1; i <= Q; i++)if (Map[B[i]] > 0)count++;
	cout << count << endl;
	



}
