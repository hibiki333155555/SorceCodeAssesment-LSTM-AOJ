#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll INF = 1e+10;

int main() {
  ll v, e, r;
  cin >> v >> e >> r;
  vector<vector<pair<ll, ll>>> graph(v);
  for(int i = 0; i < e; i++) {
    ll s, t, d;
    cin >> s >> t >> d;
    graph[s].push_back(make_pair(d, t));
  }
  
  priority_queue<pair<ll, ll>, vector<pair<ll, ll>>, greater<pair<ll, ll>>> PQ;
  vector<int> color(v, -1);
  vector<ll> d(v, INF);

  d[r] = 0;
  PQ.push(make_pair(0, r));
  color[r] = 0;
  
  while(PQ.size()) {
    pair<ll, ll> f = PQ.top();
    PQ.pop();
    ll u = f.second;
    color[u] = 1;
    if(d[u] < f.first) continue;
    for(auto g : graph[u]) {
      ll v = g.second;
      if(color[v] == 1) continue;
      if(d[v] <= d[u] + g.first) continue;
      d[v] = d[u] + g.first;
      PQ.push(make_pair(d[v], v));
      color[v] = 0;
    }
  }
  
  for(int i = 0; i < v; i++) cout << d[i] << endl;
}
