#!/usr/bin/env PyPy3

from collections import Counter, defaultdict, deque
import itertools
import re
import math
from functools import reduce
import operator
import bisect
import heapq
import functools
mod=10**9+7

import sys
input=sys.stdin.readline

n=int(input())
s=list(input().rstrip())
a=list(map(int,input().split()))
rheap = []
lheap = []
rc = 0
lc = 0
ans = 0
for i in range(n):
    if s[i] == ')':
        rc += 1
        heapq.heappush(rheap,(a[i],i))
        if rc > lc:
            cost,j = heapq.heappop(rheap)
            s[j] = '('
            ans += cost
            rc -= 1
            lc += 1
    else:
        lc += 1

s = s[::-1]
rheap = []
lheap = []
rc = 0
lc = 0
for i in range(n):
    if s[i] == '(':
        rc += 1
        heapq.heappush(rheap,(a[~i],i))
        if rc > lc:
            cost,j = heapq.heappop(rheap)
            s[j] = ')'
            ans += cost
            rc -= 1
            lc += 1
    else:
        lc += 1
print(ans)
