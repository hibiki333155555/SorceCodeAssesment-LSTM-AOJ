#include <bits/stdc++.h>
using namespace std;
int main(){
    vector<int> s(3), g(3);
    for(int i = 0; i < 3; i++) {
        cin >> s[i];
        s[i] %= 2;
    }
    for(int i = 0; i < 3; i++) {
        cin >> g[i];
        g[i] %= 2;
    }
    bool f = false;
    set<vector<int>> s1, s2, s3;
    s1.insert({0, 0, 0});
    s1.insert({-1, 1, 1});
    s1.insert({1, 1,-1});
    s1.insert({1, -1, 1});
    s2.insert({0, 0, 1});
    s2.insert({1, -1, 0});
    s2.insert({-1, 1, 0});
    s3.insert({1, 0, 0});
    s3.insert({0, 1, -1});
    s3.insert({0, -1, 1});
    for(int bit = 0; bit < (1 << 3); bit++){
        vector<int> temp = s;
        for(int i = 0; i < 3; i++){
            if((bit & (1 << i)) > 0 && temp[i] != 0){
                temp[i] *= -1;
            }
        }
        for(int b = 0; b < (1 << 3); b++){
            vector<int> te = g;
            for(int j = 0; j < 3; j++){
                if((b & (1 << j)) > 0 && te[j] != 0){
                    te[j] *= -1;
                }
            }
            if((s1.count(temp) && s1.count(te)) || (s2.count(temp) && s2.count(te)) || (s3.count(temp) && s3.count(te))) f = true;
        }
    }
    if(f) cout << "Yes" << endl;
    else cout << "No" << endl;
}
