#include <bits/stdc++.h>
#define be(v) (v).begin(),(v).end()
#define pb(q) push_back(q)
#define rep(i, n) for(int i=0;i<n;i++)
#define all(i, v) for(auto& i : v)
typedef long long ll;
using namespace std;
const ll mod=1000000007, INF=(1LL<<60);
#define doublecout(a) cout<<fixed<<setprecision(10)<<a<<endl;

int main() {
    cin.tie(0);
    cout.tie(0);
    ios::sync_with_stdio(false);
    int n;
    cin >> n;
    vector<string> s(n);
    rep(i, n) {
        cin >> s[i];
    }
    int m = s[0].length();
    vector<int> ng(n, 0);
    rep(i, n) rep(j, n) {
        if(i == j) continue;
        int cnt = 0;
        rep(k, m) {
            if(s[i][k] != s[j][k]) cnt++;
        }
        if(cnt <= 1) {
            ng[i] = ng[j] = 1;
            // cout << i << " " << j << endl;
        }
    }
    int ans = 0;
    rep(i, n) ans += ng[i];
    cout << ans << endl;

    return 0;
}
