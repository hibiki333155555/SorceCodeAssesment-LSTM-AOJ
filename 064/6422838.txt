void _insertion_sort(vll& v){
    _out(v);
    mcfor(i, 1, v.size()){
        int picup = v.at(i);
        int j = i - 1;
        while (j >= 0 && v[j] > picup){
            v.at(j+1) = v.at(j);
            j--;
        }
        v.at(j+1) = picup;
        _out(v);
    }
}
