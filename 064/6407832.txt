#include<bits/stdc++.h>
using namespace std;

int main(){
  int N;
  cin >> N;
  map<string,bool> Map;
  for(int i=0;i<N;i++){
    string S;
    cin >> S;
    Map[S] = true;
  }
  map<string,int> cnt;
  int M;
  cin >> M;
  for(int i=0;i<M;i++){
    string S;
    cin >> S;
    if(Map[S]){
      if(cnt[S]%2 == 0) cout << "Opened by ";
      else cout << "Closed by ";
      cnt[S]++;
    }
    else cout << "Unknown ";
    cout << S << endl;
  }
}
