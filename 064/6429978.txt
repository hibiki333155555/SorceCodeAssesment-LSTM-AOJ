#include <stdio.h>
#include <string.h>

int main(void)
{
    int N, K;
    int A, d[7];
    int i, j;
    int s;
    
    scanf("%d%d", &N, &K);
    memset(d, 0, sizeof(d));
    
    for (i = 0; i < N; i++){
        scanf("%d", &A);
        d[A]++;
    }
    
    while (K > 0){
        if (d[1] > 0){
            d[1]--;
            d[5]++;
        }
        else if (d[2] > 0){
            d[2]--;
            d[6]++;
        }
        else if (d[3] > 0){
            d[3]--;
            d[6]++;
        }
        else if (d[4] > 0){
            d[4]--;
            d[6]++;
        }
        else if (d[5] > 0){
            d[5]--;
            d[6]++;
        }
        else {
            break;
        }
        K--;
    }
    
    s = 0;
    for (i = 1; i <= 6; i++){
        s += (i * d[i]);
    }
    
    printf("%d\n", s);
    
    return (0);
}
