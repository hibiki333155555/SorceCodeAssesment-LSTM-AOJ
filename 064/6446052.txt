int n, ind[maxn], otd[maxn], cnt, v[maxn];
int qwq[maxn];
vector<int> e[maxn];

void dfs(int rt) {
    qwq[rt] = 1;
    ind[rt] = ++cnt;
    v[rt] = 1;
    for(auto y : e[rt]) {
        if(v[y]) continue;
        dfs(y);
    }
    otd[rt] = ++cnt;
}

        int u, k; cin>>u>>k;
        while(k--) {
            int v; cin>>v;
            e[u].push_back(v);
        }
    }
    for(int i = 1; i <= n; i++) if(qwq[i]==0) dfs(i);;
    
    for(int i = 1; i <= n; i++) {
        printf("%lld %lld %lld\n",i,ind[i],otd[i]);
    }
