#include <bits/stdc++.h>
using namespace std;

int main(){
    vector<vector<bool> > A(4,vector<bool>(12,false));
    int n;
    cin >> n;
    for(int i = 0;i < n;i++){
        char str;
        int p;
        cin >> str >> p;
        if(str=='S') A[0][p-1] = true;
        if(str=='H') A[1][p-1] = true;
        if(str=='C') A[2][p-1] = true;
        if(str=='D') A[3][p-1] = true;
    }
    for(int i = 0;i < 12;i++){
        if(!A[0][i]) cout << 'S' << ' ' << i+1 << endl;
    }
    for(int i = 0;i < 12;i++){
        if(!A[1][i]) cout << 'H' << ' ' << i+1 << endl;
    }
    for(int i = 0;i < 12;i++){
        if(!A[2][i]) cout << 'C' << ' ' << i+1 << endl;
    }
    for(int i = 0;i < 12;i++){
        if(!A[3][i]) cout << 'D' << ' ' << i+1 << endl;
    }
}
