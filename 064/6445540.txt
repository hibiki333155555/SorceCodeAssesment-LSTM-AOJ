#include <iostream>
#include <vector>
using namespace std;

int H, W;
vector< vector<int> > field;

void dfs(int h,int w){
    field[h][w] = 0;
    
    //8方向を探索
    for(int dh = -1; dh <= 1; dh++){
        for(int dw = -1; dw <= 1; dw++){
            int nh = h + dh, nw = w + dw;

            //場外or0
            if(nh < 0 || nh >= H || nw < 0 || nw >= W){
                continue;
            }
            if(field[nh][nw] == 0){
                continue;
            }

            dfs(nh,nw);
        }
    }

}

int main(){
    
    while(cin >> W >> H){
        if(H == 0 || W == 0){
            break;
        }
        field.assign(H, vector<int>(W, 0));
        for(int h = 0; h < H; h++){
            for(int w = 0; w < W; w++){
                cin >> field[h][w];
            }
        }

        int cnt = 0;
        for(int h = 0; h < H; h++){
            for(int w = 0; w < W; w++){
                if(field[h][w] == 0){
                    continue;
                }
                dfs(h,w);
                cnt++;
            }
        }
        cout << cnt << endl;

    }

}
