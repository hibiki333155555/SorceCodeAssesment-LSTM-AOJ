#include <bits/stdc++.h>
using namespace std;

int main() {
    int M,N;
    cin >> M >> N;
    vector<string>S(M);
    for(int i = 0; i < M; i++) {
        cin >> S[i];
    }
    int ans = M*N;
    for(int i = 1; i < M; i++) {
        int cnt = 0;
        for(int j = 0; j < i; j++) {
            for(int k = 0; k < N; k++) {
                cnt += S[j][k] == 'x';
            }
        }
        for(int j = i; j < M; j++) {
            for(int k = 0; k < N; k++) {
                cnt += S[j][k] == 'o';
            }
        }
        ans = min(ans,cnt);
        ans = min(ans,M*N-cnt);
    }
    for(int i = 1; i < N; i++) {
        int cnt = 0;
        for(int j = 0; j < i; j++) {
            for(int k = 0; k < M; k++) {
                cnt += S[k][j] == 'x';
            }
        }
        for(int j = i; j < N; j++) {
            for(int k = 0; k < M; k++) {
                cnt += S[k][j] == 'o';
            }
        }
        ans = min(ans,cnt);
        ans = min(ans,M*N-cnt);
    }
    cout << ans << endl;
}
