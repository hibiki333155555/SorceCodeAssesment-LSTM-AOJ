#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<map>
using namespace std;

int main(){
    int N;
    cin >> N;
    vector<string> reports(N);
    for(int i = 0; i < N; ++i) cin >> reports[i];

    int reportSize = reports[0].size();

    vector<bool> isCheater(N, false);

    for(int i = 0; i < N; ++i){
        for(int j = 0; j < i; ++j){
            int count = 0;
            for(int k = 0; k < reportSize; ++k){
                if(reports[i][k] != reports[j][k]) ++count;
            }
            if(count <= 1){
                isCheater[i] = true;
                isCheater[j] = true;
            }
        }
    }

    int ans = 0;
    for(int i = 0; i < N; ++i){
        if(isCheater[i]) ++ans;
    }
    cout << ans << endl;
}
