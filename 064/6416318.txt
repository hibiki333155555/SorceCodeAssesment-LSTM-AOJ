import java.util.Arrays;
import java.util.Scanner;

@SuppressWarnings("resource")
public class Main {
    public static void main(String[] args) {
	    Scanner sc = new Scanner(System.in);
        int[] array = new int[3];		
        for (int i = 0; i < array.length; i++) {
            array[i] = sc.nextInt();
        }
        Arrays.sort(array);
        
        for (int i = 0; i < array.length; i++) {
    	    if (i + 1 < array.length) {
    	    System.out.print(array[i] + " ");
		    } else {
    	    System.out.print(array[i]);
		    }
         }
    }
}
