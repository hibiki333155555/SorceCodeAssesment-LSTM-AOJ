#include <iostream>

using namespace std;


int main() {
    int h, w;
    do {
        cin >> h >> w;

        for (int i = 0; i < w; i++)
            cout << "#";
        cout << endl;

        for (int i = 1; i < h - 1; i++) {
            cout << "#";
            for (int j = 1; j < w - 1; j++)
                cout << ".";
            cout << "#" << endl;
        }

        for (int i = 0; i < w; i++)
            cout << "#";
        cout << endl << endl;
    } while (h != 0 && w != 0);


    return 0;
}
