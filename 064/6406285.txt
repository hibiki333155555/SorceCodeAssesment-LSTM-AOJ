#include <stdio.h>

int main(void){
    int n,x,i,j,k;
    int count=0;
    
    while(1){
    scanf("%d %d",&n,&x);
    
    if(n==0 && x==0){
        break;
    }
    
    for(i=0;i<=n;i++){
        for(j=0;j<=n;j++){
            for(k=0;k<=n;k++){
                if(i+j+k==x){
                count++;
                }
            }
        }
    }
    printf("%d\n",count);
    }
}
