#include<stdio.h>

int main(void)
{
    int n;
    int x,y,z;
    long long sum;
    scanf("%d\n",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        if(i==0||y>x)
        {
            y=x;
        }
        if(i==0||z<x)
        {
            z=x;
        }
        sum+=x;
    }
    printf("%d %d %d\n",y,z,sum);
    return 0;
}
