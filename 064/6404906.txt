#include <iostream>
#include<cstdio>
#include <numeric>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    int a[1000] = {};
    while(true) {
        int n;
        cin >> n;
        if (n == 0) {
            break;
        }
        int sum = 0;
        for (int i=0;i<n;n++) {
            scanf("%d", &a[i]);
            sum += a[i];
        }
        double average = (double)sum/(double)n;
        double aru = 0.0;
        for (int i=0;i<n;i++) {
          aru += pow(((double)a[i]-average), 2); 
        }
        cout << fixed << setprecision(7) << sqrt(aru/(double)n) << endl;
    }
    return 1;
}
