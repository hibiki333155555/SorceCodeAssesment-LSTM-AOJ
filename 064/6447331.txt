#include <iostream>
#include <cmath>
using namespace std;

void sol(int a, int b, int c)
{
    int x, y;
    x = pow(a, 2) + pow(b, 2);
    y = pow(c, 2);

    if (x == y)
    {
        cout << "YES"<<endl;
    }
    else
    {
        cout << "NO"<<endl;
    }
}
int main()
{
    int a, b, c;
    int n;
    cin >> n;
    while (n > 0)
    {
        cin >> a >> b >> c;
        sol(a, b, c);
        n--;
    }
    return 0;
}
