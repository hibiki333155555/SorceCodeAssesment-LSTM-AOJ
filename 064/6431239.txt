#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n, i, arr[100], j,count=0;
    cin>>n;
    for(i=0; i<n; i++)
        cin>>arr[i];
    for(i=0; i<(n-1); i++)
    {
        for(j=0; j<(n-i-1); j++)
        {
            if(arr[j]>arr[j+1])
            {
               swap(arr[j],arr[j+1]);
                count++;
                
            }
        }
    }
    for(i=0; i<n; i++)
        cout<<arr[i]<<" ";
    cout<<"\n"<<count;
    return 0;
}
