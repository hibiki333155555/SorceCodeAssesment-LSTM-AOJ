#include<stdio.h>
#include<string.h>
int main(){
	char s[1201],alp[26];
	int i,j;
	int count[26]={0};
	for (i = 0; i < 1200; i++) {
		scanf("%[^\n]s", s);
		getchar();
		for (i = 0; i < 26; i++) {
			alp[i] = 97 + i;
			
			for (j = 0; s[j]; j++) {
				if (alp[i] == s[j] || alp[i] - 32 == s[j]) {
					//printf("%c %c", alp[i], s[j]);
					count[i]++;
				}
			}
			printf("%c : %d\n", alp[i], count[i]);
		}
	}
	return 0;
	//2003008
}
