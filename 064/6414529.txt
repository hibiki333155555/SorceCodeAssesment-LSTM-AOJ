#include<stdio.h>

int main(void)
{
    int i,sum;
    char x[1000];
    while(1)
    {
        scanf("%s",x);
        if(x[0]-'0'==0)break;
        else
        {
            sum=0;
            for(i=0;x[i]!='\0';i++)
            {
                sum+=x[i]-'0';
            }
            printf("%d\n",sum);
        }
    }
    return 0;
}
