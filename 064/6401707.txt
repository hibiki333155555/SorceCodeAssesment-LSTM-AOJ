#include <iostream>

int main()
{
    int S,M,H,m,s;

    std::cin >> S;
    if(S>=0 && S<86400){
        if (S >= 60)
        {
            M = S / 60;
            s = S % 60;

            if (M >= 60)
            {

                H = M / 60;
                m = M % 60;
                std::cout << H << ":" << m << ":" << s << "\n";
            }
            else
            {
                std::cout << 0 << ":" << M << ":" << s << "\n";
            }
        }

        else
        {
            std::cout << 0 << ":" << 0 << ":" << S << "\n";
        }
    }


    return 0;
}
