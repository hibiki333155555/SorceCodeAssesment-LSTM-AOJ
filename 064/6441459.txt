#include<iostream>
#include<vector>
#include <climits>
using namespace std;
int n,k;
vector<int> T;
int check(int p){
    int i=0;
    for(int j=0;j<k;j++){
        int s=0;
        while(s+T[i]<=p){
            s+=T[i];
            i++;
            if(i==n)return n;
        }
    }
    return i;
}
int main(){
    cin>>n>>k;
    for(int i=0;i<n;i++){
        int temp;
        cin>>temp;
        T.push_back(temp);
    }
    int left=0,right=INT_MAX;
    int mid;
    while(left<right){
        mid=left+(right-left)/2;
        int v=check(mid);
        if(v==n){
            cout<<mid<<endl;
            return 0;
        }
        if(v>n)right=mid-1;
        else left=mid+1;
    }
    return 0;
}
