#include <iostream>

using namespace std;
int dp[10000];

int main(){
    int n,q;
    cin >> n;
    
    int A[n];
    
    for (int i = 0; i < n; i++){
        cin >> A[i];
    }
    
    for (int i = 0; i < n - 1;i++){
        for (int j = i + 1; j < n; j++){
            dp[A[i]+A[j]] = 1;
        }
    }
    cin >> q;
    for (int i = 0; i < q; i++){
        int num;
        cin >> num;
        
        if (dp[num]==1) {
            cout << "yes" << endl;
        }else{
            cout << "no" << endl;
        }
    }
}
