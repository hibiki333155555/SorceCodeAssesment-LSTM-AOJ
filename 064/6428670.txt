#include<stdio.h>
#include<math.h>

int main(void)
{
    int n,i;
    double sum1=0,sum2=0,sum3=0,max=0;
    double a[101],b[101];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%lf",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        scanf("%lf",&b[i]);
    }
    max=fabs(a[0]-b[0]);
    for(i=0;i<n;i++)
    {
        sum1+=fabs(a[i]-b[i]);
        sum2+=pow(fabs(a[i]-b[i]),2.0);
        sum3+=pow(fabs(a[i]-b[i]),3.0);
        if(max<fabs(a[i]-b[i]))
        {
            max=fabs(a[i]-b[i]);
        }   
    }
    printf("%lf\n",sum1);
    printf("%lf\n",sqrt(sum2));
    printf("%lf\n",pow(sum3,1.0/3.0));
    printf("%lf\n",max);
    return 0;
}
