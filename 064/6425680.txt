#include <iostream>
#include<stdio.h>
#include<string>

using namespace std;

int main() {
	int n, min, MAX, sum,a[100000];
	min = 1000000;
	MAX = -1000000;
	sum = 0;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> a[i];
		if (a[i] < min) {
			min = a[i];
		}
		if (a[i] > MAX) {
			MAX = a[i];
		}
		sum += a[i];
	}
	cout << min << " " << MAX << " " << sum << endl;
	return 0;
}
