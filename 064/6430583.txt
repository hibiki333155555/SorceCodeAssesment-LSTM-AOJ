#include <iostream>
#include <list>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    // 4つの建物にそれぞれ3つの階、各階に10部屋
    int n, b, f, r, v;
    cin >> n;

    int univercity[4][3][10]{};
    for (int i = 0; i < n; i++)
    {
        cin >> b >> f >> r >> v;
        univercity[b - 1][f - 1][r - 1] += v;

        // 各部屋は0人以上9人以下の範囲
        if (univercity[b - 1][f - 1][r - 1] < 0)
        {
            univercity[b - 1][f - 1][r - 1] = 0;
        }
        else if (univercity[b - 1][f - 1][r - 1] > 9)
        {
            univercity[b - 1][f - 1][r - 1] = 9;
        }
    }

    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            for (int k = 0; k < 10; k++)
            {
                cout << ' ' << univercity[i][j][k];
            }

            cout << endl;
        }
        if (i != 3)
        {
            cout << "####################" << endl;
        }
    }

    return 0;
}
