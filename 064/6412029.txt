#include <stdio.h>
#include <string.h>
int main(){
  int i=0,n=0;
  char W[10],T[1000];

  scanf("%s",&W);
  while(W[i]!='\0'){
        if(W[i]>='A' && W[i]<='Z'){W[i]=W[i]+32;}
        i++;
  }

  while(1){
    scanf("%s",&T);
    if(!strcmp(T,"END_OF_TEXT")) break;
    for(i=0;T[i]!='\0';i++){
        if(T[i]>='A' && T[i]<='Z'){
            T[i]=T[i]+32;
        }
  }
    if(!strcmp(W,T)) n++;
  }

  printf("%d\n",n);
return 0;
}
