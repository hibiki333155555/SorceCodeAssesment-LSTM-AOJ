#include <iostream>
#include <string>
#include <algorithm>
#include <cmath>
using namespace std;
int main( )
{
   int A[3];
   for(int i = 0; i < 3; ++i) cin >> A[i];
   int maxs = A[0];
   for(int i = 2; i > 0; i--){
       for(int j = 0; j < i; j++){
           if(A[j] > A[j+1]){
               maxs = A[j];
               A[j] = A[j+1];
               A[j+1] = maxs;
           }
        }
        
    }
   cout << A[0] << " " << A[1] << " " << A[2] << endl;
    
   
}
