#include <iostream>
using namespace std;


int cnt=0;

void merge(int a[], int left, int mid, int right)
{
    int n1=mid-left;
    int n2=right-mid;
    int L[n1], R[n2];
    for(int i=0; i<n1; i++){
        L[i] = a[left+i];
    }   
    for(int i=0; i<n2; i++){
        R[i] = a[mid+i];
    }
    int i=0, j=0;
    while(i<n1 && j<n2){
        a[left++] = L[i] < R[j] ? L[i++] : R[j++];
        cnt++;
    }
    while(i < n1){
        a[left++] = L[i++];
        cnt++;
    }
    while(j < n2){
        a[left++] = R[j++];
        cnt++;
    }
}

void MergeSort(int a[], int left, int right){
    if(left+1 < right){
        int mid = (left+right)/2;
        MergeSort(a, left,  mid);
        MergeSort(a, mid, right);
        merge(a, left, mid ,right);
    }
}

    int n;
    int a[10000];

        cin >> a[i];

    MergeSort(a, 0, n);
    cout << a[0];
    for(int i=1; i<n; i++){
        cout << ' ' << a[i];
    }
