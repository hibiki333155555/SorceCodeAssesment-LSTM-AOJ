#include <bits/stdc++.h>
using namespace std;
int main(){
    int test;
    cin >> test;
    while(test--){
        int N, M, v, a, b;
        cin >> N >> M >> v >> a >> b;
        v %= M;
        vector<int> cnt(M);
        bool f = false;
        while(N--){
            if(cnt[v] > 0){
                f = true;
                break;
            }
            cnt[v]++;
            v = (a * v + b) % M;
        }
        if(f) cout << "Yes" << '\n';
        else cout << "No" << '\n';
    }
}
