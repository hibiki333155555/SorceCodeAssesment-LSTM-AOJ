#include<stdio.h>

int main(){
    int a, b;
    char op;
    
    for(;op != '?';){
        scanf("%d %c %d", &a, &op, &b);
        if(op == '+')printf("%d\n",a+b);
        if(op == '-')printf("%d\n",a-b);
        if(op == '*')printf("%d\n",a*b);
        if(op == '/')printf("%d\n",a/b);
    }
    return 0;
}
