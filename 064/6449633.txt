#include <bits/stdc++.h>
using namespace std;
int main(void){
    string c1,c2;
    int r1,r2;
    cin>>c1>>r1>>c2>>r2;
    
    vector<string> f;
    
    for(char x='A';x<='Z';x++){
        string X="";
        X=x;
        f.emplace_back(X);
    }
    for(char x='A';x<='Z';x++){
        for(char y='A';y<='Z';y++){
            string X="";
            X=x;
            X+=y;
            f.emplace_back(X);
        }
    }
    
    int C1,C2;
    for(int i=0;i<f.size();i++){
        if(f[i]==c1) C1=i;
        if(f[i]==c2) C2=i;
    }
    
    cout<<(r2-r1+1)*(C2-C1+1)<<endl;
    
    
    return 0;
}
