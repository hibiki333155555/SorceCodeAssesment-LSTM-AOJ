void solve(){

        ll n,s;cin>>n>>s;
        vector<vector<ll> >a(n,vector<ll>(4));
        rep(i,n){

        }
        vector<vector<ll> >c(4);
        rep(i,n){

            rep(j,4){

            c[cnt].pb(i);
        }
        if(c[3].size()>0){
            if(c[3].size()==1){
                ll cnt=0;
                rep(i,n){
                    rep(j,4){
                        if(a[i][j]==2)cnt++;
                        else if(a[i][j]==3)cnt--;
                    }
                }
                if(cnt>0){
                    cout<<1<<' '<<1<<' '<<1<<' '<<3<<endl;
                }else{
                    cout<<1<<' '<<1<<' '<<1<<' '<<2<<endl;
                }
            }else{
                ll cnt=0;
                rep(i,n){
                    rep(j,4){
                        if(a[i][j]==2)cnt++;
                        else if(a[i][j]==3)cnt--;
                    }
                }
                if(cnt==0){
                    cout<<1<<' '<<2<<' '<<1<<' '<<3<<endl;
                }else if(cnt==2){
                    cout<<1<<' '<<3<<' '<<1<<' '<<3<<endl;
                }else if(cnt==-2){
                    cout<<1<<' '<<2<<' '<<1<<' '<<2<<endl;
                }
            }
        }else{
            if(c[2].size()==3){
                ll cnt=0;
                rep(i,n){
                    rep(j,4){
                        if(a[i][j]==2)cnt++;
                        else if(a[i][j]==3)cnt--;
                    }
                }
                if(cnt==0){
                    cout<<1<<' '<<1<<' '<<2<<' '<<3<<endl;
                }else if(cnt==2){
                    cout<<1<<' '<<1<<' '<<3<<' '<<3<<endl;
                }else if(cnt==-2){
                    cout<<1<<' '<<1<<' '<<2<<' '<<2<<endl;
                }
            }else if(c[1].size()%2==1){
                ll cnt=0;
                rep(i,n){
                    rep(j,4){
                        if(a[i][j]==2)cnt++;
                        else if(a[i][j]==3)cnt--;
                    }
                }
                ll cnt2=0;
                REP(i,1,3){
                    rep(j,c[i].size()){
                        rep(k,4){
                            if(i==1&&a[c[i][j]][(k+2)%4]==1)continue;
                            if(a[c[i][j]][k]==2)cnt2++;
                            else if(a[c[i][j]][k]==3)cnt2--;
                        }
                    }
                }
                if(abs(cnt)==3){
                    if(cnt==-3){
                        cout<<1<<' '<<2<<' '<<2<<' '<<2<<endl;
                    }else if(cnt==3){
                        cout<<1<<' '<<3<<' '<<3<<' '<<3<<endl;
                    }
                }else if(abs(cnt)==1){
                    if(cnt==-1){
                        if(cnt2==0){
                            cout<<1<<' '<<2<<' '<<2<<' '<<3<<endl;
                        }else if(cnt2==-2){
                            cout<<1<<' '<<2<<' '<<3<<' '<<2<<endl;
                        }

                            cout<<1<<' '<<3<<' '<<3<<' '<<2<<endl;

                            cout<<1<<' '<<3<<' '<<2<<' '<<3<<endl;

                        else if(a[i][j]==3)cnt--;

                        cout<<3<<' '<<3<<' '<<3<<' '<<3<<endl;

                    }else{
                        cout<<2<<' '<<3<<' '<<3<<' '<<3<<endl;
                    }

                    }else{
                        cout<<2<<' '<<2<<' '<<3<<' '<<3<<endl;
                    }
                }
            }

}
