#include <bits/stdc++.h>
using namespace std;

int main() {
    int a,b;
    cin >> a >> b;
    if(a == 0 && b == 0) {
        cout << 1 << endl;
        return 0;
    }
    int cnt = -1;
    for(int i = 30; i >= 0; i--) {
        if((1 & (a >> i)) || (1 & (b >> i))) {
            if(cnt == -1) {
                cnt = 0;
            }
        }
        if(cnt != -1) {
            if((1 & (a >> i)) == (1 & (b >> i))) {
                cnt++;
            }
            else {
                break;
            }
        }
    }
    cout << cnt << endl;
}
