#include<iostream>
using namespace std;
int main(){
  int n, x;
  int count=0;
  
  while(true){
    cin >> n >> x;
    
    //ループを抜ける条件
    if (n == 0 && x == 0){
      break;
    }
    
    //加工
    int sum=0;
    
    for (int i=1; i<=n; i++){
      for (int j=i+1; j<=n; j++){
        for (int k=j+1; k<=n; k++){
          sum = i + j + k;
          
          if (sum == x){
            count++;
          }
          
        }
      }
    }
  }
  
  cout << count << endl;
  
}
