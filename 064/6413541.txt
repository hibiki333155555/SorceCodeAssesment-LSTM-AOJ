import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
    	/*
    	 * 秒⇒時：分：秒　変換
    	 */
        Scanner sc = new Scanner(System.in);
        int input = sc.nextInt();
        int h = input / 3600;
        int m = (input % 3600) / 60;
        int s = input % 60;
        System.out.println(h + ":" + m + ":" + s); 
    }
}
