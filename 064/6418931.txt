import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        while (true){

            int x = scanner.nextInt();
            if (x == 0){
                break;
            }

            int sum = 0;

            while (x > 0){
                sum += x % 10;
                x /= 10;
            }

            System.out.println(sum);
        }
    }
}
