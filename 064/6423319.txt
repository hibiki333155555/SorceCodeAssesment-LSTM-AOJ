#include <stdio.h>

int main(void){
    int c,ctr;
    
    while(scanf("%d",&c)!=EOF){
        if(c==0) break;
        ctr=0;
        while(1){
            if(c==1) break;
            else if(c%2==0) c=c/2;
            else if(c%2==1) c=c*3+1;
            ctr++;
        }
        printf("%d\n",ctr);
    }
    return 0;
}
