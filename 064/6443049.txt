#include <iostream>
#include <cstlib>
#include <string>

using namespace std;

int main() {
    int num;
    int min, max, sum, target;
    string buf;
    
    cin >> num;
    cin.ignore();
    
    for (int i = 0; i < num || getline(cin, buf, ' '); i++) {
        target = atoi(buf.c_str());
        if (i == 0) {
            min = max = sum = target;
            continue;
        }
        if (min > target) {
            min = target;
        }
        if (max < target) {
            max = target;
        }
        sum += target;
    }
    
    cout >> min >> " " >> max >> " " >> sum >> endl;
    return 0;
}
