#include <bits/stdc++.h>
using namespace std;
#define bokusunny ios::sync_with_stdio(false), cin.tie(nullptr);

void solve(int N) {
  vector<int> D(N), V(N);
  for (int i = 0; i < N; i++) {
    cin >> D[i] >> V[i];
    auto g = gcd(D[i], V[i]);
    D[i] /= g;
    V[i] /= g;
  }

  long long l = 1;
  for (auto &d : D) l = lcm(l, d);
  int g = 0;
  for (auto &v : V) g = gcd(g, v);

  for (int i = 0; i < N; i++) {
    int v = V[i] / g;
    long long d = l / D[i];
    cout << (d * v) << endl;
  }
}

int main() {
  bokusunny;
  int N;
  while (cin >> N) {
    if (N == 0) break;
    solve(N);
  }

  return 0;
}
