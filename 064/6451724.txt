#include <iostream>
using namespace std;

int main() {
    int a,b,c; cin>>a>>b>>c;
    if(a>b) {
        int tmp = b;
        b = a;
        a = tmp;
    }
    if(b>c) {
        int tmp = c;
        c = b;
        b = tmp;
    }
    if(a>b) {
        int tmp = b;
        b = a;
        a = tmp;
    }
    cout<<a<<" "<<b<<" "<<c<<endl;
    return 0;
}
