#include<iostream>
using namespace std;
const int mod=998244353;
int N,M;
int dp[2][2001];
int main()
{
	cin>>N>>M;
	int now=0;
	dp[now][0]=1;
	for(int i=0;i<M;i++)
	{
		int A;cin>>A;
		int nxt=1-now;
		for(int j=0;j<=N;j++)dp[nxt][j]=0;
		int sum=0;
		for(int j=0;j<=N;j++)
		{
			sum=(sum+dp[now][j])%mod;
			dp[nxt][j]=sum;
			if(j>=A)sum=(sum-dp[now][j-A]+mod)%mod;
		}
		now=nxt;
	}
	int ans=dp[now][N];
	for(int i=1;i<=N;i++)ans=(long)ans*i%mod;
	cout<<ans<<endl;
}
