#include <iostream>
#include <cmath>

using namespace std;

bool isPrime(int x)
{
  for (int i = 2; i <= int(sqrt(x)); i++)
  {
    if (x % i == 0) return false;
  }
  
  return true;
}

int main(int argc, char const *argv[])
{
  int n;

  cin >> n;

  int count = 0;
  for (int i = 0; i < n; i++)
  {
    int x;
    cin >> x;
    if (isPrime(x)) count++;
  }

  cout << count << endl;

  return 0;
}
