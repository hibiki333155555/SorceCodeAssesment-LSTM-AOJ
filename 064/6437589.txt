#include <iostream>
#include <iomanip>

int main() {
  double x1, y1, x2, y2;
  std::cin >> x1 >> y1 >> x2 >> y2;
  double denom = (x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1);
  int q; std::cin >> q;
  for (int i=0; i<q; i++) {
    double x, y; std::cin >> x >> y;
    double numer_x = 0.0, numer_y = 0.0;
    numer_x += x * (x2 - x1) * (x2 - x1);
    numer_x += (y - y1) * (y2 - y1) * (x2 - x1);
    numer_x += x1 * (y2 - y1) * (y2 - y1);
    numer_y += y * (y2 - y1) * (y2 - y1);
    numer_y += (x - x1) * (x2 - x1) * (y2 - y1);
    numer_y += y1 * (x2 - x1) * (x2 - x1);
    std::cout << std::fixed << std::setprecision(9);
    std::cout << numer_x / denom << " " << numer_y / denom << std::endl;
  }
}
