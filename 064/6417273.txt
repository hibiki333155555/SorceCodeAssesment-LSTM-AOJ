#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int top,S[1000];
void push(int x)
{
    S[++top]=x;
}
int pop()
{
    top--;
    return S[top+1];
}
int main()
{
    int a,b;
    top=0;
    char s[100];
    char *q;
    q=&s[0];
    cin>>q;
    for(int i=0;i<strlen(s);i++)
    {
        if(48<=s[i]<=57)
        {
            push(s[i]-48);
        }
        else if(s[i]=='+')
        {
            a=pop();
            b=pop();
            push(a+b);
        }
        else if(s[i]=='-')
        {
            a=pop();
            b=pop();
            push(a-b);
        }
        else if(s[i]=='*')
        {
            a=pop();
            b=pop();
            push(a*b);
        }
        
    }
    printf("%d\n",pop());
    return 0;
}
