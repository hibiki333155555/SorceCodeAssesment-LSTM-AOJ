#include<iostream>
using namespace std;

int main()
{
	int n;
	int min=1000000;
	int max=-1000000;
	long long sum=0;
	int a;

	cin >> n;

	for (int i=0; i < n; i++) {
		cin >> a;
		if (min > a) {
			min = a;
		}
		else if (max < a) {
			max = a;
		}
		sum = sum + a;
	}
	cout << min << " " << max << " " << sum << "\n";

}
