using std::cin;

using std::map;

#define rep(i, a, n) for (int i = (a); i < (n); ++i)
#define bit(n, k) ((n >> k) & 1)

class Program {
private:

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

public:

    Program() {
    }

    void solve() {
        int n; cin >> n;
        map<int, int> m;
        rep(i, 0, n) {
            int a; cin >> a;
            if (m.count(a) == 1) {
                m[a]++;
            } else {
                m[a] = 1;
            }
        }

        int ans = 0;
        int frequency = 10000;
        for (auto [k, v] : m) {
            // cout << k << "," << v << endl;
            if (min(frequency, v)) {
                ans = k;
            }
        }

        cout << ans << endl;
    }
};

int main() {

#ifdef DEV
    std::ifstream in("input");
    cin.rdbuf(in.rdbuf());

    int t; cin >> t;
    for (int x = 1; x <= t; ++x) {
        Program p;
        p.solve();
    }
#else
    Program p;
    p.solve();
#endif

    return 0;
}
