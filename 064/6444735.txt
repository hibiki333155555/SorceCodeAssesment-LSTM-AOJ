#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    int N;
    cin >> N;
    int A[N];
    int max=0;
    for(int i=0; i<N; i++){
        cin >> A[i];
        if(max<=A[i]){
            max=A[i];
        }
    }
    int sum=0, sum2=0;
    for(int i=0; i<N; i++){
        if(A[i]==max){
            break;
        }
        sum+=A[i];
    }
    for(int i=N-1; i>=0; i--){
        if(A[i]==max){
            break;
        }
        sum2+=A[i];
    }
    cout << sum << endl << sum2 << endl;
}
