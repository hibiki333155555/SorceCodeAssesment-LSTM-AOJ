int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    vector<int> zy;
    int n, m;
    cin >> n >> m;
    map<i64, i64> mp;
    map<int, int> add;
    rep(i, n){
        int a, b, c;
        cin >> a >> b >> c;
        mp[b] += a;
        mp[c + 1] -= a;
        chmax(add[b], c + 1);
    }
    int q;
    cin >> q;
    map<int, int> ene;
    rep(i, q){
        int d, e;
        cin >> d >> e;
        mp[e] += 0;
        ene[e] += d;
    }
    i64 now = 0;
    set<int> s;
    i64 ans = 0;
    for (auto [i, j]: mp){
        s.erase(i);
        if (add.count(i)) s.emplace(add[i]);
        now += j;
        assert(now >= 0);
        if (now < ene[i]){
            auto k = ene[i] - now;
            if (s.empty()){
                cout << -1 << endl;
                return 0;
            }
            mp[*s.rbegin()] -= k;
            ans += k;
            now = ene[i];
        }
    }
    cout << ans << endl;
}
