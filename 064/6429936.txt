#include<iostream>
using namespace std;
int main(){
    int w,h;
    char c;
    cin>>w>>h>>c;
    for(int i = 0; i < h; i++) {
        for(int j = 0; j < w; j++) {
            if(i == 0 && j == 0 || i == 0 && j == w - 1 || i == h - 1 && j == 0 || i == h - 1 && j == w - 1) {
                cout<<"+";
            } else if (i == 0 || i == h - 1) {
                cout<<"-";
            } else if (j == 0 || j == w - 1) {
                cout<<"|";
            } else if (i == h / 2 && j == w / 2) {
                cout<<c;
            } else {
                cout<<".";
            }
        }
        cout<<endl;
    }
    return(0);
}
