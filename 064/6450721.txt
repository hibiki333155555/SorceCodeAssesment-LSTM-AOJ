#include<bits/stdc++.h>
#include <vector> 
using namespace std;

void printingA(vector<int> A, int n) {
    for (int k=0; k<n ; k++){
        if (k==n-1) {
            cout << A.at(k) << endl; 
        }
        else {
            cout << A.at(k) << " "; 
        }
    }
}

void printingCount(int count) {
    cout << count << endl;
}


void selectionSort(vector<int>& A, int n, int& count) { 
    for (int i=0; i<n; i++) {
        bool change_flag = false;
        int minj = A.at(i);
        for (int j=i+1; j<n; j++) {
            if (A.at(j) < A.at(minj)) {
                minj = j;
                change_flag = true; 
            }
        }
        
        if (change_flag==true) {
            count++;
            int tmp = A.at(i);
            A.at(i) = A.at(minj);
            A.at(minj) = tmp;
        }
    } 
}

int main () {
    int n;
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
        cin >> A.at(i);
    }
    int count = 0;
    selectionSort(A, n, count);
    printingA(A, n);
    printingCount(count);
    return 0;
}
