using System;
class program
{
    static void Main()
    {
        var n = int.Parse(Console.ReadLine());
        int line = Console.ReadLine().Split();
        for(int i = n - 1; i >= 0; i--)
        {
            var t = int.Parse(line[i]);
            if(i == 0) Console.WriteLine(t);
            else Console.Write("{0} ",t);
        }
    }
}
