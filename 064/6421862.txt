import java.util.Scanner;

public class Main{
            public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int h = 0;
        int m = 0;
        int s  = sc.nextInt();
        if (s >= 60){
            m = s / 60;
            s = s % 60; 
            if (m >= 60){
                h = m / 60;
                m = m % 60;
            }
        }
        system.out.println(h + ":" + m + ":" + s);
    }
}
