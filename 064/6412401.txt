#include <iostream>
using namespace std;

int main()
{
    int n, m;

    while ( 1 )
    {
        cin >> n >> m;
        if ( !n && !m ) break;

        int k[1000] = { 0 };
        int score;
        int max_score = 0;
        for ( int i = 0; i < m; i++ )
        {
            for ( int j = 0; j < n; j++ )
            {
                cin >> score;
                k[j] += score;
                if ( k[j] > max_score ) max_score = k[j];
            }
        }

        cout << max_score << endl;
    }

    return 0;
}
