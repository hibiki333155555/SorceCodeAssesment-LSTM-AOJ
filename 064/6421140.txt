#include <bits/stdc++.h>
using namespace std;
using ll=long long;
char f(char s,ll k){
  k%=26;
  s-='a';
  s+=k;
  s%=26;
  s+='a';
  return char(s);
}
int main(){
  string s;
  cin>>s;
  vector<ll> v(s.size(),0);
  ll now=-1;
  ll count=0;
  for(int i=0;i<s.size();i++){
    char k=s[i];
    if('0'<=k&&k<='9'){
      now++;
      v[now]=k-'0';
    }
    if(k==')'){
      //now--;
      count-=v[now];
      now--;
    }
    if(k=='('){
      //now++;
      count+=v[now];
    }
    if('a'<=k&&k<='z'){
      cout<<f(k,count);
    }
    //cerr<<now<<" ";
  }
  cout<<endl;
}
