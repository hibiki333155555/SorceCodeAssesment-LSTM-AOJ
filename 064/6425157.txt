#include<bits/stdc++.h>
using namespace std;
#define endl "\n"
#define ll long long
#define fix(ans,x) fixed << setprecision(x)<<ans
int main(){
    int a,b;
    int j=0;
    cin>>a>>b;
    vector<int> ans(a,0);
    for(int i=0;i<b;i++){
        ans[j]++;
        j++;
        if(j==a){
            j=0;
        }
    }
    sort(ans.begin(),ans.end());
    cout<<ans[a-1]<<endl;
    return 0;
}
