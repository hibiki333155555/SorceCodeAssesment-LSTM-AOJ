let input = require(`fs`).readFileSync(`/dev/stdin`, `utf-8`).split(`\n`)
let [n, arrStr] = input;
let arr = arrStr.split(` `);

console.log(arr.join(` `));

for (let i=1; i<+n; i++) {
    let v = +arr[i];
    let j = i - 1;
    while (j >= 0 && +arr[j] > v) {
        arr[j + 1] = arr[j];
        j--;
    }
    arr[j + 1] = v;
    console.log(arr.join(` `));
}
