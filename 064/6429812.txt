n = int(input())
A = list(map(int, input().split()))
q = int(input())
M = list(map(int, input().split()))
idx = {m: set() for m in M}
for i, m in enumerate(M):
    idx[m].add(i)
ans = [False]*q 

for i in range(1 << n):
    total = 0
    for j in range(n):
        if i & (1 << j):
            total += A[j]
    if total in idx:
        for k in idx[total]:
            ans[k] = True 

for a in ans:
    if a:
        print('yes')
    else:
        print('no')
