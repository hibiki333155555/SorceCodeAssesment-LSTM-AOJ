#include <iostream>
using namespace std;
#define X 2000
#define Y 2000
int count[X][Y];

int main() {
    int N = 2;
    int x,y,w,h;
    for (int i = 0; i < N; i++) {
        cin >> x >> y >> w >> h;
        for (int j = 0; j < w; j++) {
            for (int k = 0; k < h; k++) {
                count[x+j][y+k]++;
            }
        }
    }

    int answer = 0;
    for (int i = 0; i < X; i++) {
        for (int j = 0; j < Y; j++) {
            if (count[i][j] == 1) {
                answer++;
            }
        }
    }
    cout << answer << endl;
    return 0;
}
