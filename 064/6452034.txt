#include <stdio.h>

int main(){
    int n,i,j,t,min,a[101],s,cnt=0,m;
    
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    
    for(i=0;i<n;i++){
        s=i;
        for(j=i+1;j<n;j++){
            if(a[j]<a[s])s=j;
        }
        if(a[i]>a[s]){
                t=a[i];
                a[i]=a[s];
                a[s]=t;
                cnt++;
                printf("%d",a[s]);
            }
    }
    
    for(i=0;i<n;i++){
        if(i!=0)printf(" ");
        printf("%d",a[i]);
    }
    printf("\n");
    printf("%d",cnt);
}
