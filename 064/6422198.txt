#include <stdio.h>

int main(void){

    int n;
    scanf("%d", &n);

    int cards[4][13];
    int i, j;
    for(i = 0; i < 4; i++) for(j = 0; j < 13; j++) cards[i][j] = 0; // initialize

    char sym;
    int num;
    i = n;
    while(i--){
        scanf("%c %d", &sym, &num);
        if(sym == 'S') cards[0][num - 1] = 1;
        if(sym == 'H') cards[1][num - 1] = 1;
        if(sym == 'C') cards[2][num - 1] = 1;
        if(sym == 'D') cards[3][num - 1] = 1;
    }

    char syms[4] = {'S', 'H', 'C', 'D'};
    for(i = 0; j < 4; j++){
        for(j = 0; j < 13; j++) if(!cards[i][j]) printf("%c %d\n", syms[i], j);
    }



    return 0;
}
