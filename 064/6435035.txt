#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int dp[1000001];
int coins[] = {1, 5, 10, 25};

int main() {

    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    for (int j = 0; j <= n; ++j) {
        dp[j] = 1e9;
    }
    dp[0] = 0;

    for (int m = 1; m <= n; ++m) {
        int ans = 1e9;
        for (int i = 0; i < 4; ++i) {
            if (m - coins[i] >= 0)
                ans = min(ans, dp[m - coins[i]] + 1);
        }
        dp[m] = ans;
    }
    cout << dp[n] << endl;


    return 0;
}
