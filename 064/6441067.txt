#include <iostream>

using namespace std;

int binarySearch(int target, int *nums, int size)
{
  int left = 0;
  int right = size;

  while (left < right)
  {
    int mid = (left + right) / 2;
    if (target == nums[mid]) return mid;
    else if (target < nums[mid]) right = mid;
    else left = mid + 1;
  }
  
  return -1;
}

int main(int argc, char const *argv[])
{
  int n;
  cin >> n;
  int *S = new int[n];
  for (int i = 0; i < n; i++)
  {
    cin >> S[i];
  }
  
  int q;
  cin >> q;
  int count = 0;
  for (int i = 0; i < q; i++)
  {
    int num;
    cin >> num;
    if (binarySearch(num, S, n) >= 0) count++;
  }

  cout << count << endl;

  return 0;
}
