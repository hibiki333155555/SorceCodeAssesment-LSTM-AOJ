#include <bits/stdc++.h>
using namespace std;
int main(){
  int N, M, K, X;
  cin >> N >> M >> K >> X;
  vector<int> A(N);
  for (int i = 0; i < N; i++){
    cin >> A[i];
  }
  int mx = 0;
  for (int i = 0; i < N; i++){
    mx = max(mx, A[i]);
  }
  if ((long long) mx * K >= (long long) M * X){
    cout << "Yes" << endl;
  } else {
    cout << "No" << endl;
  }
}
