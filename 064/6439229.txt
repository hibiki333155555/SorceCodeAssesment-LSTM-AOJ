class UnionFind():
    # 初期化
    def __init__(self, n):
        self.par = [-1] * n # 各要素の親要素(自信が根であるばあいは-1)
        self.rank = [0] * n # 各要素の属する根付き木の高さ
        self.siz = [1] * n # 各要素の属する根付き木に含まれる頂点の数

    # 根を求める
    def root(self, x):
        if self.par[x] == -1: return x # x が根の場合は x を返す
        else:
          self.par[x] = self.root(self.par[x]) # 経路圧縮
          return self.par[x]

    # x と y が同じグループに属するか (根が一致するか)
    def issame(self, x, y):
        return self.root(x) == self.root(y)

    # x を含むグループと y を含むグループを併合する
    def unite(self, x, y):
        # x 側と y 側の根を取得する
        rx = self.root(x)
        ry = self.root(y)
        if rx == ry: return False # すでに同じグループのときは何もしない
        # union by rank
        if self.rank[rx] < self.rank[ry]: # ry 側の rank が小さくなるようにする
            rx, ry = ry, rx
        self.par[ry] = rx # ry を rx の子とする
        if self.rank[rx] == self.rank[ry]: # rx 側の rank を調整する
            self.rank[rx] += 1
        self.siz[rx] += self.siz[ry] # rx 側の siz を調整する

        return True
    
    # x を含む根付き木のサイズを求める
    def size(self, x):
        return self.siz[self.root(x)]


V, E = map(int, input().split())
STW = [list(map(int, input().split())) for _ in range(E)]

STW = sorted(STW, reverse=False, key=lambda x: x[2])
union_find = UnionFind(V)
ans = 0
for i in range(E):
    s, t, w = STW[i]
    if union_find.issame(s, t):
        continue
    union_find.unite(s, t)
    ans += w
print(ans)
