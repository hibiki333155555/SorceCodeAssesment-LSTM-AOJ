#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;
const int INF = 1e9;
const ll LINF = 1e18;
const ld pi = 3.141592653589793238;
int main(){
    int H,W;
    while(1){
        cin >> H >> W ;
        if(H==0 && W==0)break;
        for (int i = 0; i < H; i++)
        {
            for (int j = 0; j < W; j++)
            {
                if(i%2==0 && j%2==0)cout << '#' ;
                if(i%2==1 && j%2==0)cout << '.' ;
                if(i%2==0 && j%2==1)cout << '.' ;
                if(i%2==1 && j%2==1)cout << '#' ;
            }
            cout << endl;
        }
        cout << endl;
    }
}
