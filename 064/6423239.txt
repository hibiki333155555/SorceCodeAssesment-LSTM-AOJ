#include <stdio.h>

int main(void){
    while(1){
        unsigned long e[4001]={0};
        int eI,pI,qI,i,n,flag=0;
        
        scanf("%d",&n);
        if(n==0) break;
        
        for(i=0;i<n;i++){
            scanf("%d %d %d",&eI,&pI,&qI);
            e[eI]+=pI*qI;
        }
        for(i=0;i<4001;i++){
            if(e[i]>=1000000){
                printf("%d\n",i);
                flag=1;
            }
        }
        if(flag==0) puts("NA");
    }
    
    return 0;
}
