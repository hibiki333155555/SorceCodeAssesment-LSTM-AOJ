#include<iostream>
#include<string>

int main() {
    int count{0};
    while (true) {
        std::string ins;
        getline(std::cin, ins, '\n');
        if (ins != "END_OF_TEXT") {
            for (int i{0}; i < ins.length(); i++) {
                if (ins[i] == 'w') {
                    count += 1;
                }
            }
        } else {
            break;
        }
    }
    std::cout << count << std::endl;
}
