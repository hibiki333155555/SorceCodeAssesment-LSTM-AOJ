#include <stdio.h>

int main(){
    for(int i=1;;i++){
        int a, b;
        scanf("%d", &a, &b);
        if(a==0 && b==0) break;
        for(int j=1; j<=a; j++){
            for(int k=1; k<=b; k++){
                printf("#");
            }
            printf("\n");
        }
        printf("\n");
    }
}
