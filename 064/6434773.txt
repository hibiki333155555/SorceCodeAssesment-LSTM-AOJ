#include<bits/stdc++.h>
using namespace std;

#define ll long long

int main()
{
    ll n,q;
    cin >> n >> q;

    vector<vector<ll>> A(n);
    for(ll i=0;i<q;i++)
    {
        ll f;
        cin >> f;

        if(f==0)
        {
            ll t,x;
            cin >> t >> x;

            A[t].push_back(x);
        }
        else if(f==1)
        {
            ll t;
            cin >> t;

            for(ll j=0;j<(int)A[t].size();j++)
            {
                cout << A[t][j];
                if(j != A[t].size()-1)
                    cout << " ";
            }
            cout << endl;
        }
        else
        {
            ll t;
            cin >> t;

            if(A[t].size()!=0)
            {
                A[t].clear();
            }
        }
    }



    return 0;
}
