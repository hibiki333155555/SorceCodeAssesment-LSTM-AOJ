#include <bits/stdc++.h>
using namespace std;

int main() {
  int arr[10000][2];
  int i = 0;
  while (1) {
    cin >> arr[i][0] >> arr[i][1];
    if (arr[i][0] == 0 && arr[i][1] == 0) break;
    i++;
  }

  for (int m = 0; m < i; m++) {
    for (int j = 0; j < 300; j++) {
      if (j == arr[m][0]) break;
      if (j % 2 == 0) {
        for (int k = 0; k < 300; k++) {
          if (k == arr[m][1]) break;
          if (k % 2 == 0) {
            printf("#");
          } else printf(".");
        }
        printf("\n");
      } else {
        for (int l = 0; l < 300; l++) {
          if (l == arr[m][1]) break;
          if (l % 2 == 0) {
            printf(".");
          } else printf("#");
        }
        printf("\n");
      }
    }
    printf("\n");
  }

  return 0;

}
