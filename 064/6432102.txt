#include <bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);

    ll q,n,g,x,f,t;
    cin>>q;
    ll l,r;
    set<ll>v;
    while(q--)
    {
        cin>>f;
       if(f==0) {
                 cin>>x;

                 v.insert(x);
                 cout<<v.size()<<endl;}
       if(f==1) {
            cin>>x;
           if(v.count(x)) {cout<<1<<endl;}
           else {cout<<0<<endl;}
       }
       if(f==2) { cin>>x; v.erase(x);}
       if(f==3) { cin>>l>>r;

        for(auto it=v.begin();it!=v.end();it++)
        {
            if(*it>=l && *it<=r) cout<<*it<<endl;
            if(*it>r) break;
        }



    }

    }
    return 0;
}
