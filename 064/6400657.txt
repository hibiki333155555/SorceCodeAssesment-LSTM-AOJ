#include <stdio.h>
#include <iostream>
#include<algorithm>
using namespace std;
int n, m, a[100010], b[50005], ans = 0;
int main()
{

	cin >> n;
	for (int i = 0; i < n; i++) cin >> a[i];
	cin >> m;
	for (int i = 0; i < m; i++) cin >> b[i];
	sort(a, a + n);
	sort(b, b + m);
	for (int i = 0; i < n; i++)
	{
		if (a[i] == a[i + 1]) a[i] = -1;
	}
	for (int i = 0; i < m; i++)
	{
		if (b[i] == b[i + 1]) a[i] = -1;
	}
	sort(a, a + n);
	sort(b, b + m);
	for (int i = 0; i < m; i++)
	{
		int l = 0, r = n-1;
		if (b[i] == -1)continue;
		while (l < r)
		{
			int mid = l + r >> 1;
			if (b[i] <= a[mid])r = mid;
			else l = mid + 1;
		}
		if (a[l] == b[i])ans++;
	}
	cout << ans << endl;
	return 0;
}
