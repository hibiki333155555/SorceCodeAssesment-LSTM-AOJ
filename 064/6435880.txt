#include<iostream>
using namespace std;

int main() {
    int a, b, x, a_buy, b_buy;
    cin >> a >> b >> x;
    
    if ( b * 2 > a ) {
        a_buy = x / 1000;

        if ( x % 1000 > 500 ) {
            a_buy += 1;
        } else {
            b_buy = 1;
        }
    } else {
        a_buy =  x / 1000;

        if ( x % b > 500 ) {
            b_buy = 2;
        } else {
            b_buy = 1;
        }
    }
    cout << a * a_buy + b * b_buy << endl;

    return 0;
}
