#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define all(x) begin(x),end(x) 

const ll mx = 1e4+7;
std::vector<int> g[mx];
vector<int> mt;
vector<bool> used;

bool try_kuhn(int u) {
    if(used[u]) return false; 
    used[u] = true;
    for(int v: g[u]) {
        if( mt[v]==-1 || try_kuhn(v)) {
            mt[v] = u; 
            return true; 
        }
    }
    return false; 
}

int main() {
#ifndef LOCAL
    ios_base::sync_with_stdio(false),cin.tie(nullptr); 
#else
    freopen( "in.txt", "r", stdin);
    //freopen( "out.txt","w", stdout);
#endif
    int x, y, e; 
    cin >> x >> y >> e; 
    int n = max(x,y); 
    for(int i = 0, u, v; i<e; i++) {
        cin >> u >> v; 
        g[u].push_back(v); 
    }
    
    mt.assign(n,-1);
    for(int i = 0; i<n; i++) {
        used.assign(n,false); 
        try_kuhn(i); 
    }
    int cnt = 0; 
    for(int i = 0; i<n; i++) {
        if(mt[i] +=-1) {
            cnt++; 
        }
    }
    cout << cnt << endl; 
    return 0; 
}
