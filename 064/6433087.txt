#include <iostream>
using namespace std;

char field[105][105];
int H, W;
int dx[] = {1, 0, -1, 0}, dy[] = {0, 1, 0, -1};

void dfs(int x, int y) {
  char reserve = field[x][y];
  field[x][y] = ' ';
  for (int i = 0; i < 4; ++i) {
    int nx = x + dx[i], ny = y + dy[i];
    if (0 <= nx && nx < H && 0 <= ny && ny < W && field[nx][ny] == reserve)
      dfs(nx, ny);
  }
}

int main() {
  while (cin >> H >> W && H != 0) {
    for (int i = 0; i < H; ++i) {
      for (int j = 0; j < W; ++j) {
        cin >> field[i][j];
      }
    }
    int ans = 0;
    for (int i = 0; i < H; ++i) {
      for (int j = 0; j < W; ++j) {
        if (field[i][j] != ' ') {
          ++ans;
          dfs(i, j);
        }
      }
    }
    cout << ans << endl;
  }
}
