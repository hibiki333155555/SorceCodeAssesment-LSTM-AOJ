int main()
{	
	int num[30]={0};
	char c;
	int i;
	
	while((c=getchar())!=EOF)
	{
		if(c>='A'&&c<='Z')
			c=c-'A'+'a';
		
		if(c>='a'&&c<='z')
			num[c-'a']++;
	}
	
	
	
	for(i=0;i<=25;i++)
	{
		c='a'+i;
		cout<<c<<" : "<<num[i]<<endl;
	}
return 0;
}
