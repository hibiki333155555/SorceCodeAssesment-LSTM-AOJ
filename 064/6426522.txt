#include<iostream>
using namespace std;

int search(int a[],int key,int n){
    int i = 0;
    while(a[i] != key){
            i++;
        }
    return i != n;
    }
    
int main(){
    int i,n;
    int count = 0;
    cin >> n;
    int a[n];
    for(i=0;i<n;i++){
        cin >> a[i];
    }
    int m;
    cin >> m;
    int b[m];
    for(i=0;i<m;i++){
        cin >> b[i];
    }
    for(i=0;i<m;i++){
      if(search(a,b[i],m)) count++;  
    }
    cout << count << endl;
    return 0;
}
