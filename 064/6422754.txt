let [n, inputArr] = require(`fs`).readFileSync(`/dev/stdin`, `utf8`).split(`\n`);
let arr = inputArr.split(` `).map(Number);
let flag = 1;
let j = 0;
let tentouVal = 0;
while (flag) {
    flag = 0;
    for (let i = n-1; i>j; i--) {
        if (arr[i] < arr[i - 1]) {
            [arr[i], arr[i - 1]] = [arr[i - 1], arr[i]];
            tentouVal++;
            flag = 1;
        }
    }
    j++;
}

console.log(`${arr.join(` `)}\n${tentouVal}`);
