    	int x = i;
        if (x % 3 == 0) {
          cout << ' ' << i;
          continue;
        }
        if (x % 10 == 3) {
          cout << " " << i;
          continue;
        }
        while (true) {
            x /= 10;
                if (x) {
                    if (x % 10 == 3) {
                        cout<< " " << i ; 
                        break;   
                    }
                    
                }
                else break;        
        
        }
    }
}
