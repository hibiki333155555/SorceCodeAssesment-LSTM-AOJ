#include<stdio.h>
#include<math.h>
int main() {
    int i, n, s[1000];
    double h, sum;
    while (1)  {
        scanf("%d ", &n);
        if (!n) break;
        sum = 0;
        h = 0;
        for ( i=0; i<n; i++ ) {
            scanf("%d", &s[i]);
            sum += s[i];
        }
        sum /= n;
        printf("%lf", sum);
        for ( i=0; i<n; i++ ) {
            h += pow(s[i]-sum, 2.0);
            }
      h /= n;
      printf("%lf\n", pow(h, 1.0/2.0));
    }
    return 0;
}
