#include <iostream>
using namespace std;

int n;
bool ls[110][110];

int main()
{
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		int u, k;
		cin >> u >> k;
		for(int j = 0; j < k; j++)
		{
			int v;
			cin >> v;
			ls[u][v] = true;
		}
	}
	for(int i = 1; i <= n; i++)
	{
		for(int j = 1; j <= n; j++) 
		{
			if(j > 1) cout << " ";
			cout << ls[i][j];
		}
		if(i != n) cout << endl;
	}
	return 0;
}
