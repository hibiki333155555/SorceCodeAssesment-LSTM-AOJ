#include<stdio.h>
#include<vector>
using namespace std;
vector<int> vec;

int fib(int n){
  if(n == 0){return 1;}
  else if(n == 1){return 1;}
  else{
    if(vec.at(n-1) != -1){return vec.at(n-1);}
    else{
      return vec.at(n-1) = fib(n-1) + fib(n-2);
    }
  }
}


int main(){
  int n;
  scanf("%d",&n);
  vec = vector<int>(n,-1);
  printf("%d",fib(n));
}
