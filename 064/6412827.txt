#include <iostream>

using namespace std;

void selection_sort(int A[],int n)
{
    int num=0;
    int min=0;
    
    for(int i=0;i<n;i++)
    {
        min=i;
        for(int j=i;j<n;j++)
        {
            if(A[j]<A[min])
                min=j;
        }
        swap(A[i],A[min]);
        num++:
    }
    
    for(int i=0;i<n;i++)
    {
        printf("%d",A[i]);
        if(i<n-1)printf(" ");
    }printf("\n");
    printf("%d\n",num);
    
}

int main(void)
{
    int n=0;
    scanf("%d",&n);
    int A[n];
    
    for(int i=0;i<n;i++)
        scanf("%d",&A[i]);
        
    selection_sort(A,n);
}
