import java.util.Scanner;
public class Main
{
	public static void main(String[] args) 
	{
	    Scanner sc=new Scanner(System.in);
	    int a=sc.nextInt(),b=sc.nextInt(),c=sc.nextInt();
	    int count=0;
	    if(a>=1 && b>=a && c>=a && c<=10000)
	    {
	        for(int i=a;i<=b;i++)
	        {
	            if(c%i==0)
	            {
	                count++;
	            }
	        }
	        System.out.println(count);
	    }
	}
}
