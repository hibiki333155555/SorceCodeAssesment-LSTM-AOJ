import java.util.*;

class Main {
    public static void main (String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = 1 ;
        if (n < 2) {
            double a = sc.nextInt();
            String op = sc.next();
            int b = sc.nextInt();
            if (op == "?") {
                n = 2;
            }else {
                System.out.println(Math.ceil(a + op + b));
            }
        }
    }
}
