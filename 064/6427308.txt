#include<bits/stdc++.h>
using namespace std;
int main(){
    int y1,m1,d1,y2,m2,d2; cin>>y1>>m1>>d1>>y2>>m2>>d2;
    if(y1>y2){
        if(m1>m2){
            cout<<y1-y2+1<<endl;
        }else if(m1==m2){
            if(d1>d2){
                cout<<y1-y2+1<<endl;
            }else{
                cout<<y1-y2<<endl;
            }
        }else{
            cout<<y1-y2<<endl;
        }
    }else if(y1<y2){
        if(m1<m2){
            cout<<y2-y1+1<<endl;
        }else if(m1==m2){
            if(d1<d2){
                cout<<y2-y1+1<<endl;
            }else{
                cout<<y2-y1<<endl;
            }
        }else{
            cout<<y2-y1<<endl;
        }
    }else{
        if(m1>m2||m1<m2){
            cout<<1<<endl;
        }else if(m1==m2){
            if(d1==d2){
                cout<<0<<endl;
            }else{
                cout<<1<<endl;
            }
        }
    }
    return 0;
}
