using namespace std;

class Dice
{
public:
    Dice(int t, int s, int e, int w, int n, int b) : T(t), S(s), E(e), W(w), N(n), B(b){};
    void goS()
    {
        tmp = B;
        B = S;
        S = T;
        T = N;
        N = tmp;
    }
    void goN()
    {
        tmp = B;
        B = N;
        N = T;
        T = S;
        S = tmp;
    }
    void goE()
    {
        tmp = B;
        B = E;
        E = T;
        T = W;
        W = tmp;
    }
    void goW()
    {
        tmp = B;
        B = W;
        W = T;
        T = E;
        E = tmp;
    }

    int getT() { return T; }
    int getE() {return E;}
    int getB() {return B;}
    int getW() {return W;}
    int getN() {return N;}
    int getS() {return S;}

    void setT(int a) 
    {
        if(a == T) return;
        else if(a == B) {goE();goE();}
        else if(a == N) goS();
        else if(a == S) goN();
        else if(a == E) goW();
        else goE();
    }

    void rotate()
    {
        tmp = E;
        E = S;
        S = W;
        W = N;
        N = tmp;
    }

    void setS(int a){   //CALL setT before setS
        while(S != a)
            rotate();
    }

    bool check(int a)
    {
        if(a == T || a == S || a == E || a == W || a == N || a == B)
            return true;
        return false;
    }

private:
    int T, S, E, W, N, B, tmp;
};

int main()
{
    int a, b, c, d, e, f;
    cin >> a >> b >> c >> d >> e >> f;
    Dice dice1(a, b, c, d, e, f);
    
    cin >> a >> b >> c >> d >> e >> f;
    Dice dice2(a, b, c, d, e, f);

    if(dice2.check(dice1.getT()))
    {
        dice2.setT(dice1.getT());
        if(dice2.check(dice1.getS()))
        {
            dice2.setS(dice1.getS());
            if(dice2.getE() == dice1.getE() && dice2.getB() == dice1.getB() && dice2.getN() == dice1.getN())
            {
                cout << " Yes" << endl;
                return 0;
            }
        }
    }
    
    cout << " No" << endl;
    return 0;
}
