int n,q;
vector<int> s,t;

bool isOK(int index, int key){
    if(s[index] >= key) return true;
    else return false;
}

bool binary_search(int key){
    int ng = -1;
    int ok = (int)s.size();

    while(abs(ok - ng) > 1){
        int mid = (ok + ng) / 2;

        if(isOK(mid, key)) ok = mid;
        else ng = mid;
    }
    
    if(s[ok] == key) return true;
    else return false;
}

int main(void){
    cin >> n;
    s.resize(n);
    rep(i,0,n) cin >> s[i];
    cin >> q;
    t.resize(q);
    rep(i,0,q) cin >> t[i];
    int cnt = 0;

    rep(i,0,q){
        if(binary_search(t[i])) cnt++;
    }

    cout << cnt << endl;
}
