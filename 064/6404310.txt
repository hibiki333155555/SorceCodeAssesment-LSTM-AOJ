import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.next();
        String p = sc.next();

        // 文字列sを連結させて周にみせかける
        s = s.concat(s);

        if (s.contains(p)) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }
}
