//#include<bits/stdc++.h>
#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;
//
int main()
{	
	
	int n,m,l;
	int i,j,k;
	int a[101][101],b[101][101],c[101][101];
	
	cin>>n>>m>>l;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>a[i][j];
	for(i=1;i<=m;i++)
		for(j=1;j<=l;j++)
			cin>>b[i][j];
			
	for(i=1;i<=n;i++)
		for(j=1;j<=l;j++)
			for(k=1;k<=m;k++)
				c[i][j]+=a[i][k]*b[k][j];
		
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=l;j++)
		{
			if(j!=l)
				cout<<c[i][j]<<" ";
			else
				cout<<c[i][j];
		}
		cout<<endl;
	}
return 0;
}
