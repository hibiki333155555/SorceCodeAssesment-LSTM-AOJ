#include <iostream>
#include<cstdio>
using namespace std;
int A[20],q;

bool solve(int i,int m){
    if(m==0)return true;
    else if(i==q) return false;
    return solve(i+1,m-A[i])|solve(i+1,m);
}
int main(void){
    // Your code here!
    
    scanf("%d",&q);
    for (int i=0;i<q;i++){
        scanf("%d",&A[i]);
    }
    int s,m;
    scanf("%d",&s);
    for (int i=0;i<s;i++){
        scanf("%d",&m);
        if(solve(0,m))printf("yes\n");
        else printf("no\n");
    }
}
