#include<bits/stdc++.h>
using namespace std;
#define endl "\n"
#define ll long long
#define fix(ans,x) fixed << setprecision(x)<<ans
int main(){
    int a,b,x;
    int ans=0;
    cin>>a>>b>>x;
    int as=x,go=x;
    while(x>=0){
        if(x>1000){
            ans+=a;
            x-=1000;
        }else{
            ans+=b;
            x-=500;
        }
    }
    //cout<<ans<<" "<<x<<endl;
    int s=0,g=0;
    while(as>=0){
        s+=a;
        as-=1000;
    }
    //cout<<s<<" "<<as<<endl;
    while(go>=0){
        g+=b;
        go-=500;
    }
    //cout<<g<<" "<<go<<endl;
    int e;
    e=min(s,g);
    int xs=min(e,ans);
    cout<<xs<<endl;
    return 0;
}
