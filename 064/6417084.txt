import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        int n = sc.nextInt();
        String strn = n.toString();
        
        for(int i=3;i<=n;i++){
            if(i%3==0) System.out.printf(" %d",i);
            
            if(strn.contains("3")) System.out.printf(" %s",strn);
        }
    }
}
