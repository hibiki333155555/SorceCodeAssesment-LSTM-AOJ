#include <bits/stdc++.h> 
using namespace std;

int main()
{
    string s, p;
    cin>>s>>p;
    s += s;
    int len1 = s.length();
    int len2 = p.length();
    int count = 0;
    for(int i=0, j=0; i<(len1-len2); i++) {
        if(s[i] == p[j]) {
            j++;
            count++;
            if(count == len2) {
                cout<<"Yes"<<endl;
                return 0;
            }
        }
        else {
            j = 0;
            count = 0;
        }
    }
    cout<<"No"<<endl;
    return 0;
}
