// #pragma GCC target("avx2")
#pragma GCC optimize("O3")
// #pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int a, b;
    cin >> a >> b;
    if (a == 0 && b == 0) {
        cout << 1 << '\n';
        return 0;
    }
    bool s = false;
    int ans = 0;
    for (int i = 30; i >= 0; --i) {
        s |= (a >> i) || (b >> i);
        if (s && ((a >> i) & 1) == ((b >> i) & 1)) {
            ++ans;
        }
        else if (s) {
            break;
        }
    }
    cout << ans << '\n';
    return 0;
}
