#include <stdio.h>

int main(void)
{
    int a[4];
    int d,r,ans;
    (void)scanf("%d %d %d",&a[0],&a[1],&a[2]);
    if((a[1]-a[0]) == (a[2]-a[1]))
    {
        d = a[1] - a[0];
        ans = a[0] + d * 9;
    }
    else if((a[2]/a[1]) == (a[1]/a[0]))
    {
        r = a[0] / a[1];
        for(int i = 1;i >= 9;i++)
        {
            a[0] = a[0] * r; 
        }
        ans = a[0];
    }
    printf("&d\n",ans);
    return 0;
}
