counter = [0 for i in range(26)]

while True:
    try:
        chr = input()
        chr.lower()
        for i in chr:
            num = ord(i) - ord('a')
            if 0<= num <=25:
                counter[num] += 1
            else:
                pass
    except EOFError:
        break
    
for j in range(26):
    print(f'{chr(j+97)} : {counter[j]}')
