#include <iostream>
#include <vector>

//key以上の最小のindexを返す
int binary_search(int key, std::vector<int>& a) {
    int left = -1;
    int right = (int)a.size();
    while (right - left > 1) {
        int mid = (left + right) / 2;
        if (a[mid] >= key) {
            right = mid;
        }
        else {
            left = mid;
        }
    }
    return right;
}   

int main() {
    int n; std::cin >> n;
    std::vector<int> a;
    for (int i = 0; i < n; ++i) {
        int n; std::cin >> n;
        a.push_back(n);
    }

    int cnt = 0;
    int q; std::cin >> q;
    for (int i = 0; i < q; ++i) {
        int n; std::cin >> n;
        int id = binary_search(n, a);
        if (id >= (int)a.size()) continue;
        if (a[id] == n) cnt++;
    }
    std::cout << cnt << std::endl;
}
