#include <bits/stdc++.h>
using namespace std;

int main() {
    int size, type, in;
    cin >> size;
    vector<int> store;
    for (int q = 0; q < size; q++) {
        cin >> type;
        if (type == 0) {
            cin >> in;
            store.push_back(in);
        } else if (type == 1) {
            cin >> in;
            cout << store[in] << "\n";
        } else {
            store.pop_back();
        }
    }
    return 0;
}
