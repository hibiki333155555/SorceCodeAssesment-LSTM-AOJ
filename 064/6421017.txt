#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef long double LD;

#define rep(i,n) for(LL i=0;i<(n);i++)

int main(){
    string s;
    string ans;
    LL add = 0;
    LL max = 26;
    stack<LL> data;

    cin >> s;

    rep(i,s.size()){

        if(isdigit(s.at(i))){
            LL next = s.at(i) - '0';
            data.push(next);
            add += next;
        }
        else if(s.at(i) == ')'){
            LL temp = data.top();
            data.pop();
            add -= temp;
        }
        else if(isalpha(s.at(i))){
            LL num = add;
            LL current = s.at(i) - 'a';
            LL next = (current + num) % max;
            ans.push_back(next + 'a');
        }

    }

    cout << ans << endl;

    return 0;
}
