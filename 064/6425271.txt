#include<iostream>
#include<algorithm>

using namespace std;
int main() {
    int A, B, X, ans = (1<<30);
    cin >> A >> B >> X;
    for ( int a = 0; a <= 20; a++ ){
    for ( int b = 0; b <= 40; b++ ){
        if ( a*1000 + b*500 < X ) continue;
        if ( a*A + b*B < ans ) ans = a*A + b*B;
    }
    }
    cout << ans << endl;
}
