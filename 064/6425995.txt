#include <stdio.h>

int search(int A[],int n, int key){
    int i=0;
    A[n]=key;
    while(A[i]!=key){
        i++;
    }
    return i!=n;
}
int main(void){
    // Your code here!
    int n;
    scanf("%d",&n);
    int A[n];
    for (int i=0;i<n;i++) scanf("%d",&A[i]);
    int key,sum=0,q;
    scanf("%d",&q);
    for(int i=0;i<q;i++){
        scanf("%d",&key);
        if (search(A,n,key))sum++;
    }
    printf("%d\n",sum);
    return 0;
    
}
