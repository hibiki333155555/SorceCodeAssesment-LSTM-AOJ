void solve(){
    ll n,k;cin>>n>>k;
    vector<ll>a(n),p(n);
    rep(i,n)cin>>a[i];
    rep(i,n-1){
        cin>>p[i+1];
        p[i+1]--;
    }
    vector<vector<ll> >dp(n,vector<ll>(k+1));
    dp[0][a[0]]=1;
    ll ans=dp[0][k];
    REP(i,1,n){
        rep(j,k+1){
            if(j>=a[i]){
                dp[i][j]=(dp[i][j]+dp[p[i]][j-a[i]])%mod;
            }
        }
        ans=(ans+dp[i][k])%mod;
    }
    cout<<ans<<endl;
}

    solve();
