#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    // r行c列のテーブルを与えられるのでr+1行目およびc+1列目にそれぞれ各列、各行の総和sumを元のテーブルとともに出力
    // 行方向の総和はセルの値を入力する際に求められる
    // 列方向の総和をどうやって求めるか
    // r行c列の二次元配列tableにおいて
    // 行方向のi行目の総和table[i][c] = table[i][0] + table[i][1] + ...... + table[i][c-1]
    // 列方向のj行目の総和table[r][j] = table[0][j] + table[1][j] + ...... + table[r-1][j]
    int r, c;
    cin >> r >> c;

    int table[101][101];
    int rowsum[101], columnsum[101];
    for (int i = 0; i < r; i++)
    {
        for (int j = 0; j < c; j++)
        {
            cin >> table[i][j];
            table[i][c] += table[i][j];
            table[r][j] += table[i][j];
            table[r][c] += table[i][j];
        }
    }
    

    for (int i = 0; i <= r; i++)
    {
        for (int j = 0; j <= c; j++)
        {
            cout << table[i][j];
            if(j != c){
                cout<<' ';
            }
        }

        cout<<endl;
    }

    return 0;
}
