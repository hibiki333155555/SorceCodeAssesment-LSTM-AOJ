using namespace std;

int main(){
    int N;
    cin>>N;
    vector<long long> vec(N);
    for(int i=0; i<N; i++){
        cin>>vec.at(i);
    }
        long long M=vec.at(0);
        long long m=vec.at(0);
        for(int i=0; i<N; i++){
            if(i==0){
            M=max(vec.at(0),vec.at(1));
            }
            else{
            M=max(vec.at(i-1),vec.at(i)); 
            }
        }
        for(int i=0; i<N; i++){
            if(i==0){
            m=min(vec.at(0),vec.at(1));
            }
            else{
            m=min(vec.at(i-1),vec.at(i)); 
        }
        }
    long long sum=0;
    for(int i=0; i<N; i++){
        sum+=vec.at(i);
    }
    cout<< m <<" "<< M <<" "<< sum <<endl;


}
