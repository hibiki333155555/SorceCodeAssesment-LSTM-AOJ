// C - Distance II

#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    int n; cin >>n;
    int x[n];
    int y[n];
    for (int i=0; i<n; i++) {
        cin >> x[i];
    }
    for (int i=0; i<n; i++) {
        cin >> y[i];
    }

    double manhattan_distance; // p = 1
    for (int i=0; i<n; i++) {
        manhattan_distance += abs(x[i]-y[i]);
    }
    std::cout<<std::fixed<<std::setprecision(6)<<manhattan_distance <<endl;

    double euclidean_distance;
    double temp_euclid;
    for (int i=0; i<n; i++) {
        temp_euclid += abs((x[i]-y[i])*(x[i]-y[i]));
        euclidean_distance = sqrt(temp_euclid);
    }
    std::cout<<std::fixed<<std::setprecision(6)<< euclidean_distance <<endl;

}
