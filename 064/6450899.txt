vector< int64_t > divisor(int64_t n) {
  vector< int64_t > ret;
  for(int64_t i = 1; i * i <= n; i++) {
    if(n % i == 0) {
      ret.push_back(i);
      // if(i * i != n) ret.push_back(n / i);
    }
  }
  sort(begin(ret), end(ret));
  return ret;
}

void solve(int n) {
  int s = input();
  auto divs = divisor((int64_t)(n + 1));

  vector< int > cnt(3);
  range(_, 0, 4 * n) {
    int a = input();
    cnt[a - 1]++;
  }

  for (int a : divs) {
    int b = (n + 1) / a;

    int e1 = 2 * (a + b);
    int e2 = 2 * (n + 1) - (a + b);

    int t1 = e1 - cnt[0];
    int t2 = e2 - cnt[1];
    int t3 = e2 - cnt[2];

    if (min({t1, t2, t3}) < 0) continue;
    if (max({t1, t2, t3}) > 4) continue;
    if (t1 == 4) continue;
    if (t1 + t2 + t3 != 4) continue;

    vector< int > ans;
    range(i, 0, t1) ans.emplace_back(1);
    range(i, 0, t2) ans.emplace_back(2);
    range(i, 0, t3) ans.emplace_back(3);

    if (s and max({t1 & 1, t2 & 1, t3 & 1}) == 0) {
      swap(ans[1], ans[2]);
    }

    cout << ans << endl;
    return;
  }

}

signed main() {
  int t = input();

  range(ti, 0, t) {
    int n = input();
    solve(n);
  }
}
