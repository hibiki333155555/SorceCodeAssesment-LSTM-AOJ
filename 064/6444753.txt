#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    int N,M,sum=0;
    cin >> N >> M;
    int A[N],B[M];
    for(int i=0; i<N; i++){
        cin >> A[i];
    }
    for(int i=0; i<M; i++){
        cin >> B[i];
    }
    for(int i=0; i<N; i++){
        for(int j=0 ;j<M; j++){
            if(A[i]<=B[j]){
                sum++;
            }
        }
    }
    cout << sum << endl;
}
