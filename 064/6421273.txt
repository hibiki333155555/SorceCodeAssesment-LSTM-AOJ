#include <iostream>
#include <string>

using namespace std;
int R[2000010];
void manacher(string &s){
    int i = 0,j = 0;
    string S;
    S.push_back('A');
    for(i=0;i<s.size();i++){
        S.push_back(s[i]);
        S.push_back('A');
    }
    s = S;
    for(i=0;i<S.size();i++){
        R[i] = 0;
    }
    i = 0; j = 0;
    while (i<S.size()) {
        while (i-j >= 0 && i+j < S.size() && S[i-j] == S[i+j]) ++j;
        R[i] = j;
        int k = 1;
        while (i-k >= 0 && i+k < S.size() && k+R[i-k] < j){
            R[i+k] = R[i-k];
            ++k;
        }
        i += k; j -= k;
    }
}

int main(){
    int i,n; cin >> n;
    string s; cin >> s;
    manacher(s);
    int c = 0;
    for(i=1;i<s.size() - 1;i++){
        if(s[i]=='A'){
            if((R[i] - 1)==min(i,(int)s.size() - i - 1)) c++;
        }
    }
    cout << c << endl;
