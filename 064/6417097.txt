import java.util.Scanner;

class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        
        int n = sc.nextInt();
        
        for(int i=3;i<=n;i++){
            String strn = String.valueOf(i);
            if(i%3==0) System.out.printf(" %d",i);
            else if(strn.contains("3")) System.out.printf(" %s",strn);
        }
        System.out.println("");
    }
}
