import java.util.Scanner;

public class Main{
    public static void main(String[] args){
        Scanner scanner = new Scanner(System.in);
        int a = scanner.nextInt();
        int b = scanner.nextInt();
        
        int d = a/b;
        int r = a%b;
        double f = (double)a/b;

        System.out.println(d + " " + r + " " + String.format("%.6f", f));
    }
}
