#include <bits/stdc++.h>

#define ll long long
#define ld long double
#define rep(i,a,b) for (int i = a; i < b; i++)

using namespace std;

int main(){
    int n;
    cin >> n;
    vector<int> a(n);
    ll sum = 0;
    rep(i,0,n){ cin >> a[i]; sum += a[i];}
    int max = *max_element(a.begin(),a.end());
    int min = *min_element(a.begin(),a.end());
    cout << min << " " << max << " " << sum << endl;
    //system("pause");
    return 0;
}
