import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
    	Scanner sc = new Scanner(System.in);
    	int x;
    	
    	for (int i = 1; i < 10001; i++) {
    		x = sc.nextInt();
    		if (x == 0) {
    			break;
    		} else {
    			System.out.println("Case " + i + ": " + x);
    		}
    	}
    }
}
