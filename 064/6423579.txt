#include <iostream>

using namespace std;

int main()
{
    int  n = 1;
    cin>>n;
    int  arr[n];
    for(int i = 0;i < n;i++)
    {
        cin>>arr[i];
    }
    int a = 0;
    int b = n-1;
    while(a < b)
    {
        int c = arr[a];
        arr[a] = arr[b];
        arr[b] = c;
        a++;
        b--;
    }
    for(int j = 0; j < n;j++)
    {
        if( j != n - 1 )
        {
            cout<<arr[j]<<" ";
        }
        else if(j == n - 1)
        {
        cout<<arr[j];
        }
    }
    return 0;
}
