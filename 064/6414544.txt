#include <iostream>
#include <string>

using namespace std;
string s[110];
int main(){
    int i,j,k,m,n; cin >> m >> n;
    for(i=0;i<m;i++) cin >> s[i];
    int ans = 1000000000;
    for(i=0;i<m - 1;i++){
        int c1 = 0,c2 = 0,c3 = 0,c4 = 0;
        for(k=0;k<m;k++){
            if(k<=i){
                for(j=0;j<n;j++){
                    if(s[k][j]=='o') c1++;
                    else c2++;
                }
            }else{
                for(j=0;j<n;j++){
                    if(s[k][j]=='o') c3++;
                    else c4++;
                }
            }
        }
        ans = min(ans,c2 + c3);
        ans = min(ans,c1 + c4);
    }
    for(j=0;j<n - 1;j++){
        int c1 = 0,c2 = 0,c3 = 0,c4 = 0;
        for(k=0;k<n;k++){
            if(k<=j){
                for(i=0;i<m;i++){
                    if(s[i][k]=='o') c1++;
                    else c2++;
                }
            }else{
                for(i=0;i<m;i++){
                    if(s[i][k]=='o') c3++;
                    else c4++;
                }
            }
        }
        ans = min(ans,c1 + c4);
        ans = min(ans,c2 + c3);
    }
    cout << ans << endl;
}
