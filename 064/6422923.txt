bool is_prime(ll x){
    switch(x){
        case 0:
        case 1:
        return false;
        case 2:
        case 3:
        return true;
    }
    if (x%2 == 0 || x%3 == 0) return false;

    for(ll i = 5; i * i <= x; i += 6){
        if(x%i == 0) return false;
        if(x%(i+2) == 0) return false;
    }
    return true;
}

int main(){
    ll n = cinll();
    ll c = 0;
    while(n--){
        if(is_prime(cinll())){
            c++;
        }
    }
    coutll(c);

    return 0;
}
