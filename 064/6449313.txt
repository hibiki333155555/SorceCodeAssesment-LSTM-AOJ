#include <cstdio>
#include <iostream>
#include <utility> // pair, make_pair
#include <algorithm> // pair, make_pair
#include<list>
using namespace std;
typedef long long ll;

int main() {
    int n;
    list<ll> l;
    cin >> n;
    for(int i = 0;i < n;i++){
        string s;
        ll x;
        cin >> s;
        if(s.at(0) =='i'){
            cin >> x;
            l.push_front(x);
        }
        else{
            if(s.size() < 8){
                cin >>x;
                for(list<int>::interator it = l.begin();it != l.end();it++){
                    if(*it == x){
                        l.erase(it);
                        break;
                    }
                }
            }
            else if(s.at(6) == 'F'){
                l.pop_front();
            }
            else if(s.at(6) == 'L'){
                l.pop_back();
            }

        }

    }
    ll i = 0;
    for(list<int>::interator it = l.begin();it != l.end();it++){
        if(i++){
            cout << " ";
        }
        cout << *it;
    }
    cout << endl;
}
