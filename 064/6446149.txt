#include <iostream>
using namespace std;

int main(){
    int n;
    char type[53];
    char stock[] ={'S', 'H', 'C', 'D'};
    int num[53];
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> type[i] >> num[i];
    }
    
    for(int i = 0; i < 4; i++){
        for(int j = 1; j <= 13; j++){
            bool result = true;
            for(int k = 0; k < n; k++){               
                if(type[k] == stock[i] && num[k] == j) result = false;
            }
            if(result) cout << stock[i] << " " << j << endl;
        }
    }
    return 0;
}
