#include<bits/stdc++.h>
using namespace std;
using ll=long long;
int main(){
	ll a,b;
	cin>>a>>b;
	vector<ll> s(60,-1),t(60,-1);;
	s[0]=0;
	t[0]=0;
	for(int i=0;i<60;i++){
		if(a%2) s[i]=1;
		else{
			if(a>0) s[i]=0;
		}
		if(b%2) t[i]=1;
		else{
			if(b>0) t[i]=0;
		}
		a/=2;
		b/=2;
	}
	ll ans=0;
	bool g=0;
	for(int i=59;i>=0;i--){
		if(s[i]+t[i]>-2) g=true;
		if(g){
			if(s[i]==t[i]) ans++;
			else{
				cout<<ans<<endl;
				return 0;
			}
		}
	}
	cout<<ans<<endl;
}
