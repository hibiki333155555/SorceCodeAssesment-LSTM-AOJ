#include <iostream>
#include <bits/stdc++.h>
#include <string>
using namespace std;

int main() {
   string t;

   while(cin >> t) {
      for(int i=0; i<t.size(); i++) {
         if(islower(t[i])) {
            t[i] = toupper(t[i]);
         }else if(isupper(t[i])) {
            t[i] = tolower(t[i]);
         }
      }
      cout << t << " ";
   }
   cout << "\n";
}
