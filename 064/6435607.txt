#include <bits/stdc++.h>
using namespace std;

//Answer
int main() {
    long long n;cin>>n;
    vector<long long> r(n);
    for(int i = 0;i<n;i++) cin >> r[i];

    long long maxv = -2000000000;
    long long minv = r[0];

    //現在探索中までのr[i]について、それまでに出てきたr[]の中で
    //最も最小のものとのみ比較すれば良い
    for(int i = 1;i<n;i++) {
        maxv = max(maxv, r[i] - minv);
        minv = min(minv,r[i]);
    }
    cout << maxv << endl;
    return 0;
}
