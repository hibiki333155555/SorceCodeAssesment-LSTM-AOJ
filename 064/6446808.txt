#include <stdio.h>
#include <string.h>
//safd
int main(void)
{
    char s[1200];
    int freq[26];
    int i;
    
    memset(freq, 0, sizeof(freq));
    while (fgets(s, 1200, stdin) != NULL){
        i = 0;
        while (s[i] != '\0'){
            if ('A' <= s[i] && s[i] <= 'Z'){
                freq[s[i] - 'A']++;
            }
            else if ('a' <= s[i] && s[i] <= 'z'){
                freq[s[i] - 'a']++;
            }
            i++;
        }
    }
    for (i = 0; i < 26; i++){
        printf("%c : %d\n", 'a' + i, freq[i]);
    }
    
    return (0);
}
