#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,i;
    cin>>n;
    vector<int> k(n),p(n);
    for(i=0;i<n;i++){
        cin>>k[i]>>p[i];
    }
    for(i=0;i<n;i++){
        if(k[i]%p[i]==0){
            cout<<k[i]/p[i]<<endl;
        }else{
            cout<<k[i]%p[i]<<endl;
        }
    }
    return(0);
}
