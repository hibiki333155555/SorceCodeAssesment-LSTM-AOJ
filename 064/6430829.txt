#include<stdio.h>
#include<math.h>
#include<iostream>

using namespace std;

int main() {
	int n,x,i, j, k, sum;
	
	while(1){
		sum = 0;
		cin >> n >> x;
		if (n == 0 && x == 0) {
			break;
		}
		for (i = 1; i <= n; ++i) {
			for (j = 1; j <= n; ++j) {
				for (k = 1; k <= n; ++k) {
					if (i + j + k == x && i < j && j < k) {
						sum += 1;
					}
				}
			}
		}
		cout << sum << endl;
	}
}
