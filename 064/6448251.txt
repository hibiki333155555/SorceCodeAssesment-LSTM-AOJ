#include <stdio.h>

int main() {
    int num[100]={0};
    int count=0;
    int sum=0;
    int divisor = 0;
    
    while(1) {
        scanf("%d",&num[count]);
        if(num[count] == 0) {
            break;
        }
        count++;
    }
    
    for(int i=0;i<count;i++) {
        while(1) {
            divisor = num[i] % 10;
            sum = sum + divisor;
            num[i] = num[i] / 10;
            if(num[i] == 0) {
                break;
            }
        }
        printf("%d\n", sum);
        sum = 0;
    }
      
    return 0;
}
