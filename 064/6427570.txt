#include <stdio.h>
#include <math.h>

int main(void){
    unsigned long n,a[30],i,d,y[30];
    
    a[0]=1;a[1]=2;a[2]=4;
    for(i=3;i<30;i++){
        a[i]=a[i-1]+a[i-2]+a[i-3];
    }
    for(i=0;i<30;i++){
        d=(unsigned long)ceil((double)a[i]/10);
        y[i]=(unsigned long)ceil((double)d/365);
        //printf("%lu,%lu \n",i,y[i]);
    }
    
    while(1){
        scanf("%lu",&n);
        if(n==0) break;
        printf("%lu\n",y[n-1]);
    }
}
