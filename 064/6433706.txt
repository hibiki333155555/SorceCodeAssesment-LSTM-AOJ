#include <iostream>
#include <algorithm>
using namespace std;

#define MAX 300000

int c[MAX];

int main() {
    int N, Q;
    cin >> N >> Q;
    for (int i = 0; i < N; i++) {
        cin >> c[i];
    }
    sort(c, c+N);

    for (int i = 0; i < Q; i++) {
        int ans = 0;
        int q;
        cin >> q;

        for(int j = 0; j <= MAX + q; j += q){
            int index = lower_bound(c, c+N, j) - c;
            if (index != 0) index--;
            ans = max(ans, c[index] % q);
        }
        
        cout << ans << endl;
    }

    return 0;
}
