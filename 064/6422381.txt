int bubbleSort(vector<int>&A){
    int cnt = 0;
    for (int i = 0; i < A.size(); i++){
        for (int j = A.size()-1; j >= (i+1); j--){
            if (A[j] < A[j-1]){
                swap(A[j],A[j-1]);
                cnt++;
            }
        }
    }
    return cnt;
}

int main(){
    int n;
    cin >> n;
    vector<int>a(n);
    for (int i = 0; i < n; i++){
        cin >> a[i];
    }
    cout << bubbleSort(a) << endl;
}
