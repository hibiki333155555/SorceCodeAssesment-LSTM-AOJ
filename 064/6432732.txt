#include<iostream>
using namespace std;



int main(){
  int a[300];
  int n;
  cin>>n;
  for(int i=0;i<n;i++){
    cin>>a[i];
  }
  int q;
  cin>>q;
  for(int i=0;i<q;i++){
    int query1,query2;
    cin>>query1>>query2;
    int tmp=a[query1-1];
    a[query1-1]=a[query2-1];
    a[query2-1]=tmp;
    bool isSorted=1;
    for(int j=0;j<n-1;j++){
        if(a[j]>a[j+1]){
            isSorted=0;
            break;
        }
    }
    if(isSorted){
      cout<<i+1<<endl;
      return 0;
    }
  }
  cout<<-1<<endl;
  return 0;
}
