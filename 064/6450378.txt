// #pragma GCC target("avx2")
#pragma GCC optimize("O3")
// #pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;

using P = pair<int, int>;

int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int n, m;
    cin >> n >> m;
    vector<long long> d(m + 1);
    vector<int> mr(m, -1);
    for (int i = 0; i < n; ++i) {
        int a, l, r;
        cin >> a >> l >> r;
        --l;
        d[l] += a;
        d[r] -= a;
        mr[l] = max(mr[l], r);
    }
    for (int i = 1; i < m; ++i) {
        mr[i] = max(mr[i], mr[i - 1]);
    }
    int q;
    cin >> q;
    vector<long long> e(m);
    for (int i = 0; i < q; ++i) {
        int a, j;
        cin >> a >> j;
        --j;
        e[j] += a;
    }
    long long p = 0;
    long long ans = 0;
    for (int i = 0; i < m; ++i) {
        p += d[i];
        long long f = max(e[i] - p, 0LL);
        if (f > 0 && i >= mr[i]) {
            cout << -1 << '\n';
            return 0;
        }
        p += f;
        d[mr[i]] -= f;
        ans += f;
    }
    cout << ans << '\n';



    return 0;
}
