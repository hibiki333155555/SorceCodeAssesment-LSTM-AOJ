#define all(x) (x).begin(),(x).end()

#define si(x) int(x.size())

int main(){
    
    std::ifstream in("text.txt");
    std::cin.rdbuf(in.rdbuf());
    cin.tie(0);
    ios::sync_with_stdio(false);
    
    int N;cin>>N;
    vector<string> S(N);
    for(int i=0;i<N;i++) cin>>S[i];
    int ans=0;
    
    for(int i=0;i<N;i++){
        bool f=false;
        for(int j=0;j<N;j++){
            if(i==j) continue;
            int cn=0;
            for(int k=0;k<si(S[i]);k++){
                if(S[i][k]!=S[j][k]) cn++;
            }
            if(cn<=1) f=true;
        }
        if(f) ans++;
    }
    
    cout<<ans<<endl;
}
