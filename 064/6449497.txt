int main() {
	lint N;
	cin >> N;
	V<string> arr(N);
	cin >> arr;
	lint ans = 0;
	REP(i, N) {
		bool flag = false;
		REP(j, N) {
			if (i == j) continue;
			lint cnt = 0;
			REP(k, SZ(arr[i])) {
				if (arr[i][k] != arr[j][k]) cnt++;
			}
			if (cnt <= 1) flag = true;
		}
		ans += flag;
	}
	cout << ans << endk;
}
