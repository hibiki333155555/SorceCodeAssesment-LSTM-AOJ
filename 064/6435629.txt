#include <iostream>
using namespace std;

const int INF = 1 << 30;

int main()
{
    int N, num[2], price[2];
    cin >> N;

    for ( int i = 0; i < 2; i++ )
    {
        cin >> num[i] >> price[i];
    }
    
    int min_price = INF;
    for ( int i = 0; i < 2; i++ )
    {
        int count = 0;
        int total_price = 0;
        while ( count < N )
        {
            count += num[i];
            total_price += price[i];
        }
        if ( total_price < min_price ) min_price = total_price;
    }
    
    cout << min_price << endl;
    
    return 0;
}
