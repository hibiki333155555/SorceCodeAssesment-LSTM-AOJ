#include <iostream>
using namespace std;

void bubblesort(int a[],int n){
    int i,flag;
    flag = 1;
    while(flag){
        flag = 0;
        for(i=n-1;i>0;i--){
           if(a[i] < a[i-1]){
               flag = 1;
               swap(a[i],a[i-1]);
           }
        }
    }
    for(i=0;i<n;i++){
        cout << a[i] << " ";
    }
    cout << endl;
}

int main(){
    int i,n;
    cin >> n;
    int a[n];
    for(i=0;i<n;i++){
        cin >> a[i];
    }
    bubblesort(a,n);
    return 0;
}
