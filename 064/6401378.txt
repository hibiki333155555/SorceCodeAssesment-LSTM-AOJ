import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 値を読み込む
        int a = sc.nextInt();
        int b = sc.nextInt();

        // 割算を求める
        int d = a / b;

        // 余りを求める
        int r = a % b;

        // 浮動小数点の割算
        double f = ((double) a) / ((double) b);

        System.out.println(d + " " + r + " " + String.format("%.8f", f));
    }
}
