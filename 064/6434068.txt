#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> e(12);
    for(int i = 0; i < 12; i++){
        cin >> e[i];
    }
    sort(e.begin(), e.end());
    if(e[0] == e[3] && e[4] == e[7] && e[8] == e[11]){
        cout << "yes" << endl;
        return 0;
    }
    cout << "no" << endl;
    return 0;
}
