#include <iostream>
#include <algorithm>
#include "vector"

using namespace std;

struct P
{
    double x, y;
};
int q;

P p0;
vector<P> ls;

void cal()
{
    double sum = 0;
    for (int i = 0; i < q; ++i)
    {
        sum += ls[i].x*ls[i+1].y - ls[i+1].x*ls[i].y;
    }
    cout << 0.5*abs(sum) << endl;
}

signed main()
{
    cout.setf(ios::fixed);
    cout.precision(1);
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    cin >> q;
    for (int i = 0; i < q; ++i)
    {
        cin >> p0.x >> p0.y;
        ls.push_back(p0);
    }
    ls.push_back(ls[0]);
    cal();
    return 0;
}
