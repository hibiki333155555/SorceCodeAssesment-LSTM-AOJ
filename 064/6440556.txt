#include <iostream>
#include <vector> // vector
#include <utility> // pair, make_pair
using namespace std;

void bubblesort(vector<int> R, int n){
    int tmp = 0;
    for(int i = 0;i < n;i++){
        for(int j = n-1;j > i;j--) {
            if (R.at(j) < R.at(j-1)){
                swap(R.at(j), R.at(j-1));
                tmp++;
            }
        }
    }
    for(int i = 0;i < n;i++)cout << R.at(i);
    cout << endl << tmp;
}



int main() {
    int  n;

    cin >> n;
    vector<int> R(n);
    for(int i = 0;i < n;i++)cin >> R.at(i);

    bubblesort(R, n);



}
