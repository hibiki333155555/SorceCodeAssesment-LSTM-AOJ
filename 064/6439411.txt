#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n;
  int a[n];

  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }

  long long min = a[0];
  long long max = a[0];
  long long sum = 0;
  for (int i = 0; i < n; i++) {
    if (min > a[i]) min = a[i];
    if (max < a[i]) max = a[i];
    sum = sum + a[i];
  }

  cout << min << " " << max << " " << sum << endl;

  return 0;

}
