int main(){
  ll n;cin >> n;
  vl ans(n);rep(i,n)cin >> ans[i];
  ll q;cin >> q;
  stack<pl> st;
  vl sig(n);rep(i,n)sig[i]=i;
  //vl ans(n);rep(i,n)ans[i]=i;
  while(q--){
    ll t;cin >> t;
    if(t==1){
      ll x,y;cin >> x >> y;x--;y--;
      st.push({x,y});
      swap(sig[x],sig[y]);
    }
    else if(t==2){
      auto [x,y]=st.top();st.pop();
      swap(sig[x],sig[y]);
    } 
    else{
      vvl dp(33,vl(n));
      dp[0]=sig;
      rep(x,32){
        rep(i,n){
          dp[x+1][i]=dp[x][dp[x][i]];
        }
      }
      /*rep(ff,3){
        rep(gg,n)cout << dp[ff][gg] <<" ";cout << endl;
      }*/
      ll k;cin >> k;
      vl nans(n);
      rep(i,n){
        ll now=i;
        per(x,33){
          if(k>>x&1){
            now=dp[x][now];
          }
        }
        nans[i]=ans[now];
      }
      swap(ans,nans);
      //cout << endl;
      //rep(i,n)cout << ans[i] <<" ";cout << endl;
    }
  }
  rep(i,n)cout << ans[i] <<" ";cout << endl;
}
