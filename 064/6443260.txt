#include <iostream>
using namespace std;

int main() {
   int n, min, max, sum=0;
   cin >> n;

   for(int i=0; i<n; i++) {
      int x = 0;
      cin >> x;

      if(i == 0) min = x;
      else if(min > x) min = x;
      
      if(i == 0) max = x;
      else if(max < x) max = x;

      sum += x;
   }

   cout << min << " " << max << " " << sum << "\n";
}
