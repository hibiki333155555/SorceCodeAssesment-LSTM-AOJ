#include<iostream>
using namespace std;
int main(){
    int A[3];
    cin >> A[0] >> A[1] >> A[2];
    int max=0;
    int min=101;
    for(int i=0; i<3; i++){
        if(A[i]>=max){
            max=A[i];
        }
        if(A[i]<=min){
            min=A[i];
        }
    }
    int sum=A[0]+A[1]+A[2];
    cout << sum-max-min << endl;
}
