#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;
#define drep(i, cc, n) for (ll i = (cc); i <= (n); ++i)
#define rep(i, n) drep(i, 0, n - 1)
#define all(a) (a).begin(), (a).end()
#define pb push_back
#define fi first
#define se second

const ll MOD = 1000000007;
const ll MOD2 = 998244353;
const ll INF = 1LL << 60;
const ll N_MAX = 2e5;

int main(){
    ll a, b;
    cin >> a >> b;

    if(a == 0 && b == 0){
        cout << 1 << endl;
        return 0;
    }else if(a == 0 || b == 0){
        cout << 0 << endl;
        return 0;
    }

    ll bit = max((ll)ceil(log2(a)), (ll)ceil(log2(b)));
    ll ans = 0;
    for(ll i=bit-1; i>=0; i--){
        if((a>>i) == (b>>i)){
            ans++;
        }else{
            break;
        }
    }

    cout << ans << endl;
}
