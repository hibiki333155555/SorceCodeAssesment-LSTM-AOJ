#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int ans[n],sum=1;
    int k[n],p[n];
    for(int i=0;i<n;i++) cin>>k[i]>>p[i];
    for(int i=0;i<n;i++){
        for(int j=1;j<k[i];j++){
            sum++;
            if(sum==p[i]+1)sum=1;
            //cout<<i<<" "<<sum<<endl;
        }
        ans[i]=sum;
        sum=1;
    }
    for(int i=0;i<n;i++){
        cout<<ans[i]<<endl;
    }
}
