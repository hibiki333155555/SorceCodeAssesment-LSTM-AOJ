typedef long long ll;
typedef unsigned long long ull;
const ll maxn = 1e2 +7;
const ll maxm = 2e5 +7;
const ll inf = 0x3f3f3f3f;
const ll mod = 1e9 +7;

int n;
int row[maxn], col[maxn];
int dp[maxn][maxn], p[maxn][maxn];

int DP(int l, int r) {
    if(l == r) return 0;
    if(dp[l][r]) return dp[l][r];
    int mn = inf;
    int pos = l;
    for(int k = l; k < r; k++) { // split at k [l,k] (k,r]
        int v = DP(l,k) + DP(k+1,r) + row[l] * col[k] * col[r];
        if(v < mn) {
            mn = v;
            pos = k;
        }
    }
    p[l][r] = pos;
    return dp[l][r] = mn;
}
