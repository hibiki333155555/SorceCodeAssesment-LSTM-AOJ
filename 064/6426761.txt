#include <iostream>
#include <vector>

using namespace std;

void f(vector<int>& A, const vector<vector<pair<int, int>>>& C, int c, int p, int d)
{
	if (A[c] < d) A[c] = d;
	for (auto [n, w] : C[c]) if (n != p) f(A, C, n, c, d + w);
}

int main()
{
	int n; cin >> n;
	vector<vector<pair<int, int>>> C(n);
	vector<int> A(n, 0);
	for (int i = 1, s, t, w; i < n; ++i) {
		cin >> s >> t >> w;
		C[t].push_back({s, w});
		C[s].push_back({t, w});
		auto As = A[s], At = A[t];
		f(A, C, s, t, At + w);
		f(A, C, t, s, As + w);
	}
	for (int a : A) cout << a << endl;
	return 0;
}
