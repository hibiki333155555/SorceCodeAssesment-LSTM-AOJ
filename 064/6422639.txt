    int binserch(vector<int>& vec, int key){
        int l = 0, r = vec.size();
        while(r - l > 1){
            int mid = (r + l) / 2;
            if(vec[mid] < key){
                l = mid;
            }else{
                r = mid;
            }
        }
        return l;
    }
