#include<bits/stdc++.h>
#include<cmath>
#include<numeric>
using namespace std;

int main(){

    int score[1000]={},n;
    double ans;
    
    while(1){

        double h=0,b=0;
        cin>>n;
        for(int i=0;i<n;++i)cin>>score[i];
        if(n==0)break;

        h=accumulate(score,score+n,0)/n;

        for(int i=0;i<n;++i){
            b+=pow(score[i]-h,2);
        }

        ans=sqrt(b/n);

        cout<<fixed<<setprecision(10)<<ans<<endl;
    }

    return 0;
}
