#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main()
{
	int nV, nE; cin >> nV >> nE;
	vector<pair<int, vector<int>>> C(nV, {0, vector<int>()});
	for (int s, t; --nE >= 0; C[s].second.push_back(t), ++C[t].first) cin >> s >> t;
	vector<int> A;
	queue<int> S;
	for (int n = 0; n < (int) C.size(); ++n) if (C[n].first == 0) S.push(n);
	for (; !S.empty(); S.pop()) {
		int n = S.front();
		A.push_back(n);
		for (int i : C[n].second) if (--C[i].first == 0) S.push(i);
	}
	for (int a : A) cout << a << endl;
	return 0;
}
