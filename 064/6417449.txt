#include <stdio.h>

int main()
{
	int i, N, K, A, num[7] = {};
	scanf("%d %d", &N, &K);
	for (i = 1; i <= N; i++) {
		scanf("%d", &A);
		num[A]++;
	}
	
	if (num[1] <= K) {
		num[5] += num[1];
		K -= num[1];
		num[1] = 0;
	} else {
		num[5] += K;
		num[1] -= K;
		K = 0;
	}
	if (num[2] <= K) {
		num[6] += num[2];
		K -= num[2];
		num[2] = 0;
	} else {
		num[6] += K;
		num[2] -= K;
		K = 0;
	}
	if (num[3] <= K) {
		num[6] += num[3];
		K -= num[3];
		num[3] = 0;
	} else {
		num[6] += K;
		num[3] -= K;
		K = 0;
	}
	if (num[4] <= K) {
		num[6] += num[4];
		K -= num[4];
		num[4] = 0;
	} else {
		num[6] += K;
		num[4] -= K;
		K = 0;
	}
	if (num[5] <= K) {
		num[6] += num[5];
		K -= num[5];
		num[5] = 0;
	} else {
		num[6] += K;
		num[5] -= K;
		K = 0;
	}
	
	int sum = 0;
	for (i = 1; i <= 6; i++) sum += num[i] * i;
	printf("%d\n", sum);
	fflush(stdout);
	return 0;
}
