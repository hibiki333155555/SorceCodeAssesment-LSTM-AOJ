#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)
#define Inf 1000000001

int main(){
	
	int n;
	cin>>n;
	vector<int> a(n);
	rep(i,n)cin>>a[i];
	
	vector<int> t;
	rep(i,n){
		t.push_back(a[i]+i);
	}
	sort(t.begin(),t.end());
	long long ans = 1;
	rep(i,a.size()){
		ans *= t[i] - i;
		ans %= 998244353;
	}
	cout<<ans<<endl;
	
	return 0;
}
