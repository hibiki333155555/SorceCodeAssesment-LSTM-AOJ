#include <iostream>
using namespace std;

void call(int n);
int main()
{
    // void型関数call、引数はint値n
    // 呼び出されたらi=1で初期化
    // 変数xにiの値を入れる
    // xが３の倍数の時はiをそのまま出力、終了判定を行う
    // 終了判定:iをインクリメントし、それがn以下ならばまた3の倍数か判定
    // 3の倍数でなければxが3で終わる数か判定:真ならばその時のiの値を出力、終了判定を行う
    // 偽ならばx/=10し、xが0でなければまた3で終わる値か調べなおす

    int n;
    cin >> n;
    call(n);
    return 0;
}

void call(int n)
{
    for (int i = 1; ++i <= n;)
    {
        int x = i;
        if (x % 3 == 0)
        {
            cout << " " << i;
            continue;
        }
        do
        {
            if (x % 10 == 3)
            {
                cout << " " << i;
                break;
            }
            x /= 10;
        } while (x);
    }

    cout << endl;
}
