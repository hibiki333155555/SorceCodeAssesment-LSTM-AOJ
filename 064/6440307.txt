#include <bits/stdc++.h>
using namespace std;
void printKmin(int arr[], int n, int k)
{
	int j,min;
	vector<int>v;
    for (int i = 0; i <= n - k; i++)
	{
		min = arr[i];
    for (j = 1; j < k; j++)
		{
			if (arr[i+j]<min)
				min = arr[i+j];
		}
	v.push_back(min);
	}
	//cout<<" ";
	for(int i=0;i<v.size();i++)
	cout<<v[i]<<" ";
	cout<<"\n";
}
int main()
{
	int n,k;
	cin>>n>>k;
	int a[n];
	for(int i=0;i<n;i++) cin>>a[i];
	printKmin(a, n, k);
	return 0;
}
