#include <bits/stdc++.h>
using namespace std;

int main() {
  
  ios::sync_with_stdio(false);
  cin.tie(0);
  
  int m,n;
  cin >> m >> n;
  vector<vector<char>> s(m,vector<char>(n));
  for (int i=0;i<m;i++) {
    for (int j=0;j<n;j++) {
      cin >> s[i][j];
    }
  }
  vector<int> ows(n,0);
  vector<int> ohs(m,0);
  for (int i=0;i<m;i++) {
    for (int j=0;j<n;j++) {
      if (s[i][j] == 'o') {
        ows[j]++;
        ohs[i]++;
      }
    }
  }
  int oa = 0;
  int ob = 0;
  int ans = 1020304050;
  for (int i=0;i<n;i++) {
    ob += ows[i];
  }
  for (int i=0;i<n-1;i++) {
    oa += ows[i];
    ob -= ows[i];
    // yoko
    ans = min(ans,oa+(n-1-i)*m-ob);
    ans = min(ans,(i+1)*m-oa+ob);
  }
  oa = 0;
  ob = 0;
  for (int i=0;i<m;i++) {
    ob += ohs[i];
  }
  for (int i=0;i<m-1;i++) {
    oa += ohs[i];
    ob -= ohs[i];
    // tate
    ans = min(ans,oa+(m-1-i)*n-ob);
    ans = min(ans,(i+1)*n-oa+ob);
  }
  cout << ans << endl;
}
