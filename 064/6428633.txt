#include <iostream>
#include <list>
using namespace std;

int *a;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    int temp[1000];
    a = temp;
    for(int i = 0; i <n;i++){
        cin>>*a;
        a++;
    }

    a--;
    for(int i = 0; i < n-1; i++){
        cout<<*a <<' ';
        a--;
    }

    cout<<*a;

    return 0;
}
