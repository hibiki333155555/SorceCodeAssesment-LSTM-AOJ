#include <stdio.h>

int main(void){
    int n;
    int x, i;
    
    scanf("%d", &n);
    
    while(i <= n){
        x = i;
        // CHECK_NUM
        if(x % 3 == 0) printf(" %d", x);
        // INCLUDE3
        do {
            if(x % 10 == 3) printf(" &d", x);
        } while(x);
        
        i++;
    }
}
