#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main(void){
    int N; cin >> N;
    vector<int> R(N);
    for (int i = 0; i < N; i++) cin >> R[i];
    
    int max_profit = R[1] - R[0];
    int minr = R[0];
    for (int t = 1; t < N; t++) {
        max_profit = max(max_profit, R[t] - minr);
        minr = min(R[t], minr);
    }
    
    cout << max_profit << endl;
    
}
