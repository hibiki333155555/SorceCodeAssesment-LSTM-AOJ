#include <bits/stdc++.h>
using namespace std;
void print (vector<int>v){
    for(int i=0;i<v.size();i++){
       cout<<v[i]<<" ";
        
    }
     cout<<"\n";
}


void bubblesort (vector <int>&arr){
    int n= arr.size();
    int count=0;
    for(int i=0;i<n-1
    
    ;i++){
        for(int j=n-1;j>i;j--){
            if (arr[j-1]>arr[j]){
                swap(arr[j-1],arr[j]);
                count++;
                
        }
        
    }
    
    }
    print(arr);
    cout<<count;
     cout<<"\n";
    
}

int main(){
    int n;
    cin>>n;
    vector<int>v(n);
    for(int i=0; i<n;i++){
        cin>>v[i];
       
    }
     bubblesort(v);
      return 0;
}
