#include<iostream>

using namespace std;

int main()
{
    int n;
    int a[n];

    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    for (int i = 0; i < n/2; i++) {
        int t = a[i];
        a[i] = a[n-i-1];
        a[n-i-1] = t;
    }

    for (int i = 0; i < n; i++) {
        if (i != n-1) {
            printf("%d ", a[i]);
        } else {
            printf("%d\n", a[i]);
        }
    }
}
