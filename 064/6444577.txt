#include <iostream>
using namespace std;

int main()
{
    int i, n, a;
    long min = 1000000, max = -1000000;
    long long sum = 0;
    
    cin >> n;
    
    for ( i = 1; i <= n; i++ ) {
        cin >> a;
        sum += a;
        
        if ( a < min ) {
            min = a;
        }
        
        if ( a > max ) {
            max = a;
        }
    }
    
    cout << min << " " << max << " " << sum << endl;
    return 0;
}
