#include <iostream>

using namespace std;

int bubble_sort(int A[],int n)
{
    int flag=1;
    int num=0;
    
    while(flag)
    {
        flag=0;
        
        for(int j=0;j<n;j++)
        {
            if(A[j]<A[j-1])
            {
                swap(A[j],A[j-1]);
                flag=1;
                num++;
            }
        }
    }
    
    for(int i=0;i<n;i++)
    {
        printf("%d",A[i]);
        if(i%2==1)printf(" ");
    }printf("\n");
    
    return num;
}

int main(void)
{
    int n;
    scanf("%d",&n);
    int A[n];
    
    for(int i=0;i<n;i++)
        scanf("%d",&A[i]);
        
    printf("%d\n",bubble_sort());
}
