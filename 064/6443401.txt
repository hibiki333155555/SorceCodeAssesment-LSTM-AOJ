#include <iostream>
using namespace std;

int main() {
   int n, i=1;
   cin >> n;

   while(true) {
      int x = i;
      int flag = 0;

      if(x % 3 == 0) {
         cout << " " << i;

         if(i >= n) break;
         else i++;
      }else{
         while(flag==0 && x!=0) {
            if(x % 10 == 3) {
               cout << " " << i;
               flag = 1;

               if(i >= n) break;
               else i++;
            }
            x /= 10;
         }

         if(i >= n) break;
         else if(flag == 0) i++;
      }
   }
}
