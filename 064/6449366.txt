#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

bool bit(int x, int pos) { return (x & (1 << pos)) != 0; }

int main() {
    int a, b;
    cin >> a >> b;

    int i = 30;
    while (i > 0 && !bit(a, i) && !bit(b, i))i--;

    int ret = 0;
    for (; i >= 0; i--) {
        if (bit(a, i) == bit(b, i))ret++;
        else break;
    }

    cout << ret << endl;
    return 0;
}
