#include<bits/stdc++.h>
using namespace std;

#define MAX 1000

int lcs(string X, string Y){
    int c[MAX + 1][MAX + 1];
    int m = X.length();
    int n = Y.length();
    int maxl = 0;
    X = ' ' + X;
    Y = ' ' + Y;
    for(int i = 0; i <= m; i++){
        c[i][0] = 0;
    }
    for(int i = 0; i <= n; i++){
        c[0][i] = 0;
    }
    for(int i = 1; i <= m; i++){
        for(int j = 1; j <= n; j++){
            if(X[i] == Y[j]){
                c[i][j] = c[i - 1][j - 1] + 1;
            }
            else if(c[i - 1][j] >= c[i][j - 1]){
                c[i][j] = c[i - 1][j];
            }
            else{
                c[i][j] = c[i][j - 1];
            }
            maxl = max(maxl, c[i][j]);
        }
    }

    return maxl;
}


int main(){
    string X, Y;
    int q;
    cin >> q;
    for(int i = 0; i < q; i++){
        cin >> X >> Y;
        cout << lcs(X, Y) << endl;
    }
}
