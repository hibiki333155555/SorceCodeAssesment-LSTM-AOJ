using namespace std;

using ll=long long;

int main(){
  ll M,N;
  cin>>M>>N;
  vector<string> S(M);
  for(int i=0;i<M;i++){
    cin>>S[i];
  }
  vector<ll> X(M),Y(N);
  for(int i=0;i<M;i++){
    for(int j=0;j<N;j++){
      if(S[i][j]=='o'){
        X[i]++;
        Y[j]++;
      }
    }
  }
  vector<ll> XX(M+1),YY(N+1);
  for(int  i=0;i<M;i++){
    XX[i+1]=XX[i]+X[i];
  }
  for(int i=0;i<N;i++){
    YY[i+1]=YY[i]+Y[i];
  }
  ll ans=N*M;
  for(int i=1;i<M;i++){
    ans=min(ans,min(XX[i]+(M-i)*N-(XX[M]-XX[i]),i*N-XX[i]+XX[M]-XX[i]));
  }
  for(int i=1;i<N;i++){
    ans=min(ans,min(YY[i]+(N-i)*M-(YY[N]-YY[i]),i*M-YY[i]+YY[N]-YY[i]));
  }
  cout<<ans<<endl;
  return 0;
}
