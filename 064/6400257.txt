#include <iostream>
using namespace std;

int main(){
    int r, c, a[101][101], sum = 0;
    for(int i = 0; i < 101; i++){
        for(int j = 0; j < 101; j++) a[i][j] = 0;
    }
    cin >> r >> c;
    for(int i = 0; i < r; i++){
        for(int j = 0; j < c; j++){
            cin >> a[i][j];
            sum += a[i][j];
            a[i][c] += a[i][j];
            a[r][j] += a[i][j];
        }
        
    }
    a[r][c] = sum;
    
    for(int i = 0; i <= r; i++){
        for(int j = 0; j <= c; j++){
            if(j) cout << " ";
            cout << a[i][j];
        }
        cout << endl;
    }
    
    
    return 0;
}
