#include <iostream>
#include <iomanip>

int main() {
  double x0, y0, x1, y1;
  std::cin >> x0 >> y0 >> x1 >> y1;
  double denom = (x1-x0)*(x1-x0) + (y1-y0)*(y1-y0);
  int q; std::cin >> q;
  for (int i=0; i<q; i++) {
    double x2, y2; std::cin >> x2 >> y2;
    double norm_x = ((x1-x0)*(x2-x0) + (y1-y0)*(y2-y0)) / denom;
    double norm_y = ((x2-x0)*(y0-y1) + (x1-x0)*(y2-y0)) / denom;
    if (norm_y > 0.0) {
      std::cout << "COUNTER_CLOCKWISE" << std::endl;
    } else if (norm_y < 0.0) {
      std::cout << "CLOCKWISE" << std::endl;
    } else if (norm_x < 0) {
      std::cout << "ONLINE_BACK" << std::endl;
    } else if (norm_x > 1) {
      std::cout << "ONLINE_FRONT" << std::endl;
    } else if (norm_x < 1) {
      std::cout << "ON_SEGMENT2" << std::endl;
    } else {
      std::cout << "UNKNOWN" << std::endl;
    }
  }
}
