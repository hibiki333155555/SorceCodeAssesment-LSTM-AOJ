#include <iostream>
#include <vector>
#include <string>

using namespace std;

int judge(int n) {
    vector <long long>numb(n);
    int a;
    for (int i = 0; i < n; ++i) {
        cin >> a;
        if (a % 2 != 0) return 0;
        numb.at(i) = a;
    }
    int cnt=0;
    for (int i = 0;; ++i) {
        for (int j = 0; j < n; ++j) {
            if (numb.at(j) % 2 != 0) return cnt;
            numb.at(j) /= 2;
           
        }
        cnt += 1;
        
    }
    
}
int main()
{
    int n;
    cin >> n;
    cout << judge(n);

}
