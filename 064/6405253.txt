#define MAX_LINE 1000

int* splitstr(char* str,int *n){ 
    char* s=(char*)malloc(sizeof(char)*2);
    int* ret=(int*)malloc(sizeof(int)*(*n));
    char c=' ';
    sprintf(s,"%c",c);
    int cn=0; for(char* p=strtok(str,s);p;p=strtok(NULL,s)){
        ret[cn]=strtol(p,NULL,10);
        cn++; 
    }
    *n=cn;
    s=NULL;
    return ret; 
}
