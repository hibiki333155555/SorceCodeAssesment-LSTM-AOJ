#include<bits/stdc++.h>    
using namespace std;
int main()
{  
int a[10][10],b[10][10],mul[10][10],c,d,e;    
cin>>c>>d>>e;
for(int i=0;i<c;i++)    
{    
 for(int j=0;j<d;j++)    
 {    
 	cin>>a[i][j];
  }    
}    
for(int i=0;i<d;i++)    
{    
 for(int j=0;j<e;j++)    
 {    
 	cin>>b[i][j];
 }    
}    
for(int i=0;i<c;i++)    
{    
 for(int j=0;j<e;j++)    
 {    
  mul[i][j]=0;    
  for(int k=0;k<d;k++)    
   {    
    mul[i][j]+=a[i][k]*b[k][j];    
   }    
 }    
}    
for(int i=0;i<c;i++)    
{    
 for(int j=0;j<e;j++)    
 {    
 	cout<<mul[i][j]<<" ";
 }    
 cout<<endl;  
}    
return 0;  
}
