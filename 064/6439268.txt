#include<iostream>

using namespace std;

int main() {
    int S;
    int h, m, s;
    m = 0;
    cin >> S;
    s = S % 60;

    if (S > 60) {
        while (S > 60) {
            m += 1;
            S = S / 60;
        }

    }

    if (m > 60) {
        while (m > 60) {
            h += 1;
            m = m / 60;
        }
    }
    cout << h << ":" << m << ":" << s << endl;
}
