#include <bits/stdc++.h>
using namespace std;

int main() {
    int n; cin >> n;
    vector<int> a(n);
    for(int i = 0;i < n; i++) cin >> a[i];
    // 5 2 4 6 1 3

    for(int i = 0;i < n; i++) {
        int v  = a[i]; // v = 5
        int j  = i - 1; // j == -1
        while(j >= 0 && a[j] > v) {
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = v; // a[0] = 5
        for(int k = 0;k < n - 1;k++) {
            cout << a[k] << " ";
        }
        cout << a[n-1] << endl; 
    }
}
