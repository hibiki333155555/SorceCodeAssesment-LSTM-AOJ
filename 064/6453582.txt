#include <iostream>
using namespace std;

int main() {
    while(1) {
        string s; cin>>s;
        if(s=="-") break;
        int m; cin>>m;
        while(m--) {
            int h; cin>>h;
            string f="",b="";
            for(int i=0;i<s.size();i++) {
                if(i<h) f+=s[i];
                else b+=s[i];
            }
            s=b+f;
        }
        cout<<s<<endl;
    }
    return 0;
}
