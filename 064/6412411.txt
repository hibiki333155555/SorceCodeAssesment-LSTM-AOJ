#include<bits/stdc++.h>
using namespace std;

int main() {
	int n,m,l;
	cin>>n>>m>>l;
	int r1=n,c1=m,r2=m,c2=l;
	vector<vector<int>>v1(r1,vector<int>(c1));
	for(int i=0;i<r1;i++){
		for(int j=0;j<c1;j++){
			cin>>v1[i][j];
		}
	}
	vector<vector<int>>v2(r2,vector<int>(c2));
	for(int i=0;i<r2;i++){
		for(int j=0;j<c2;j++){
			cin>>v2[i][j];
		}
	}
	if(c1==r2){
	vector<vector<int>>v(r1,vector<int>(c2));
	for(int i=0;i<r1;i++){
		for(int j=0;j<c2;j++){
			v[i][j]=0;
			for(int k=0;k<r2;k++){
					v[i][j]=v[i][j]+(v1[i][k]*v2[k][j]);
			}
		}	
	}
	for(int i=0;i<r1;i++){
		for(int j=0;j<c2;j++){
			cout<<v[i][j]<<" ";
		}
	
		cout<<"\n";
	}
	}
	return 0;
}
