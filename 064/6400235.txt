#include<stdio.h>

int main(void)
{
    int n;
    scanf("%d\n",&n);
    int x,min,max;
    long long sum=0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        if(i==0||x<min)
        {
            min=x;
        }
        if(i==0||x>max)
        {
            max=x;
        }
        sum+=x;
    }
    printf("%d %d %lld\n",min,max,sum);
    return 0;
}
