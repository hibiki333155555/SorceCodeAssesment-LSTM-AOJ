#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    while(n!=0){
        int k,ans=0,x=0;
        for(int i=0;i<n;i++){
            cin>>k;
            if(k!=0) ans++;
            x=max(k,x);
        }
        if(ans<2&&n!=1||k<2) cout<<"NA"<<endl;
        else if(n==1) cout<<2<<endl;
        else if(ans==2||ans==n) cout<<ans+1<<endl;
        else cout<<ans<<endl;
        cin>>n;
    }
    return(0);
}
