#include <iostream>
using namespace std;

int A[400000];
int N;

bool is_sorted() {
    for (int i = 1; i < N; i++) {
        if (A[i-1] > A[i]) {
            return false;
        }
    }
    return true;
}

int main() {
    cin >> N;
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }

    if (is_sorted()) {
        cout << 0 << endl;
        return 0;
    }

    int x, y, Q; cin >> Q;
    int count = 0;
    for (int i = 0; i < Q; i++) {
        cin >> x >> y;
        x--; y--;
        // swap
        int temp = A[x];
        A[x] = A[y];
        A[y] = temp;

        count++;

        if (is_sorted()) {
            cout << count << endl;
            return 0;
        }
    }

    cout << -1 << endl;

    return 0;
}
