#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
#include <cmath>
#include <stdio.h>
#include <string.h>
#define max 1000000007
using namespace std;



int main(){
  long  long int n,r=0,i,j,l,tmp=0,flag=0,count=0,c=1,m=0;
  cin>>n;
  long long int a[n];
  for(i=0;i<n;i++){
     cin>>a[i];
     c*=2;
  }
  cin>>m;
  long long int b[m];
  for(i=0;i<m;i++){
      cin>>b[i];
  }
  for(i=0;i<m;i++){
      flag=0;
      for(j=0;j<c;j++){
          tmp=0;
          for(l=0;l<n;l++){
             if((j&(1<<l))==0){
                tmp+=a[l];
             }
          }
         
          if(tmp==b[i]){
              flag=1;
              cout<<"yes";
              j=c;
          }
        
      }
      if(flag==0){
          cout<<"no";
      }
      putchar('\n');
  }
  
  
  
  
 
  return 0;
}
