#include <iostream>
#include <string>
using namespace std;
int main() {
	int N;
	cin >> N;
	int A[202002];
	for (int i = 0; i < N; i++)cin >> A[i];
	int count = 0;
	for (int i = 0; i < N; i++) {
		for (int j = N - 1; j >= i + 1; j--) {
			if (A[j] < A[j - 1]) {
				count++;
				swap(A[j], A[j - 1]);
			}
		}
	}
	for (int i = 0; i < N; i++)cout << i+1 << " ";
	cout << " " << endl;
	cout << count << endl;
  
}
