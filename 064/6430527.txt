#include <bits/stdc++.h>
using namespace std;

void print(vector<int> v,int n,int c)
{
	for(int i=0;i<n;i++)
	{
		if(i<n-1)
		cout<<v[i]<<" ";
		else
		cout<<v[i]<<"\n"<<c<<"\n";
	}
}

void bubblesort(vector<int> v,int n)
{
	int count=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if (v[j] < v[j-1])
			{
             swap (v[j],v[j-1]);
			count++	;
			}
		}
	}
	print(v,n,count);
}

int main() {
	int n;cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++)
	{
		cin>>v[i];
	}
	bubblesort(v,n);
	return 0;
}
