#include <bits/stdc++.h>

#define ll long long
#define ld long double
#define rep(i,a,b) for (int i = a; i < b; i++)
#define irep(i,a,b) for (int i = a; i > b; i--)

using namespace std;

int main(){
    int n,m;
    cin >> n >> m;
    vector<vector<int>> A(n,vector<int>(m));
    vector<int> b(m);

    rep(i,0,n) rep(j,0,m) cin >> A[i][j];
    rep(i,0,m) cin >> b[i];

    int sum = 0;
    rep(i,0,n){
        sum = 0;
        rep(j,0,m){
            sum += A[i][j] * b[j];
        }
        cout  << sum << endl;
    }

    // system("pause");
    return 0;
}
