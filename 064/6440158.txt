#include <stdio.h>
#include <string.h>

#define MIN(a, b) ((a) > (b)?(b):(a))

char S[1000001];
char SR[1000001];

int main(void)
{
    int N;
    int n;
    int i;
    
    scanf("%d%s", &N, S);
    for (i = N - 1; i >= 0; i--){
        SR[N - 1 - i] = S[i];
    }
    SR[N] = '\0';
    n = 0;
    for (i = 0; i < N - 1; i++){
        if (S[i] == S[i + 1]){
            if (i + 1 < N - (i + 1)){
//printf("<%d>", N - (i + 1) - (i + 1));
                if (S[0] == S[(i + 1) * 2 - 1]){
                    if (strncmp(S, &SR[N - (i + 1) - (i + 1)], i + 1) == 0){
                        n++;
                    }
                }
            }
            else {
                if (SR[0] == SR[(N - (i + 1)) * 2 - 1]){
                    if (strncmp(&S[(i + 1) - (N - (i + 1))], SR, N - (i + 1)) == 0){
                        n++;
                    }
                }
            }
        }
    }
    
    printf("%d\n", n);
    
    return (0);
}
