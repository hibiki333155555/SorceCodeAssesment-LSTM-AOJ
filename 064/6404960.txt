#include <bits/stdc++.h>
using namespace std;

int main(){
    
   while(1){ 
    int H,W;
    cin>> H>> W;
    if(H !=0 && W !=0){
        for(int i=0; i<H; i++){
            cout<<endl;
            for(int j=0; j<W; j++){
                if(i==0 || i==H-1 || j==0 || j==W-1){
                cout<<'#';
                }
                else{
                cout<< '.';
                }    
            }
        }
    }
    else{
        break;
    } 
    cout<<endl;
   } 
}
