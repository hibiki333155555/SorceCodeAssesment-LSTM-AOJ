signed main(){
    using P=pair<int,int>;
    int N;cin>>N;
    map<ll, P> mp;
    rep(i,N){
        rep(j,N){
            ll x;cin>>x;
            if(mp.count(x)) chmin(mp[x], P(i, j));
            else mp[x]=P(i, j);
        }
    }
    int Q;cin>>Q;
    while(Q--){
        ll t;cin>>t;
        if(mp.count(t)){
            auto [i, j]=mp[t];
            cout<<i+1<<" "<<j+1<<"\n";
        }else{
            cout<<-1<<"\n";
        }
    }
    return 0;
}
