import java.util.Scanner;
class Main{

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int a = sc.nextInt();
		int b = sc.nextInt();
		int c = sc.nextInt();
		int d = sc.nextInt();
		if(a==c && b==d) {
			System.out.println(1);
		}else{
			if(a==b && c==d) {
			System.out.println(1);
		}else {
		    if(a==d && b==c) {
			System.out.println(1);
		}else{
			System.out.println(0);
		}
		}
		}
	}
}
