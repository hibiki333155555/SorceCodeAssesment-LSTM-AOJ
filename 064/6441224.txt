#include "iostream"
#include "algorithm"
#include "string"

std::string convertToBinary(int x)
{
  if (x == 0)
    return "0";
  std::string ans;
  while (x != 0)
  {
    ans.push_back(x % 2 + '0');
    x /= 2;
  }
  std::reverse(begin(ans), end(ans));
  return ans;
}

int main()
{
  int a, b;
  std::cin >> a >> b;

  std::string alpha = convertToBinary(a);
  std::string beta = convertToBinary(b);
  while (alpha.length() > beta.length())
  {
    beta.insert(begin(beta), '0');
  }
  while (alpha.length() < beta.length())
  {
    alpha.insert(begin(alpha), '0');
  }

  int ans = 0;
  for (int i = 0; i < alpha.size(); ++i)
  {
    if (alpha[i] == beta[i])
    {
      ans++;
    }
    else
    {
      break;
    }
  }

  std::cout << ans << std::endl;
  return 0;
}
