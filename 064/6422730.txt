#include <iostream>
using namespace std;
int main(void){
    // Your code here!
    while(1)
    {
        char op;
        int a,b;
        cin >> a >> op >> b;
        
        if(op == '?') break;
        
        else if(op == '+')
        {
            int t;
            t=a+b;
            cout << t << "\n";
        }
        
        else if(op == '-')
        {
            int t;
            t=a-b;
            cout << t << "\n";
        }
        
        else if(op == '*')
        {
            int t;
            t=a*b;
            cout << t << "\n";
        }
        
        else if(op == '/')
        {
            int t;
            t=a/b;
            cout << t << "\n";
        }
    }
    
}
