#include <bits/stdc++.h>

using namespace std;

int bubble_sort(vector<int>& xs) {
  int n = xs.size();
  int num_operation = 0;
  for (int i = 0; i < n - 1; ++i) {
    for (int j = n - 1; j >= i + 1; --j) {
      if (xs[j - 1] > xs[j]) {
        num_operation++;
        swap(xs[j - 1], xs[j]);
      }
    }
  }
  return num_operation;
}

int main() {
  int N;
  cin >> N;
  vector<int> xs(N);
  for (auto& x : xs) cin >> x;

  int num_operation = bubble_sort(xs);
  
  for (int i = 0; i < N; ++i) {
    if (i == N - 1) cout << xs[i] << endl;
    else cout << xs[i] << ' ';
  }

  cout << num_operation << endl;
  return 0;
}
