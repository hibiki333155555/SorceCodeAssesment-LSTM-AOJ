#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    int n,p,t; cin>>n>>t;
    vector<int>v[n];
    while(t--)
    {
       int x; cin>>x;
       if(x==0)
       {
           int p,k; cin>>p>>k;
           v[p].push_back(k);
       }
       else if(x==1)
       {
           int p; cin>>p;
           for(int i=0; i<v[p].size(); i++)
           {
               if(i!=v[p].size()-1) cout<<v[p][i]<<" ";
               else cout<<v[p][i];
           }cout<<endl;
       }
       else if(x==2)
       {
           int p; cin>>p;
           v[p].clear();
       }
    }
}
