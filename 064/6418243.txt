N, K = map(int, input().split())
A = list(map(int, input().split()))

maxi = {1:5, 2:6, 3:6, 4:6, 5:6, 6:5}
used = [False]*N

sum = 0

for i in range(1, 6):
    for j in range(N):
        if used[j]==False:
            if K>0 and A[j]==i and A[j]!=6:
                used[j] = True
                K -= 1
                sum += maxi[A[j]]
for i in range(N):
    if used[i]==False:
        if K>0 and A[i]==1:
            K -= 1
            sum += 6
        else:
            sum += A[i]

print(sum)
