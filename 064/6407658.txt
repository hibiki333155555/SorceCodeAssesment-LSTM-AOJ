#include <iostream>
#include <vector>

using namespace std;

void trace(vector<int>& A){
    for(auto x : A){
        cout << x;
    }
    cout << endl;
}

void insertionSort(vector<int>& A, int N){
    int v;
    for(int i = 1; i < N; i++){
        v = A[i];
        int j = i - 1;
        while(j>=0 && v < A[j]){
            A[j + 1] = A[j];
            j--;
        }
        A[j + 1] = v;
        trace(A);
    }
}

int main()
{
    ios::sync_with_stdio(false);
    int N, i, j;
    vector<int> A;
    cin >> N;
    for(i = 0; i < N; i++){
        int x;
        cin >> x;
        A.push_back(x);
    }
    
    trace(A);
    insertionSort(A, N);

    return 0;
}
