import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        String w = scan.next().toLowerCase();

        String str = "";
        int count = 0;
        while (scan.hasNext()){
            str = scan.nextLine();
            if (str.equals("END_OF_TEXT")){
                break;
            }else {
                str = str.toLowerCase();
            }
            while (str.contains(w)){
                str = str.substring(str.indexOf(w) + w.length());
                count++;
            }
        }

        System.out.println(count);
    }
}
