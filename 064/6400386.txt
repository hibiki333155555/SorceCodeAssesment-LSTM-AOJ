#include<stdio.h>


void output(int);



main()
{
	int h, m, s, s1, s2, s3;
	while (1) {
		scanf("%d %d %d", &h, &m, &s);
		if (h == -1 && m == -1 && s == -1)
		{
			break;
		}
		s1 = (120 * 60) - (h * 3600 + m * 60 + s);
		s2 = s1 * 3;
		output(s1);
		output(s2);
	}
	return 0;
}

void output(int s) {
	int a, b, c, d, e, f;
	a = (s / 3600) / 10;
	b = (s / 3600) % 10;
	c = (s % 3600 / 60) / 10;
	d = (s % 3600 / 60) % 10;
	e = (s % 60) / 10;
	f = (s % 60) % 10;

	printf("%d%d:%d%d:%d%d\n", a, b, c, d, e, f);
}
