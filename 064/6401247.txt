#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int n;
int arr[4][3][10] = {0};

int main(){
  cin >> n;
  for (int i = 0; i < n ; i++){
    int b, f, r, v;
    cin >> b >> f >> r >> v;
    arr[b-1][f-1][r-1] += v;
  }

  for (int i = 0; i < 4; i++){
    for (int j = 0; j<3; j++){
      for (int k = 0; k<10; k++){
        if (k != 9) cout << arr[i][j][k] << ' ';
        else cout << arr[i][j][k];
      }
      cout << endl;
    }

    if (i != 3) cout << "###################" << endl;
  }
}
