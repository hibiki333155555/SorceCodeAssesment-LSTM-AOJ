const ll maxm = 2e5 +7;
const ll inf = 0x3f3f3f3f;

int n, ans;
int d[maxn], a[maxn][maxn], v[maxn];

signed main() {
    #ifdef moyi_qwq
        freopen("D:/source file/intxt/in.txt","r",stdin);
    #endif
    
    cin>>n;
    for(int i = 1; i <= n; i++) {
        d[i] = inf;
        for(int j = 1; j <= n; j++) {
            cin>>a[i][j];
        }
    }
    d[1] = 0;
    v[1] = 1;
    int p = 1;
    int qwq = 1;
    while(1) {
        for(int i = 1; i <= n; i++) {
            if(a[p][i] == -1) continue;
            if(v[i]) continue;
            d[i] = min(d[i], d[p] + a[p][i]);
        }
        int mn = inf;
        p = -1;
        //for(int i = 1; i <= n; i++) cout<<d[i]<<(i==n?"\n":" ");
        for(int i = 1; i <= n; i++) {
            if(v[i] == 0 && mn > d[i]) {
                mn = d[i];
                p = i;
            }
        }
        //cout<<p<<" "<<d[p]<<endl;
        if(p == -1) break;
        v[p] = 1;
        ans += d[p];
        d[p] = 0;
    }
    cout<<ans<<endl;
