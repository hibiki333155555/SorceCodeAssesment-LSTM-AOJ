#include <iostream>
#include <queue>
using namespace std;
struct Task{
public:
    int remain;
    string name;
    Task(string name, int rem):name(name), remain(rem){
    }
};

int main(){
    int n, s, sum=0;
    cin >> n >> s;
    queue<Task> q;
    while(n--){
        string name;
        int rem;
        cin >> name >> rem;
        q.push(Task(name, rem));
    }
    int t=s;
    while(q.size()){
        Task cur = q.front();
        q.pop();
        if(t >= cur.remain){
            sum += cur.remain;
            cout << cur.name << ' ' << sum << endl;
            t = s;
        }
        else {
            cur.remain-=t;
            sum += t;
            q.push(cur);
            t = s;
        }
    }
    return 0;
}
