vector<ll> cnt(1000,0);

void solve(){
  ll N,M,v,a,b;
  cin>>N>>M>>v>>a>>b;
  v%=M;
  a%=M;
  b%=M;
  if(N>M){cout<<"Yes"<<endl; return;}
  for(int i=0;i<M;i++){cnt[i]=0;}
  for(int i=1;i<=N;i++){
    if(cnt[v]){cout<<"Yes"<<endl; return;}
    cnt[v]++;
    v*=a; v+=b;
    v%=M;
  }
  cout<<"No"<<endl;
}

int main(){
  cin.tie(0);
  ios::sync_with_stdio(false);

  ll T;
  cin>>T;
  while(T--){solve();}


  return 0;
}
