#include<iostream>
#include<cstdio>
#include<string>
#include<cmath>
#include<algorithm>
#include<map>
#include<vector>
using namespace std;


int main() {
	int n;
	cin >> n;
	n++;
	int u, k;
	int adjL[100][101];
	for (int i = 1;i < n;i++) {
		scanf("%d %d ", &u, &k);
		adjL[i][0]= k;
		for (int j = 1;j <= k;j++) {
			scanf("%d", &adjL[u][j]);

		}
	}

	vector<vector<bool>> adjM(n, vector<bool>(n,false));
	int m;
	for (int i = 0;i < n;i++) {
		m = adjL[i][0];
		for (int j = 1;j <= m;j++) {
			adjM[i][adjL[i][j]] = true;
		}
	}
	for (int i = 1;i < n;i++) {
		if (i-1)printf("\n");
		for (int j = 0;j < n;j++) {
			if (j)printf(" ");
			if (adjM[i][j])	printf("1");
			else printf("0");
		}
	}

	return 0;
}
