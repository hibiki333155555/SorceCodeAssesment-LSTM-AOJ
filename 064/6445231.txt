import java.util.*;

public class Main {
    public static void main(String[] args){
        Scanner scan = new Scanner(System.in);
        int cards = scan.nextInt();
        int tScore = 0, hScore = 0;
        
        for (int i = 0; i < cards; i++) {
        	String tCard = scan.next().toLowerCase();
        	String hCard = scan.next().toLowerCase();
//        	花子が勝者の場合
        	if (tCard.compareTo(hCard) < 0) {
				hScore += 3;
//        	太郎が勝者の場合
			} else if (tCard.compareTo(hCard) > 0) {
				tScore += 3;
			} else {
				tScore += 1;
				hScore += 1;
			}
        }
        System.out.println(tScore + " " + hScore);
    }
}
