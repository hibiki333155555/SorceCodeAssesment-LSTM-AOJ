#include<iostream>
#include<string>
#include<stack>
using namespace std;
int main(){
    string temp;
    stack<int> st;
    while(cin.peek()!='\n'){
        cin>>temp;
        if(temp=="+"){
            int a=st.top();
            st.pop();
            int b=st.top();
            st.pop();
            st.push(a+b);
        }
        else if(temp=="*"){
            int a=st.top();
            st.pop();
            int b=st.top();
            st.pop();
            st.push(a*b);
        }else if(temp=="-"){
            int a=st.top();
            st.pop();
            int b=st.top();
            st.pop();
            st.push(b-a);
        }else{
            st.push(atoi(temp.c_str()));
        }
    }
    cout<<st.top()<<endl;
    return 0;
}
