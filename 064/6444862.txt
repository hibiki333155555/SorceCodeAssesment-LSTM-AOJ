#include <bits/stdc++.h>
using namespace std;
struct Point{
    int x,y;
    Point operator-(Point &b) {return {x-b.x,y-b.y};}
    double len() {return sqrt(x*x+y*y);}
};
double cross(Point a,Point b) {return a.x*b.y-a.y*b.x;}
double dot(Point a,Point b) {return a.x*b.x+a.y*b.y;}
int main()
{
    while (1)
    {
        Point p[405];
        int s,vis[405]={0},flag=1;
        scanf("%d",&s);
        if (s==0) break;
        for (int i=0;i<s;i++) scanf("%d%d",&p[i].x,&p[i].y);
        Point a={0,-1},b={0,0};
        double ans=0;
        while (flag)
        {
            flag=0;
            double c=-2;
            int next;
            for (int i=0;i<s;i++)
            {
                if (vis[i]||cross(b-a,p[i]-a)>0) continue;
                double tmp=dot(b-a,p[i]-b)/(b-a).len()/(p[i]-b).len();
                if (tmp>c||(tmp==c&&(p[i]-b).len()<(p[next]-b).len()))
                    next=i,c=tmp,flag=1;
            }
            vis[next]=1;
            a=b,b=p[next];
            ans+=(b-a).len();
        }
        printf("%.1lf\n",ans);
    }
}
