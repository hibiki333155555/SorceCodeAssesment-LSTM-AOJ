#include <bits/stdc++.h>
using namespace std;

int main() {
  int H,W;

  while (1){
  
  cin >> H >> W;

  if(W == 0 && H == 0){
    break;
  }
  
  for(int i = 1;i <= H ;i++){
    for(int j = 1;j <= W;j++){
      if(i % 2 == 1){
        if(j % 2 == 1){
          cout << "#";
        }else if(j % 2 == 0){
          cout << ".";
        }
      }else if(i % 2 == 0){
        if(j % 2 == 0){
          cout << "#";
        }else if(j % 2 == 1){
          cout << ".";
        }
      }
    }
    cout << endl;
  }
  cout << endl;
}

  return 0;
}
