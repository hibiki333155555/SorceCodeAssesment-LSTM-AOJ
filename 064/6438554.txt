import math
n = int(input())
a = []
for i in range(n):
    s = input().split()
    a.append(s)
def PrintNonExistetCard(type):
    for i in range(1, 14):
        ref = [j for j in a if j == [type, f'{i}']]
        if(len(ref) < 1):
            print(f'{type} {i}')
PrintNonExistetCard('S')
PrintNonExistetCard('H')
PrintNonExistetCard('C')
PrintNonExistetCard('D')
