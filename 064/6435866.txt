#include <bits/stdc++.h>
using namespace std;

int isPrime(int N) {
    int count = 0;
    bool primeFlag = true;
    for(int i = 2;i * i <= N;i++) {
        primeFlag = true;
        if(N % i == 0)primeFlag = false;
    }
    if(primeFlag) count++;
    return count;
}

int main() {
    int n; cin >> n;
    int number;
    for(int i = 0;i<n;i++) {
        cin >> number;
        count += isPrime(number);
    }
    cout << count << endl;
}
