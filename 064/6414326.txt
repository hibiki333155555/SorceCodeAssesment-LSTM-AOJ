#include <iostream>
using namespace std;

int bubbleSort(int A[], int N){
    int x = 0;
    for(int i = 0; i < N - 1; i++){
        for(int j = N - 1; j >= i + 1; j--){
            if(A[j] < A[j - 1]){
                swap(A[j], A[j - 1]);
                x++;
            }
        }
    }
    return x;
}

int main(){
    int N, A[100], x;
    for(int i = 0; i < N; i++) cin >> A[i];
    x = bubbleSort(A[], N);
    for(int i = 0; i < N; i++){
        if(i) cout << " ";
        cout << A[i];
    }
    cout << "\n" << x << endl; 
    return 0;
}
