fscanf(STDIN, '%d', $N);
$tmp = explode(' ', trim(fgets(STDIN)));

$flag = 1;
$count = 0;
for ($i=0; $flag ; $i++) { 
    $flag = 0;
    for ($j=$N-1; $j > $i; $j--) { 
        if($tmp[$j] < $tmp[$j-1]){
            $hoge = $tmp[$j-1];
            $tmp[$j-1] = $tmp[$j];
            $tmp[$j] = $hoge;
            $flag = 1;
            $count++;
        }
    }
}

}
