#include <algorithm>
#include <iostream>
#include <iterator>
#include <vector>
#include <numeric>
using namespace std;


int main()
{
    int n, m, l;
    cin >> n>>m>>l;
    auto mat_nm = vector<vector<long>>(n, vector<long>(m)),
         mat_ml = vector<vector<long>>(l, vector<long>(m));
    for(int i = 0; i < n; i++)
        for(int j = 0; j < m ;j++)
            cin >> mat_nm[i][j];

    for(int i = 0; i < m; i++)
        for(int j = 0; j < l ;j++)
            cin >> mat_ml[j][i];

    auto mat_nl = vector<vector<long>>(n, vector<long>(l));
    for(int i = 0; i < n; i++)
        for(int j = 0; j < l ;j++)
            mat_nl[i][j] = inner_product(mat_nm[i].begin(), mat_nm[i].end(), mat_ml[j].begin(), 0L);

    for(int i = 0; i < n; i++)
    {
        for(int j = 0; j < l-1; j++)
            cout << mat_nl[i][j] << " ";
        cout << mat_nl[i][l-1];
        cout << endl;
    }
    return 0;
}
