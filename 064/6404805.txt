#include <cstdio>
#include <cmath>

int main() {
    double PI = acos(-1);
    double a,b,C;
    scanf("%lf %lf %lf", &a, &b, &C);
    double rad = C*PI/180;
    double m = a*b*sin(rad)/2;
    double c = sqrt(a*a+b*b-2*a*b*cos(rad));
    double h = b*sin(rad);
    printf("%.4lf\n%.4lf\n%.4lf\n", m, a+b+c, h);
}
