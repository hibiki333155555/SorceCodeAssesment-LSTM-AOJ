#include<iostream>
#include<stdio.h>
using namespace std;
int main() {
    char ch;
    while(1) {
        scanf("%c", &ch);
        if(ch == '\n') {
            break;
        }
        if(isalpha(ch)) {
            if(isupper(ch)) {
                ch = ch + 32;
                printf("%c", ch);
            } else if(islower(ch)) {
                ch = ch - 32;
                printf("%c", ch);
            }
        } else {
            printf("%c", ch);
        }
    }
}
