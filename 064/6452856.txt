#include <iostream>
#include <vector>
using namespace std;

int main() {
    while(1) {
        int n,x,ans=0; cin>>n>>x;
        if(n==0&&x==0) break;
        for(int i=1;i<=n-2;i++) {
            for(int j=i+1;j<=n-1;j++) {
                if(x-i-j>=j+1&&x-i-j<=n) ans++;
                else if(x-i-j>n) continue;
                else break;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
