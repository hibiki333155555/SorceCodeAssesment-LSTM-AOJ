// #pragma GCC target("avx2")
#pragma GCC optimize("O3")
// #pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    int n;
    cin >> n;
    vector<string> s(n);
    for (int i = 0; i < n; ++i) {
        cin >> s[i];
    }
    int ans = 0;
    for (int i = 0; i < n; ++i) {
        bool f = false;
        for (int j = 0; j < n; ++j) {
            if (i == j) continue;
            int c = 0;
            for (int k = 0; k < (int)s[i].size(); ++k) {
                c += s[i][k] != s[j][k];
            }
            if (c <= 1) {
                f = true;
                break;
            }
        }
        if (f) {
            ++ans;
        }
    }
    cout << ans << '\n';
    return 0;
}
