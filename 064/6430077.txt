#include <iostream>
#include <math.h>
using namespace std;
int main(){
    while(1){
        int a;
        cin>>a;
        if(a==0)return(0);
        else{
        int f[a],k=0,p=0;
        for(int i=0;i<a;i++){
            cin>>f[i];
            if(f[i]>=2)k=1;
        }
        if(k==0)cout<<"NA"<<endl;
        else{
            for(int i=0;i<a;i++){
                if(f[i]!=0)p++;
            }
            cout<<p+1<<endl;
        }
    }
    }
}
