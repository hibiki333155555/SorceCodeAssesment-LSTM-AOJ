#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int r,c;
vector<vector<int> > v;

int main(){
  cin >> r >> c;
  for (int i = 0; i < r; i++){
    vector<int> tmp;
    for (int j = 0; j < c; j++){
      int input;
      cin >> input;
      tmp.push_back(input);
    }
    v.push_back(tmp);
  }

  for (int i = 0; i < r; i++){
    int sum = 0;
    for (int j = 0; j < c + 1; j++){
      if(j == c) cout << sum << endl;
      else {
        cout << v[i][j] << ' ';
        sum += v[i][j];
      }
    }
  }
  int total = 0;
  for (int j = 0; j < c + 1; j++){
    if(j == c) cout << total << endl;
    else {
      int sum = 0;
      for (int i = 0; i < r; i++){
        sum += v[i][j];
      }
      cout << sum << ' ';
      total += sum;
    }
  }
}
