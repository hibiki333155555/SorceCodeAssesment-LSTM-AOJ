#include <iostream>
using namespace std;

int main()
{
    in i, j;
    int n;
    
    cin >> n;
    
    int a[n];
    
    for ( i = 0, i < n, i++ ) {
        cin >> a[i];
    }
    
    for ( j = n-1, i >= 0, j-- ) {
        if ( j == n-1 ) {
            cout << a[j];
        } else {
            cout << " " << a[j];
        }
    }
    
    cout <<endl;
    return 0;
}
