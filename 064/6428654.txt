#include <iostream>
#include <list>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    int a[100];
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
    }

    for (int i = n-1; i > 0; i--)
    {
        cout << a[i] << ' ';
    }

    cout << a[0];

    return 0;
}
