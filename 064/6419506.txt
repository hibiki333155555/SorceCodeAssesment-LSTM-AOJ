#include <iostream>
#include <string>

const int distance = 'A' - 'a';

int main()
{
    int result[26] = { 0 };

    std::string str;
    
    while ( std::getline(std::cin, str) ) {
        for (int i = 0;i < str.length();++i) {
            if ('A' <= str[i] && str[i] <= 'Z')
                str[i] += distance;
    
            int index = str[i] - 'a';
            if (0 <= index && index < 26)
                ++result[index];
        }
    }

    for (int i = 0;i < 26;++i) {
        std::string buff;

        buff += 'a' + i;
        buff += " : ";
        std::cout << buff << result[i] << std::endl;
    }

    return 0;
}
