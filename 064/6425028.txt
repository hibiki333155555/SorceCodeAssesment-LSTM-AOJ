#include <stdio.h>

int main(void){
    int i,j;
    
    int ctr[2][4]={0};
    float sight[2]={0};
    
    while(scanf("%f %f",&sight[0],&sight[1])!=EOF){
        for(i=0;i<=1;i++){
            if(sight[i]>=1.1){
                ctr[i][0]++;
            }
            else if(sight[i]<1.1&&sight[i]>=0.6){
                ctr[i][1]++;
            }
            else if(sight[i]<0.6&&sight[i]>=0.2){
                ctr[i][2]++;
            }
            else if(sight[i]<0.2){
                ctr[i][3]++;
            }
        }
    }
    
    for(i=0;i<4;i++){
        for(j=0;j<2;j++){
            printf("%d",ctr[j][i]);
            if(j==0) printf(" ");
        }
        puts("");
    }
    
    return 0;
}
