#include<bits/stdc++.h>
using namespace std;
int main(){
    int hy,hm,hd,ty,tm,td; cin>>hy>>hm>>hd>>ty>>tm>>td;
    if(hy==ty){
        if(hm==tm){
            if(hd==td){
                cout<<0<<endl;
            }else cout<<1<<endl;
        }else{
            cout<<1<<endl;
        }
    }else if(hy<ty){
        if(hm==tm){
            if(hd==td){
                cout<<ty-hy<<endl;
            }else if(hd<td){
                cout<<ty-hy+1<<endl;
            }else if(td<hd){
                cout<<ty-hy<<endl;
            }
        }else if(hm<tm){
            cout<<ty-hy+1<<endl;
        }else if(tm<hm){
            cout<<ty-hy<<endl;
        }
    }else if(ty<hy){
        if(tm==hm){
            if(hd==td){
                cout<<hy-ty<<endl;
            }else if(td<hd){
                cout<<hy-ty+1<<endl;
            }else if(hd<td){
                cout<<hy-ty<<endl;
            }
        }else if(tm<hm){
            cout<<hy-ty+1<<endl;
        }else if(hm<tm){
            cout<<hy-ty<<endl;
        }
    }
    return 0;
}
