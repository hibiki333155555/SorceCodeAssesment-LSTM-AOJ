/* function */
template<class T>
T GCD(T a, T b) {
	if (b != 0) return GCD(b, a % b);
	return a;
}
template<class T>
T LCM(T a, T b) {
	return a / GCD(a, b) * b;
}

const int dy[4] = { 0, 1, -1, 0 };
const int dx[4] = { 1, 0, 0, -1 };

int main(){
	int n,k,tu=0;
	cin >> n >> k;
	vector<int> A(n),cnt(6);
	rep(i,0,n) {
		cin >> A[i];
		cnt[A[i]-1]++;
		if(A[i]==1) tu+=2;
		else tu++;
	}
	if(tu<=k) {
		cout << 6*n << endl;
		return 0;
	}
	ll ans=0,index=0;
	bool flag=false;
	sort(ALL(A));
	rep(i,0,n){
		if(A[i]==1){
			ans+=5;
			k--;
			if(k<=0){
				index=i+1;
				break;
			}
			continue;
		}else if(A[i]!=6){
			ans+=6;
			k--;
			if(k<=0){
				index=i+1;
				break;
			}
			continue;
		}
		flag=true;
		ans+=6;
	}
	if(flag) cout << ans << endl;
	else {
		for(int i=index;i<n;i++){
			ans+=A[i];
		}
		cout << ans << endl;
	}
}
