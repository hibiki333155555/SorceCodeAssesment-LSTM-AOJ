#include <iostream>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

int check(int i, int n) {
   if(++i > n) return 1;
}

int main() {
   cin.tie(0);
   ios::sync_with_stdio(false);

   int n, i=1;
   cin >> n;
   // n = 30;

   while(true) {
      int x = i;
      int m = 0;

      if(x % 3 == 0) {
         cout << " " << i;
         if(check(i, n) == 1) break;
         else i++;
      }else{
         while(x!=0) {
            if(x % 10 == 3) {
               cout << " " << i;
               m = 1;
               if(check(i, n) == 1) break;
               else i++;
            }
            x /= 10;
         }
         if(check(i, n) == 1) break;
         else if(m == 0) i++;
      }
   }
}
