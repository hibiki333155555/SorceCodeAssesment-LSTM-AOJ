#include <iostream>
#include <string>
using namespace std;

int main() {
    string a;
    string b;
    cin >> a >> b;
    int distance = 0;
    int arr[a.size()][b.size()];
    for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            arr[i][j] = 0;
        }
    }
    for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            if (a.at(i) == b.at(j)) {
                //if (arr[i-1][j-1] != 0) arr[i-1][j-1] = arr[i-2][j-2];
                ++arr[i][j];
            }
        }
    }
    for (size_t i = 1; i < a.size(); i++) {
        for (size_t j = 1; j < b.size(); j++) {
            int min = i;
            if (j < i) min = j;
            for (size_t k = 1; k <= min ; k++) {
                if (arr[i][j] == 1 && arr[i-k][j-k] >= 1) {
                    arr[i][j] += arr[i-k][j-k];
                    break;
                }
            }
        }
    }
    distance = arr[0][0];
    for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            if (arr[i][j] > distance) {
                distance = arr[i][j];
            }
        }
    }
    if (a.size() > b.size()) distance = a.size() - distance;
    else distance = b.size() - distance;
    /*for (size_t i = 0; i < a.size(); i++) {
        for (size_t j = 0; j < b.size(); j++) {
            cout << arr[i][j] << " ";
        }
        cout << endl;
    }*/
    cout << distance << endl;
    return 0;
}
