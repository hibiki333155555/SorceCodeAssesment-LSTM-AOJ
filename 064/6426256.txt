#include<stdio.h>

int main(){
    int max, min, n, tem;
    double sum;
    
    scanf("%d",&n);
    
    for(int i = 0; i < n; i++){
        scanf("%d",&tem);
        sum += tem;
        if(i == 0)max = min = tem;
        if(max < tem)max = tem;
        if(min > tem)min = tem;
    }
    printf("%d %d %.0lf\n", min, max, sum);
    return 0;
}
