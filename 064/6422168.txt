#include <bits/stdc++.h>
using namespace std;

int main(){
  int n,m,l;
  cin >> n >> m >> l;
  vector<vector<int>> a(n,vector<int>(m)), b(m,vector<int>(l));
  
  for(int i=0; i<n; i++){
    for(int j=0; j<m; j++) cin >> a.at(i).at(j);
  }
  for(int i=0; i<m; i++){
    for(int j=0; j<l; j++) cin >> b.at(i).at(j);
  }
  
  for(int i=0; i<n; i++){
    for(int j=0; j<l; j++){
      long long x=0;
      for(int k=0; k<m; k++){
        x += a.at(i).at(k)*b.at(k).at(j);
      }
      cout << x;
      if(j<l-1) cout << " ";
      else cout << endl;
    }
  }
  return 0;
}
