int main()
{	
	int n,sum=0,i,j;
	int q;
	
	cin>>n;
	
	for(i=0;i<n;i++)
	{
		cin>>q;
		
		for(j=floor(sqrt(q));j>=2;j--)
			if(q%j==0)
				break;
		if(j==1)
			sum++;
	
	}
	
	
	
	cout<<sum<<endl;
	
	
return 0;
}
