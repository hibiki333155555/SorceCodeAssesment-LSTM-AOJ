#include <bits/stdc++.h>
#define rep(i, n) for (int i = 0; i < n; ++i)
typedef long long ll;
using namespace std;

void solve(int N) {
    vector<int> A(N);
    rep(i, N) cin >> A[i];

    int ans = 0;
    rep(i, N - 3) {
        if (A[i] == 2 && A[i + 1] == 0 && A[i + 2] == 2 && A[i + 3] == 0)
            ans++;
    }
    cout << ans << "\n";
}

int main() {
    cin.tie(0);
    ios_base::sync_with_stdio(false);

    int N;
    while (cin >> N, N)
        solve(N);

    return 0;
}
