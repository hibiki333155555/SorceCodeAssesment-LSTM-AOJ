#include <iostream>
#include <list>

using namespace std;
int main(void)
{
    list<int> A;
    list<int>::iterator itr = A.begin();
    
    int d, x;
    int op;
    
    int n;
    cin >> n;
    for( int i = 0; i < n; i++)
    {
        cin >> op;
        switch( op)
        {
            case 0:
                // insert
                cin >> x;
                A.insert(itr, x);
                ++itr;
                break;
            case 1:
                // move
                cin >> d;
                if(d > 0)
                {
                    for(int j = 0; j < d; j++) itr++;
                }
                else
                {
                    for(int j = 0; j > d; j--) itr--;
                }
                break;
            case 2:
                // erase
                A.erase(itr);
                --itr;
                break;
        }
    }
    for(auto itr = A.begin(); itr != A.end(); ++itr) 
    {
        std::cout << *itr << "\n";
    }
    
}
