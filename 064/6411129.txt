#include <bits/stdc++.h>

using namespace std;

int main(){
    int n;
    cin>>n;
    vector<tuple<int,int,char,long long,string>> t;
    int a,b;
    long long d;
    char c;
    string e;
    while(n--){
        cin>>a>>b>>c>>d>>e;
        t.push_back(make_tuple(a,b,c,d,e));
        
    }
    sort(t.begin(), t.end());
    for(int i = 0; i<t.size(); i++){
        cout << get<0>(t.at(i))<<" "<< get<1>(t.at(i))<<" " << get<2>(t.at(i))<<" "<< get<3>(t.at(i))<<" "<< get<4>(t.at(i))<<endl;
    }
    return 0;
}
