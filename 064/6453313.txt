#include<stdio.h>
int main(void){
    int a[3];
    int t;
    scanf("%d %d %d",&a[0],&a[1],&a[2]);
    for(int i=0;i<2;i++){
        for(int j=1;j<3;j++){
         if(a[i]>a[j]){
             t=a[j];
             a[j]=a[i];
             a[i]=t;
         }
        }
    }
    printf("%d %d %d\n",a[0],a[1],a[2]);
    return 0;
}
