template<class A,class...B>void pr(const A &a,const B&...b){cout<<a<<(sizeof...(b)?" ":"");pr(b...);}
template<class A>void PR(A a,ll n){rep(i,n)cout<<(i?" ":"")<<a[i];ln;}
const ll MAX=1e9+7,MAXL=1LL<<61,dx[8]={-1,0,1,0,-1,-1,1,1},dy[8]={0,1,0,-1,-1,1,1,-1};
typedef pair<ll,ll> P;
typedef pair<ll,P> PP;

void Main() {
  ll n,m;
  cin >> n >> m;
  string s[n];
  rep(i,n) R s[i];
  ll d[n][m];
  rep(i,n)rep(j,m) d[i][j]=MAX;
  priority_queue<PP,vector<PP>,greater<PP>> que;
  rep(i,n) {
    rep(j,m) {
      if(s[i][j]=='S') {
        d[i][j]=0;
        que.push(PP(0,P(i,j)));
      }
    }
  }
  while(!que.empty()) {
    PP p=que.top();que.pop();
    ll x=p.S.F,y=p.S.S,c=p.F;
    if(s[x][y]=='G') {
      pr(c);
      return;
    }
    if(d[x][y]<c) continue;
    if(s[x][y]=='.'||s[x][y]=='S') {
      rep(i,4) {
        ll xx=x+dx[i],yy=y+dy[i];
        if(check(n,m,xx,yy)&&s[xx][yy]!='#'&&d[xx][yy]>c+1) {
          d[xx][yy]=c+1;
          que.push(PP(c+1,P(xx,yy)));
        }
      }
    } else {
      ll xx=x,yy=y;
      if(s[x][y]=='U') xx--;
      else if(s[x][y]=='D') xx++;
      else if(s[x][y]=='L') yy--;
      else yy++;
      if(check(n,m,xx,yy)&&s[xx][yy]!='#'&&d[xx][yy]>c) {
        d[xx][yy]=c;
        que.push(PP(c,P(xx,yy)));
      }
    }
  }
  pr(-1);
}
