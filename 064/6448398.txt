#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    string p;
    cin>>p;
    int sum=0;
    while(true){
        cin>>s;
        if(s == "END_OF_TEXT"){
            break;
        }
        for(int i=0;i<s.size();i++){
            if('A'<=s[i] && s[i]<='Z'){
                s[i] += 'a' - 'A';
            }
        }
        if(s == p){
            sum++;
        }
    }
    cout<<sum<<"\n";
    return(0);
}
