#include <stdio.h>

int p(int a){
    if(a>=0) return a;
    else return 3;
}

int main(void){
    double x[4],y[4],n,xA,xB,yA,yB;
    int pre,main,next,i,flg;
    
    while(1){
        flg=0;
        for(i=0;i<4;i++){
            if(scanf("%lf,%lf,",&x[i],&y[i])==EOF) goto finish;
        }
        
        for(i=0;i<4;i++){
            pre=p(i-1);
            main=i;
            next=(i+1)%4;
            xA=x[pre]-x[main];
            yA=y[pre]-y[main];
            xB=x[next]-x[main];
            yB=y[next]-y[main];
            n=xA*xB+yA*yB;
            //printf("%d,%d,%d,xA:%lf,yA:%lf,xB:%lf,yB:%lf,naiseki:%lf\n",pre,main,next,xA,xB,yA,yB,n);
            if(n<0){flg=1;}
        }
        if(flg==1) puts("NO");
        else if(flg==0) puts("YES");
    }
    
    finish:
    
    return 0;
}
