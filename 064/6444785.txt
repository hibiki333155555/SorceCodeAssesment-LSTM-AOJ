#include <iostream>

template <typename T>
T powmod(T a, T n, T mod) {
    T result = 1;
    for (; n > 0; n >>= 1, a = (a * a) % mod) {
        if (n & 1) result = (result * a) % mod;
    }
    return result;
}

int main() {
    int m, n; std::cin >> m >> n;
    std::cout << powmod(m,n,1000000007) << std::endl;
}
