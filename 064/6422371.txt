#include <iostream>

using namespace std;

int H, W;
char maze[105][105];
int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};

void dfs(int x, int y) {
  char tmp = maze[x][y];
  maze[x][y] = ' ';
  for (int i = 0; i < 4; ++i) {
    int nx = x + dx[i], ny = y + dy[i];
    if (0 <= nx && nx < H && 0 <= ny && ny < W && tmp == maze[nx][ny]) {
      dfs(nx, ny);
    }
  }
}

void solve() {
  int ans = 0;
  for (int i = 0; i < H; ++i) {
    for (int j = 0; j < W; ++j) {
      if (maze[i][j] != ' ') {
        ++ans;
        dfs(i, j);
      }
    }
  }
  cout << ans << endl;
}

int main() {
  while (cin >> H >> W && H != 0) {
    for (int i = 0; i < H; ++i) {
      for (int j = 0; j < W; ++j)
        cin >> maze[i][j];
    }
    solve();
  }
}
