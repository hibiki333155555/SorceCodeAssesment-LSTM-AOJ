#define _GLIBCXX_DEBUG
#include<iostream>
#include<stdio.h>
using namespace std;
    
    void selectionSort(int N,int A[]){
        int cnt=0,min_idx;
        
        for(int i=0;i<N;i++){
            min_idx=i;
            
            for(int j=i;j<N;j++)
                if(A[min_idx] > A[j])
                    min_idx = j;
            
            if(min_idx != i){
                swap(A[min_idx],A[i]);
                cnt++;
            }
        }
        
        for(int i=0;i<N;i++){
            if(i!=0)
                printf(" ");
            printf("%d",A[i]);
        }
        
        printf("\n");
        printf("%d\n",cnt);
        
    }
    
    int main(){
        int N;
        int A[100] = {};
        
        scanf("%d",&N);
        for(int i=0;i<N;i++)
            scanf("%d",&A[i]);
        
        selectionSort(N,A);
        
        return 0;
        
    }
