int main() {
    int n;
    cin >> n;
    std::vector<int> vec(n);
    std::vector<int> x(n);
    for (int i = 0; i < n; i++)cin >> vec.at(i);

    for(int i = 0;i < n;i++){
        std::vector<int> a(n);
        for(int j = i + 1;j < n;j++){
            a.at(j) = vec.at(j);
        }
        sort(a.begin(), a.end());
        x.at(i) = a.at(n - 1) - vec.at(i);
    }
sort(x.begin(), x.end());
    cout << x.at(n - 1) << endl;

}
