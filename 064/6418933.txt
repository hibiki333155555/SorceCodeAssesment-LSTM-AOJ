#include <cctype>
#include <iostream>
int main() {
    char ch;
    while (true) {
        scanf("%c", &ch);
        if (ch == '\n') break;
        if (islower(ch)) {
            printf("%c", toupper(ch));
        } else if (isupper(ch)) {
            printf("%c", tolower(ch));
        } else printf("%c", ch);
    }
    return 0;
}
