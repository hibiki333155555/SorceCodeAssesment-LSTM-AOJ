
import java.util.*;
public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		ArrayList<Integer> arr = new ArrayList<>();
		for(int i=0;i<N;i++) {
			int a = sc.nextInt();
			if(a == 0) {
				int b = sc.nextInt();
				arr.add(b);
			}
			if(a == 1) {
				int b = sc.nextInt();
				System.out.println(arr.get(b));
			}
			if(a == 2) {
				arr.remove(arr.size()-1);
			}
		}

	}

}
