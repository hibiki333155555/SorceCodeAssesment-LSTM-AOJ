using namespace std;

long long pow2(long long x, long long n,long long MOD) {
  long long ret = 1;
  while (n > 0) {
    if (n & 1) ret = ret * x % MOD;  // n の最下位bitが 1 ならば x^(2^i) をかける
    x = x * x % MOD;
    n >>= 1;  // n を1bit 左にずらす
  }
  return ret;
}
int main() {
int m,n;
cin >>m>>n;
cout << pow2(m,n,1000000007) << endl;
return 0;
}
