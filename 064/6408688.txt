#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    int n, e;
    cin>> n;
    vector<int> v;
    while(n--){
        cin>>e;
        v.push_back(e);
    }
    cin>>n;
    int a, b;
    while(n--){
        cin>>a>>b;
        reverse(v.begin()+a, v.begin()+b);
    }
    
    for(int i = 0; i < v.size(); i++){
        cout << v.at(i) << ((i+1 == v.size()) ? "\n" : " ");
    }
    
    return 0;
}
