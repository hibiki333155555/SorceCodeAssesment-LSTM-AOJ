#include<iostream>
#include<string>
using namespace std;
int main() {
    string W;
    int count = 0;
    while(1) {
        string T;
        int n = T.size();
        cin >> T;
        if (T == "END_OF_TEXT") {
            break;
        } else {
            for(int num = 0; num < n; num++) {
                if(W == T) {
                    count++;
                    break;
                } else {
                    if(isupper(W[n])) {
                        W[n] = tolower(W[n]);
                        if(W != T) {
                            break;
                        }
                    } else {
                        W[n] = toupper(W[n]);
                        if(W != T) {
                            break;
                        }
                    }
                }
            }
        }
    }
    cout << count << endl;
}
