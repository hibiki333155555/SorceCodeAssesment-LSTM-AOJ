#pragma GCC target("avx2")
#pragma GCC optimize("O3")
#pragma GCC optimize("unroll-loops")
#include<bits/stdc++.h>
using namespace std;

using ll = long long;
const int INF = 1e9;
const ll inf = 1LL<<62;

void solve() {
  int n; cin >> n;
  vector<vector<ll>> a(n, vector<ll>(n));
  for (int i=0; i<n; i++) for (int j=0; j<n; j++) cin >> a[i][j];
  map<ll, pair<ll, ll>> m;
  for (int i=n-1; i>=0; i--) for (int j=n-1; j>=0; j--) m[a[i][j]] = {i+1, j+1};
  int q; cin >> q;
  while (q--) {
    ll t; cin >> t;
    if (m.count(t)) cout << m[t].first << " " << m[t].second << '\n';
    else cout << -1 << '\n';
  }
}

int main() {

  std::cin.tie(nullptr);

}
