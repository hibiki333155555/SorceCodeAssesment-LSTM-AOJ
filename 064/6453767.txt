#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    cout<<fixed<<setprecision(6);
    int n; cin>>n;
    vector<int> x(n),y(n);
    double p1=0.0,p2=0.0,p3=0.0,p_inf=0.0;
    for(int i=0;i<n;i++) cin>>x[i];
    for(int i=0;i<n;i++) cin>>y[i];
    for(int i=0;i<n;i++) {
        p1 += abs(x[i]-y[i]);
        p2 += pow(abs(x[i]-y[i]),2);
        p3 += pow(abs(x[i]-y[i]),3);
        p_inf = max(p_inf,(double)abs(x[i]-y[i]));
    }
    cout<<p1<<endl;
    cout<<sqrt(p2)<<endl;
    cout<<cbrt(p3)<<endl;
    cout<<p_inf<<endl;
    return 0;
}
