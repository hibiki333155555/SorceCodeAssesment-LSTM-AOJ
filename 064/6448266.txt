#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a,b,c,t;
    cin>>t;
    for(int i=0;i<t;i++){
        scanf("%d %d %d", &a, &b, &c );
        vector<int>sides={a,b,c};
        sort(sides.begin(), sides.end());
        if(pow(c,2)==pow(a,2)+pow(b,2)){
            cout<<"YES"<<endl;
        }
        else{
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
