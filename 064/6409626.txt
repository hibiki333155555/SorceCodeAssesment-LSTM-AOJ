#include <stdio.h>
int main(){
    int h, w; 
    while(1){
        scanf("%d%d", &h, &w);
        printf("%d\n", h);
        if (h==0 && w==0)break;
        for(int j=0; j < w;j++){
            for(int i=0; i < h; i++){
                putchar('#');
            }
            putchar('\n');
        }
        putchar('\n');
    }
    return 0;
}
