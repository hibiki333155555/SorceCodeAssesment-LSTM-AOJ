#include <iostream>
using namespace std;

long long modpow(long long a, long long b, long long m){
  long long p = a, ans = 1;
  for(int i=0 ; i<30 ; i++){
    if((b & (1 << i)) != 0){
      ans *= p;
      ans %= m;
    }
    p *= p;
    p %= m;
  }
  return ans;
}


int main(void){
  const long long mod = 1000000007;
  long long m, n;
  cin >> m >> n;
  cout << modpow(m, n, mod) << endl;
  return 0;
}
