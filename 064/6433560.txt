#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,n1;
	cin>>n;
	int count=0;
	vector<int> v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
	}
	cin>>n1;
	vector<int> v1(n1);
	for(int i=0;i<n1;i++){
		cin>>v1[i];
	}
	for(int i=0;i<n1;i++){
		for(int j=0;j<n;j++){
			if(v1[i]==v[j]){
				count++;
			}
		}
	}
	if(count==n1){
		cout<<"1";
	}
	else{
		cout<<"0";
	}
	cout<<"\n";
	return 0;
}
