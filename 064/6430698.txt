#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int n; cin >> n;
	vector<pair<int, int>> T(n, {0, 0});
	for (int i = 0, k, c; i < n; ++i) for (cin >> k; --k >= 0; T[c].first = i) cin >> c;
	int q, c, v, w;
	for (cin >> q >> c >> v; --q >= 0; cin >> c >> v) switch (c) {
	case 0: cin >> w; T[v].second += w; break;
	case 1: {
		int r = 0;
		for (int i = v; i >= 0; i = T[i].first) r += T[i].second;
		cout << r << endl;
	} break;
	}
	return 0;
}
