#include <iostream>
#include <vector>
#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int r,c,n;
	cin>>r>>c>>n;
	int k;int long long e=0;
	vector<vector<long long> > A(r,vector<long long> (c));
	vector<vector<long long> > B(c,vector<long long> (n));
	vector<vector<long long> > C(r,vector<long long> (n));
	for(int i=0;i<r;i++)
	{
		for(int j=0;j<c;j++)
		{
			cin>>A[i][j];
		}
	}
	for(int i=0;i<c;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>B[i][j];
		}
	}
	for(int i=0;i<r;i++)
	{
		for(int j=0;j<n;j++)
		{
			for(k=0;k<c;k++)
			{
				e=e+(A[i][k]*B[k][j]);
			}
			C[i][j]=e;
			e=0;
		}
	}
	for(int i=0;i<r;i++)
	{
		for(int j=0;j<n;j++)
		{
			cout<<C[i][j];
			if(j!=n-1)
				cout<<" ";
		}
		cout<<"\n";
	}
	return 0;
}
