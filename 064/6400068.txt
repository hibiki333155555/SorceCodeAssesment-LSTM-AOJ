#include<stdio.h>

int main(void)
{
    int n;
    int x,min,max;
    long long sum;
    scanf("%d\n",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        if(i==0||min>x)
        {
            min=x;
        }
        if(i==0||max<x)
        {
            max=x;
        }
        sum+=x;
    }
    printf("%d %d %.01d\n",min,max,sum);
    return 0;
}
