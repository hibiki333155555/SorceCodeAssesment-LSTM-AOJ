#include<stdio.h>

int main(void)
{
    int n;
    long x,min,max;
    long long  sum=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        if(i==0)
        {
            min=x;
            max=x;
        }
        if(max<x)
        {
            max=x;
        }
        if(min>x)
        {
            min=x;
        }
        sum=sum+x;
    }
    printf("%d %d %11d\n",min,max,sum);
    return 0;
}
