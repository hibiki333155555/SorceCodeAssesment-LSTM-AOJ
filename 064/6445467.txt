#include<stdio.h>
#include<ctype.h>
#include<string.h>

int main(void){
    int i,counter=0;
    char w[11],str[1000];
    
    scanf("%s",w);
    
    scanf("%s",str);
    
    while(strcmp(str,"END_OF_TEXT") != 0){
        for(i=0;str[i]=='\0';i++){
            str[i] = tolower(str[i]);
        }
        if(strcmp(str,w) == 0) counter++;
        scanf("%s",str);
    }
    
    printf("%d",counter);
    
    return 0;
}
