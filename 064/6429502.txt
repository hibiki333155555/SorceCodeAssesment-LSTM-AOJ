#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
#define rep(i,a,b) for(int i=a;i<b;i++)
#define fore(i,a) for(auto &i:a)

vector<vector<int> > dist;
vector<vector<int> > dp;
int INF = 1000000000;
int V, E;
int end_v;

int rec(int bit, int v) {
  if (bit==(1<<v)) {
    if (dist.at(end_v).at(v) != -1) return dist.at(end_v).at(v);
    else return INF;
  }
  
  int res = INF;
  int bit2 = bit & ~(1<<v);
  
  rep(i,0,V) {
    if ((bit2 & (1<<i)) && dist.at(i).at(v) != -1) {
      if (res > rec(bit2, i) + dist.at(i).at(v)) {
        res = rec(bit2, i) + dist.at(i).at(v);
      }
    }
  }
  
  return dp.at(bit).at(v) = res;
}

int main() {
  cin >> V >> E;
  
  dist.assign(V, vector<int>(V, -1));
  rep(i,0,E) {
    int s,t,d;
    cin >> s >> t >> d;
    dist.at(s).at(t) = d;
  }
  
  dp.assign((1<<V)+1, vector<int>(V,-1));
  
  int res = INF;
  rep(i,0,V) {
    end_v = i;
    res = min(res, rec((1<<V)-1, i));
  }
  
  if (res == INF) cout << -1 << endl;
  else cout << res << endl;
}
