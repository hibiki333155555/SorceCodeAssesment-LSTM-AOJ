#include <bits/stdc++.h>
using namespace std;

void outp (vector<int> vec ,int N){
    for ( int i = 0 ; i < N ; i++ ){
        if ( i == N - 1){
            cout << vec.at(i) << endl;
        }

        else{
            cout << vec.at(i) << ' ';
        }

}
return;
}

void insertionSort(vector<int> vec , int N){
    for( int i = 1 ; i < N ; i++ ){
        int v = vec.at(i) , j = i - 1;
        while ( j >= 0 && vec.at(j) > v ){
            vec.at( j + 1 ) = vec.at( j );
            j--;
        }
        vec.at( j + 1 ) = v;
        outp ( vec , N );
    }
    return;
}

int main() {
   int N; cin >> N;
   vector<int> vec(N);

   for ( int i = 0 ; i < N ; i++ ){
       cin >> vec.at(i);
   }
   outp ( vec , N );
   insertionSort ( vec , N );
}
