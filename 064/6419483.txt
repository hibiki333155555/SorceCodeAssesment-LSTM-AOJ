#include <iostream>
#include <string>
#include <cmath>

const int distance = 'A' - 'a';

int main()
{
    int result[26] = { 0 };
    
    std::string str;
    std::getline(std::cin, str);
    
    for (int i = 0;i < str.length();++i) {
        if ('A' <= str[i] && str[i] <= 'Z')
            str[i] += distance;
            
        int index = str[i] - 'a';
        ++result[index];
    }
    
    for (int i = 0;i < 26;++i) {
        std::string buff;
        
        buff += 'a' + i;
        buff += " : ";
        
        std::cout << buff << result[i] << std::endl;
    }
    
    return 0;
}
