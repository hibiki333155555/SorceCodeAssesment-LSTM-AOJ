#include <bits/stdc++.h>

const int distance = 'a' - 'A';

int main()
{
    int result[26] = { 0 };

    while ( true ) {
        std::string str;
        std::getline(std::cin, str);
        if (str == "")
            break;
        
        for (int i = 0;i < str.length();++i) {
            if ('A' <= str[i] && str[i] <= 'Z')
                str[i] += distance;
    
            int index = str[i] - 'a';
            if (0 <= index && index < 26)
                ++result[index];
        }
    }

    for (int i = 0;i < 26;++i) {
        std::string buff;

        buff += 'a' + i;
        buff += " : ";
        std::cout << buff << result[i] << std::endl;
    }

    return 0;
}
