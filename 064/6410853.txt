//int N, M, K, T, H, W, L, R;
long long int N, M, K, T, H, W, L, R;

void Solve() {
	int v, a, b;
	cin >> N >> M >> v >> a >> b;
	if (N > M) {
		cout << "Yes\n";
		return;
	}
	vector<int>w(1, v % M);
	for (int i = 1; i < N; i++) {
		w.push_back((w.back() * a + b) % M);
	}
	sort(w.begin(), w.end());
	for (int i = 1; i < w.size(); i++) {
		if (w[i - 1] == w[i]) {
			cout << "Yes\n";
			return;
		}
	}
	cout << "No\n";
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);

	cin >> T;
	while (T--) {
		Solve();
	}
}
