#include <iostream>
#include <set>
using namespace std;

int main(){
    set<int> s;
    int choice, x;
    int q, total = 0;
    cin >> q;
    for(int i=0; i<q; i++){
        cin >> choice >> x;
        if(choice == 0){
            if(!s.count(x)){
                s.insert(x);
                total++;
            }
            cout << total << endl;
        }
        else if(choice == 1){
            cout << s.count(x) << endl;
        }
        else if(choice == 2){
            s.erase(s.find(x));
            total--;
        }
    }
    return 0;
}
