using namespace std;
//挺复杂 
int smi(int x,int y)
{
	if(x<=y)
		return x;
	else
		return y;
}

int main()
{	
	int sc1=0,sc2=0,n,i,j,mi;
	string a,b;
	 
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a>>b;
		
		mi=smi(a.length(),b.length());
		for(j=0;j<mi;j++)
		{
			if(a[j]==b[j])
				continue;
			else if(a[j]<b[j])
			{
				sc2+=3;
				break;
			}
			else
			{
				sc1+=3;
				break;
			}
		}
		
		if(j==mi)
		{
			if(a.length()==b.length())
			{
				sc1++;sc2++;	
			}	
			else if(a.length()<b.length())
				sc2+=3;
			else if(a.length()>b.length())
				sc1+=3;
		}		
	
	}
	
	cout<<sc1<<" "<<sc2<<endl;
	
return 0;
}
