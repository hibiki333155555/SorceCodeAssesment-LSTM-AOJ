#include <iostream>
#include<queue>
#include <iomanip>
#include<math.h>

using namespace std;

/*
 配列Aの中のA[p]からA[r]の間で、A[r]よりも小さいものを左、大きいものを右に分ける（partation）
 
 */
int partition(vector<int>&A, int p, int r){
    int x = A[r];                       //配列の最後の数
    int i = p-1;                        //配列の指定の先頭ー１
    for (int j = p; j < r; j++){        //pからrまで処理
        if (A[j] <= x){
            i++;
            swap(A[i],A[j]);            //該当するものを左からどんどん詰めていく。
        }
    }
    swap(A[i + 1],A[r]);                //最後に基準となるA[r]そのものを詰める。
    return i+1;                         //A [r]の入れた場所を返す
    
}
int main(){
    //情報の入力
    int n;
    cin >> n;
    vector<int>A(n);
    for (int i = 0; i < n; i++) cin >> A[i];
    
    int k = partition(A, 0, n-1);
    
    for(int i = 0; i < n; i++){
      if(i) cout<<" ";
      if(i==k) cout<<"[";
      cout<<A[i];
      if(i==k) cout<<"]";
    }
    cout << endl;
    return 0;
}
