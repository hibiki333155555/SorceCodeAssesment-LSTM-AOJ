vvl dp;
vl v, w;

ll rec(ll i, ll j)
{
    ll res = 0;
    if (i == 0)
    {
        dp[i][j] = 0;
        return dp[i][j];
    }
    if (dp[i][j] >= 0)
        res = dp[i][j];
    else if (j < w[i])
        res = rec(i - 1, j);
    else
    {
        dp[i][j] = max(rec(i - 1, j), rec(i, j - w[i]) + v[i]);
        res = dp[i][j];
    }
    return res;
}

int main()
{
    ll n, W;
    cin >> n >> W;
    dp.assign(n + 1, vl(W + 1, -1));
    v.resize(n + 1), w.resize(n + 1);
    rep1(i, n) cin >> v[i] >> w[i];
    cout << rec(n, W) << endl;
    return 0;
}
