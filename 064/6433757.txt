#include<stdio.h>
int da[12];
void sort(){
    int ind;
    for(int s=0;s<12;s++){
        int m=105;
        for(int i=s;i<12;i++){
            if(m>da[i]){
                m=da[i];
                ind=i;
            };
        }
        int t=da[s];
        da[s]=m;
        da[ind]=t;
    }
}
int main(void){
    for(int i=0;i<12;i++) scanf("%d",&da[i]);
    sort();
    int j=1;
    for(int i=0;i<12;i+=4){
        if(da[i]!=da[i+3]) j=0;
    }
    printf(j?"yes\n":"no\n");
    return 0;
}
