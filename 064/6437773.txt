#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    int ans=1,max=a[n-1];
    for(int i=n-1;i>=0;i--){
        if(max<a[i]){
            ans++;
            max=a[i];
        }
    }
    cout<<ans<<endl;
    return 0;
}
