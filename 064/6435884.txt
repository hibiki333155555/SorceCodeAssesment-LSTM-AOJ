#include <bits/stdc++.h>
using namespace std;

bool isPrime(int N) {
    for(int i = 2;i * i <= N;i++) {
      if(N % i == 0) return false;
    }
  return true;
}

int main() {
    int n; cin >> n;
    vector<int> a(n);
    int count = 0;
    for(int i = 0;i < n;i++) cin >> a[i];
    for(int i = 0;i<n;i++) {
        bool flag = isPrime(a[i]);
        if(flag) {
            // cout << a[i] << " is prime" << endl;
            count++;
        }
    }
    cout << count << endl;
}
