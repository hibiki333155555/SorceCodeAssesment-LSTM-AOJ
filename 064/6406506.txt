#include <cstdio>
#include <iostream>
#include <string>
#include <list>
using namespace std;
int n;
list <int> a;
list <int> :: iterator it = a.begin ();
int main () {
    scanf ("%d",&n);
    while (n--) {
    	char s[20];
    	int x;
		scanf ("%s",s);
    	if (s[0] == 'i') {
			scanf ("%d",&x);
    		a.push_front (x);
		}
		else if (s[6] == 'L') a.pop_back ();
		else if (s[6] == 'F') a.pop_front ();
		else {
			scanf ("%d",&x);
			for (it = a.begin ();it != a.end ();it++) {
				if (*it == x) {
					a.erase (it);
					break;
				}
			}
		}
	}
	it = a.begin ();
	printf ("%d",*it);
	it++;
	for (;it != a.end ();it++) printf (" %d",*it);
	printf ("\n");
    return 0;
}
