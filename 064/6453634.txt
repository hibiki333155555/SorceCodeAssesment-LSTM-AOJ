#include <bits/stdc++.h>
#include <vector>
using namespace std;
int main()
{
	vector<int>vec;
	int n;
	cin>>n;
	int m,temp;
	for(int i=0;i<n;i++)
	{
		cin>>m;
		if(m==0){
			cin>>temp;
			vec.push_back(temp);
		}else if(m==1){
			cin>>temp;
			cout<<vec[temp]<<endl;
		}else if(m==2){
			vec.pop_back();
		}
	}
}
