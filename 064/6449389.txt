int main(){
    ll a,b; cin >> a >> b;
    if(a == 0 && b == 0){
        cout << 1 << endl;
        return 0;
    }
    vl A,B;
    while(a > 0){
        A.push_back(a%2);
        a /= 2;
    }
    while(b > 0){
        B.push_back(b%2);
        b /= 2;
    }
    int ans = 0;
    while(!A.empty() || !B.empty()){
        if(A.size() > B.size()){
            if(A.back() != 0) break;
            A.pop_back();
        }else if(A.size() == B.size()){
            if(A.back() != B.back()) break;
            A.pop_back(); B.pop_back();
        }else{
            if(B.back() != 0) break;
            B.pop_back();
        }
        ans++;
    }
    cout << ans << endl;
}
