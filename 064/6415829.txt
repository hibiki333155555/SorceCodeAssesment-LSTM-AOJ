#include<iostream>
#include<string>
using namespace std;
int main() {
    int Taro = 0;
    int Hanako = 0;
    int turn;
    cin >> turn;
    for (int n = 0; n < turn; n++) {
        string taro, hanako;
        cin >> taro >> hanako;
        if(taro == hanako) {
            Taro++;
            Hanako++;
        } else {
            int Taro2 = 0;
            int Hanako2 = 0;
            int t = taro.size();
            int h = hanako.size();
            int size;
            if(t < h) {
                Taro2 = 3;
                size = h;
            } else if(t > h) {
                Hanako2 = 3;
                size = t;
            }
            for(int m = 0; m < t; m++) {
                if(taro[m] > hanako[m]) {
                    if(Taro2 == 0) {
                        Taro2 = 3;
                        Hanako2 = 0;
                        break;
                    }
                } else if(taro[m] < hanako[m]) {
                    if(Hanako2 == 0) {
                        Hanako2 = 3;
                        Taro2 = 0;
                        break;
                    }
                } else {
                    continue;
                }
            }
            Taro += Taro2;
            Hanako += Hanako2;
        }
    }
    cout << Taro << " " << Hanako << endl;
}
