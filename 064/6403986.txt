#include <iostream>
#include <math.h>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;
int main() {

	while (true) {
		int a = 0;
		int n, x;
		cin >> n >> x;
		if (n == 0 && x == 0)break;
		for (int i = 1; i <= n; ++i) {
			for (int j = i + 1; j <= n; ++j) {
				for (int k = j + 1; k <= n; ++k) {
					if (i + j + k == x)a++;
				}
			}
		}
		cout << a << endl;
	}
	return 0;
}
