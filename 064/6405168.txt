import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
	Scanner sc=new Scanner(System.in);
	int k=0,m=0,y=0, n=sc.nextInt();
	int[] a=new int[1001];
	for(int i=1;i<n+1;i++) {
		a[i]=sc.nextInt();
	}
	for(int i=0;i<n;i++) {
		m=a[i]-a[i+1];
		y=a[i+1]-a[i];
		if(y<0) {
			y=y*-1;
		}
		if(m<0) {
			m=m*-1;
		}
		if(m>y) {
			k+=y;
		}else {
			k+=m;
		}
	}
	System.out.println(k);
	sc.close();
	}
}
