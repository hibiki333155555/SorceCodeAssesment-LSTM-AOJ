#include <iostream>
#include <string>
using namespace std;

int main()
{
string x;
int count[26] = {};
cin >> x;
for  (int i = 0; x[i] != '\0';i++)
  {
    if (x[i] >= 'a' and x[i] <='z')
    {
      count[x[i] - 'a']++;
    }
    else if (x[i] >= 'A' and x[i] <= 'z')
    {
      count[x[i] - 'A']++;
    }
  }
for (int j = 0; j < 26; j++)
  {
    x[j] = 'a' + j;
    cout << x[j] << " : " << count[j] << endl;
  }
}
