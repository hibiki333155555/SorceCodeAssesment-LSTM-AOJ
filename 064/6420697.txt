n = int(input())
A = [list(map(int,input().split())) for i in range(n)]
dic = {}
for i in range(n):
    for j in range(n):
        a = A[i][j]
        if a not in dic:
            dic[a] = (i+1,j+1)

q = int(input())
for _ in range(q):
    t = int(input())
    if t in dic:
        x,y = dic[t]
        print(x,y)
    else:
        print(-1)
