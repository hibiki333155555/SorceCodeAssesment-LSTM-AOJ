let [n, ...inputArr] = require(`fs`).readFileSync(`/dev/stdin`, `utf8`).split(`\n`);

let minVal = +inputArr[0];
let maxVal = inpuArr[1] - minVal;
for (let i=1; i<+n; i++) {
    maxVal = Math.max(maxVal, inputArr[i] - minVal);
    minVal = Math.min(minVal, +inputArr[i]);
}

console.log(maxVal - minVal);
