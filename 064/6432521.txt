#include <iostream>

using namespace std;

bool isPrime(int n)
{

  bool is_prime = true;
  // 0 and 1 are not prime numbers
  if (n == 0 || n == 1) {
    is_prime = false;
  }

  // loop to check if n is prime
  for (int i = 2; i <= n/2; ++i) {
    if (n % i == 0) {
      is_prime = false;
      break;
    }
  }
  return is_prime;
}
int main()
{
    int n, cnt = 0, temp;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> temp;
        if (isPrime(temp)) {
            ++cnt;
        }
    }

    cout << cnt << endl;
    return 0;
}
