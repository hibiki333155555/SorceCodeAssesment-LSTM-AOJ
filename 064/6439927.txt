#!/usr/bin/python3

from collections import defaultdict

N = int(input())
reports = []
for i in range(N):
    report = input()
    reports.append(report)
reportSize = len(reports[0])
dpLeft = []
dpRight = []
for i in range(N):
    dpLeft.append([])
    dpRight.append([])
    for j in range(reportSize + 1):
        dpLeft[i].append(0)
        dpRight[i].append(0)
for j in range(reportSize):
    lPs = []
    for i in range(N):
        lPs.append((dpLeft[i][j], reports[i][j], i))
    leftPairs = sorted(lPs)
    dpLeft[leftPairs[0][2]][j+1] = 0
    for i in range(1, N):
        val = dpLeft[leftPairs[i-1][2]][j+1]
        if leftPairs[i][0] != leftPairs[i-1][0] or leftPairs[i][1] != leftPairs[i-1][1] :
            val += 1
        dpLeft[leftPairs[i][2]][j+1] = val
for j in range(reportSize-1, -1, -1):
    rightPairs = []
    for i in range(N):
        rightPairs.append((dpRight[i][j+1], reports[i][j], i))
    rightPairs.sort()
    dpRight[rightPairs[0][2]][j] = 0
    for i in range(1, N):
        val = dpRight[rightPairs[i-1][2]][j]
        if rightPairs[i][0] != rightPairs[i-1][0] or rightPairs[i][1] != rightPairs[i-1][1] :
            val += 1
        dpRight[rightPairs[i][2]][j] = val
isCheater = []
for i in range(N):
    isCheater.append(0)
for j in range(reportSize):
    cnt = defaultdict(int)
    for i in range(N):
        cnt[(dpLeft[i][j], dpRight[i][j+1])] += 1
    for i in range(N):
        if cnt[(dpLeft[i][j], dpRight[i][j+1])] > 1:
            isCheater[i] = 1
ans = 0
for i in range(N):
    ans += isCheater[i]
print(ans)
