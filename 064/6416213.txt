import scala.io.StdIn

object Main {
  def main(args: Array[String]): Unit = {
    println(
      StdIn
        .readLine()
        .split(" ")
        .map(_.toLong)
        .zip(StdIn.readLine().split(" ").map(_.toLong))
        .map { case (a, b) =>
          (a + b - 1) / b
        }
        .reduce(_ * _)
    )
  }
}
