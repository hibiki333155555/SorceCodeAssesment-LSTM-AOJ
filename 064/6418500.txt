#include <bits/stdc++.h>
using namespace std;

int main(){
  int r,c;
  cin >> r >> c;
  vector<vector<int>> table(r+1,vector<int>(c+1));
  
  for(int i=0; i<r; i++){
    int s=0;
    for(int j=0; j<c; j++){
      int x;
      cin >> x;
      table.at(i).at(j) = x;
      s += table.at(i).at(j);
    }
    table.at(i).at(c) = s;
  }
  
  for(int i=0; i<c+1; i++){
    int s=0;
    for(int j=0; j<r; j++){
      s += table.at(j).at(i);
    }
    table.at(r).at(i) = s;
  }
  
  for(int i=0; i<r+1; i++){
    for(int j=0; j<c+1; j++){
      if(j==0) cout << table.at(i).at(j);
      else cout << " " << table.at(i).at(j);
    }
    cout << endl;
  }
  return 0;
        
}
