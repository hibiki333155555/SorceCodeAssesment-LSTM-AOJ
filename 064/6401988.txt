let x = require(`fs`).readFileSync(`/dev/stdin`, `utf-8`).split(`\n`).map(item => item.split(` `));
//console.log(x);
for (let i=0; i<x.length; i++) {
    if (x[i][0] == 0 && x[i][1] == 0) {
        break;
    } else {
        x[i].sort((a, b) => a - b);
        console.log(`${x[i][0]} ${x[i][1]}`);
    }
}
