//分析见书《挑战程序设计竞赛2》上P96

int binarySearch(int A[],int left,int right,int x)
{
	int mid=(left+right)/2;
	while(left<right)//搜索区间不为空 
	{
		if(x==A[mid])
			return 1;
		else if(A[mid]>x)
			return binarySearch(A,left,mid,x);
		else if(A[mid]<x)
			return binarySearch(A,mid+1,right,x);//注意+1 
	}
	return 0;
}

int main()
{	
	int n,q,i,num;
	int sum=0;
	int S[100005];
	
	cin>>n;
	for(i=0;i<n;i++)
		cin>>S[i];
	cin>>q;
	for(i=0;i<q;i++)
	{
		cin>>num;
		if(binarySearch(S,0,n,num)==1)
			sum++;
	}
	cout<<sum<<endl;
	
	
return 0;
}
