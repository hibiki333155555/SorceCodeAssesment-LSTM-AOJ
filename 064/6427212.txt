// G - Sum of Numbers

#include <iostream>
using namespace std;

int main() {
    int n;
    int sum =0;
    while (cin >> n)
    {
        if (n!=0) {
            int l = to_string(n).length();
            for (int i=0; i<l; i++) {
                sum = sum + (n%10);
                n = n/10;
            }
        }
        else break;
        cout << sum << endl;
        sum =0;
    }
}
