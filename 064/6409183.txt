#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include<string.h>

int main()
{
    int chtable[256]={0};
    char ch;
    while(ch=getchar() != EOF){//chに文字が読み込まれている間
        chtable[ch]++;//すべての種類の文字がその文字コードに対応する場所がインクリメントされる
    }
    for(int i=97; i<=122; i++){//chtableの中で小文字は97~122
    printf("%c : %d\n",i,chtable[i]+chtable[i-32]);//大文字部分も足す

    }
	return 0;
}
