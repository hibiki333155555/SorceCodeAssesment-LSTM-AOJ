#include<iostream>
#include<algorithm>
#include<cstdio>
#include<string>
#include<climits>
#include<vector>
#define check cout<<"check"<<endl
#define rep(a,b,c)   for(int a = b ; a < c ;a++)
using namespace std;
#define maxw 1e9 + 1
using REF = vector< vector<int> > ;

int main(){
    int n,w;
    cin>>n>>w;
    vector< int > V(n);
    vector< int > W(n);
    int vsam = 0;
    rep(i,0,n){
        cin>>V[i]>>W[i];
        vsam += V[i];
    }
    vsam ++;
    //cout<<vsam<<endl;
    REF dp(n + 1,vector< int > (vsam,w + 1));
    rep(i,0,n + 1)dp[i][0] = 0;
    rep(i,0,n)
    {
        rep(j,1,vsam)
        {
            dp[i + 1][j] = dp[i][j];
            int x = j - V[i];
            if(x < 0)continue;
            int a = dp[i][x] + W[i];
            if(a < dp[i + 1][j])dp[i + 1][j] = a;
        }
    }
    int max = 0;
    rep(i,0,vsam)if(dp[n][i] != w + 1)max = i;
    cout<<max<<endl;

    return 0;
}
