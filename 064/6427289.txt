#include <bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    int i = 0;
    int a,b;
    while (i < n) {
        cin>>a>>b;
        if(a<b){
            swap(a,b);
        }
        if(a % b != 0) {
            cout<<a%b<<endl;
        } else {
            cout<<b<<endl;
        }
        i++;
    }
    return(0);
}
