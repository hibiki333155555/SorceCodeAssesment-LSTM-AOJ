int main(){
    cin.tie(nullptr);
    ios::sync_with_stdio(false);
    //cout << fixed << setprecision(20);
    string c1, c2;
    int r1, r2;
    cin >> c1 >> r1 >> c2 >> r2;
    int ans = r2 - r1 + 1;

    vector<string> s;
    s.reserve(27 + 27 * 27);
    for (char c = 'A';c <= 'Z'; c++) {
        string tmp = "";
        tmp += c;
        s.push_back(tmp);
    }
    for (char c1 = 'A';c1 <= 'Z'; c1++) {
        for (char c2 = 'A';c2 <= 'Z'; c2++) {
            string tmp = "";
            tmp += c1;
            tmp += c2;
            s.push_back(tmp);
        }
    }

    map<string, int> mp;
    for (int i = 0;i < (int)s.size(); i++) mp[s[i]] = i;
    ans *= mp[c2] - mp[c1] + 1;
    cout << ans << endl;



    return 0;
}
