using namespace std;
//分析见书《挑战程序设计竞赛2》上P

int main()
{	
	int n,i,j,t,sum=0;
	int a[200];
	int mi,flag,k;
	
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	
	for(i=0;i<n-1;i++)
	{
		mi=a[i];
		flag=0;
		for(j=i+1;j<n;j++)
		{
			if(a[j]<mi)
			{
				
				mi=a[j];
				flag=1;
				k=j;
			}
		
		}
		if(flag)
		{
			sum++;
			t=a[k];
			a[k]=a[i];
			a[i]=t;
		}
		
	}
	
	
	for(i=0;i<n;i++)
	{
		if(i==n-1)
			cout<<a[i]<<endl;
		else
		 	cout<<a[i]<<" ";
	
	}
	cout<<sum<<endl;
	
return 0;
}
