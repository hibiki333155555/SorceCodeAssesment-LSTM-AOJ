int main()
{	
	
	int n,i;
	double sum,a,m,vsum,v;
	double sc[1005];
	
	cin>>n;
	
	while(n!=0)
	{
		sum=0;
		vsum=0;
		for(i=0;i<n;i++)
		{
			cin>>sc[i];
			sum+=sc[i];
		}
		m=sum/n;
		
		for(i=0;i<n;i++)
		{
			vsum+=(sc[i]-m)*(sc[i]-m);
		}
		v=vsum/n;
		
		a=sqrt(v);
		
		printf("%.6f\n",a);
		
		cin>>n;
	}
	
	
return 0;
}
