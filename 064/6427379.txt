#include <bits/stdc++.h>
using namespace std;
int main(){
    bool b=1;
    while(b){
        int ans=INT_MAX;
        int n;
        cin>>n;
        if(n == 0) {
            b = 0;
        }
        int a;
        for(int i=0;i<n;i++){
            cin>>a;
            ans=min(a,ans);
        }
            if(ans>=2 && b){
                cout<<n+1<<endl;
            }else if(b){
                cout<<"NA"<<endl;
            }
    }
    return(0);
}
