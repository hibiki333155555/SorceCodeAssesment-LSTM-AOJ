#include<bits/stdc++.h>
using namespace std;
using ll = long long;
#define rep(i,n) for(int i=0; i<n; ++i)

vector<ll> memo(50, -1);

ll fib(ll n){
    if(memo[n] != -1) return memo[n];
    else return memo[n] = fib(n-1) + fib(n-2); 
}

int main(){
    ll n;
    cin >> n;

    memo[0] = memo[1] = (ll)1;

    cout << fib(n) << endl;
    return 0;
}
