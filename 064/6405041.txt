import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n=sc.nextInt();
        int a[]=new int[n];
        int z=0;
        int w=10;
        for(int i=0;i<n;i++){
            a[i]=sc.nextInt();
	}
        for(int x=0;x<n;x++) {
        	int y=w-a[x];
        	if(a[x]<y) {
        		z=+y;
        		w=w-y;
        	}else {
        		z=-a[x];
        		w=w+a[x];
        	}
        }
        System.out.println(z);
}
}
