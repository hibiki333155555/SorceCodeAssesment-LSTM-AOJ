#include <bits/stdc++.h>
#include <string>
using namespace std;

int main() {

    int a,b;
    char op;
    cin>>a>>op>>b;

while(true){
   if(op=='+'){
       cout<<a+b<<endl;
   }
   else if(op=='-'){
       cout<<a-b<<endl;
   }
   else if(op=='*'){
       cout<<a*b<<endl;
   }
   else if(op=='/'){
       cout<<a/b<<endl;
   }
   else if(op=='?')break;

}



    return 0;
}
