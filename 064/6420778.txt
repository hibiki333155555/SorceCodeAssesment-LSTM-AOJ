#include <stdio.h>

#define HASH 1000003
const int H_Mod = HASH;

typedef struct List {
	struct List *next;
	int v, x, y;
} list;

int main()
{
	int h, i, j, N, hn = 0, A;
	list *hash[HASH] = {}, hd[1000001], *p;
	scanf("%d", &N);
	for (i = 1; i <= N; i++) {
		for (j = 1; j <= N; j++) {
			scanf("%d", &A);
			h = A % H_Mod;
			for (p = hash[h]; p != NULL; p = p->next) if (p->v == A) break;
			if (p == NULL) {
				hd[hn].v = A;
				hd[hn].x = i;
				hd[hn].y = j;
				hd[hn].next = hash[h];
				hash[h] = &(hd[hn++]);
			}
		}
	}
	
	int Q, T;
	scanf("%d", &Q);
	for (i = 1; i <= Q; i++) {
		scanf("%d", &T);
		h = T % H_Mod;
		for (p = hash[h]; p != NULL; p = p->next) if (p->v == T) break;
		if (p != NULL) printf("%d %d\n", p->x, p->y);
		else printf("-1\n");
	}
	fflush(stdout);
	return 0;
}
