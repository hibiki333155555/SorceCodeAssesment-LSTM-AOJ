#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    string n;
    int m;
    cin >> n;
    cin >> m;
    while(n != "-")
    {
    for(int i = 0; i < m; i++)
    {
        int h = 0;
        cin >> h;
        n = n.substr(h,n.size()-1) + n.substr(0,h);
    }
    cout << n << endl;
    cin >> n;
    }

    return 0;
}
