#include<bits/stdc++.h>
using namespace std;

int main(){
    pair<int,int > a[6];
    for(int i=0;i<6;i++){
        cin>>a[i].first>>a[i].second;
        if(a[i].first>a[i].second)swap(a[i].first,a[i].second);
    }
    sort(a,a+6);
    if(a[0]==a[1]&&a[2]==a[3]&&a[4]==a[5]&&a[0].first==a[2].first&&a[2].second==a[4].second&&a[0].second==a[4].first){
        cout<<"yes"<<endl;
    }
    else cout<<"no"<<endl;
}
