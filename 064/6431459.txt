#define rep(i, n) for (int i = 0; i < (int)(n); i++)
#define srep(i, s, n) for (int i = (s); i < (int)(n); i++)
#include <bits/stdc++.h>
using namespace std;
using pii = pair<int, int>;
using vi = vector<int>;
using vvi = vector<vi>;
using ll = long long;

int main() {
    
    bool flag = true;
    
    while(flag) {
        int n, k;
        cin >> n >> k;
        
        if(n == 00) {
            flag = false;
        }
        
        vi a(n);
        rep(i, n) cin >> a[i];

        vi s(n + 1);
        rep(i, n) {
            s[i + 1] = s[i] + a[i];
        }


        int max = -10010;

        rep(l, n - k + 1) {
            if(max <= s[l + k] - s[l]) {
                max = s[l + k] - s[l];
            }
        }

        cout << max << endl;
    }
    
    
}
