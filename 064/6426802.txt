#include <bits/stdc++.h>

using namespace std;

int main() {
  deque<int> deq;
  char c;
  int first, second, result;
  while ((c = getchar()) != EOF) {
    if (c == ' ' || c == '\n') continue;
    if (c == '+') {
      first = deq.back();
      deq.pop_back();
      second = deq.back();
      deq.pop_back();
      result = first + second;
      deq.push_back(result);
    } else if (c == '-') {
      first = deq.back();
      deq.pop_back();
      second = deq.back();
      deq.pop_back();
      result = second - first;
      deq.push_back(result);
    } else if (c == '*') {
      first = deq.back();
      deq.pop_back();
      second = deq.back();
      deq.pop_back();
      result = second * first;
      deq.push_back(result);
    } else if (c == '/') {
      first = deq.back();
      deq.pop_back();
      second = deq.back();
      deq.pop_back();
      result = second / first;
      deq.push_back(result);
    } else {
      deq.push_back(c - '0');
    }
  }
  int print = deq.back();
  cout << print << endl;
}
