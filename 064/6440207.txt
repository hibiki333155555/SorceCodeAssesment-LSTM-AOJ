#include <iostream>
#include <math.h>
#include <cmath>
using namespace std;

int isP(int x){
    if (x < 2) {
        return 0;
    }
    
    if (x%2==0) {
        return 1;
    }
    
    int sqrtNum = trunc(sqrt(x)); 
    for(int j=3;j<=sqrtNum;j++) {
        if (x%j == 0) {
            return 0;
        }
    }
    return 1;
}

int main()
{
    int n,num,count=0;
    cin >> n;
    for(int i=0;i<n;i++) {
        cin >> num;
        if (isP(num) > 0) {
            count++;
        }
    }
    cout << count << endl;
}
