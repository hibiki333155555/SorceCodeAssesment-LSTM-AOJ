#include <iostream>
#include <string>
using namespace std;

void print(string s, int a, int b) {
    cout << s.substr(a, b - a + 1) << endl;
    return;
}

string reverse(string s, int a, int b) {
    string temp = s.substr(a, b - a + 1);
    for (int i = a; i <= b; i++) {
        s[a] = temp[b - i];
    }
    return s;
}

string replace(string s, int a, int b, string p) {
    s = s.replace(a, b - a + 1, p);
    return s;
}

int main() {
    string s, cmd, p;
    int q, a, b;
    cin >> s >> q;
    for (int i = 0; i < q; i++) {
        cin >> cmd >> a >> b;
        if (cmd == "print") print(s, a, b);
        else if (cmd == "reverse") reverse(s, a, b);
        else {
            cin >> p;
            replace(s, a, b, p);
        }
    }
    return 0;
}
