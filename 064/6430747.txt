#include<stdio.h>
#include<math.h>
#include<iostream>

using namespace std;

int main() {
	int n, m;
	cin >> n >> m;
	int a[100][100],b[100];
	int A, B;
	int sum = 0;
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			cin >> A;
			a[i][j] = A;
		}
	}
	for (int i = 0; i < m; ++i) {
		cin >> B;
		b[i] = B;
	}
	for (int i = 0; i < n; ++i) {
		sum = 0;
		for (int j = 0; j < m; ++j) {
			sum += a[i][j] * b[j];
		}
		cout << sum <<endl;
	}
	return 0;
}
