#include <stdio.h>
#include <math.h>
#include <complex.h>

int main()
{
    int deg,length,addDeg;
    double _Complex ang,sum,add;
    sum = 0;deg = 0;
    
    while(1)
    {
        scanf("%d,",&length);
        ang=I*deg/180*M_PI;
        add = length*cexp(ang);
        sum += add;
        //printf("len=%d,ang=%d,ang=%f,re=%f,im=%f\n",length,deg,cimag(ang),creal(add),cimag(add));
        scanf("%d",&addDeg);
        deg-=addDeg;
        if(length*addDeg==0) break;
    }
    printf("%d\n%d\n",(int)(-1*cimag(sum)),(int)(creal(sum)));
}
