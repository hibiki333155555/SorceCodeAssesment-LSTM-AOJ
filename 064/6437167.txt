#include <bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin >> n;

  char pat;
  vector<int> S(14);
  vector<int> H(14);
  vector<int> C(14);
  vector<int> D(14);
  int iS = 1;
  int iH = 1;
  int iC = 1;
  int iD = 1;

  for(int i = 0;i < n;i++){
    cin >> pat;
    if(pat == 'S'){
      cin >> S[iS];
      iS++;
    }else if(pat == 'H'){
      cin >> H[iH];
      iH++;
    }else if(pat == 'C'){
      cin >> C[iC];
      iC++;
    }else if (pat == 'D'){
      cin >> D[iD];
      iD++;
    }
  }

  iS = 1;
  iH = 1;
  iC = 1;
  iD = 1;

  sort(S.begin(),S.end());
  sort(H.begin(),H.end());
  sort(C.begin(),C.end());
  sort(D.begin(),D.end());

  for(int i = 1;i <= S.size();i++){
    if(S[iS] != i){
      cout << 'S' << " " << i << endl;
    }else{
      iS++;
    }
  }
  for(int i = 1;i <= H.size();i++){
    if(H[iH] != i){
      cout << 'H' << " " << i << endl;
    }else{
      iH++;
    }
  }
  for(int i = 1;i <= C.size();i++){
    if(C[iC] != i){
      cout << 'C'<< " " << i << endl;
    }else{
      iC++;
    }
  }
  for(int i = 1;i <= D.size();i++){
    if(D[iS] != i){
      cout << 'D'<< " " << i << endl;
    }else{
      iD++;
    }
  }

  return 0;
}
