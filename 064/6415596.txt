 #include <iostream>
    #include <cstdio>
    #include <algorithm>
    using namespace std;

    const int maxn = 100000 + 8;

    int n,sum = 0;
    int a[maxn];

    bool find(int key){
      int l = 0,r = n;
      while(l < r){
        int mid = (l+r)>>1;
        if(a[mid] < key){
          l = mid+1;
        }
        else if(a[mid] > key){
          r = mid;
        }
        else{
          return true;
        }
      }
      return false;
    }

    int main(){
      scanf("%d",&n);
      for(int i = 0;i < n; i++){
        scanf("%d",&a[i]);
      }
      sort(a+1,a+n+1);
      int q;
      scanf("%d",&q);
      for(int i = 0;i < q; i++){
        int key;
        scanf("%d",&key);
        if(find(key)){
          sum++;
        }
      }
      printf("%d\n",sum);
    }
