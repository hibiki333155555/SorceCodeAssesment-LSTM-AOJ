def calculate_area_to_same_depth(grounds, fromIndex):
    allArea = 0
    depth = 0
    for i in range(fromIndex, len(grounds)):
        nowGround = grounds[i]
        prevDepth = depth
        if nowGround == "\\":
            depth -= 1
        elif nowGround == "/":
            depth += 1
        elif nowGround == "_":
            pass

        allArea += ((abs(prevDepth) + abs(depth)) * 1 / 2)

        if i != fromIndex and depth == 0:
            return [i, int(allArea)]

    return [None, None]


def main():
    from sys import stdin
    readline = stdin.readline
    grounds = str(readline())

    allAreas = []
    i = 0
    while i < len(grounds):
        ground = grounds[i]
        if ground == "\\":
            # 毎回探すのではなく、存在しない場合は存在する位置に移動するようにする
            endIndex, area = calculate_area_to_same_depth(grounds, i)
            if endIndex is not None and area is not None:
                allAreas.append(area)
                i = endIndex + 1
                continue
        elif ground == "/":
            pass
        elif ground == "_":
            pass
        i += 1

    print(sum(allAreas))
    print(' '.join([str(len(allAreas))] + list(map(lambda x: str(x), allAreas))))


if __name__ == '__main__':
    main()
