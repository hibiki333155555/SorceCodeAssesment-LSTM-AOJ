#include <iostream>
using namespace std;

int w[1000000] = {0};
int k, n;
int load_num(int p){
    int cnt=0, v=0;
    for(int i=0; i<n; i++){
        int sum=0;
        while(sum+w[v] <= p){
            sum += w[v];
            v++;
            if(v==k) return k;
        }
    }
    return v;
}
int main(){
    cin >> k >> n;
    int sum=0;
    for(int i=0; i<k; i++){
        cin >> w[i];
        sum += w[i];
    }
    int l=0, r=sum;
    while(l<r){
        int mid=(l+r)/2;
        if(load_num(mid) >= k) r=mid;
        else l = mid+1;
    }
    cout << r << endl;
    return 0;
}
