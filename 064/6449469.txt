#include<bits/stdc++.h>
using namespace std;


int main() {
    cin.tie(0);
    ios::sync_with_stdio(0);
    string sa, sb;
    int ia, ib;
    cin >> sa >> ia >> sb >> ib;
    int a;
    if (sa.size() == 1) {
        a = sa[0] - 'A';
    }
    else {
        a = (sa[0] - 'A' + 1) * 26 + (sa[1] - 'A');
    }
    int b;
    if (sb.size() == 1) {
        b = sb[0] - 'A';
    }
    else {
        b = (sb[0] - 'A' + 1) * 26 + (sb[1] - 'A');
    }
    cout << (b - a + 1) * (ib - ia + 1) << '\n';
    return 0;
}
