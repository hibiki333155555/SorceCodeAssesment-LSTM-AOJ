#include <iostream>
#include <iomanip>
#include <vector>
#include <cmath>

using std::cin;
using std::cout; using std::endl;
using std::vector;

int main()
{
    while ( true ) {
        int n;
        cin >> n;
        if ( !n )
            break;

        vector<int> s(n);
        int sum = 0;
        for (int i = 0;i < n;++i) {
            int tmp;
            cin >> tmp;
            sum += tmp;
            s[i] = tmp;
        }

        double m = (double)sum / n;

        double sum_ = 0;
        for (int i = 0;i < n;++i)
            sum_ += pow(s[i] - m, 2);

        double alpha = sqrt(sum_ / n);

        cout << std::fixed << std::setprecision(8)
             << alpha
             << endl;
    }
}
