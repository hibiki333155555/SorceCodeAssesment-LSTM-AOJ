#include<iostream>

using namespace std;

void print_edgeline(int n) {
    for (int i = 0; i < n; i++) {
        cout << "#";
    }
    cout << endl;
}

void print_middleline(int n) {
    cout << "#";
    for (int i = 1; i < n - 1; i++) {
        cout << ".";
    }
    cout << "#" << endl;
}

void print_rect_with_flame(int h, int w) {
    
    print_edgeline(w);
    for (int i = 1; i < h - 1; i++) {
        print_middleline(w);
    }
    print_edgeline(w);
}

int main() {
    int h, w;
    
    while (true) {
        cin >> h >> w;
        if (h == 0 && w == 0) {
            break;
        }
        print_rect_with_flame(h, w);
        cout << endl;
    }
    
    return 0;
}
