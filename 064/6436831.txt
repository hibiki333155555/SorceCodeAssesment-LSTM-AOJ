def resolve(s, t)
  count = 0
  t.each do |val|
    if contain?(s, val)
      count += 1
    end
  end
  puts count
end

def contain?(arr, target)
  map = {}
  left = 0
  right = arr.size - 1

  loop do
    mid = (left + right) / 2
    return true if arr[mid] == target

    if arr[mid] > target
      left = mid + 1
    else
      right = mid
    end

    return false if left >= right

    map[mid] = true
  end
end
