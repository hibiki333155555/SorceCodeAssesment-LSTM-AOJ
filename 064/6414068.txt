import java.util.Scanner;
class Main
{
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        Long[] num = new Long[n];
        
        for(int i = 0; i < num.length; i++)
        {
            num[i] = sc.nextInt();
        }
        //Minimun value
        int minVal = num[0];
        for(int i = 1; i < num.length; i++)
        {
            if(num[i] < minVal) minVal = num[i];
        }
        //Maxmun value
        int maxVal = num[0];
        for(int i = 1; i < num.length; i++)
        {
            if(num[i] > maxVal) maxVal = num[i];
        }
        //Summary value
        Long sumVal = 0;
        for(int i = 0; i < num.length; i++)
        {
            sumVal = sumVal + num[i];
        }
        System.out.println(minVal + " " + maxVal + " " + sumVal);
    }
}
