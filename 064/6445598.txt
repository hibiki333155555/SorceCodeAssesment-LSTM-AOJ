#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef unsigned long long ul;
typedef long double ld;
#define rep(i, n) for (int i = 0; i < n; i++)

int main() {
  ll n;
  while (cin >> n, n) {
    ll count = 0;
    int migi = 0, hidari = 0;
    rep(i, n) {
      string s;
      cin >> s;
      if (s == "lu") {
        migi++;
        if (hidari == 1) {
          count++;
          migi = 0;
          hidari = 0;
        }
      } else if (s == "ru") {
        hidari++;
        if (migi == 1) {
          count++;
          migi = 0;
          hidari = 0;
        }
      } else if (s == "ld") {
        migi--;
        if (hidari == -1) {
          count++;
          migi = 0;
          hidari = 0;
        }
      } else {
        hidari--;
        if (migi == -1) {
          count++;
          migi = 0;
          hidari = 0;
        }
      }
    }
    cout << count << endl;
  }
}
