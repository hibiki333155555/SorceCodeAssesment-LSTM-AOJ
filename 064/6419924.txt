class Program {
private:

    bool max(int &a, const int b) {
        if (a < b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

    bool min(int &a, const int b) {
        if (a > b) {
            a = b;
            return true;
        } else {
            return false;
        }
    }

public:

    Program() {
    }

    void solve() {
        int a, b; cin >> a >> b;

        if (b > 0) {
            cout << a + b << endl;
            cout << a - b << endl;
        } else {
            cout << a - b << endl;
            cout << a + b << endl;
        }
    }
};
